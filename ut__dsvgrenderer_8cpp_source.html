89 static bool testPixmapHasData(const QPixmap &pixmap)
90 {
91  QImage image = pixmap.toImage();
92 
93  image.reinterpretAsFormat(QImage::Format_RGB32);
94  const QRgb *bits = reinterpret_cast<const QRgb *>(image.constBits());
95  const QRgb *end = bits + image.byteCount() / sizeof(QRgb);
96  return !std::all_of(bits, end, [](QRgb r) { return r == QColor(Qt::green).rgb(); });
97 }
