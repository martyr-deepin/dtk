98 
99  if (DWindowManagerHelper::instance()->hasBlurWindow()) {
100  EXPECT_TRUE(pHandle->setWindowBlurAreaByWM(window, wmAreaVector));
101 
102  QPainterPath pPath;

104  EXPECT_TRUE(pHandle->setWindowBlurAreaByWM(window, {pPath}));
105  EXPECT_TRUE(pHandle->setWindowBlurAreaByWM(wmAreaVector));

113  }
