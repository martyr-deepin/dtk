49 void TestPrintPreviewDialog::slotPaintRequestSync(DPrinter *printer)
50 {
51  enum { pageCount = 4 };
52 
53  QPainter painter;
54  painter.begin(printer);
55  for (int i = 0; i < pageCount; ++i) {
56  const QRect f = printer->pageRect(QPrinter::DevicePixel).toRect();
57  painter.fillRect(f, Qt::white);
58  painter.drawText(f.center(), QString::fromLatin1("Page %1").arg(i + 1));
59  if (i != pageCount - 1)
60  printer->newPage();
61  }
62  painter.end();
63 }

65 void TestPrintPreviewDialog::slotPaintRequestAync(DPrinter *printer, const QVector<int> &pageRange)
66 {
67  enum { pageCount = 5 };
68 
69  QPainter painter;
70  painter.begin(printer);
71  for (int i = 0; i < pageCount; ++i) {
72  if (!pageRange.contains(i + 1))
73  continue;
74 
75  const QRect f = printer->pageRect(QPrinter::DevicePixel).toRect();
76  painter.fillRect(f, Qt::white);
77  painter.drawText(f.center(), QString::fromLatin1("Page %1").arg(i + 1));
78  if (i != pageCount - 1)
79  printer->newPage();
80  }
81  painter.end();
82 }
