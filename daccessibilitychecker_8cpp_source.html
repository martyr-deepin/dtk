84 bool DAccessibilityCheckerPrivate::check()
85 {
86  if (topLevelWidgets.isEmpty()) {

88  return true;

105 void DAccessibilityCheckerPrivate::checkWidgetName()
106 {
107  D_Q(DAccessibilityChecker);
108 
109  QWidgetList childrenList(topLevelWidgets);
110  for (const QWidget *topLevelWidget : topLevelWidgets)
111  childrenList.append(topLevelWidget->findChildren<QWidget *>());
112 
113  for (auto child : qAsConst(childrenList)) {
114  if (q->isIgnore(DAccessibilityChecker::Widget, child)) {
115  widgetIgnoredCount++;
116  continue;
117  }
118 
119  const QAccessibleInterface *interface = QAccessible::queryAccessibleInterface(child);
120  bool hasNoText;
121  if (interface && interface->isValid()) {
122  hasNoText = interface->text(QAccessible::Name).isEmpty();
123  } else {
124  hasNoText = child->accessibleName().isEmpty();
125  }
126 
127  switch (outputFormat) {
128  case DAccessibilityChecker::AssertFormat: {
129  Q_ASSERT_X(hasNoText, "Widget Accessible Name Check", widgetOutputLog(child).toLocal8Bit());
130  }
131  break;
132  case DAccessibilityChecker::FullFormat: {
133  if (hasNoText)
134  widgetsWarningList.append(widgetOutputLog(child));
135  }
136  break;
137  default:
138  break;
139  }
140  }
141 }

196 bool DAccessibilityCheckerPrivate::checkViewItemNameFromAccessibleInteface(QAbstractItemView *listview)
197 {
198  auto tableAbsInterface = QAccessible::queryAccessibleInterface(listview);
199  bool ret = false;
200 
201  if (!tableAbsInterface || !tableAbsInterface->isValid())
202  return ret;
203 
204  QAccessibleTableInterface *tableInterface = tableAbsInterface->tableInterface();
205  if (!tableInterface)
206  return ret;
207 
208  ret = true;
209  for (int rowIdx = 0; rowIdx < tableInterface->rowCount(); ++rowIdx) {
210  for (int columnIdx = 0; columnIdx < tableInterface->columnCount(); ++columnIdx) {
211  QAccessibleInterface *cellAbsInterface = tableInterface->cellAt(rowIdx, columnIdx);
212  if (cellAbsInterface && cellAbsInterface->isValid()) {
213  switch (outputFormat) {
214  case DAccessibilityChecker::AssertFormat: {
215  Q_ASSERT_X(!cellAbsInterface->text(QAccessible::Name).isEmpty(), "Item Accessible Text Check", viewItemOutputLog(rowIdx, columnIdx, listview).toLocal8Bit());
216  }
217  break;
218  case DAccessibilityChecker::FullFormat: {
219  if (cellAbsInterface->text(QAccessible::Name).isEmpty()) {
220  itemWariningList.append(viewItemOutputLog(rowIdx, columnIdx, listview));
221  }
222  }
223  break;
224  default:
225  break;
226  }
227  }
228  }
229  }
230 
231  return ret;
232 }

240 bool DAccessibilityCheckerPrivate::checkViewItemNameFromAccessibleText(QAbstractItemView *listview)
241 {
242  QStandardItemModel *model = qobject_cast<QStandardItemModel *>(listview->model());
243  bool ret = false;
244 
245  if (!model)
246  return ret;
247 
248  ret = true;
249  for (int rowIdx = 0; rowIdx < model->rowCount(); ++rowIdx) {
250  for (int columnIdx = 0; columnIdx < model->columnCount(); ++columnIdx) {
251  auto standardItem = model->item(rowIdx, columnIdx);
252 
253  if (!standardItem)
254  continue;
255 
256  switch (outputFormat) {
257  case DAccessibilityChecker::AssertFormat: {
258  Q_ASSERT_X(!standardItem->accessibleText().isEmpty(), "Item Accessible Text Check", QStringLiteral("\n%1\n").arg(viewItemOutputLog(rowIdx, columnIdx, listview, standardItem->text())).toLocal8Bit());
259  }
260  break;
261  case DAccessibilityChecker::FullFormat: {
262  if (standardItem->accessibleText().isEmpty())
263  itemWariningList.append(viewItemOutputLog(rowIdx, columnIdx, listview, standardItem->text()));
264  }
265  break;
266  default:
267  break;
268  }
269  }
270  }
271 
272  return ret;
273 }

280 bool DAccessibilityCheckerPrivate::isDefaultIgnoreWidget(const QWidget *w) const
281 {
282  static QStringList defaultIgnoredWidgetObjectName = {
283  QLatin1String("qt_scrollarea_hcontainer"),
284  QLatin1String("qt_scrollarea_vcontainer"),
285  QLatin1String("qt_scrollarea_viewport"),
286  QLatin1String("qt_spinbox_lineedit"),
287  QLatin1String("qt_toolbar_ext_button")
288  };
289 
290  static QByteArrayList defaultIgnoredWidgetClassName = {
291  QByteArrayLiteral("QScrollBar"),
292  QByteArrayLiteral("QToolTip"),
293  QByteArrayLiteral("QComboBoxPrivateContainer"),
294  QByteArrayLiteral("QComboBoxListView"),
295  QByteArrayLiteral("QComboBoxPrivateScroller"),
296  QByteArrayLiteral("QColumnViewGrip")
297  };
298 
299  bool ret = defaultIgnoredWidgetObjectName.contains(w->objectName());
300 
301  if (!ret)
302  ret = std::any_of(defaultIgnoredWidgetClassName.begin(), defaultIgnoredWidgetClassName.end(), [w](const QByteArray &ignoreClass) -> bool { return w->inherits(ignoreClass); });
303 
304  return ret;

314 QString DAccessibilityCheckerPrivate::widgetInfoString(const QWidget *w) const
315 {
316  QString splices("%1 Widget Path: %2");
317 
318  QString classDetailString(QStringLiteral("Class Name: ") + w->metaObject()->className());
319  if (!w->objectName().isEmpty())
320  classDetailString.append(QStringLiteral(", Object Name: ") + w->objectName());
321 
322  QString classPathString(QStringLiteral(" ——► ") + w->metaObject()->className());
323  QWidget *parentWidget = w->parentWidget();
324  if (!parentWidget) {
325  classPathString = w->metaObject()->className();
326  } else {
327  while (parentWidget->parentWidget()) {
328  classPathString.prepend(parentWidget->metaObject()->className()).prepend(QStringLiteral(" ——► "));
329  parentWidget = parentWidget->parentWidget();
330  }
331 
332  classPathString.prepend(parentWidget->metaObject()->className());
333  }
334 
335  splices = splices.arg(classDetailString).arg(classPathString);
336  return splices;
337 }

365 void DAccessibilityCheckerPrivate::printSummaryResults()
366 {
367  int totalWidgetsCount = std::accumulate(this->topLevelWidgets.begin(), this->topLevelWidgets.end(), 0, [](int before, const QWidget *after) -> int { return before + after->findChildren<QWidget *>().count(); }) + this->topLevelWidgets.count();
368 
369  QString summary("[=============]Result Summary: Total Widgets Number: %1 Succeeded: %2 Failed: %3 Ignored: %4");
370  qWarning().noquote() << summary.arg(totalWidgetsCount).arg(totalWidgetsCount - this->widgetsWarningList.count() - this->widgetIgnoredCount).arg(this->widgetsWarningList.count()).arg(this->widgetIgnoredCount);
371 }
