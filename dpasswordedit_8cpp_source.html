125  q->lineEdit()->setEchoMode(QLineEdit::Password);

140 void DPasswordEditPrivate::_q_toggleEchoMode()
141 {
142  D_Q(DPasswordEdit);
143 
144  if (q->isEchoMode()) {
145  togglePasswordVisibleButton->setIcon(DStyle::standardIcon(q->style(),DStyle::SP_ShowPassword));
146  q->setEchoMode(q->lineEdit()->Password);

148  togglePasswordVisibleButton->setIcon(DStyle::standardIcon(q->style(), DStyle::SP_HidePassword));
