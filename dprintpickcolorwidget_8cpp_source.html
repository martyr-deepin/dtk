206  if (!btnColor) {
207  if (color.isValid()) {

286 QColor ColorLabel::getColor(qreal h, qreal s, qreal v)
287 {
288  int hi = int(h / 60) % 6;
289  qreal f = h / 60 - hi;
290 
291  qreal p = v * (1 - s);
292  qreal q = v * (1 - f * s);
293  qreal t = v * (1 - (1 - f) * s);
294 
295  if (hi == 0) {
296  return QColor(std::min(int(255 * p), 255), std::min(int(255 * q), 255), std::min(int(255 * v), 255));
297  } else if (hi == 1) {
298  return QColor(std::min(int(255 * t), 255), std::min(int(255 * p), 255), std::min(int(255 * v), 255));
299  } else if (hi == 2) {
300  return QColor(std::min(int(255 * v), 255), std::min(int(255 * p), 255), int(255 * q));
301  } else if (hi == 3) {
302  return QColor(std::min(int(255 * v), 255), std::min(int(255 * t), 255), std::min(int(255 * p), 255));
303  } else if (hi == 4) {
304  return QColor(std::min(int(255 * q), 255), std::min(int(255 * v), 255), std::min(int(255 * p), 255));
305  } else {
306  return QColor(std::min(int(255 * p), 255), std::min(int(255 * v), 255), std::min(int(255 * t), 255));
307  }
308 }

323 void ColorLabel::pickColor(QPoint pos)
324 {
325  if (pos.x() < 0 || pos.y() < 0 || pos.x() >= this->width() || pos.y() >= this->height()) {
326  return;
327  }
328 
329  QPixmap pickPixmap;
330  pickPixmap = this->grab(QRect(0, 0, this->width(), this->height()));
331  QImage pickImg = pickPixmap.toImage();
332 
333  if (!pickImg.isNull()) {
334  QRgb pickRgb = pickImg.pixel(pos);
335  m_pickedColor = QColor(qRed(pickRgb), qGreen(pickRgb), qBlue(pickRgb));
336  } else {
337  m_pickedColor = QColor(0, 0, 0);
338  }
339  Q_EMIT pickedColor(m_pickedColor);
340 }

345 QCursor ColorLabel::pickColorCursor()
346 {
347  int tipWidth = 11;
348  QPixmap cursorPix = QPixmap(QSize(tipWidth, tipWidth));
349  cursorPix.fill(QColor(Qt::transparent));
350 
351  QPen whitePen;
352  whitePen.setWidth(1);
353  whitePen.setCapStyle(Qt::FlatCap);
354  whitePen.setJoinStyle(Qt::RoundJoin);
355  whitePen.setColor(QColor(255, 255, 255, 255));
356 
357  QPen blackPen;
358  blackPen.setWidth(1);
359  blackPen.setCapStyle(Qt::FlatCap);
360  blackPen.setJoinStyle(Qt::RoundJoin);
361  blackPen.setColor(QColor(0, 0, 0, 125));
362 
363  QPainter painter(&cursorPix);
364  painter.setRenderHints(QPainter::Antialiasing | QPainter::SmoothPixmapTransform);
365  painter.setPen(blackPen);
366  painter.drawEllipse(cursorPix.rect().center(), tipWidth / 2 - 1, tipWidth / 2 - 1);
367  painter.drawEllipse(cursorPix.rect().center(), tipWidth / 2 - 3, tipWidth / 2 - 3);
368  painter.setPen(whitePen);
369  painter.drawEllipse(cursorPix.rect().center(), tipWidth / 2 - 2, tipWidth / 2 - 2);
370 
371  return QCursor(cursorPix, -1, -1);
372 }

374 void ColorLabel::paintEvent(QPaintEvent *)
375 {
376  QPainter painter(this);
377  painter.setRenderHints(QPainter::Antialiasing | QPainter::SmoothPixmapTransform);
378  QImage backgroundImage(this->width(), this->height(), QImage::Format_ARGB32);
379  for (qreal s = 0; s < this->width(); s++) {
380  for (qreal v = 0; v < this->height(); v++) {
381  QColor penColor = getColor(m_hue, s / this->width(), v / this->height());
382  if (!penColor.isValid()) {
383  continue;
384  }
385  backgroundImage.setPixelColor(int(s), this->height() - 1 - int(v), penColor);
386  }
387  }
388 
389  painter.drawImage(this->rect(), backgroundImage);
390 }
