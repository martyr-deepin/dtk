82 QString DLogManager::getlogFilePath()
83 {
84  // 不再构造时去设置默认logpath(且mkdir), 而在getlogPath时再去判断是否设置默认值
85  // 修复设置了日志路径还是会在默认的位置创建目录的问题
86  if (DLogManager::instance()->m_logPath.isEmpty()) {
87  if (QDir::homePath() == QDir::rootPath()) {
88  qWarning() << "unable to locate the cache directory."
89  << "logfile path is empty, the log will not be written.\r\n"
90  << (qgetenv("HOME").isEmpty() ? "the HOME environment variable not set" : "");
91  return QString();
92  }
93 
94  QString cachePath = QStandardPaths::standardLocations(QStandardPaths::CacheLocation).at(0);
95  if (!QDir(cachePath).exists()) {
96  QDir(cachePath).mkpath(cachePath);
97  }
98  DLogManager::instance()->m_logPath = DLogManager::instance()->joinPath(cachePath, QString("%1.log").arg(qApp->applicationName()));
99  }
100 
101  return QDir::toNativeSeparators(DLogManager::instance()->m_logPath);
102 }
