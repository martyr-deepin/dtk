180 QMap<int, QString> DDialogPrivate::scanTags(QString origin) const
181 {
182  QMap<int, QString> result;
183 
184  QRegExp re("<.*?>");
185  int index = origin.indexOf(re, 0);
186  int matchLength = 0;
187  while (index >= 0) {
188  result[index] = re.cap();
189  matchLength = re.matchedLength();
190  index = origin.indexOf(re, index + matchLength);
191  }
192 
193  return result;
194 }

196 // It's not a perfect solution to elide rich text, but it should be sufficient
197 // for working with rich text only change text token color.
198 QString DDialogPrivate::elideString(QString str, const QFontMetrics &fm, int width) const
199 {
200  QString trimmed = trimTag(str);
201  if (fm.width(trimmed) > width) {
202  QMap<int, QString> info = scanTags(str);
203  QString elided = fm.elidedText(trimmed, Qt::ElideMiddle, width);
204  int elideStart = elided.indexOf("…");
205  int elideLength = trimmed.length() - elided.length();
206  QList<int> indexes = info.keys();
207  std::sort(indexes.begin(), indexes.end());
208  for (int index : indexes) {
209  if (index <= elideStart) {
210  elided.insert(index, info[index]);
211  } else if (elideStart < index && index <= elideStart + elideLength) {
212  elided.insert(elideStart, info[index]);
213  elideStart += 1;
214  } else {
215  elided.insert(index - elideLength + 1, info[index]);
216  }
217  }
218  return elided;
219  } else {
220  return str;
221  }
222 }

598 void DDialog::insertButton(int index, QAbstractButton *button, bool isDefault)
599 {
600  D_D(DDialog);
601 
602  DVerticalLine *line = new DVerticalLine;
603  line->setObjectName("VLine");
604  line->setFixedHeight(30);
605 
606  d->buttonLayout->insertWidget(index * 2 , line);
607  d->buttonLayout->insertWidget(index * 2 + 1, button);
608  d->buttonList << button;
609  if (d->buttonList.count() == 1) {
610  int spacing = DStyleHelper(style()).pixelMetric(DStyle::PM_ContentsSpacing);
611  d->setSpacer(spacing);
612  }
613  line->show();
614  d->buttonLayout->itemAt(0)->widget()->hide();
615 
616  connect(button, SIGNAL(clicked(bool)), this, SLOT(_q_onButtonClicked()));
617 
618  if (isDefault) {
619  setDefaultButton(button);
620  }
621 
622  const QString &text = button->text();
623 
624  if (text.count() == 2) {
625  for (const QChar &ch : text) {
626  switch (ch.script()) {
627  case QChar::Script_Han:
628  case QChar::Script_Katakana:
629  case QChar::Script_Hiragana:
630  case QChar::Script_Hangul:
631  break;
632  default:
633  return;
634  }
635  }
636 
637  button->setText(QString().append(text.at(0)).append(QChar::Nbsp).append(text.at(1)));
638  }
639 }
