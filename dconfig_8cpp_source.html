135  virtual bool isValid() const override

140  virtual bool load(const QString &appid) override
141  {
142  if (configFile)

144 
145  configFile.reset(new DConfigFile(appid,owner->name, owner->subpath));
146  configCache.reset(configFile->createUserCache(getuid()));

157 
158  virtual QVariant value(const QString &key, const QVariant &fallback) const override

161  return v.isValid() ? v : fallback;

167  Q_EMIT owner->q_func()->valueChanged(key);

169  }
170 
171  virtual QString name() const override

174  }

210 class Q_DECL_HIDDEN DBusBackend : public DConfigBackend
211 {
212 public:
213  explicit DBusBackend(DConfigPrivate* o):
214  owner(o)
215  {
216  }
217 
218  virtual ~DBusBackend() override;
219 
220  static bool isServiceRegistered()
221  {
222  return QDBusConnection::systemBus().interface()->isServiceRegistered(DSG_CONFIG);
223  }
224 
225  virtual bool isValid() const override
226  {
227  return config && config->isValid();
228  }
229 
236  virtual bool load(const QString &appid) override
237  {
238  if (config)
239  return true;
240 
241  qCDebug(cfLog, "Try acquire config manager object form DBus");
242  DSGConfig dsg_config(DSG_CONFIG, "/", QDBusConnection::systemBus());
243  QDBusPendingReply<QDBusObjectPath> dbus_reply = dsg_config.acquireManager(appid, owner->name, owner->subpath);
244  const QDBusObjectPath dbus_path = dbus_reply.value();
245  if (dbus_reply.isError() || dbus_path.path().isEmpty()) {
246  qCWarning(cfLog, "Can't acquire config manager. error:\"%s\"", qPrintable(dbus_reply.error().message()));
247  return false;
248  } else {
249  qCWarning(cfLog(), "dbus path=\"%s\"", qPrintable(dbus_path.path()));
250  config.reset(new DSGConfigManager(DSG_CONFIG_MANAGER, dbus_path.path(),
251  QDBusConnection::systemBus(), owner->q_func()));
252  if (!config->isValid()) {
253  qCWarning(cfLog(), "Can't acquire config path=\"%s\"", qPrintable(dbus_path.path()));
254  config.reset();
255  return false;
256  } else {
257  QObject::connect(config.data(), &DSGConfigManager::valueChanged, owner->q_func(), &DConfig::valueChanged);
258  }
259  }
260  return true;
261  }
262 
263  virtual QStringList keyList() const override
264  {
265  return config->keyList();
266  }
267 
268  virtual QVariant value(const QString &key, const QVariant &fallback) const override
269  {
270  const QDBusVariant &dv = config->value(key);
271  const QVariant &v = dv.variant();
272  return v.isValid() ? v : fallback;
273  }
274 
275  virtual void setValue(const QString &key, const QVariant &value) override
276  {
277  config->setValue(key, QDBusVariant(value));
278  }
279 
280  virtual QString name() const override
281  {
282  return QString("DBusBackend");
283  }
284 
285 private:
286  QScopedPointer<DSGConfigManager> config;
287  DConfigPrivate* owner;
288 };

293  config->release();
294  }

371 DConfigBackend *DConfigPrivate::getOrCreateBackend()
372 {
373  if (backend) {
374  return backend.data();
375  }
376  if (auto backendEnv = createBackendByEnv()) {
377  backend.reset(backendEnv);
378  return backend.data();
379  }
380 #ifndef D_DISABLE_DCONFIG
381 #ifndef D_DISABLE_DBUS_CONFIG
382  if (DBusBackend::isServiceRegistered()) {
383  qCDebug(cfLog, "Fallback to DBus mode");
384  backend.reset(new DBusBackend(this));
385  } else {
386  qCDebug(cfLog, "Can't use DBus config service, fallback to DConfigFile mode");
387  backend.reset(new FileBackend(this));
388  }
389 #else
390  backend.reset(new FileBackend(this));
391 #endif //D_DISABLE_DBUS_CONFIG
392 #else
393  qCDebug(cfLog, "Fallback to QSettings mode");
394  backend.reset(new QSettingBackend(this));
395 #endif //D_DISABLE_DCONFIG
396  return backend.data();
397 }

406 DConfigBackend *DConfigPrivate::createBackendByEnv()
407 {
408  const QByteArray &envBackend = qgetenv("DSG_DCONFIG_BACKEND_TYPE");
409  if (!envBackend.isEmpty()) {
410  if (envBackend == "DBusBackend") {
411 
412 #ifndef D_DISABLE_DCONFIG
413 #ifndef D_DISABLE_DBUS_CONFIG
414  if (DBusBackend::isServiceRegistered()) {
415  qCDebug(cfLog, "Fallback to DBus mode");
416  return new DBusBackend(this);
417  }
418 #endif //D_DISABLE_DBUS_CONFIG
419 #endif //D_DISABLE_DCONFIG
420  } else if (envBackend == "FileBackend") {
421 
422 #ifndef D_DISABLE_DCONFIG
423  qCDebug(cfLog, "Fallback to DConfigFile mode");
424  return new FileBackend(this);
425 #endif //D_DISABLE_DCONFIG
426  } else {
427 
428 #ifndef D_DISABLE_DCONFIG
429 #else
430  qCDebug(cfLog, "Fallback to QSettings mode");
431  return new QSettingBackend(this);
432 #endif //D_DISABLE_DCONFIG
433  }
434  }
435  return nullptr;
436 }
