641 void DVideoWidget::paintEvent(QPaintEvent *)
642 {
643  D_DC(DVideoWidget);
644 
645  QPainter painter(this);
646 
647  const QMediaPlaylist *pl = d->player ? d->player->playlist() : NULL;
648  bool loop = pl && (pl->playbackMode() == QMediaPlaylist::Loop || pl->playbackMode() == QMediaPlaylist::CurrentItemInLoop);
649  QVideoFrame frame = (!loop || d->formatProxy->m_currentFrame.isValid()) ? d->formatProxy->m_currentFrame : d->formatProxy->m_lastFrame;
650 
651  frame.map(QAbstractVideoBuffer::ReadOnly);
652  QImage image(
653  frame.bits(),
654  frame.width(),
655  frame.height(),
656  frame.bytesPerLine(),
657  QVideoFrame::imageFormatFromPixelFormat(frame.pixelFormat()));
658  frame.unmap();
659 
660  if (image.isNull()) {
661  return;
662  }
663 
664  image.setDevicePixelRatio(d->ratio);
665  image = image.scaled(size() * d->scale * d->ratio, d->aspectRatioMode, Qt::SmoothTransformation);
666  image = image.mirrored(d->mirroredHorizontal, d->mirroredVertical);
667 
668  if (d->round) {
669  QPainterPath path;
670  int diameter = qMin(width(), height());
671  path.addEllipse(width()/2.0-diameter/2.0, height()/2.0-diameter/2.0, diameter, diameter);
672  painter.setRenderHint(QPainter::Antialiasing);
673  painter.setRenderHint(QPainter::SmoothPixmapTransform);
674  painter.setClipPath(path);
675  }
676 
677  painter.drawImage(QRectF(rect()).center() - QRectF(image.rect()).center() / d->ratio, image);
678 }
