74  void SelectedTest(bool isSelect) {
75  switch (type) {
76  case selectAll: {
77  ASSERT_TRUE(isSelect);
78  }
79  break;
80  case selectPrev: {
81  if (idx == 2) {
82  ASSERT_TRUE(isSelect);
83  } else {
84  ASSERT_FALSE(isSelect);
85  }
86  }
87  break;
88  case selectLast: {
89  if (idx == 3) {
90  ASSERT_TRUE(isSelect);
91  } else {
92  ASSERT_FALSE(isSelect);
93  }
94  }
95  break;
96  case selectFirst: {
97  if (idx == 0) {
98  ASSERT_TRUE(isSelect);
99  } else {
100  ASSERT_FALSE(isSelect);
101  }
102  }
103  break;
104  case selectSingle: {
105  static int count = 0;
106  if (isSelect) {
107  count++;
108  }
109  ASSERT_TRUE(count < 2);
110  }
111  break;
112  default:
113  ASSERT_TRUE(false);
114  }
115  }
