77  static QString resolveFromPSM(const QString &moduleName)
78  {
79  QFile f("/proc/self/maps");
80  if (!f.open(QIODevice::ReadOnly))
81  qFatal("%s", f.errorString().toLocal8Bit().data());
82 
83  QString versoinName;
84  QByteArray data = f.readAll();
85  QTextStream ts(data);
86  while (Q_UNLIKELY(!ts.atEnd())) {
87  const QString line = ts.readLine();
88  const QStringList &maps = line.split(' ', QString::SplitBehavior::SkipEmptyParts);
89  if (Q_UNLIKELY(maps.size() < 6))
90  continue;
91 
92  QFileInfo info(maps.value(5));
93  if (!info.fileName().contains(moduleName))
94  continue;
95 
96  int idx = info.fileName().indexOf(moduleName) + moduleName.length();
97  int idx2 = info.fileName().indexOf(QLatin1String(".so"));
98  versoinName = info.fileName().mid(idx, idx2 - idx);
99  break;
100  }
101  return versoinName;
102  }

104  static QString pluginName(const QString &baseName)
105  {
106  QString pluginName = baseName;
107  QFunctionPointer versionStrFunc = resolve(VERSION_STR_SYMBOL);
108  if (versionStrFunc) {
109  QString versionStr = reinterpret_cast<const char *(*)()>(versionStrFunc)();
110  // 5.5.3.01 ==> 5.5
111  const QRegularExpression reg(R"(\d+\.\d+)");
112  QRegularExpressionMatch rem = reg.match(versionStr);
113  if (rem.hasMatch())
114  pluginName += rem.captured();
115  else
116  qCDebug(lcDPlugin) << versionStr << "is invalid";
117  } else {
118  qCDebug(lcDPlugin) << VERSION_STR_SYMBOL << "resolve failed, trying to read self maps";
119  QString versoinName = resolveFromPSM(QLatin1String("dtkcore"));
120  if (!versoinName.isEmpty())
121  pluginName += versoinName;
122  else
123  qCDebug(lcDPlugin) << "versoinName is empty";
124  }
125 
126  qCDebug(lcDPlugin) << "baseName:" << baseName << "pluginName:" << pluginName;
127  return pluginName;
128  }

130  template<class FactoryInterface>
131  static FactoryInterface *load(const QString &baseName)
132  {
133  FactoryInterface *infc = nullptr;
134  QFileInfo soInfo(soAbsoluteFilePath());
135  if (!soInfo.exists()) {
136  qCWarning(lcDPlugin) << soInfo.absoluteFilePath() << " not existed!";
137  return infc;
138  }
139 
140  QString realTargetPath = soInfo.absolutePath() + QString("/%1/").arg(BASED_DTK_DIR) + pluginName(baseName) + QLatin1String(".so");
141  QPluginLoader pluginLoader(realTargetPath);
142  QObject *plugin = pluginLoader.instance();
143  if (plugin) {
144  qCDebug(lcDPlugin) << realTargetPath << "loaded successfully";
145 
146  infc = qobject_cast<FactoryInterface *>(plugin);
147  /*
148  * Unless unload() was called explicitly, the plugin stays in memory
149  * until the application terminates.
150  */
151  if (!infc) {
152  QString msg = QString("qobject_cast to %1 failed").arg(typeid(FactoryInterface).name());
153  qCDebug(lcDPlugin) << msg;
154 
155  pluginLoader.unload();
156  }
157  } else {
158  qCWarning(lcDPlugin) << pluginLoader.errorString();
159  }
160 
161  return infc;
162  }
