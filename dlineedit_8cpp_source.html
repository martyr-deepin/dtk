179 {
180  Q_D(DLineEdit);
181 
182  if (d->leftWidget != nullptr) {
183  d->leftWidget->hide();
184  d->leftWidget->deleteLater();

186  }
187 
188  if (list.isEmpty())
189  return;
190 
191  d->leftWidget = new QWidget;

193  layout->setContentsMargins(0, 0, 0, 0);
194  d->hLayout->insertWidget(0, d->leftWidget);
195 
196  QList<QWidget *>::const_iterator itor;
197 
198  for (itor = list.constBegin(); itor != list.constEnd(); itor++) {
199  layout->addWidget(*itor);
200  }
201 
202  d->leftWidget->adjustSize();
203 }

474 bool DLineEdit::eventFilter(QObject *watched, QEvent *event)
475 {
476  D_D(DLineEdit);
477 
478  if (event->type() == QEvent::FocusIn) {
479  if (DGuiApplicationHelper::IsTableEnvironment && QGuiApplication::inputMethod())
480  QGuiApplication::inputMethod()->show();
481  Q_EMIT focusChanged(true);
482  } else if (event->type() == QEvent::FocusOut) {
483  Q_EMIT focusChanged(false);
484  } else if (watched == lineEdit() && event->type() == QEvent::KeyPress) {
485  QKeyEvent *pKeyEvent = static_cast<QKeyEvent *>(event);
486 
487  if (pKeyEvent == QKeySequence::Copy && !copyEnabled()) {
488  return true;
489  }
490  if (pKeyEvent == QKeySequence::Cut && !cutEnabled()) {
491  return true;
492  }
493 
494  if (pKeyEvent == QKeySequence::SelectAll) {
495  QApplication::clipboard()->setText(lineEdit()->text(), QClipboard::Mode::Selection);
496  }
497  } else if (watched == lineEdit() && event->type() == QEvent::ContextMenu && lineEdit()->contextMenuPolicy() == Qt::DefaultContextMenu) {
498  QLineEdit *le = static_cast<QLineEdit *>(watched);
499  if (!le->isEnabled() || le->echoMode() == QLineEdit::Password ||
500  (!textToSpeechIsEnabled() && !textToTranslateIsEnabled() && !speechToTextIsEnabled())) {
501  return QWidget::eventFilter(watched, event);
502  }
503 
504  QDBusInterface testSpeech("com.iflytek.aiassistant",
505  "/aiassistant/tts",
506  "com.iflytek.aiassistant.tts",
507  QDBusConnection::sessionBus());
508  //测试朗读接口是否开启
509  QDBusReply<bool> speechReply = testSpeech.call(QDBus::AutoDetect, "getTTSEnable");
510 
511  QDBusInterface testReading("com.iflytek.aiassistant",
512  "/aiassistant/tts",
513  "com.iflytek.aiassistant.tts",
514  QDBusConnection::sessionBus());
515  //测试朗读是否在进行
516  QDBusReply<bool> readingReply = testReading.call(QDBus::AutoDetect, "isTTSInWorking");
517 
518  QDBusInterface testTranslate("com.iflytek.aiassistant",
519  "/aiassistant/trans",
520  "com.iflytek.aiassistant.trans",
521  QDBusConnection::sessionBus());
522  //测试翻译接口是否开启
523  QDBusReply<bool> translateReply = testTranslate.call(QDBus::AutoDetect, "getTransEnable");
524 
525  QDBusInterface testSpeechToText("com.iflytek.aiassistant",
526  "/aiassistant/iat",
527  "com.iflytek.aiassistant.iat",
528  QDBusConnection::sessionBus());
529  //测试听写接口是否开启
530  QDBusReply<bool> speechToTextReply = testSpeechToText.call(QDBus::AutoDetect, "getIatEnable");
531 
532  QLineEdit *pLineEdit = static_cast<QLineEdit*>(watched);
533  QMenu *menu = pLineEdit->createStandardContextMenu();
534 
535  for (QAction *action : menu->actions()) {
536  if (action->text().startsWith(QLineEdit::tr("&Copy"))) {
537  action->setEnabled(copyEnabled());
538  }
539  if (action->text().startsWith(QLineEdit::tr("Cu&t"))) {
540  action->setEnabled(cutEnabled());
541  }
542  }
543 
544  connect(menu, &QMenu::triggered, this, [pLineEdit](QAction *pAction) {
545  if (pAction->text().startsWith(QLineEdit::tr("Select All"))) {
546  QApplication::clipboard()->setText(pLineEdit->text(), QClipboard::Mode::Selection);
547  }
548  });
549 
550  //朗读,翻译,听写都没有开启，则弹出默认菜单
551  if (!speechReply.value() && !translateReply.value() && !speechToTextReply.value()) {
552  menu->popup(static_cast<QContextMenuEvent*>(event)->globalPos());
553  event->accept();
554  return true;
555  }
556 
557  menu->addSeparator();
558 
559  if (speechReply.value() && textToSpeechIsEnabled()) {
560  QAction *pAction_1 = nullptr;
561  if (readingReply.value()) {
562  pAction_1 = menu->addAction(QCoreApplication::translate("DLineEdit", "Stop reading"));
563  } else {
564  pAction_1 = menu->addAction(QCoreApplication::translate("DLineEdit", "Text to Speech"));
565  }
566 
567  //没有选中文本，则菜单置灰色
568  if (!pLineEdit->hasSelectedText()) {
569  pAction_1->setEnabled(false);
570  }
571 
572  connect(pAction_1, &QAction::triggered, this, [] {
573  QDBusInterface speechInterface("com.iflytek.aiassistant",
574  "/aiassistant/deepinmain",
575  "com.iflytek.aiassistant.mainWindow",
576  QDBusConnection::sessionBus());
577 
578  if (speechInterface.isValid()) {
579  speechInterface.call(QDBus::BlockWithGui, "TextToSpeech");//此函在第一次调用时朗读，在朗读状态下再次调用为停止朗读
580  } else {
581  qWarning() << "[DLineEdit] TextToSpeech ERROR";
582  }
583  });
584  }
585 
586  if (translateReply.value() && textToTranslateIsEnabled()) {
587  QAction *pAction_2 = menu->addAction(QCoreApplication::translate("DLineEdit", "Translate"));
588 
589  //没有选中文本，则菜单置灰色
590  if (!pLineEdit->hasSelectedText()) {
591  pAction_2->setEnabled(false);
592  }
593 
594  connect(pAction_2, &QAction::triggered, this, [] {
595  QDBusInterface translationInterface("com.iflytek.aiassistant",
596  "/aiassistant/deepinmain",
597  "com.iflytek.aiassistant.mainWindow",
598  QDBusConnection::sessionBus());
599 
600  if (translationInterface.isValid()) {
601  translationInterface.call(QDBus::BlockWithGui, "TextToTranslate");//执行翻译
602  } else {
603  qWarning() << "[DLineEdit] Translation ERROR";
604  }
605  });
606  }
607 
608  if (speechToTextReply.value() && speechToTextIsEnabled()) {
609  QAction *pAction_3 = menu->addAction(QCoreApplication::translate("DLineEdit", "Speech To Text"));
610  connect(pAction_3, &QAction::triggered, this, [] {
611  QDBusInterface speechToTextInterface("com.iflytek.aiassistant",
612  "/aiassistant/deepinmain",
613  "com.iflytek.aiassistant.mainWindow",
614  QDBusConnection::sessionBus());
615 
616  if (speechToTextInterface.isValid()) {
617  speechToTextInterface.call(QDBus::BlockWithGui, "SpeechToText");//执行听写
618  } else {
619  qWarning() << "[DLineEdit] SpeechToText ERROR";
620  }
621  });
622  }
623 
624  //FIXME: 由于Qt在UOS系统环境下不明原因的bug,使用menu->setAttribute(Qt::WA_DeleteOnClose) 销毁menu会在特定情况下出现崩溃的问题，这里采用一种变通的做法
625  connect(menu, &QMenu::aboutToHide, this, [=] {
626  if (menu->activeAction()) {

628  } else {
629  QTimer::singleShot(0, this, [=] {

631  });

633  });
634

636  event->accept();
637  pLineEdit->setFocus();
638  return true;
639  }
