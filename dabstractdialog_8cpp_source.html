48 void DAbstractDialogPrivate::init(bool blurIfPossible)
49 {
50  D_Q(DAbstractDialog);
51 
52  if (qApp->isDXcbPlatform()) {
53  handle = new DPlatformWindowHandle(q, q);
54 
55  handle->setEnableSystemMove(false);
56  handle->setEnableSystemResize(false);
57 
58  bgBlurWidget = new DBlurEffectWidget(q);
59  bgBlurWidget->setAccessibleName("DAbstractDialogBlurEffectWidget");
60  bgBlurWidget->lower();
61  bgBlurWidget->setBlendMode(DBlurEffectWidget::BehindWindowBlend);
62  bgBlurWidget->setFull(true);
63  bgBlurWidget->setMaskColor(DBlurEffectWidget::AutoColor);
64  bgBlurWidget->setMaskAlpha(204); // 80%
65 
66  if (!DWindowManagerHelper::instance()->hasBlurWindow()
67  && DGuiApplicationHelper::instance()->isTabletEnvironment()) {
68  blurIfPossible = false;
69  }
70 
71  bgBlurWidget->setBlurEnabled(blurIfPossible);
72  q->setAttribute(Qt::WA_TranslucentBackground, blurIfPossible);
73  } else {
74  // fix wayland no titlebar
75  //q->setWindowFlags(q->windowFlags() | Qt::FramelessWindowHint);
76  }
77 
78  q->resize(DIALOG::DEFAULT_WIDTH, DIALOG::DEFAULT_HEIGHT);
79  q->setAttribute(Qt::WA_Resized, false);
80 
81  if (DGuiApplicationHelper::isTabletEnvironment()) {
82  if (QScreen *screen = QGuiApplication::primaryScreen()) {
83  q->connect(screen, &QScreen::primaryOrientationChanged, q, [ = ] {
84  q->moveToCenter();
85  });
86  }
87  }
88 }

355 {

396 void DAbstractDialog::showEvent(QShowEvent *event)
397 {
398  D_DC(DAbstractDialog);
399 
400  // 不要在resizeEvent中改变窗口geometry，可能会导致以下问题。在双屏开启不同缩放比时：
401  // 屏幕A缩放为1，屏幕B缩放为2。窗口初始在屏幕B上，此时大小为 100x100，则其缩放
402  // 后的真实大小为200x200，由于窗口管理器会对未设置过位置的窗口自动布局，接下来
403  // 窗口可能会因为位置改变而被移动到屏幕A上，则此对话框的大小将变化为 200x200，
404  // 触发了resize事件，紧接着又会尝试将窗口位置更新到鼠标所在屏幕的中心，如果鼠标
405  // 所在屏幕为B，则在移动窗口位置时又会根据B的缩放比重新设置窗口大小，窗口将变为：
406  // 400x400，将形成一个循环。。。
407  if (!d->mouseMoved) {
408  setDisplayPosition(displayPosition());
409  }
410 
411  QDialog::showEvent(event);
412 }
