40 void DAlertControlPrivate::updateTooltipPos()
41 {
42  if (!target || !target->parentWidget() || !frame || !frame->parentWidget()) {
43  qWarning("target or frame is nullptr.");
44  return;
45  }
46  QWidget *p = target->parentWidget();
47 
48  int w = DStyle::pixelMetric(p->style(), DStyle::PM_FloatingWidgetShadowMargins) / 2;
49  QPoint point = QPoint(target->x() - w, target->y() + target->height() - w);
50  frame->move(p->mapTo(qobject_cast<QWidget *>(frame->parentWidget()), point));
51  int tipWidth= frame->parentWidget()->width() - 20;
52  tooltip->setMaximumWidth(tipWidth);
53  frame->setMinimumHeight(tooltip->heightForWidth(tipWidth) + frame->layout()->spacing() *2);
54  frame->adjustSize();
55 
56  int tw = target->width();
57  int fw = frame->width();
58  int leftPending = tw - fw;
59  QPoint objPoint;
60  switch (alignment) {
61  case Qt::AlignLeft:
62  objPoint = p->mapTo(qobject_cast<QWidget *>(frame->parentWidget()), point);
63  if (objPoint.x() + frame->width() > (frame->parentWidget()->width())) {
64  int shift = (objPoint.x() + frame->width()) - frame->parentWidget()->width();
65  objPoint.setX(objPoint.x() - shift);
66  }
67  break;
68  case Qt::AlignRight:
69  point += QPoint(leftPending, 0);
70  objPoint = p->mapTo(qobject_cast<QWidget *>(frame->parentWidget()), point);
71  if (objPoint.x() < 0) {
72  objPoint.setX(0);
73  }
74  break;
75  case Qt::AlignHCenter:
76  case Qt::AlignCenter:
77  point += QPoint(leftPending/2, 0);
78  objPoint = p->mapTo(qobject_cast<QWidget *>(frame->parentWidget()), point);
79  break;
80  default:
81  return;
82  }
83 
84  frame->move(objPoint);
85 }

214 void DAlertControl::showAlertMessage(const QString &text, QWidget *follower, int duration)
215 {
216  D_D(DAlertControl);
217  if (!d->target)
218  return;
219 
220  if (!d->tooltip) {
221  d->tooltip = new DToolTip(text);
222  d->tooltip->setObjectName("AlertTooltip");
223  d->tooltip->setAccessibleName("DAlertControlAlertToolTip");
224  d->tooltip->setForegroundRole(DPalette::TextWarning);
225  d->tooltip->setWordWrap(true);
226 
227  d->frame = new DFloatingWidget;
228  d->frame->setAccessibleName("DAlertControlFloatingWidget");
229  d->frame->setFramRadius(DStyle::pixelMetric(d->target->style(), DStyle::PM_FrameRadius));
230  d->frame->setBackgroundRole(QPalette::ToolTipBase);
231  d->frame->setWidget(d->tooltip);
232  }
233 
234  if (follower) {
235  d->frame->setParent(follower->topLevelWidget());
236  d->follower = follower;
237  d->follower->installEventFilter(this);
238  } else {
239  d->frame->setParent(d->target->topLevelWidget());
240  d->follower = d->target;
241  d->follower->installEventFilter(this);
242  }
243 
244  d->tooltip->setText(text);
245  if (d->frame->parent()) {
246  d->updateTooltipPos();
247  // 如果 target 都隐藏了,显示警告消息毫无意义，只会让人困惑
248  d->frame->setVisible(d->target->isVisibleTo(d->target->topLevelWidget()));
249  d->frame->adjustSize();
250  d->frame->raise();
251  }
252  if (duration < 0) {
253  d->timer.stop();
254  return;
255  }
256 
257  d->timer.start(duration);
258 }

284 bool DAlertControl::eventFilter(QObject *watched, QEvent *event)
285 {
286  Q_D(DAlertControl);
287  if (watched == d->follower) {
288  if (event->type() == QEvent::Move || event->type() == QEvent::Resize)
289  d->updateTooltipPos();
290 
291  if (event->type() == QEvent::Hide || event->type() == QEvent::HideToParent)
292  hideAlertMessage();
293  }
294 
295 
296  return QObject::eventFilter(watched, event);
297 }
