101 TEST_F(TDThumbnailProvider, TestCreate)
102 {
103  if (qgetenv("QT_QPA_PLATFORM").contains("offscreen"))
104  return;
105 
106  QFileInfo fi(TESTRES_PATH);
107  QSignalSpy finishedSpy(provider, SIGNAL(createThumbnailFinished(const QString &, const QString &)));
108  QString ret = provider->createThumbnail(fi, DThumbnailProvider::Normal);
109  ASSERT_FALSE(ret.isEmpty());
110  ASSERT_EQ(finishedSpy.count(), 1);
111  ASSERT_FALSE(provider->thumbnailFilePath(fi, DThumbnailProvider::Normal).isEmpty());
112 
113  QFileInfo fi_notexisted(TESTRES_PATH_1);
114  QSignalSpy failedSpy(provider, SIGNAL(createThumbnailFailed(const QString &)));
115  ret = provider->createThumbnail(fi_notexisted, DThumbnailProvider::Normal);
116  ASSERT_TRUE(ret.isEmpty());
117  ASSERT_EQ(finishedSpy.count(), 1);
118  ASSERT_FALSE(provider->errorString().isEmpty());
119  ASSERT_TRUE(provider->thumbnailFilePath(fi_notexisted, DThumbnailProvider::Normal).isEmpty());
120 }
