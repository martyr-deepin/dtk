734 bool DPlatformHandle::setWindowBlurAreaByWM(QWindow *window, const QVector<DPlatformHandle::WMBlurArea> &area)
735 {
736  if (!window) {
737  return false;
738  }
739

741  setWindowProperty(window, _windowBlurAreas, QVariant::fromValue(*(reinterpret_cast<const QVector<quint32>*>(&area))));
742

746  QFunctionPointer setWmBlurWindowBackgroundArea = Q_NULLPTR;
747 
748 #if QT_VERSION >= QT_VERSION_CHECK(5, 4, 0)

760  format.setAlphaBufferSize(8);

764

766  return reinterpret_cast<bool(*)(const quint32, const QVector<WMBlurArea>&)>(setWmBlurWindowBackgroundArea)(window->winId(), area);
767  }
768 
769  QVector<WMBlurArea> new_areas;
770 
771  new_areas.reserve(area.size());
772

774  new_areas.append(a * device_ratio);
775  }
776 
777  return reinterpret_cast<bool(*)(const quint32, const QVector<WMBlurArea>&)>(setWmBlurWindowBackgroundArea)(window->winId(), new_areas);

1198 bool DPlatformHandle::eventFilter(QObject *obj, QEvent *event)
1199 {
1200  if (obj == m_window) {
1201  if (event->type() == QEvent::DynamicPropertyChange) {
1202  QDynamicPropertyChangeEvent *e = static_cast<QDynamicPropertyChangeEvent *>(event);
1203 
1204  if (e->propertyName() == _windowRadius) {
1205  Q_EMIT windowRadiusChanged();
1206  } else if (e->propertyName() == _borderWidth) {
1207  Q_EMIT borderWidthChanged();
1208  } else if (e->propertyName() == _borderColor) {
1209  Q_EMIT borderColorChanged();
1210  } else if (e->propertyName() == _shadowRadius) {
1211  Q_EMIT shadowRadiusChanged();
1212  } else if (e->propertyName() == _shadowOffset) {
1213  Q_EMIT shadowOffsetChanged();
1214  } else if (e->propertyName() == _shadowColor) {
1215  Q_EMIT shadowColorChanged();
1216  } else if (e->propertyName() == _clipPath) {
1217  Q_EMIT clipPathChanged();
1218  } else if (e->propertyName() == _frameMask) {
1219  Q_EMIT frameMaskChanged();
1220  } else if (e->propertyName() == _frameMargins) {
1221  Q_EMIT frameMarginsChanged();
1222  } else if (e->propertyName() == _translucentBackground) {
1223  Q_EMIT translucentBackgroundChanged();
1224  } else if (e->propertyName() == _enableSystemResize) {
1225  Q_EMIT enableSystemResizeChanged();
1226  } else if (e->propertyName() == _enableSystemMove) {
1227  Q_EMIT enableSystemMoveChanged();
1228  } else if (e->propertyName() == _enableBlurWindow) {
1229  Q_EMIT enableBlurWindowChanged();
1230  } else if (e->propertyName() == _autoInputMaskByClipPath) {
1231  Q_EMIT autoInputMaskByClipPathChanged();

1236  return false;
