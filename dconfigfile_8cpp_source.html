64 inline QString getFile(const QString &baseDir, const QString &subpath, const QString &name,
65  bool canFallbackUp = true) {
66  qCDebug(cfLog, "load json file from base dir:\"%s\", subpath = \"%s\", file name =\"%s\".",
67  qPrintable(baseDir), qPrintable(subpath), qPrintable(name));
68 
69  const QDir base_dir(baseDir);
70  QDir target_dir = base_dir;
71 
72  if (!subpath.isEmpty())
73  target_dir.cd(subpath.mid(1));
74 
75  do {
76  qCDebug(cfLog, "load json file from: \"%s\"", qPrintable(target_dir.path()));
77 
78  if (QFile::exists(target_dir.filePath(name))) {
79  return target_dir.filePath(name);
80  }
81 
82  if (base_dir == target_dir)
83  break;
84  } while (canFallbackUp && target_dir.cdUp());
85 
86  return QString();
87 }

99 static QJsonDocument loadJsonFile(QIODevice *data)
100 {
101  if (!data->open(QIODevice::ReadOnly)) {
102  if (auto file = qobject_cast<QFile*>(data)) {
103  qCDebug(cfLog, "Falied on open file: \"%s\", error message: \"%s\"",
104  qPrintable(file->fileName()), qPrintable(file->errorString()));
105  }
106  return QJsonDocument();
107  }
108 
109  QJsonParseError error;
110  auto document = QJsonDocument::fromJson(data->readAll(), &error);
111  data->close();
112 
113  if (error.error != QJsonParseError::NoError) {
114  qCWarning(cfLog, "%s", qPrintable(error.errorString()));
115  return QJsonDocument();
116  }
117 
118  return document;
119 }

121 static DConfigFile::Version parseVersion(const QJsonObject &obj) {
122  DConfigFile::Version version {0, 0};
123  const QString &verStr = obj[QLatin1String("version")].toString();
124 
125  if (verStr.isEmpty()) {
126  return version;
127  }
128 
129  const QStringList &items = verStr.split(QLatin1Char('.'));
130 
131  if (items.size() != 2)
132  return version;
133 
134  bool ok = false;
135  quint16 major = items.first().toUShort(&ok);
136 
137  if (!ok)
138  return version;
139 
140  quint16 minor = items.last().toUShort(&ok);
141 
142  if (!ok)
143  return version;
144 
145  version.major = major;
146  version.minor = minor;
147 
148  return version;
149 }

652  bool load(QIODevice *meta, const QList<QIODevice*> &overrides) override
653  {
654  {
655  const QJsonDocument &doc = loadJsonFile(meta);
656  if (!doc.isObject())
657  return false;
658 
659  // 检查标识
660  const QJsonObject &root = doc.object();
661  if (!checkMagic(root, MAGIC_META)) {
662  qCWarning(cfLog, "The meta magic does not match");
663  return false;
664  }
665 
666  // 检查版本兼容性
667  const auto &v = parseVersion(root);
668  if (!versionIsValid(v) || v.major > DConfigFile::supportedVersion().major) {
669  qCWarning(cfLog, "The meta version number does not match, "
670  "the file major version=%i, supported major version<=%i",
671  v.major, DConfigFile::supportedVersion().major);
672  return false;
673  }
674 
675  m_version = v;
676 
677  const auto &contents = root[QLatin1String("contents")].toObject();
678  auto i = contents.constBegin();
679 
680  // 初始化原始值
681  for (; i != contents.constEnd(); ++i) {
682  values.update(i.key(), i.value().toObject().toVariantHash());
683  }
684  }
685 
686  // for override
687  Q_FOREACH(auto override, overrides) {
688  const QJsonDocument &doc = loadJsonFile(override);
689 
690  if (doc.isObject()) {
691  const QJsonObject &root = doc.object();
692  if (!checkMagic(root, MAGIC_OVERRIDE)) {
693  if (auto file = static_cast<QFile*>(override)) {
694  qCWarning(cfLog, "The override magic does not match, file: \"%s\", error message: \"%s\"",
695  qPrintable(file->fileName()), qPrintable(file->errorString()));
696  } else {
697  qCWarning(cfLog, "The override magic does not match");
698  }
699  break; //TODO don't continue parse?
700  }
701  if (!checkVersion(root, m_version)) {
702  qCWarning(cfLog, "The override version number does not match");
703  break;
704  }
705 
706  if (auto file = static_cast<QFile*>(override)) {
707  qCDebug(cfLog, "The override will be applied, file: \"%s\"", qPrintable(file->fileName()));
708  }
709 
710  const auto &contents = root[QLatin1String("contents")].toObject();
711  auto i = contents.constBegin();
712 
713  for (; i != contents.constEnd(); ++i) {
714  // 检查是否允许 override
715  if (values.flags(i.key()) & DConfigFile::NoOverride)
716  continue;
717 
718  values.updateValue(i.key(), i.value());
719  values.updateSerial(i.key(), i.value());
720  values.updatePermissions(i.key(), i.value());
721  }
722  }
723  }
724 
725  return true;
726  }

744  inline QStringList allOverrideDirs(const bool useAppId, const QString &prefix) const override
745  {
746  QStringList dirs;
747  // 只有当允许不使用 appid 时才能回退到通用目录
748  if (!useAppId) {
749  dirs << overrideDirs(prefix, false);
750  }

764  QList<QIODevice *> loadOverrides(const QString &prefix, bool useAppId) const
765  {
766  auto filters = QDir::Files | QDir::NoDotAndDotDot | QDir::Readable;
767  const QStringList nameFilters {"*" + FILE_SUFFIX};
768 
769  QStringList dirs = allOverrideDirs(useAppId, prefix);
770 
771  QList<QIODevice*> list;
772  list.reserve(50);
773  QCollator collator(QLocale::English);
774  collator.setNumericMode(true);
775  collator.setIgnorePunctuation(true);
776 
777  Q_FOREACH(const auto &dir, dirs) {
778  const QDir base_dir(QDir::cleanPath(dir));
779 
780  if (!base_dir.exists())
781  continue;
782 
783  QDir target_dir = base_dir;
784  target_dir.setFilter(filters);
785  target_dir.setNameFilters(nameFilters);
786 
787  if (!configKey.subpath.isEmpty())
788  target_dir.cd(configKey.subpath.mid(1));
789 
790  do {
791  qCDebug(cfLog, "load override file from: \"%s\"", qPrintable(target_dir.path()));
792 
793  QDirIterator iterator(target_dir);
794  QList<QIODevice*> sublist;
795  sublist.reserve(50);
796  while(iterator.hasNext()) {
797  sublist.append(new QFile(iterator.next()));
798  }
799 
800  // 从小到大排序
801  std::sort(sublist.begin(), sublist.end(), [&collator](QIODevice *f1, QIODevice *f2){
802  if (collator.compare(static_cast<const QFile*>(f1)->fileName(),
803  static_cast<const QFile*>(f2)->fileName()) < 0)
804  return true;
805  return false;
806  });
807 
808  list = sublist + list;
809 
810  if (base_dir.path() == target_dir.path())
811  break;
812  } while (target_dir.cdUp());
813  }
814 
815  return list;
816  }
