53 TEST_F(ut_DBlurEffectWidget, testBlurEffectWidgetFrontEnd)
54 {
55  // 测试信号槽的有效性
56  QObject::connect(widget, &DBlurEffectWidget::modeChanged, widget, [this](DBlurEffectWidget::BlurMode mode){
57  ASSERT_TRUE(widget->mode() == DBlurEffectWidget::BlurMode::GaussianBlur);
58  });
59 
60  QObject::connect(widget, &DBlurEffectWidget::fullChanged, widget, [this](bool isFull){
61  ASSERT_TRUE(widget->isFull() == isFull);
62  });
63  widget->setFull(true);
64 
65  // 测试 set mode 以及 mode change 的有效性
66  widget->setMode(DBlurEffectWidget::BlurMode::GaussianBlur);
67  ASSERT_TRUE(widget->mode() == DBlurEffectWidget::BlurMode::GaussianBlur);
68 
69  widget->setRadius(5);
70  ASSERT_TRUE(widget->radius() == 5);
71 
72  // 测试 setMaskPath
73  QPainter painter;
74  painter.setPen(Qt::yellow); //设置画笔颜色
75  QPainterPath path;
76  path.moveTo(widget->width()/2, widget->height()/2);
77  path.arcTo(widget->width()/2 - 100, widget->height()/2 - 100, 200, 200, 30, 120);
78  painter.drawPath(path);
79 
80  widget->setMaskPath(path);
81  ASSERT_TRUE(widget->d_func()->maskPath == path);
82 
83  widget->setBlendMode(DBlurEffectWidget::InWindowBlend);
84  ASSERT_TRUE(widget->blendMode() == DBlurEffectWidget::InWindowBlend);
