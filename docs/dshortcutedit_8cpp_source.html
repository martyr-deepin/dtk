71 bool DShortcutEdit::eventFilter(QObject *o, QEvent *e)
72 {
73  if (o == m_keysEdit && e->type() == QEvent::FocusOut && m_keysEdit->isVisible()) {
74  toEchoMode();
75  } else if (o == m_keysLabel && e->type() == QEvent::MouseButtonRelease && m_keysLabel->isVisible()) {
76  toInputMode();
77  } else if (o == m_keysEdit && e->type() == QEvent::KeyPress) {
78  shortcutKeyPress(static_cast<QKeyEvent *>(e));
79  }
80 
81  return false;
82 }

192 void DShortcutEdit::shortcutKeyPress(QKeyEvent *e)
193 {
194  //qDebug() << e->key() << e->text() << e->count() << Qt::CTRL << e->nativeScanCode() << e->nativeVirtualKey();
195  m_shortcutKeys.clear();
196 
197  int state = 0;
198  if ((e->modifiers() & Qt::ShiftModifier) && (e->text().isEmpty() ||
199  !e->text().at(0).isPrint() ||
200  e->text().at(0).isLetterOrNumber() ||
201  e->text().at(0).isSpace())) {
202  state |= Qt::SHIFT;
203  }
204  if (e->modifiers() & Qt::ControlModifier) {
205  state |= Qt::CTRL;
206  }
207  if (e->modifiers() & Qt::MetaModifier) {
208  state |= Qt::META;
209  }
210  if (e->modifiers() & Qt::AltModifier) {
211  state |= Qt::ALT;
212  }
213 
214  int key = e->key() | state;
215 // qDebug() << "keys: " << QKeySequence(key, 0, 0, 0).toString(QKeySequence::NativeText);
216  m_shortcutKeys = QKeySequence(key, 0, 0, 0).toString(QKeySequence::NativeText);
217 
218  //qDebug() << m_shortcutKeys << e->text() << e->key();
219  //qDebug() << "keys: " << int(e->key() & ~(Qt::SHIFT | Qt::ControlModifier | Qt::META | Qt::ALT));
220 
221  // if pressed key is "Backspace", its means "clear all"
222  if (m_shortcutKeys == "Backspace") {
223  return clearShortcutKey();
224  }
225 
226  setShortcutKey(m_shortcutKeys);
227 
228  QString lastKey = m_shortcutKeys.split("+").last();
229  qDebug() << "last: " << lastKey << lastKey.size();
230 
231  // if lastKey is not alt, shift, ctrl, meta, shortcut key is finished.
232  if (lastKey.size() == 2 && !lastKey.at(0).isLetter()) {
233  return;
234  }
235  if (lastKey != "Meta" && lastKey != "Ctrl" &&
236  lastKey != "Shift" && lastKey != "Alt") {
237  toEchoMode();
238  }
239 }
