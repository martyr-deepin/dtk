261  QString fallbackWidgetThemeName(const QWidget *widget, QWidget **baseWidget = nullptr) const
262  {
263  QString theme;
264 
265  if (baseWidget) {
266  *baseWidget = nullptr;
267  }
268 
269  do {
270  theme = widget->property("_d_dtk_theme").toString();
271  if (!theme.isEmpty()) {
272  if (baseWidget) {
273  *baseWidget = const_cast<QWidget *>(widget);
274  }
275 
276  break;
277  }
278  widget = widget->isWindow() ? 0 : widget->parentWidget();
279  } while (widget);
280 
281  return theme.isEmpty() ? themeName : theme;
282  }

284  QString fallbackWidgetThemeFilename(QWidget *w) const
285  {
286  QStringList fallbackList;
287 
288  QString overriveName = w->property("_d_dtk_theme_filename").toString();
289  if (!overriveName.isEmpty()) {
290  fallbackList << overriveName;
291  }
292 
293  QString className = getThemeNameByClassName(w);
294  if (!className.isEmpty()) {
295  fallbackList << className;
296  }
297 
298  QString rawClassName = getThemeNameByRawClassName(w);
299  if (!rawClassName.isEmpty()) {
300  fallbackList << rawClassName;
301  }
302 
303  auto themeName = fallbackWidgetThemeName(w, nullptr);
304  for (auto filename : fallbackList) {
305  if (themeFileExist(themeURL(themeName, filename))) {
306  return filename;
307  }
308  }
309 
310  return "";
311  }

313  void registerWidget(QWidget *widget, const QString &filename, const QStringList &properties)
314  {
315  if (filename.isEmpty()) {
316  qWarning() << "can not load qss with out filename" << widget;
317  return;
318  }
319 
320  auto themeurl = themeURL(fallbackWidgetThemeName(widget), filename);
321 
322  auto dtm = DThemeManager::instance();
323  widget->setStyleSheet(widget->styleSheet() + dtm->d_func()->getQssContent(themeurl));
324  widget->style()->polish(widget);
325 
326  auto reloadTheme = [this, dtm](QWidget * widget, const QString & filename, const QString & themename) {
327  const char *baseClassReloadThemeProp = "_dtk_theme_base_calss_reload_theme";
328  auto themeurl = themeURL(fallbackWidgetThemeName(widget), filename);
329  auto reloadTheme = widget->property(baseClassReloadThemeProp).toString();
330  if (reloadTheme != themename) {
331  widget->setStyleSheet(dtm->d_func()->getQssContent(themeurl));
332  widget->setProperty(baseClassReloadThemeProp, themename);
333  } else {
334  widget->setStyleSheet(widget->styleSheet() + dtm->d_func()->getQssContent(themeurl));
335  }
336  };
337 
338  dtm->connect(dtm, &DThemeManager::themeChanged,
339  widget, [reloadTheme, widget, filename](QString theme) {
340  reloadTheme(widget, filename, theme);
341  });
342 
343  dtm->connect(dtm, &DThemeManager::widgetThemeChanged, widget,
344  [reloadTheme, widget, filename](QWidget * w, QString theme) {
345  if (widget == w) {
346  reloadTheme(widget, filename, theme);
347  }
348  });
349 
350  auto meta = widget->metaObject();
351  QMap<QString, QString> dynamicProperties;
352  for (auto &prop : properties) {
353  auto propIndex = meta->indexOfProperty(prop.toLatin1().data());
354  if (propIndex < 0) {
355  dynamicProperties.insert(prop, prop);
356  continue;
357  }
358  dtm->connect(widget, meta->property(propIndex).notifySignal(),
359  dtm, dtm->metaObject()->method(dtm->metaObject()->indexOfMethod("updateQss()")));
360  }
361 
362  if (!dynamicProperties.isEmpty()) {
363  widget->installEventFilter(dtm);
364  if (dtm->d_func()->watchedDynamicProperties.contains(widget)) {
365  QMap<QString, QString> oldProps = dtm->d_func()->watchedDynamicProperties.value(widget);
366  for (auto &key : oldProps.keys()) {
367  dynamicProperties.insert(key, oldProps.value(key));
368  }
369  }
370  dtm->d_func()->watchedDynamicProperties.insert(widget, dynamicProperties);
371 
372  dtm->connect(widget, &QObject::destroyed, dtm, [ = ]() {
373  dtm->d_func()->watchedDynamicProperties.remove(widget);
374  });
375  }
376  }

410  void setTheme(QWidget *widget, const QString theme)
411  {
412  D_Q(DThemeManager);
413 
414  Q_ASSERT(widget);
415 
416  if (theme.isEmpty()) {
417  QString old_theme = fallbackWidgetThemeName(widget);
418 
419  widget->setProperty("_d_dtk_theme", QVariant());
420 
421  QWidget *baseWidget = nullptr;
422 
423  if (fallbackWidgetThemeName(widget, &baseWidget) != old_theme) {
424  emitThemeChanged(q, widget, fallbackWidgetThemeName(widget));
425  }
426 
427  inseritStyle(widget, baseWidget);
428 
429  return;
430  }
431 
432  const QString &old_theme = fallbackWidgetThemeName(widget);
433  QStyle *style = Q_NULLPTR;
434 
435  // TODO: remove this shit in the future.
436  // It's just a trick to make all DApplications use dde qt5 styles,
437  // if dlight or ddark style is set to default style of dde, those
438  // ugly code will no longer needed.
439  if (theme == "light") {
440  style = QStyleFactory::create("dlight");
441  widget->setProperty("_d_dtk_theme", theme);
442  } else if (theme == "dark") {
443  style = QStyleFactory::create("ddark");
444  widget->setProperty("_d_dtk_theme", theme);
445  } else if (theme == "semilight") {
446  style = QStyleFactory::create("dsemilight");
447  widget->setProperty("_d_dtk_theme", "light");
448  } else if (theme == "semidark") {
449  style = QStyleFactory::create("dsemidark");
450  widget->setProperty("_d_dtk_theme", "dark");
451  }
452 
453  if (style) {
454  setStyle(widget, style);
455  }
456 
457  if (old_theme != theme) {
458  emitThemeChanged(q, widget, theme);
459  }
460  }

617 bool DThemeManager::eventFilter(QObject *watched, QEvent *event)
618 {
619  D_D(DThemeManager);
620  if (event->type() != QEvent::DynamicPropertyChange) {
621  return QObject::eventFilter(watched, event);
622  }
623 
624  auto widget = qobject_cast<QWidget *>(watched);
625  if (!d->watchedDynamicProperties.contains(widget)) {
626  return QObject::eventFilter(watched, event);
627  }
628 
629  auto propEvent = reinterpret_cast<QDynamicPropertyChangeEvent *>(event);
630  if (!propEvent) {
631  return QObject::eventFilter(watched, event);
632  }
633 
634  auto props = d->watchedDynamicProperties.value(widget);
635  auto propName = QString::fromLatin1(propEvent->propertyName().data());
636  if (props.contains(propName) && widget) {
637  widget->setStyleSheet(widget->styleSheet());
638  widget->style()->unpolish(widget);
639  widget->style()->polish(widget);
640  widget->update();
641  }
642 
643  return QObject::eventFilter(watched, event);
644 }
