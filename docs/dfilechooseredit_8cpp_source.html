251 void DFileChooserEditPrivate::_q_showFileChooserDialog()
252 {
253  D_Q(DFileChooserEdit);
254 
255  if (!dialog) {
256  q->initDialog();
257  }
258 
259  if (dialogDisplayPosition == DFileChooserEdit::CurrentMonitorCenter) {
260  QPoint pos = QCursor::pos();
261 
262  for (QScreen *screen : qApp->screens()) {
263  if (screen->geometry().contains(pos)) {
264  QRect rect = dialog->geometry();
265  rect.moveCenter(screen->geometry().center());
266  dialog->move(rect.topLeft());
267  break;
268  }
269  }
270  }
271 
272  q->dialogOpened();
273 
274  int code = dialog->exec();
275 
276  if (code == QDialog::Accepted && !dialog->selectedFiles().isEmpty()) {
277  const QString fileName = dialog->selectedFiles().first();
278 
279  q->setText(fileName);
280  Q_EMIT q->fileChoosed(fileName);
281  }
282 
283  q->dialogClosed(code);
284 }
