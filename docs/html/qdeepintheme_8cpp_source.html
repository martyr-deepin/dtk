91  static quint64 entryCacheKey(const ScalableEntry *color_entry, const QIcon::Mode mode, const QIcon::State state)
92  {
93  return quint64(color_entry) ^ (quint64(mode) << 56) ^ (quint64(state) << 48);
94  }
95 
96  QPixmap followColorPixmap(ScalableEntry *color_entry, const QSize &size, QIcon::Mode mode, QIcon::State state)
97  {
98  quint64 cache_key = entryCacheKey(color_entry, mode, state);
99  const QString &cache_color_scheme = entryToColorScheme.value(cache_key);
100 
101  // 当size为1时表示此svg文件不需要处理ColorScheme标签
102  if (cache_color_scheme.size() == 1)
103  return color_entry->pixmap(size, mode, state);
104 
105  const QString &color_scheme = DEEPIN_QT_THEME::colorScheme.localData();
106  QPixmap pm = color_scheme == cache_color_scheme ? color_entry->svgIcon.pixmap(size, mode, state) : QPixmap();
107  // Note: not checking the QIcon::isNull(), because in Qt5.10 the isNull() is not reliable
108  // for svg icons desierialized from stream (see https://codereview.qt-project.org/#/c/216086/)
109  if (pm.isNull())
110  {
111  // The following lines are adapted and updated from KDE's "kiconloader.cpp" ->
112  // KIconLoaderPrivate::processSvg() and KIconLoaderPrivate::createIconImage().
113  // They read the SVG color scheme of SVG icons and give images based on the icon mode.
114  QHash<int, QByteArray> svg_buffers;
115  bool invalidBuffers = true;
116  QFile device{color_entry->filename};
117  if (device.open(QIODevice::ReadOnly))
118  {
119  // Note: indexes are assembled as in qtsvg (QSvgIconEnginePrivate::hashKey())
120  QPair<int, QString> style_sheet;
121  style_sheet = qMakePair((mode<<4)|state, QStringLiteral(".ColorScheme-Text{color:%1;}").arg(color_scheme));
122  QSharedPointer<QXmlStreamWriter> writer(new QXmlStreamWriter{&svg_buffers[style_sheet.first]});
123 
124  QXmlStreamReader xmlReader(&device);
125  while (!xmlReader.atEnd())
126  {
127  if (xmlReader.readNext() == QXmlStreamReader::StartElement
128  && xmlReader.qualifiedName() == QLatin1String("style")
129  && xmlReader.attributes().value(QLatin1String("id")) == QLatin1String("current-color-scheme"))
130  {
131  invalidBuffers = false;
132 
133  writer->writeStartElement(QLatin1String("style"));
134  writer->writeAttributes(xmlReader.attributes());
135  writer->writeCharacters(style_sheet.second);
136  writer->writeEndElement();
137 
138  while (xmlReader.tokenType() != QXmlStreamReader::EndElement)
139  xmlReader.readNext();
140  } else if (xmlReader.tokenType() != QXmlStreamReader::Invalid)
141  {
142  writer->writeCurrentToken(xmlReader);
143  }
144  }
145  // duplicate the contents also for opposite state
146 // svg_buffers[(QIcon::Normal<<4)|QIcon::On] = svg_buffers[(QIcon::Normal<<4)|QIcon::Off];
147 // svg_buffers[(QIcon::Selected<<4)|QIcon::On] = svg_buffers[(QIcon::Selected<<4)|QIcon::Off];
148  }
149 
150  if (invalidBuffers) {
151  // 此svg图标无ColorScheme标签时不应该再下面的操作，且应该记录下来，避免后续再处理svg文件内容
152  entryToColorScheme[cache_key] = QStringLiteral("#");
153  return color_entry->pixmap(size, mode, state);
154  }
155 
156  // use the QSvgIconEngine
157  // - assemble the content as it is done by the operator <<(QDataStream &s, const QIcon &icon)
158  // (the QSvgIconEngine::key() + QSvgIconEngine::write())
159  // - create the QIcon from the content by usage of the QIcon::operator >>(QDataStream &s, const QIcon &icon)
160  // (icon with the (QSvgIconEngine) will be used)
161  QByteArray icon_arr;
162  QDataStream str{&icon_arr, QIODevice::WriteOnly};
163  str.setVersion(QDataStream::Qt_4_4);
164  QHash<int, QString> filenames;
165  filenames[0] = color_entry->filename; // Note: filenames are ignored in the QSvgIconEngine::read()
166  filenames[-1] = color_scheme; // 在dsvg插件中会为svg图标做缓存，此处是为其添加额外的缓存文件key标识，避免不同color的svg图标会命中同一个缓存文件
167  str << QStringLiteral("svg") << filenames << static_cast<int>(0)/*isCompressed*/ << svg_buffers << static_cast<int>(0)/*hasAddedPimaps*/;
168 
169  QDataStream str_read{&icon_arr, QIODevice::ReadOnly};
170  str_read.setVersion(QDataStream::Qt_4_4);
171 
172  str_read >> color_entry->svgIcon;
173  pm = color_entry->svgIcon.pixmap(size, mode, state);
174 
175  // load the icon directly from file, if still null
176  if (pm.isNull())
177  {
178  color_entry->svgIcon = QIcon(color_entry->filename);
179  pm = color_entry->svgIcon.pixmap(size, mode, state);
180  }
181 
182  entryToColorScheme[cache_key] = color_scheme;
183  }
184 
185  return pm;
186  }

188  QPixmap pixmapByEntry(QIconLoaderEngineEntry *entry, const QSize &size, QIcon::Mode mode, QIcon::State state)
189  {
190  if (!XdgIcon::followColorScheme()) {
191  DEEPIN_QT_THEME::colorScheme.setLocalData(QString());
192 
193  return entry->pixmap(size, mode, state);
194  }
195 
196  QPixmap pixmap;
197  char *type_name = abi::__cxa_demangle(typeid(*entry).name(), 0, 0, 0);
198 
199  if (type_name == QByteArrayLiteral("ScalableFollowsColorEntry")) {
200  if (DEEPIN_QT_THEME::colorScheme.localData().isEmpty()) {
201  const QPalette &pal = qApp->palette();
202  DEEPIN_QT_THEME::colorScheme.setLocalData(mode == QIcon::Selected ? pal.highlightedText().color().name() : pal.windowText().color().name());
203  }
204 
205  pixmap = followColorPixmap(static_cast<ScalableEntry*>(entry), size, mode, state);
206  } else {
207  pixmap = entry->pixmap(size, mode, state);
208  }
209 
210  free(type_name);
211  DEEPIN_QT_THEME::colorScheme.setLocalData(QString());
212 
213  return pixmap;
214  }

225  if (pix.isNull())

265 #if QT_VERSION >= QT_VERSION_CHECK(5, 9, 0)
266  if (id == QIconEngine::ScaledPixmapHook)
267  {
268  engine->ensureLoaded();
269 
270  QIconEngine::ScaledPixmapArgument &arg = *reinterpret_cast<QIconEngine::ScaledPixmapArgument*>(data);
271  // QIcon::pixmap() multiplies size by the device pixel ratio.
272  const int integerScale = qCeil(arg.scale);
273  QIconLoaderEngineEntry *entry = engine->entryForSize(arg.size / integerScale, integerScale);
274  // 先禁用缩放，因为此size是已经缩放过的
275  bool useHighDpiPixmap = qGuiApp->testAttribute(Qt::AA_UseHighDpiPixmaps);
276  qGuiApp->setAttribute(Qt::AA_UseHighDpiPixmaps, false);
277  arg.pixmap = entry ? pixmapByEntry(entry, arg.size, arg.mode, arg.state) : QPixmap();
278  qGuiApp->setAttribute(Qt::AA_UseHighDpiPixmaps, useHighDpiPixmap);
279  DEEPIN_QT_THEME::colorScheme.setLocalData(QString());
280 
281  return;
282  }
283 #endif
284 
285  return engine->virtual_hook(id, data);
286  }

342 static void updateWindowGeometry(QWindow *w)
343 {
344  if (w->type() == Qt::ForeignWindow || w->type() == Qt::Desktop) {
345  return;
346  }
347 
348  if (!w->handle() || !w->isTopLevel())
349  return;
350 
351  if (w->property(DNOT_UPDATE_WINDOW_GEOMETRY).toBool()) {
352  QWindowSystemInterfacePrivate::GeometryChangeEvent gce(w, QHighDpi::fromNativePixels(w->handle()->geometry(), w)
353  #if QT_VERSION < QT_VERSION_CHECK(5, 10, 0)
354  , QRect()
355  #endif
356  );
357  QGuiApplicationPrivate::processGeometryChangeEvent(&gce);
358  } else {// 通知窗口大小发送改变
359  const QRect currentGeo = QWindowPrivate::get(w)->geometry;
360 
361  if (!currentGeo.isValid())
362  return;
363 
364  // 对外提供能hook自动更新窗口大小的方法
365  QObject *hook_obj = qvariant_cast<QObject*>(w->property(HOOK_UPDATE_WINDOW_GEOMETRY_OBJECT));
366 
367  if (!hook_obj) {
368  hook_obj = w;
369  }
370 
371  bool accept = true;
372  // 通知窗口即将开始更新其geometry
373  if (hook_obj->metaObject()->indexOfMethod(UPDATE_WINDOW_GEOMETRY_ENTRY "()") >= 0) {
374  QMetaObject::invokeMethod(hook_obj, UPDATE_WINDOW_GEOMETRY_ENTRY,
375  Qt::DirectConnection, Q_RETURN_ARG(bool, accept));
376  }
377 
378  if (!accept) {
379  // 中断操作
380  return;
381  }
382 
383  QRect nativeGeo;
384  // 尝试从窗口对象获取新的geometry
385  if (hook_obj->metaObject()->indexOfMethod(UPDATE_WINDOW_GEOMETRY_GEOMETRY "()") >= 0) {
386  QMetaObject::invokeMethod(hook_obj, UPDATE_WINDOW_GEOMETRY_GEOMETRY,
387  Qt::DirectConnection, Q_RETURN_ARG(QRect, nativeGeo));
388  }
389 
390  if (!nativeGeo.isValid()) {
391  nativeGeo = w->handle()->geometry();
392 
393  qreal scale = QHighDpiScaling::factor(w);
394  const QPoint &cursor_pos = w->screen()->handle()->cursor()->pos();
395  // 如果窗口是active的，且鼠标处于窗口内，应当移动窗口位置以保持鼠标相对窗口的位置
396  if (w->isActive()) {
397  const QMargins &frame_margins = w->handle()->frameMargins();
398  const QRect &frame_rect = nativeGeo.marginsAdded(frame_margins);
399 
400  if (frame_rect.contains(cursor_pos)) {
401  nativeGeo.moveTopLeft(cursor_pos + (nativeGeo.topLeft() - cursor_pos) * currentGeo.width() * scale / nativeGeo.width());
402  }
403  }
404 
405  nativeGeo.setSize(currentGeo.size() * scale);
406  }
407 
408  bool positionAutomaticSave = QWindowPrivate::get(w)->positionAutomatic;
409  QWindowPrivate::get(w)->positionAutomatic = false;
410  w->handle()->setGeometry(nativeGeo);
411  QWindowPrivate::get(w)->positionAutomatic = positionAutomaticSave;
412  // 请求重绘
413  QGuiApplication::sendEvent(w, new QEvent(QEvent::UpdateRequest));
414  // 通知窗口geometry更新结束
415  if (hook_obj->metaObject()->indexOfMethod(UPDATE_WINDOW_GEOMETRY_EXIT "()") >= 0) {
416  QMetaObject::invokeMethod(hook_obj, UPDATE_WINDOW_GEOMETRY_EXIT);
417  }
418  }
419 }

557 // 使用 QT_SCREEN_SCALE_FACTORS 为每个屏幕设置不同的缩放比之后，Qt会自动将 dpi 除以主屏的
558 // 缩放倍数，以此来避免字体被放大。font dpi会影响未设置pixel size的QFont，默认情况下，
559 // QGuiApplication::font() 不会设置pixel size，因此，使用分屏幕设置不同缩放比后，字体却还
560 // 是缩放前的大小。
561 // 此处，如果设置了 ScreenScaleFactors，但未指定 ScaleLogcailDpi 时，默认将其重设回主屏
562 // 的 logicalDpi。
563 static bool updateScaleLogcailDpi(const QPair<qreal, qreal> &dpi)
564 {
565  bool ok = dpi.first >= 0 && dpi.second >= 0;
566 #if QT_VERSION < QT_VERSION_CHECK(5,14,0)
567  if (dpi.first > 0) {
568  QHighDpiScaling::m_logicalDpi.first = dpi.first;
569  } else if (qIsNull(dpi.first)) {
570  QHighDpiScaling::m_logicalDpi.first = qGuiApp->primaryScreen()->handle()->logicalDpi().first;
571  }
572 
573  if (dpi.second > 0) {
574  QHighDpiScaling::m_logicalDpi.second = dpi.second;
575  } else if (qIsNull(dpi.second)) {
576  QHighDpiScaling::m_logicalDpi.second = qGuiApp->primaryScreen()->handle()->logicalDpi().second;
577  }
578 #else
579  QHighDpiScaling::m_usePixelDensity = false;
580 #endif
581  return ok;
582 }

584 static QByteArray getEnvValueByScreenScaleFactors(const QByteArray &data)
585 {
586  QByteArray envValue;
587
