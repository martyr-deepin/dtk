91  static quint64 entryCacheKey(const ScalableEntry *color_entry, const QIcon::Mode mode, const QIcon::State state)
92  {
93  return quint64(color_entry) ^ (quint64(mode) << 56) ^ (quint64(state) << 48);
94  }

96  QPixmap followColorPixmap(ScalableEntry *color_entry, const QSize &size, QIcon::Mode mode, QIcon::State state)
97  {
98  const QString &color_scheme = DEEPIN_QT_THEME::colorScheme.localData();
99  quint64 cache_key = entryCacheKey(color_entry, mode, state);
100  const QString &cache_color_scheme = entryToColorScheme.value(cache_key);
101  QPixmap pm = color_scheme == cache_color_scheme ? color_entry->svgIcon.pixmap(size, mode, state) : QPixmap();
102  // Note: not checking the QIcon::isNull(), because in Qt5.10 the isNull() is not reliable
103  // for svg icons desierialized from stream (see https://codereview.qt-project.org/#/c/216086/)
104  if (pm.isNull())
105  {
106  // The following lines are adapted and updated from KDE's "kiconloader.cpp" ->
107  // KIconLoaderPrivate::processSvg() and KIconLoaderPrivate::createIconImage().
108  // They read the SVG color scheme of SVG icons and give images based on the icon mode.
109  QHash<int, QByteArray> svg_buffers;
110  QFile device{color_entry->filename};
111  if (device.open(QIODevice::ReadOnly))
112  {
113  // Note: indexes are assembled as in qtsvg (QSvgIconEnginePrivate::hashKey())
114  QMap<int, QString> style_sheets;
115  style_sheets[(mode<<4)|state] = QStringLiteral(".ColorScheme-Text{color:%1;}").arg(color_scheme);
116  QMap<int, QSharedPointer<QXmlStreamWriter> > writers;
117  for (auto i = style_sheets.cbegin(); i != style_sheets.cend(); ++i)
118  {
119  writers[i.key()].reset(new QXmlStreamWriter{&svg_buffers[i.key()]});
120  }
121 
122  QXmlStreamReader xmlReader(&device);
123  while (!xmlReader.atEnd())
124  {
125  if (xmlReader.readNext() == QXmlStreamReader::StartElement
126  && xmlReader.qualifiedName() == QLatin1String("style")
127  && xmlReader.attributes().value(QLatin1String("id")) == QLatin1String("current-color-scheme"))
128  {
129  for (auto i = style_sheets.cbegin(); i != style_sheets.cend(); ++i)
130  {
131  QXmlStreamWriter & writer = *writers[i.key()];
132  writer.writeStartElement(QLatin1String("style"));
133  writer.writeAttributes(xmlReader.attributes());
134  writer.writeCharacters(*i);
135  writer.writeEndElement();
136  }
137  while (xmlReader.tokenType() != QXmlStreamReader::EndElement)
138  xmlReader.readNext();
139  } else if (xmlReader.tokenType() != QXmlStreamReader::Invalid)
140  {
141  for (auto i = style_sheets.cbegin(); i != style_sheets.cend(); ++i)
142  {
143  writers[i.key()]->writeCurrentToken(xmlReader);
144  }
145  }
146  }
147  // duplicate the contents also for opposite state
148 // svg_buffers[(QIcon::Normal<<4)|QIcon::On] = svg_buffers[(QIcon::Normal<<4)|QIcon::Off];
149 // svg_buffers[(QIcon::Selected<<4)|QIcon::On] = svg_buffers[(QIcon::Selected<<4)|QIcon::Off];
150  }
151  // use the QSvgIconEngine
152  // - assemble the content as it is done by the operator <<(QDataStream &s, const QIcon &icon)
153  // (the QSvgIconEngine::key() + QSvgIconEngine::write())
154  // - create the QIcon from the content by usage of the QIcon::operator >>(QDataStream &s, const QIcon &icon)
155  // (icon with the (QSvgIconEngine) will be used)
156  QByteArray icon_arr;
157  QDataStream str{&icon_arr, QIODevice::WriteOnly};
158  str.setVersion(QDataStream::Qt_4_4);
159  QHash<int, QString> filenames;
160  filenames[0] = color_entry->filename; // Note: filenames are ignored in the QSvgIconEngine::read()
161  str << QStringLiteral("svg") << filenames << static_cast<int>(0)/*isCompressed*/ << svg_buffers << static_cast<int>(0)/*hasAddedPimaps*/;
162 
163  QDataStream str_read{&icon_arr, QIODevice::ReadOnly};
164  str_read.setVersion(QDataStream::Qt_4_4);
165 
166  str_read >> color_entry->svgIcon;
167  pm = color_entry->svgIcon.pixmap(size, mode, state);
168 
169  // load the icon directly from file, if still null
170  if (pm.isNull())
171  {
172  color_entry->svgIcon = QIcon(color_entry->filename);
173  pm = color_entry->svgIcon.pixmap(size, mode, state);
174  }
175 
176  entryToColorScheme[cache_key] = color_scheme;
177  }
178 
179  return pm;
180  }

182  QPixmap pixmapByEntry(QIconLoaderEngineEntry *entry, const QSize &size, QIcon::Mode mode, QIcon::State state)
183  {
184  if (!XdgIcon::followColorScheme()) {
185  DEEPIN_QT_THEME::colorScheme.setLocalData(QString());
186 
187  return entry->pixmap(size, mode, state);
188  }
189 
190  QPixmap pixmap;
191  char *type_name = abi::__cxa_demangle(typeid(*entry).name(), 0, 0, 0);
192 
193  if (type_name == QByteArrayLiteral("ScalableFollowsColorEntry")) {
194  if (DEEPIN_QT_THEME::colorScheme.localData().isEmpty()) {
195  const QPalette &pal = qApp->palette();
196  DEEPIN_QT_THEME::colorScheme.setLocalData(mode == QIcon::Selected ? pal.highlightedText().color().name() : pal.windowText().color().name());
197  }
198 
199  pixmap = followColorPixmap(static_cast<ScalableEntry*>(entry), size, mode, state);
200  } else {
201  pixmap = entry->pixmap(size, mode, state);
202  }
203 
204  free(type_name);
205  DEEPIN_QT_THEME::colorScheme.setLocalData(QString());
206 
207  return pixmap;
208  }

258  void virtual_hook(int id, void *data) override {
259 #if QT_VERSION >= QT_VERSION_CHECK(5, 9, 0)
260  if (id == QIconEngine::ScaledPixmapHook)
261  {
262  engine->ensureLoaded();
263 
264  QIconEngine::ScaledPixmapArgument &arg = *reinterpret_cast<QIconEngine::ScaledPixmapArgument*>(data);
265  // QIcon::pixmap() multiplies size by the device pixel ratio.
266  const int integerScale = qCeil(arg.scale);
267  QIconLoaderEngineEntry *entry = engine->entryForSize(arg.size / integerScale, integerScale);
268  // 先禁用缩放，因为此size是已经缩放过的
269  bool useHighDpiPixmap = qGuiApp->testAttribute(Qt::AA_UseHighDpiPixmaps);
270  qGuiApp->setAttribute(Qt::AA_UseHighDpiPixmaps, false);
271  arg.pixmap = entry ? pixmapByEntry(entry, arg.size, arg.mode, arg.state) : QPixmap();
272  qGuiApp->setAttribute(Qt::AA_UseHighDpiPixmaps, useHighDpiPixmap);
273  DEEPIN_QT_THEME::colorScheme.setLocalData(QString());
274 
275  return;
276  }
277 #endif
278 
279  return engine->virtual_hook(id, data);
280  }

336 static void updateWindowGeometry(QWindow *w)
337 {
338  if (w->type() == Qt::ForeignWindow || w->type() == Qt::Desktop) {
339  return;
340  }
341 
342  if (!w->handle() || !w->isTopLevel())
343  return;
344 
345  if (w->property(DNOT_UPDATE_WINDOW_GEOMETRY).toBool()) {
346  QWindowSystemInterfacePrivate::GeometryChangeEvent gce(w, QHighDpi::fromNativePixels(w->handle()->geometry(), w)
347  #if QT_VERSION < QT_VERSION_CHECK(5, 10, 0)
348  , QRect()
349  #endif
350  );
351  QGuiApplicationPrivate::processGeometryChangeEvent(&gce);
352  } else {// 通知窗口大小发送改变
353  const QRect currentGeo = QWindowPrivate::get(w)->geometry;
354 
355  if (!currentGeo.isValid())
356  return;
357 
358  // 对外提供能hook自动更新窗口大小的方法
359  QObject *hook_obj = qvariant_cast<QObject*>(w->property(HOOK_UPDATE_WINDOW_GEOMETRY_OBJECT));
360 
361  if (!hook_obj) {
362  hook_obj = w;
363  }
364 
365  bool accept = true;
366  // 通知窗口即将开始更新其geometry
367  if (hook_obj->metaObject()->indexOfMethod(UPDATE_WINDOW_GEOMETRY_ENTRY "()") >= 0) {
368  QMetaObject::invokeMethod(hook_obj, UPDATE_WINDOW_GEOMETRY_ENTRY,
369  Qt::DirectConnection, Q_RETURN_ARG(bool, accept));
370  }
371 
372  if (!accept) {
373  // 中断操作
374  return;
375  }
376 
377  QRect nativeGeo;
378  // 尝试从窗口对象获取新的geometry
379  if (hook_obj->metaObject()->indexOfMethod(UPDATE_WINDOW_GEOMETRY_GEOMETRY "()") >= 0) {
380  QMetaObject::invokeMethod(hook_obj, UPDATE_WINDOW_GEOMETRY_GEOMETRY,
381  Qt::DirectConnection, Q_RETURN_ARG(QRect, nativeGeo));
382  }
383 
384  if (!nativeGeo.isValid()) {
385  nativeGeo = w->handle()->geometry();
386 
387  qreal scale = QHighDpiScaling::factor(w);
388  const QPoint &cursor_pos = w->screen()->handle()->cursor()->pos();
389  // 如果窗口是active的，且鼠标处于窗口内，应当移动窗口位置以保持鼠标相对窗口的位置
390  if (w->isActive()) {
391  const QMargins &frame_margins = w->handle()->frameMargins();
392  const QRect &frame_rect = nativeGeo.marginsAdded(frame_margins);
393 
394  if (frame_rect.contains(cursor_pos)) {
395  nativeGeo.moveTopLeft(cursor_pos + (nativeGeo.topLeft() - cursor_pos) * currentGeo.width() * scale / nativeGeo.width());
396  }
397  }
398 
399  nativeGeo.setSize(currentGeo.size() * scale);
400  }
401 
402  bool positionAutomaticSave = QWindowPrivate::get(w)->positionAutomatic;
403  QWindowPrivate::get(w)->positionAutomatic = false;
404  w->handle()->setGeometry(nativeGeo);
405  QWindowPrivate::get(w)->positionAutomatic = positionAutomaticSave;
406  // 请求重绘
407  QGuiApplication::sendEvent(w, new QEvent(QEvent::UpdateRequest));
408  // 通知窗口geometry更新结束
409  if (hook_obj->metaObject()->indexOfMethod(UPDATE_WINDOW_GEOMETRY_EXIT "()") >= 0) {
410  QMetaObject::invokeMethod(hook_obj, UPDATE_WINDOW_GEOMETRY_EXIT);
411  }
412  }
413 }

551 // 使用 QT_SCREEN_SCALE_FACTORS 为每个屏幕设置不同的缩放比之后，Qt会自动将 dpi 除以主屏的
552 // 缩放倍数，以此来避免字体被放大。font dpi会影响未设置pixel size的QFont，默认情况下，
553 // QGuiApplication::font() 不会设置pixel size，因此，使用分屏幕设置不同缩放比后，字体却还
554 // 是缩放前的大小。
555 // 此处，如果设置了 ScreenScaleFactors，但未指定 ScaleLogcailDpi 时，默认将其重设回主屏
556 // 的 logicalDpi。
557 static bool updateScaleLogcailDpi(const QPair<qreal, qreal> &dpi)
558 {
559  bool ok = dpi.first >= 0 && dpi.second >= 0;
560 #if QT_VERSION < QT_VERSION_CHECK(5,14,0)
561  if (dpi.first > 0) {
562  QHighDpiScaling::m_logicalDpi.first = dpi.first;
563  } else if (qIsNull(dpi.first)) {
564  QHighDpiScaling::m_logicalDpi.first = qGuiApp->primaryScreen()->handle()->logicalDpi().first;
565  }
566 
567  if (dpi.second > 0) {
568  QHighDpiScaling::m_logicalDpi.second = dpi.second;
569  } else if (qIsNull(dpi.second)) {
570  QHighDpiScaling::m_logicalDpi.second = qGuiApp->primaryScreen()->handle()->logicalDpi().second;
571  }
572 #else
573  QHighDpiScaling::m_usePixelDensity = false;
574 #endif
575  return ok;
576 }

578 static QByteArray getEnvValueByScreenScaleFactors(const QByteArray &data)
579 {
580  QByteArray envValue;
581 
582  if (!isGenericScreenScaleFactors(data)) {
583  envValue = data;
584  } else if (!data.isEmpty()) {
585  // 这种情况下，在Qt的实现中，只会将值应用给第一个屏幕，因此此处需要给所有屏幕都设置通用的值
586  int screen_count = qApp->screens().count();
587 
588  // 为除了最后一个屏幕的其它屏幕添加缩放比
589  while (--screen_count) {
590  envValue.append(data).append(';');
