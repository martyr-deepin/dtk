38 void DInputDialogPrivate::init()
39 {
40  D_Q(DInputDialog);
41 
42  q->addButtons(QStringList() << QApplication::translate("DInputDialog", "Cancel")
43  << QApplication::translate("DInputDialog", "Confirm"));
44  q->setOnButtonClickedClose(false);
45  q->setDefaultButton(1);
46 
47  cancelButton = q->getButton(0);
48  okButton = q->getButton(1);
49 
50  lineEdit = new DLineEdit;
51  spinBox = new DSpinBox;
52  doubleSpinBox = new DDoubleSpinBox;
53  comboBox = new QComboBox;
54 
55  q->addSpacing(10);
56  q->addContent(lineEdit);
57  q->addContent(spinBox);
58  q->addContent(doubleSpinBox);
59  q->addContent(comboBox);
60 
61  q->setInputMode(DInputDialog::TextInput);
62 
63  q->connect(lineEdit, &DLineEdit::textChanged, q, &DInputDialog::textValueChanged);
64  q->connect(lineEdit, &DLineEdit::alertChanged, q, &DInputDialog::textAlertChanged);
65  q->connect(comboBox, &QComboBox::currentTextChanged, q, &DInputDialog::textValueChanged);
66  q->connect(comboBox, SIGNAL(currentTextChanged(QString)), q, SIGNAL(textValueChanged(QString)));
67  q->connect(spinBox, SIGNAL(valueChanged(int)), q, SIGNAL(intValueChanged(int)));
68  q->connect(doubleSpinBox, SIGNAL(valueChanged(double)), q, SIGNAL(doubleValueChanged(double)));
69  q->connect(cancelButton, &QAbstractButton::clicked, q, [q] {
70  q->done(QDialog::Rejected);
71  q->cancelButtonClicked();
72  });
73  q->connect(okButton, &QAbstractButton::clicked, q, [q, this] {
74  q->done(QDialog::Accepted);
75  q->okButtonClicked();
76 
77  switch (inputMode) {
78  case DInputDialog::TextInput:
79  q->textValueSelected(lineEdit->text());
80  break;
81  case DInputDialog::ComboBox:
82  q->textValueSelected(comboBox->currentText());
83  break;
84  case DInputDialog::IntInput:
85  q->intValueSelected(spinBox->value());
86  break;
87  case DInputDialog::DoubleInput:
88  q->doubleValueSelected(doubleSpinBox->value());
89  break;
90  default:
91  break;
92  }
93  });
94 }

260 void DInputDialog::setTextEchoMode(QLineEdit::EchoMode mode)
261 {
262  D_D(DInputDialog);
263 
264  if(mode == d->lineEdit->echoMode())
265  return;
266 
267  DLineEdit *edit;
268 
269  if(mode == QLineEdit::Normal) {
270  edit = new DLineEdit;
271  } else {
272  edit = new DPasswordEdit;
273 
274  edit->setEchoMode(mode);
275  }
276 
277  edit->setClearButtonEnabled(true);
278 
279  disconnect(d->lineEdit, &DLineEdit::alertChanged, this, &DInputDialog::textAlertChanged);
280  disconnect(d->lineEdit, &DLineEdit::textChanged, this, &DInputDialog::textValueChanged);
281  connect(edit, &DLineEdit::alertChanged, this, &DInputDialog::textAlertChanged);
282  connect(edit, &DLineEdit::textChanged, this, &DInputDialog::textValueChanged);
283 
284  edit->setText(d->lineEdit->text());
285  insertContent(d->getContentLayout()->indexOf(d->lineEdit), edit);
286  removeContent(d->lineEdit);
287 
288  d->lineEdit = edit;
289 }
