137 void DLineEdit::showAlertMessage(const QString &text, int duration)
138 {
139  D_D(DLineEdit);
140 
141  if (!d->tooltip) {
142  d->tooltip = new DArrowRectangle(DArrowRectangle::ArrowTop, this);
143  d->tooltip->setObjectName("AlertTooltip");
144 
145  QLabel *label = new QLabel(d->tooltip);
146 
147  label->setWordWrap(true);
148  label->setMaximumWidth(width());
149  d->tooltip->setContent(label);
150  d->tooltip->setBackgroundColor(DThemeManager::instance()->theme(this) == "light" ? Qt::white : Qt::black);
151  d->tooltip->setArrowX(15);
152  d->tooltip->setArrowHeight(5);
153 
154  QTimer::singleShot(duration, d->tooltip, [d] {
155  d->tooltip->deleteLater();
156  d->tooltip = Q_NULLPTR;
157  });
158  }
159 
160  QLabel *label = qobject_cast<QLabel *>(d->tooltip->getContent());
161 
162  if (!label) {
163  return;
164  }
165 
166  label->setText(text);
167  label->adjustSize();
168 
169  const QPoint &pos = mapToGlobal(QPoint(15, height()));
170 
171  d->tooltip->show(pos.x(), pos.y());
172 }

192 void DLineEdit::setIconVisible(bool visible)
193 {
194  Q_D(DLineEdit);
195 
196  if (visible == d->m_rightIcon->isVisible()) {
197  return;
198  }
199 
200  d->m_rightIcon->setVisible(visible);
201 
202  if (visible) {
203 
204  addAction(d->m_iconAction, TrailingPosition);
205 #ifndef Q_OS_WIN
206  QLineEditPrivate *d_d = reinterpret_cast<QLineEditPrivate*>(d_ptr.data());
207  if (d_d->trailingSideWidgets.size() > 1) {
208  if ((*(d_d->trailingSideWidgets.end() - 1)).action == d->m_iconAction) {
209  d_d->trailingSideWidgets.insert(d_d->trailingSideWidgets.begin(), *d_d->trailingSideWidgets.erase(d_d->trailingSideWidgets.end() - 1));
210  QResizeEvent resize_event(size(), size());
211  qApp->sendEvent(this, &resize_event);
212  }
213  }
214 #endif
215  } else {
216  removeAction(d->m_iconAction);
217  }
218 }
