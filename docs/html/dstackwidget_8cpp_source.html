87 void DSlideStackWidgetTransition::beginTransition(const TransitionInfo &info)
88 {
89  DAbstractStackWidgetTransition::beginTransition(info);
90 
91  info.oldWidget->move(0, 0);
92  if(info.type == Push) {
93  info.newWidget->move(info.stackWidget->width(), 0);
94  } else if(info.type == Pop) {
95  info.newWidget->move(-info.stackWidget->width(), 0);
96  }
97 
98  info.oldWidget->show();
99  info.newWidget->show();
100 
101  animation()->setStartValue(0);
102  animation()->setEndValue(-info.newWidget->x());
103  animation()->start();
104 }

346 void DStackWidget::popWidget(QWidget *widget, bool isDelete, int count, bool enableTransition)
347 {
348  Q_D(DStackWidget);
349 
350  int i = widget ? indexOf(widget) : currentIndex();
351 
352  if(i < 0 || i >= depth())
353  return;
354 
355  bool current_widget_deleted = false;
356 
357  while(count-- > 0){
358  QWidget *tmp_widget = d->widgetList[i];
359 
360  if(tmp_widget == currentWidget()) {
361  current_widget_deleted = true;
362  } else if(isDelete) {
363  tmp_widget->deleteLater();
364  }
365 
366  d->widgetList.removeAt(i);
367  }
368 
369  if(current_widget_deleted && isDelete){
370  if(enableTransition && depth()){
371  d->trashWidgetList << d->currentWidget;
372  } else if(d->currentWidget) {
373  d->currentWidget->deleteLater();
374  d->currentWidget = nullptr;
375  }
376  }
377 
378  setCurrentIndex(depth() - 1, DAbstractStackWidgetTransition::Pop, enableTransition && current_widget_deleted);
379 }

421 void DStackWidget::setTransition(DAbstractStackWidgetTransition *transition)
422 {
423  Q_D(DStackWidget);
424 
425  if(d->transition){
426  d->transition->deleteLater();
427  }
428 
429  transition->setParent(this);
430  d->transition = transition;
431 
432  connect(transition->animation(), &QVariantAnimation::stateChanged,
433  this, [this, d](QAbstractAnimation::State newState, QAbstractAnimation::State oldState){
434  if(newState == QVariantAnimation::Stopped) {
435  busyChanged(false);
436  qDeleteAll(d->trashWidgetList);
437  d->trashWidgetList.clear();
438 
439  Q_EMIT switchWidgetFinished();
440  } else if(oldState == QVariantAnimation::Stopped) {
441  busyChanged(true);
442  }
443  });
444 }

475 void DStackWidget::setCurrentIndex(int currentIndex, DAbstractStackWidgetTransition::TransitionType type,
476  bool enableTransition)
477 {
478  Q_D(DStackWidget);
479 
480  if(enableTransition && currentWidget() && currentIndex >= 0) {
481  DAbstractStackWidgetTransition::TransitionInfo info;
482  info.stackWidget = this;
483  info.oldWidget = currentWidget();
484  info.newWidget = getWidgetByIndex(depth() - 1);
485  info.type = type;
486 
487  d->setCurrentIndex(currentIndex);
488  d->transition->beginTransition(info);
489  } else {
490  if(currentWidget()) {
491  currentWidget()->hide();
492  }
493 
494  d->setCurrentIndex(currentIndex);
495 
496  if(currentWidget()) {
497  currentWidget()->move(0, 0);
498  currentWidget()->show();
499  }
500 
501  Q_EMIT switchWidgetFinished();
502  }
