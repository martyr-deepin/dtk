232 QSize DArrowRectangle::getFixedSize()
233 {
234  D_D(DArrowRectangle);
235 
236  if (d->m_content) {
237  qreal delta = (d->m_handle ? 0 : shadowBlurRadius() + d->m_shadowDistance) + margin();
238 
239  switch (d->m_arrowDirection) {
240  case ArrowLeft:
241  case ArrowRight:
242  return QSize(d->m_content->width() + delta * 2 + d->m_arrowHeight - ((d->floatMode == FloatWidget) ? delta : 0),
243  d->m_content->height() + delta * 2);
244  case ArrowTop:
245  case ArrowBottom:
246  return QSize(d->m_content->width() + delta * 2,
247  d->m_content->height() + delta * 2 + d->m_arrowHeight - ((d->floatMode == FloatWidget) ? delta : 0));
248  }
249  }
250 
251  return QSize(0, 0);
252 }

819 QPainterPath DArrowRectanglePrivate::getLeftCornerPath()
820 {
821  D_Q(DArrowRectangle);
822 
823  QRect rect = q->rect();
824 
825  if (!m_handle) {
826  qreal delta = q->shadowBlurRadius() + m_shadowDistance;
827 
828  if (DApplication::isDXcbPlatform())
829  rect = rect.marginsRemoved(QMargins((DArrowRectangle::FloatWidget == floatMode) ? 0 : delta, delta, delta - q->margin(), delta));
830  else
831  rect = rect.marginsRemoved(QMargins((DArrowRectangle::FloatWidget == floatMode) ? 0 : 0, delta,
832  (DArrowRectangle::FloatWidget == floatMode) ? delta - q->margin() : delta *2, delta));
833  }
834 
835  QPoint cornerPoint(rect.x(), rect.y() + (m_arrowY > 0 ? m_arrowY : (rect.height() / 2)));
836  QPoint topLeft(rect.x() + m_arrowHeight, rect.y());
837  QPoint topRight(rect.x() + rect.width(), rect.y());
838  QPoint bottomRight(rect.x() + rect.width(), rect.y() + rect.height());
839  QPoint bottomLeft(rect.x() + m_arrowHeight, rect.y() + rect.height());
840  int radius = m_radius > (rect.height() / 2) ? (rect.height() / 2) : m_radius;
841  int arrowWidth = m_arrowWidth - m_arrowHeight;
842  int widgetRadius = rect.height() > 4 * radius ? 2 * radius : radius;
843  if (!radiusEnabled()) {
844  radius = 0;
845  }
846 
847  QPainterPath border;
848  border.moveTo(topLeft.x() + arrowWidth, topLeft.y());
849  border.lineTo(topRight.x() - radius, topRight.y());
850  border.arcTo(topRight.x() - 2 * radius, topRight.y(), 2 * radius, 2 * radius, 90, -90);
851  border.lineTo(bottomRight.x(), bottomRight.y() - radius);
852  border.arcTo(bottomRight.x() - 2 * radius, bottomRight.y() - 2 * radius, 2 * radius, 2 * radius, 0, -90);
853 
854  if (radiusArrowStyleEnable) {
855  border.lineTo(bottomLeft.x() + arrowWidth, bottomLeft.y());
856  border.arcTo(bottomLeft.x(), bottomLeft.y() - widgetRadius, widgetRadius, widgetRadius, -90, -90);
857  border.lineTo(bottomLeft.x(), cornerPoint.y() + m_arrowWidth / 2 + radius / 2);
858 
859  border.cubicTo(QPointF(bottomLeft.x(), cornerPoint.y() + m_arrowWidth / 2 + radius / 2),
860  QPointF(bottomLeft.x(), cornerPoint.y() + m_arrowWidth / 2),
861  QPointF(cornerPoint.x() + m_arrowHeight / 2, cornerPoint.y() + m_arrowWidth / 4.5));
862 
863  border.cubicTo(QPointF(cornerPoint.x() + m_arrowHeight / 2, cornerPoint.y() + m_arrowWidth / 4.5),
864  QPointF(cornerPoint),
865  QPointF(cornerPoint.x() + m_arrowHeight / 2, cornerPoint.y() - m_arrowWidth / 4.5));
866 
867  border.cubicTo(QPointF(cornerPoint.x() + m_arrowHeight / 2, cornerPoint.y() - m_arrowWidth / 4.5),
868  QPointF(cornerPoint.x() + m_arrowHeight, cornerPoint.y() - m_arrowWidth / 2),
869  QPointF(topLeft.x(), cornerPoint.y() - m_arrowWidth / 2 - radius / 2));
870 
871  border.lineTo(topLeft.x(), cornerPoint.y() - m_arrowWidth / 2 - radius / 2);
872  border.arcTo(topLeft.x(), topLeft.y(), widgetRadius, widgetRadius, -180, -90);
873  } else {
874  if (leftRightRadius) {
875  border.lineTo(bottomLeft.x() + arrowWidth, bottomLeft.y());
876  border.arcTo(bottomLeft.x(), bottomLeft.y() - widgetRadius, widgetRadius, widgetRadius, -90, -90);
877  border.lineTo(bottomLeft.x(), cornerPoint.y() + m_arrowWidth / 2);
878  } else {
879  border.lineTo(bottomLeft.x(), bottomLeft.y());
880 
881  if (cornerPoint.y() > m_arrowWidth)
882  border.lineTo(bottomLeft.x(), cornerPoint.y() + m_arrowWidth / 2);
883  }
884 
885  border.lineTo(cornerPoint);
886 
887  if (leftRightRadius) {
888  border.lineTo(topLeft.x(), cornerPoint.y() - m_arrowWidth / 2);
889  border.lineTo(topLeft.x(), topLeft.y() + radius);
890  border.arcTo(topLeft.x(), topLeft.y(), widgetRadius, widgetRadius, -180, -90);
891  } else {
892  if (cornerPoint.y() > m_arrowWidth)
893  border.lineTo(topLeft.x(), cornerPoint.y() - m_arrowWidth / 2);
894 
895  border.lineTo(topLeft.x(), topLeft.y());
896  }
897  }
898 
899  return border;
900 }

902 QPainterPath DArrowRectanglePrivate::getRightCornerPath()

905 
906  QRect rect = q->rect();
907 
908  if (!m_handle) {
909  qreal delta = q->shadowBlurRadius() + m_shadowDistance;
910 
911  if (DApplication::isDXcbPlatform())
912  rect = rect.marginsRemoved(QMargins((DArrowRectangle::FloatWidget == floatMode) ? delta - q->margin() : delta, delta, (DArrowRectangle::FloatWidget == floatMode) ? 0 : delta, delta));
913  else
914  rect = rect.marginsRemoved(QMargins((DArrowRectangle::FloatWidget == floatMode) ? delta -q->margin() : delta *2, delta, (DArrowRectangle::FloatWidget == floatMode) ? 0 : 0, delta));
915  }
916 
917  QPoint cornerPoint(rect.x() + rect.width(), rect.y() + (m_arrowY > 0 ? m_arrowY : rect.height() / 2));
918  QPoint topLeft(rect.x(), rect.y());
919  QPoint topRight(rect.x() + rect.width() - m_arrowHeight, rect.y());
920  QPoint bottomRight(rect.x() + rect.width() - m_arrowHeight, rect.y() + rect.height());
921  QPoint bottomLeft(rect.x(), rect.y() + rect.height());
922  int radius = this->m_radius > (rect.height() / 2) ? rect.height() / 2 : this->m_radius;
923  int widgetRadius = rect.height() >= 4 * radius ? 2 * radius : radius;
924 
925  if (!radiusEnabled()) {
926  radius = 0;
927  }
928 
929  QPainterPath border;
930  border.moveTo(topLeft.x(), topLeft.y());
931 
932  if (radiusArrowStyleEnable) {
933  border.lineTo(topRight.x() - radius, topRight.y());
934  border.arcTo(topRight.x() - widgetRadius, topRight.y(), widgetRadius, widgetRadius, 90, -90);
935  border.lineTo(topRight.x(), cornerPoint.y() - m_arrowWidth / 2 - radius / 2);
936 
937  border.cubicTo(QPointF(topRight.x(), cornerPoint.y() - m_arrowWidth / 2 - radius / 2),
938  QPointF(topRight.x(), cornerPoint.y() - m_arrowWidth / 2),
939  QPointF(cornerPoint.x() - m_arrowHeight / 2, cornerPoint.y() - m_arrowWidth / 4.5));
940 
941  border.cubicTo(QPointF(cornerPoint.x() - m_arrowHeight / 2, cornerPoint.y() - m_arrowWidth / 4.5),
942  QPointF(cornerPoint),
943  QPointF(cornerPoint.x() - m_arrowHeight / 2, cornerPoint.y() + m_arrowWidth / 4.5));
944 
945  border.cubicTo(QPointF(cornerPoint.x() - m_arrowHeight / 2, cornerPoint.y() + m_arrowWidth / 4.5),
946  QPointF(bottomRight.x(), cornerPoint.y() + m_arrowWidth / 2),
947  QPointF(bottomRight.x(), cornerPoint.y() + m_arrowWidth / 2 + radius / 2));
948 
949  border.lineTo(bottomRight.x(), cornerPoint.y() + m_arrowWidth / 2 + radius / 2);
950  border.arcTo(bottomRight.x() - widgetRadius, bottomRight.y() - widgetRadius, widgetRadius, widgetRadius, 0, -90);
951  } else {
952  if (leftRightRadius) {
953  border.lineTo(topRight.x() - radius, topRight.y());
954  border.arcTo(topRight.x() - widgetRadius, topRight.y(), widgetRadius, widgetRadius, 90, -90);
955  border.lineTo(topRight.x(), cornerPoint.y() - m_arrowWidth / 2);
956  } else {
957  border.lineTo(topRight.x(), topRight.y());
958 
959  if (cornerPoint.y() > m_arrowWidth)
960  border.lineTo(topRight.x(), cornerPoint.y() - m_arrowWidth / 2);
961  }
962 
963  border.lineTo(cornerPoint);
964 
965  if (leftRightRadius) {
966  border.lineTo(bottomRight.x(), cornerPoint.y() + m_arrowWidth / 2);
967  border.lineTo(bottomRight.x(), bottomRight.y() - radius);
968  border.arcTo(bottomRight.x() - widgetRadius, bottomRight.y() - widgetRadius, widgetRadius, widgetRadius, 0, -90);
969 
970  } else if (!leftRightRadius){
971  if (cornerPoint.y() > m_arrowWidth)
972  border.lineTo(bottomRight.x(), cornerPoint.y() + m_arrowWidth / 2);
973 
974  border.lineTo(bottomRight.x(), bottomRight.y());
975  }
976  }
977 
978  border.lineTo(bottomLeft.x() + radius, bottomLeft.y());
979  border.arcTo(bottomLeft.x(), bottomLeft.y() - 2 * radius, 2 * radius, 2 * radius, -90, -90);
980  border.lineTo(topLeft.x(), topLeft.y() + radius);
981  border.arcTo(topLeft.x(), topLeft.y(), 2 * radius, 2 * radius, 180, -90);
982 
983  return border;

986 QPainterPath DArrowRectanglePrivate::getTopCornerPath()
987 {
988  D_Q(DArrowRectangle);
989 
990  QRect rect = q->rect();
991 
992  if (!m_handle) {
993  qreal delta = q->shadowBlurRadius() + m_shadowDistance;
994  if (DApplication::isDXcbPlatform())
995  rect = rect.marginsRemoved(QMargins(delta, (DArrowRectangle::FloatWidget == floatMode) ? 0 : delta,
996  delta, (DArrowRectangle::FloatWidget == floatMode) ? delta -q->margin() : delta));
997  else
998  rect = rect.marginsRemoved(QMargins(delta, 0, delta, (DArrowRectangle::FloatWidget == floatMode) ? delta - q->margin() : delta * 2));
999  }
1000 
1001  QPoint cornerPoint(rect.x() + (m_arrowX > 0 ? m_arrowX : rect.width() / 2), rect.y());
1002  QPoint topLeft(rect.x(), rect.y() + m_arrowHeight);
1003  QPoint topRight(rect.x() + rect.width(), rect.y() + m_arrowHeight);
1004  QPoint bottomRight(rect.x() + rect.width(), rect.y() + rect.height());
1005  QPoint bottomLeft(rect.x(), rect.y() + rect.height());
1006  int radius = this->m_radius > (rect.height() / 2 - m_arrowHeight) ? rect.height() / 2 - m_arrowHeight : this->m_radius;
1007  if (!radiusEnabled()) {
1008  radius = 0;
1009  }
1010 
1011  QPainterPath border;
1012  border.moveTo(topLeft.x() + radius, topLeft.y());
1013 
1014  if (radiusArrowStyleEnable) {
1015  border.lineTo(cornerPoint.x() - m_arrowWidth / 2 - radius / 2, cornerPoint.y() + m_arrowHeight);
1016  border.cubicTo(QPointF(cornerPoint.x() - m_arrowWidth / 2 - radius / 2, topLeft.y()),
1017  QPointF(cornerPoint.x() - m_arrowWidth / 2, topLeft.y()),
1018  QPointF(cornerPoint.x() - m_arrowWidth / 4.5, cornerPoint.y() + m_arrowHeight / 2));
1019 
1020  border.cubicTo(QPointF(cornerPoint.x() - m_arrowWidth / 4.5, cornerPoint.y() + m_arrowHeight / 2),
1021  QPointF(cornerPoint),
1022  QPointF(cornerPoint.x() + m_arrowWidth / 4.5, cornerPoint.y() + m_arrowHeight / 2));
1023 
1024  border.cubicTo(QPointF(cornerPoint.x() + m_arrowWidth / 4.5, cornerPoint.y() + m_arrowHeight / 2),
1025  QPointF(cornerPoint.x() + m_arrowWidth / 2, topLeft.y()),
1026  QPointF(cornerPoint.x() + m_arrowWidth / 2 + radius / 2, cornerPoint.y() + m_arrowHeight));
1027  border.lineTo(cornerPoint.x() + m_arrowWidth / 2 + radius / 2, cornerPoint.y() + m_arrowHeight);
1028  } else {
1029  border.lineTo(cornerPoint.x() - m_arrowWidth / 2, cornerPoint.y() + m_arrowHeight);
1030  border.lineTo(cornerPoint);
1031  border.lineTo(cornerPoint.x() + m_arrowWidth / 2, cornerPoint.y() + m_arrowHeight);
