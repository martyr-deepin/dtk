85  template<typename Fun1, typename Fun2>
86  static bool overrideVfptrFun(quintptr *vfptr_t1, Fun1 fun1, quintptr *vfptr_t2, Fun2 fun2, bool forceWrite)
87  {
88  typedef QtPrivate::FunctionPointer<Fun1> FunInfo1;
89  typedef QtPrivate::FunctionPointer<Fun2> FunInfo2;
90 
91  //compilation error if the arguments does not match.
92  Q_STATIC_ASSERT_X((CheckCompatibleArguments<typename FunInfo1::Arguments, typename FunInfo2::Arguments>::value),
93  "Function1 and Function2 arguments are not compatible.");
94  Q_STATIC_ASSERT_X((CheckCompatibleArguments<QtPrivate::List<typename FunInfo1::ReturnType>, QtPrivate::List<typename FunInfo2::ReturnType>>::value),
95  "Function1 and Function2 return type are not compatible..");
96 
98  quintptr fun1_offset = toQuintptr(&fun1);
99  quintptr fun2_offset = toQuintptr(&fun2);
100 
101  if (fun1_offset < 0 || fun1_offset > UINT_LEAST16_MAX)
102  return false;
103 
104  quintptr *vfun = vfptr_t1 + fun1_offset / sizeof(quintptr);
105 
106  // if the fun2 is not virtual function
107  if (fun2_offset <= UINT_LEAST16_MAX) {
108  fun2_offset = *(vfptr_t2 + fun2_offset / sizeof(quintptr));
109  }
110 
111  if (forceWrite)
112  return forceWriteMemory(vfun, &fun2_offset, sizeof(fun2_offset));
113 
114  *vfun = fun2_offset;
115 
116  return true;
117  }
