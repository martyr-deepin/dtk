73 bool Style::drawMenuItemControl(const QStyleOption *option, QPainter *painter, const QWidget *widget) const
74 {
75  Q_D(const Style);
76 
77  painter->save();
78  // Draws one item in a popup menu.
79  if (const QStyleOptionMenuItem *menuItem = qstyleoption_cast<const QStyleOptionMenuItem *>(option)) {
80 // QColor highlightOutline = d->highlightedOutline(option->palette);
81  QColor highlight = option->palette.highlight().color();
82  if (menuItem->menuItemType == QStyleOptionMenuItem::Separator) {
83  int w = 0;
84  if (!menuItem->text.isEmpty()) {
85  painter->setFont(menuItem->font);
86  proxy()->drawItemText(painter, menuItem->rect.adjusted(Menu_SeparatorItemHMargin, 0, -Menu_SeparatorItemHMargin, 0), Qt::AlignLeft | Qt::AlignVCenter,
87  menuItem->palette, menuItem->state & State_Enabled, menuItem->text,
88  QPalette::Text);
89  w = menuItem->fontMetrics.width(menuItem->text) + Menu_SeparatorItemHMargin;
90  }
91  painter->setPen(m_palette->brush(PaletteExtended::Menu_SeparatorColor, option).color());
92  bool reverse = menuItem->direction == Qt::RightToLeft;
93  painter->drawLine(menuItem->rect.left() + Menu_SeparatorItemHMargin + (reverse ? 0 : w), menuItem->rect.center().y(),
94  menuItem->rect.right() - Menu_SeparatorItemHMargin - (reverse ? w : 0), menuItem->rect.center().y());
95  painter->restore();
96  return true;
97  }
98  bool selected = menuItem->state & State_Selected && menuItem->state & State_Enabled;
99  if (selected) {
100  QRect r = option->rect;
101  painter->fillRect(r, highlight);
102 // painter->setPen(QPen(highlightOutline));
103 // painter->drawRect(QRectF(r));
104  }
105  bool checkable = menuItem->checkType != QStyleOptionMenuItem::NotCheckable;
106  bool checked = menuItem->checked;
107 // bool sunken = menuItem->state & State_Sunken;
108 // bool enabled = menuItem->state & State_Enabled;
109 
110  bool ignoreCheckMark = false;
111  int checkcol = qMax(menuItem->maxIconWidth, 20);
112  QRect iconRect;
113 
114  if (qobject_cast<const QComboBox*>(widget) ||
115  (option->styleObject && option->styleObject->property("_q_isComboBoxPopupItem").toBool()))
116  ignoreCheckMark = true; //ignore the checkmarks provided by the QComboMenuDelegate
117 
118  if (!ignoreCheckMark) {
119  // Check
120  QRect checkRect(option->rect.left() + 10, option->rect.center().y() - 4, 14, 14);
121  checkRect = visualRect(menuItem->direction, menuItem->rect, checkRect);
122  if (checkable) {
123  if (!isVisibleMenuCheckBox() || (menuItem->checkType & QStyleOptionMenuItem::Exclusive)) {
124  // Radio button
125  if (checked/* || sunken*/) {
126  QStyleOptionMenuItem newMI = *menuItem;
127  newMI.rect = checkRect;
128 
129  drawDeepinStyleIcon("check", &newMI, painter, widget);
130  }
131  } else {
132  // Check box
133  if (menuItem->icon.isNull()) {
134  QStyleOptionButton box;
135  box.QStyleOption::operator=(*option);
136  box.rect = checkRect;
137  if (checked)
138  box.state |= State_On;
139  proxy()->drawPrimitive(PE_IndicatorCheckBox, &box, painter, widget);
140  }
141  }
142  }
143 
144  // draw check label
145  if (checkable) {
146  if (!menuItem->icon.isNull()) {
147  checkcol = qMax(menuItem->maxIconWidth * 2, 20);
148  }
149  else {
150  checkcol = qMax(menuItem->maxIconWidth, 20);
151  }
152 
153  QRect vCheckRect = visualRect(option->direction, menuItem->rect,
154  QRect(menuItem->rect.x() + 4, menuItem->rect.y(),
155  checkcol, menuItem->rect.height()));
156  iconRect = QRect(vCheckRect.x() + vCheckRect.width() / 2, vCheckRect.y(), vCheckRect.width() / 2, vCheckRect.height());
157  }
158  else {
159  checkcol = qMax(menuItem->maxIconWidth, 20);
160  iconRect = visualRect(option->direction, menuItem->rect,
161  QRect(menuItem->rect.x() + 4, menuItem->rect.y(),
162  checkcol, menuItem->rect.height()));
163  }
164 
165 
166  } else { //ignore checkmark
167  if (menuItem->icon.isNull())
168  checkcol = 0;
169  else
170  checkcol = menuItem->maxIconWidth;
171  }
172 
173  // Text and icon, ripped from windows style
174  bool dis = !(menuItem->state & State_Enabled);
175  bool act = menuItem->state & State_Selected;
176  const QStyleOption *opt = option;
177  const QStyleOptionMenuItem *menuitem = menuItem;
178 
179  QPainter *p = painter;
180 
181  // NOTE: If widget is QLineEdit and qApp is DApplication, will hide menu icon
182  if (!menuItem->icon.isNull() && (!isEdit(widget) || isVisibleMenuIcon())) {
183  QIcon::Mode mode = dis ? QIcon::Disabled : QIcon::Normal;
184  if (act && !dis)
185  mode = QIcon::Active;
186  QPixmap pixmap;
187 
188  int smallIconSize = proxy()->pixelMetric(PM_SmallIconSize, option, widget);
189  QSize iconSize(smallIconSize, smallIconSize);
190  if (const QComboBox *combo = qobject_cast<const QComboBox*>(widget))
191  iconSize = combo->iconSize();
192  if (checked)
193  pixmap = menuItem->icon.pixmap(iconSize, mode, QIcon::On);
194  else
195  pixmap = menuItem->icon.pixmap(iconSize, mode);
196 
197  const int pixw = pixmap.width() / pixmap.devicePixelRatio();
198  const int pixh = pixmap.height() / pixmap.devicePixelRatio();
199 
200  QRect pmr(0, 0, pixw, pixh);
201  pmr.moveCenter(iconRect.center());
202  painter->setPen(menuItem->palette.text().color());
203 // if (!ignoreCheckMark && checkable && checked) {
204 // QStyleOption opt = *option;
205 // if (act) {
206 // QColor activeColor = mergedColors(option->palette.background().color(),
207 // option->palette.highlight().color());
208 // opt.palette.setBrush(QPalette::Button, activeColor);
209 // }
210 // opt.state |= State_Sunken;
211 // opt.rect = vCheckRect;
212 // proxy()->drawPrimitive(PE_PanelButtonCommand, &opt, painter, widget);
213 // }
214  painter->drawPixmap(pmr.topLeft(), pixmap);
215  }
216  if (selected) {
217  painter->setPen(menuItem->palette.highlightedText().color());
218  } else {
219  painter->setPen(menuItem->palette.text().color());
220  }
221  int x, y, w, h;
222  menuitem->rect.getRect(&x, &y, &w, &h);
223  int tab = menuitem->tabWidth;
224  QColor discol;
225  if (dis) {
226  discol = menuitem->palette.brush(QPalette::Disabled, QPalette::Text).color();
227  p->setPen(discol);
228  }
229  int xm = Menu_ItemFrameWidth + checkcol + Menu_ItemHTextMargin + 2;
230  int xpos = menuitem->rect.x() + xm;
231 
232  QRect textRect(xpos, y + Menu_ItemVTextMargin, w - xm - Menu_PanelRightPadding - tab + 1, h - 2 * Menu_ItemVTextMargin);
233  QRect vTextRect = visualRect(opt->direction, menuitem->rect, textRect);
234  QString s = menuitem->text;
235  if (!s.isEmpty()) { // draw text
236  p->save();
237  QFont font = menuitem->font;
238  // font may not have any "hard" flags set. We override
239  // the point size so that when it is resolved against the device, this font will win.
240  // This is mainly to handle cases where someone sets the font on the window
241  // and then the combo inherits it and passes it onward. At that point the resolve mask
242  // is very, very weak. This makes it stonger.
243  font.setPointSizeF(QFontInfo(menuItem->font).pointSizeF());
244 
245  if (menuitem->menuItemType == QStyleOptionMenuItem::DefaultItem)
246  font.setBold(true);
247 
248  p->setFont(font);
249 
250  int t = s.indexOf(QLatin1Char('\t'));
251  int text_flags = Qt::AlignVCenter | Qt::TextShowMnemonic | Qt::TextDontClip | Qt::TextSingleLine;
252  if (!styleHint(SH_UnderlineShortcut, menuitem, widget))
253  text_flags |= Qt::TextHideMnemonic;
254  text_flags |= Qt::AlignLeft;
255 
256  // NOTE: hide short text
257  if (t >= 0 && isVisibleMenuShortText()) {
258  QRect vShortcutRect = visualRect(opt->direction, menuitem->rect,
259  QRect(textRect.topRight(), QPoint(menuitem->rect.right(), textRect.bottom())));
260  if (dis && !act && proxy()->styleHint(SH_EtchDisabledText, option, widget)) {
261  p->setPen(menuitem->palette.light().color());
262  p->drawText(vShortcutRect.adjusted(1, 1, 1, 1), text_flags, s.mid(t + 1));
263  p->setPen(discol);
264  }
265  p->drawText(vShortcutRect, text_flags, s.mid(t + 1));
266  s = s.left(t);
267  }
268 
269  if (dis && !act && proxy()->styleHint(SH_EtchDisabledText, option, widget)) {
270  p->setPen(menuitem->palette.light().color());
271  p->drawText(vTextRect.adjusted(1, 1, 1, 1), text_flags, s.left(t));
272  p->setPen(discol);
273  }
274  p->drawText(vTextRect, text_flags, s.left(t));
275  p->restore();
276  }
277 
278  // Arrow
279  if (menuItem->menuItemType == QStyleOptionMenuItem::SubMenu) {// draw sub menu arrow
280  int dim = (menuItem->rect.height() - 4) / 2;
281 // PrimitiveElement arrow;
282 // arrow = option->direction == Qt::RightToLeft ? PE_IndicatorArrowLeft : PE_IndicatorArrowRight;
283  int xpos = menuItem->rect.left() + menuItem->rect.width() - 3 - dim;
284  QRect vSubMenuRect = visualRect(option->direction, menuItem->rect,
285  QRect(xpos, menuItem->rect.top() + menuItem->rect.height() / 2 - dim / 2, dim, dim));
286  QStyleOptionMenuItem newMI = *menuItem;
287  newMI.rect = vSubMenuRect;
288  if (selected)
289  newMI.palette.setColor(QPalette::Foreground,
290  newMI.palette.highlightedText().color());
291 
292  drawDeepinStyleIcon("arrow-right", &newMI, painter, widget);
293  }
294  }
295  painter->restore();
296 
297  return true;
298 }

300 bool Style::drawMenuBarItemControl(const QStyleOption *option, QPainter *painter, const QWidget *widget) const
301 {
302  Style *style = CommonHelper::widgetStyle(widget);
303  if (!style) return false;
304 
305  const bool enabled(option->state & QStyle::State_Enabled);
306  const bool mouseOver(option->state & QStyle::State_MouseOver);
307  const bool sunken((option->state | QStyle::State_Sunken) == option->state);
308 
309  if (mouseOver || sunken) {
310  const QColor shadow( Qt::transparent );
311  const QBrush outline(style->m_palette->brush(PaletteExtended::PushButton_BorderBrush, option));
312  const QBrush background(style->m_palette->brush(PaletteExtended::PushButton_BackgroundBrush, option));
313 
314  // render
315  drawPushButtonFrame(painter, option->rect, background, outline, shadow );
316  }
317 
318  if (const QStyleOptionMenuItem *mbi = qstyleoption_cast<const QStyleOptionMenuItem *>(option)) {
319  uint alignment = Qt::AlignCenter | Qt::TextShowMnemonic | Qt::TextDontClip
320  | Qt::TextSingleLine;
321  if (!proxy()->styleHint(SH_UnderlineShortcut, mbi, widget))
322  alignment |= Qt::TextHideMnemonic;
323  int iconExtent = proxy()->pixelMetric(PM_SmallIconSize);
324  QPixmap pix = mbi->icon.pixmap(qt_getWindow(widget), QSize(iconExtent, iconExtent), (enabled) ? (mouseOver ? QIcon::Active : QIcon::Normal) : QIcon::Disabled);
325  if (!pix.isNull())
326  proxy()->drawItemPixmap(painter, mbi->rect, alignment, pix);
327  else
328  proxy()->drawItemText(painter, mbi->rect, alignment, mbi->palette, enabled,
329  mbi->text, QPalette::ButtonText);
330  }
331 
332  return true;
333 }
