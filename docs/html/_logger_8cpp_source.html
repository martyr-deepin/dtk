825 void Logger::write(const QDateTime& timeStamp, LogLevel logLevel, const char* file, int line, const char* function, const char* category,
826  const QString& message, bool fromLocalInstance)
827 {
828  Q_D(Logger);
829 
830  QMutexLocker locker(&d->loggerMutex);
831 
832  QString logCategory = QString::fromLatin1(category);
833  if (logCategory.isNull() && !d->defaultCategory.isNull())
834  logCategory = d->defaultCategory;
835 
836  bool wasWritten = false;
837  bool isGlobalInstance = this == globalInstance();
838  bool linkedToGlobal = isGlobalInstance && d->categories.value(logCategory, false);
839 
840  if (!logCategory.isNull())
841  {
842  QList<AbstractAppender*> appenders = d->categoryAppenders.values(logCategory);
843  if (appenders.length() == 0)
844  {
845  if (logCategory != d->defaultCategory && !linkedToGlobal && !fromLocalInstance)
846  std::cerr << "No appenders assotiated with category " << qPrintable(logCategory) << std::endl;
847  }
848  else
849  {
850  Q_FOREACH (AbstractAppender* appender, appenders)
851  appender->write(timeStamp, logLevel, file, line, function, logCategory, message);
852  wasWritten = true;
853  }
854  }
855 
856  // the default category is linked to the main logger appenders
857  // global logger instance also writes all linked categories to the main appenders
858  if (logCategory.isNull() || logCategory == d->defaultCategory || linkedToGlobal)
859  {
860  if (!d->appenders.isEmpty())
861  {
862  Q_FOREACH (AbstractAppender* appender, d->appenders)
863  appender->write(timeStamp, logLevel, file, line, function, logCategory, message);
864  wasWritten = true;
865  }
866  else
867  {
868  static bool noAppendersWarningShown = false;
869  if (!noAppendersWarningShown)
870  {
871 #if defined(Q_OS_ANDROID)
872  __android_ write(ANDROID_LOG_WARN, "Logger", "No appenders registered with logger");
873 #else
874  std::cerr << "No appenders registered with logger" << std::endl;
875 #endif
876  noAppendersWarningShown = true;
877  }
878  }
879  }
880 
881  // local logger instances send category messages to the global instance
882  if (!logCategory.isNull() && !isGlobalInstance)
883  globalInstance()->write(timeStamp, logLevel, file, line, function, logCategory.toLatin1(), message, true);
884 
885  if (!wasWritten && !fromLocalInstance)
886  {
887  // Fallback
888 #if defined(Q_OS_ANDROID)
889  QString result = QString(QLatin1String("<%2> %3")).arg(AbstractStringAppender::stripFunctionName(function)).arg(message);
890  __android_log_write(AndroidAppender::androidLogPriority(logLevel), "Logger", qPrintable(result));
891 #else
892  QString result = QString(QLatin1String("[%1] <%2> %3")).arg(levelToString(logLevel), -7)
893  .arg(AbstractStringAppender::stripFunctionName(function)).arg(message);
894  std::cerr << qPrintable(result) << std::endl;
895 #endif
896  }
897 
898  if (logLevel == Logger::Fatal)
899  abort();
900 }
