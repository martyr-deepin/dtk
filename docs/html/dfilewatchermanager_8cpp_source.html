72 DFileWatcher *DFileWatcherManager::add(const QString &filePath)
73 {
74  Q_D(DFileWatcherManager);
75 
76  DFileWatcher *watcher = d->watchersMap.value(filePath);
77 
78  if (watcher) {
79  return watcher;
80  }
81 
82  watcher = new DFileWatcher(filePath, this);
83 
84  connect(watcher, &DFileWatcher::fileAttributeChanged, this, [this](const QUrl & url) {
85  Q_EMIT fileAttributeChanged(url.toLocalFile());
86  });
87  connect(watcher, &DFileWatcher::fileClosed, this, [this](const QUrl & url) {
88  Q_EMIT fileClosed(url.toLocalFile());
89  });
90  connect(watcher, &DFileWatcher::fileDeleted, this, [this](const QUrl & url) {
91  Q_EMIT fileDeleted(url.toLocalFile());
92  });
93  connect(watcher, &DFileWatcher::fileModified, this, [this](const QUrl & url) {
94  Q_EMIT fileModified(url.toLocalFile());
95  });
96  connect(watcher, &DFileWatcher::fileMoved, this, [this](const QUrl & fromUrl, const QUrl & toUrl) {
97  Q_EMIT fileMoved(fromUrl.toLocalFile(), toUrl.toLocalFile());
98  });
99  connect(watcher, &DFileWatcher::subfileCreated, this, [this](const QUrl & url) {
100  Q_EMIT subfileCreated(url.toLocalFile());
101  });
102 
103  d->watchersMap[filePath] = watcher;
104  watcher->startWatcher();
105 
106  return watcher;
107 }
