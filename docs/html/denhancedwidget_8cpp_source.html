213 bool DEnhancedWidget::eventFilter(QObject *o, QEvent *e)
214 {
215  Q_D(DEnhancedWidget);
216 
217  if (o == d->target) {
218  switch (e->type()) {
219  case QEvent::Resize: {
220  QResizeEvent *event = static_cast<QResizeEvent *>(e);
221  if (event) {
222  QSize size = event->size();
223 
224  if (size.width() != d->old_size.width()) {
225  Q_EMIT widthChanged(size.width());
226  }
227 
228  if (size.height() != d->old_size.height()) {
229  Q_EMIT heightChanged(size.height());
230  }
231 
232  if (size != d->old_size) {
233  Q_EMIT sizeChanged(size);
234  }
235 
236  d->old_size = size;
237  }
238 
239  break;
240  }
241  case QEvent::Move: {
242  QMoveEvent *event = static_cast<QMoveEvent *>(e);
243 
244  if (event) {
245  QPoint pos = event->pos();
246 
247  if (pos.x() != d->old_pos.x()) {
248  Q_EMIT xChanged(pos.x());
249  }
250 
251  if (pos.y() != d->old_pos.y()) {
252  Q_EMIT yChanged(pos.y());
253  }
254 
255  if (pos != d->old_pos) {
256  Q_EMIT positionChanged(pos);
257  }
258 
259  d->old_pos = pos;
260  }
261 
262  break;
263  }
264  case QEvent::Show: {
265  Q_EMIT showed();
266 
267  break;
268  }
269  default:
270  break;
271  }
272  }
273 
274  return false;
275 }
