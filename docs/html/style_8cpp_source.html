159 void Style::polish(QWidget *w)
160 {
161  QCommonStyle::polish(w);
162 
163  if (qobject_cast<QPushButton *>(w)
164  || qobject_cast<QComboBox *>(w)
165  || qobject_cast<QScrollBar *>(w)
166  || qobject_cast<QCheckBox*>(w)
167  || qobject_cast<QRadioButton*>(w)
168  || qobject_cast<QToolButton*>(w)
169  || qobject_cast<QAbstractSpinBox*>(w)
170  || qobject_cast<QTabBar*>(w)
171  || isTabBarToolButton(w))
172  w->setAttribute(Qt::WA_Hover, true);
173 
174  if (qobject_cast<QScrollBar *>(w)) {
175  w->setAttribute(Qt::WA_OpaquePaintEvent, false);
176  }
177 
178  if (w->inherits("dcc::widgets::SliderAnnotation")) {
179  QFont font = w->font();
180  font.setPointSizeF(qApp->font().pointSizeF() - 1);
181  w->setFont(font);
182  }
183 
184  if (DApplication::isDXcbPlatform()) {
185  bool is_menu = qobject_cast<QMenu*>(w);
186  bool is_tip = w->inherits("QTipLabel");
187 
188  // 当窗口已经创建对应的native窗口，要判断当前是否已经设置了窗口背景透明
189  // Bug: https://github.com/linuxdeepin/internal-discussion/issues/323
190  if ((is_menu || is_tip) && w->windowHandle()) {
191  if (const QPlatformWindow *handle = w->windowHandle()->handle()) {
192  if (!w->testAttribute(Qt::WA_TranslucentBackground) && !handle->isExposed()) {
193  // 销毁现有的native窗口，否则设置Qt::WA_TranslucentBackground不会生效
194  class DQWidget : public QWidget {public: using QWidget::destroy;};
195  reinterpret_cast<DQWidget*>(w)->destroy(true, false);
196  }
197  }
198  }
199 
200  if (is_menu) {
201  DPlatformWindowHandle handle(w);
202 
203  if (DPlatformWindowHandle::isEnabledDXcb(w)) {
204  const QColor &color = m_palette->brush(PaletteExtended::Menu_BorderColor).color();
205 
206  if (color.isValid())
207  handle.setBorderColor(color);
208 
209  handle.setShadowOffset(QPoint(0, 4));
210  handle.setShadowRadius(15);
211  handle.setShadowColor(QColor(0, 0, 0, 100));
212  #ifdef DTK_SUPPORT_BLUR_WINDOW
213  handle.setEnableBlurWindow(true);
214  #endif
215  handle.setTranslucentBackground(true);
216 
217  w->setAttribute(Qt::WA_TranslucentBackground);
218  }
219  } else if (is_tip) {
220  DPlatformWindowHandle handle(w);
221 
222  if (DPlatformWindowHandle::isEnabledDXcb(w)) {
223  handle.setShadowOffset(QPoint(0, 2));
224  handle.setShadowRadius(4);
225 
226  w->setAttribute(Qt::WA_TranslucentBackground);
227  }
228  }
229  }
230 
231  // NOTE(lxz): QLineEdit not refresh style after theme changed
232  if (QLineEdit *edit = qobject_cast<QLineEdit*>(w)) {
233  bool clear_button_visible = edit->isClearButtonEnabled();
234 
235  if (clear_button_visible) {
236  const QList<QAction*> &actions = edit->findChildren<QAction*>("_q_qlineeditclearaction");
237 
238  if (!actions.isEmpty()) {
239  actions.first()->setIcon(standardIcon(QStyle::SP_LineEditClearButton, nullptr, w));
240  }
241  }
242  }
243 
244  QPalette palette = w->palette();
245 
246  if (w->testAttribute(Qt::WA_SetStyle)) {
247  polish(palette);
248  }
249 
250  // combo box drop-list.
251  // line edit completer drop-list
252  if (widgetIsComboBoxPopupFramePrimitive(w)) {
253  polish(palette);
254  palette.setBrush(QPalette::Background, m_palette->brush(PaletteExtended::Menu_BackgroundBrush,
255  PaletteExtended::PseudoClass_Unspecified,
256  palette.brush(QPalette::Background)));
257  }
258 
259  // TODO(zccrs): 临时解决方案，用于支持应用程序中自定义DTabBar的被选中Tab的文本颜色
260  DTabBar *tabBar = qobject_cast<DTabBar*>(w);
261 
262  if (tabBar) {
263  if (!tabBar->testAttribute(Qt::WA_SetPalette))
264  palette.setBrush(QPalette::Active, QPalette::Text, m_palette->brush(PaletteExtended::TabBarTab_TextColor, PaletteExtended::PseudoClass_Selected));
265  }
266 
267  w->setPalette(palette);
268 }

320 int Style::pixelMetric(QStyle::PixelMetric metric, const QStyleOption *option, const QWidget *widget) const
321 {
322  // handle special cases
323  switch( metric )
324  {
325 
326  // frame width
327  case PM_DefaultFrameWidth: {
328  if( qobject_cast<const QMenu*>( widget ) ) return Metrics::Menu_FrameWidth;
329  if( qobject_cast<const QLineEdit*>( widget ) ) return Metrics::LineEdit_FrameWidth;
330 
331  // fallback
332  return Metrics::Frame_FrameWidth;
333  }
334 
335  case PM_ComboBoxFrameWidth: {
336  const QStyleOptionComboBox* comboBoxOption( qstyleoption_cast< const QStyleOptionComboBox*>( option ) );
337  return comboBoxOption && comboBoxOption->editable ? Metrics::LineEdit_FrameWidth : Metrics::ComboBox_FrameWidth;
338  }
339 
340  case PM_SpinBoxFrameWidth: return Metrics::SpinBox_FrameWidth;
341  case PM_ToolBarFrameWidth: return Metrics::ToolBar_FrameWidth;
342  case PM_ToolTipLabelFrameWidth: return Metrics::ToolTip_FrameWidth;
343 
344  case PM_LayoutHorizontalSpacing: return Metrics::Layout_DefaultSpacing;
345  case PM_LayoutVerticalSpacing: return Metrics::Layout_DefaultSpacing;
346 
347  // buttons
348  case PM_ButtonMargin:
349  {
350  // needs special case for kcalc buttons, to prevent the application to set too small margins
351  if( widget && widget->inherits( "KCalcButton" ) ) return Metrics::Button_MarginWidth + 4;
352  else return Metrics::Button_MarginWidth;
353  }
354 
355  case PM_ButtonDefaultIndicator: return 0;
356  case PM_ButtonShiftHorizontal: return 0;
357  case PM_ButtonShiftVertical: return 0;
358 
359  //menu
360  case PM_MenuHMargin: return Menu_HMargin;
361  case PM_MenuVMargin: return Menu_VMargin;
362 
363  // menubars
364  case PM_MenuBarPanelWidth: return 0;
365  case PM_MenuBarHMargin: return 0;
366  case PM_MenuBarVMargin: return 0;
367  case PM_MenuBarItemSpacing: return Metrics::MenuBar_ItemSpacing;
368  case PM_MenuDesktopFrameWidth: return 0;
369 
370  // menu buttons
371  case PM_MenuButtonIndicator: return Metrics::MenuButton_IndicatorWidth;
372 
373  // toolbars
374  case PM_ToolBarHandleExtent: return Metrics::ToolBar_HandleExtent;
375  case PM_ToolBarSeparatorExtent: return Metrics::ToolBar_SeparatorWidth;
376  case PM_ToolBarExtensionExtent:
377  return pixelMetric( PM_SmallIconSize, option, widget ) + 2*Metrics::ToolButton_MarginWidth;
378 
379  case PM_ToolBarItemMargin: return 0;
380  case PM_ToolBarItemSpacing: return Metrics::ToolBar_ItemSpacing;
381 
382  // tabbars
383  case PM_TabBarTabShiftVertical: return 0;
384  case PM_TabBarTabShiftHorizontal: return 0;
385  case PM_TabBarTabOverlap: return Metrics::TabBar_TabOverlap;
386  case PM_TabBarBaseOverlap: return Metrics::TabBar_BaseOverlap;
387  case PM_TabBarTabHSpace: return 2*Metrics::TabBar_TabMarginWidth;
388  case PM_TabBarTabVSpace: return 2*Metrics::TabBar_TabMarginHeight;
389  case PM_TabCloseIndicatorWidth:
390  case PM_TabCloseIndicatorHeight:
391  return pixelMetric( PM_SmallIconSize, option, widget );
392 
393  // scrollbars
394  case PM_ScrollBarExtent: return Metrics::ScrollBar_Extend;
395  case PM_ScrollBarSliderMin: return Metrics::ScrollBar_MinSliderHeight;
396 
397  // title bar
398  case PM_TitleBarHeight: return 2*Metrics::TitleBar_MarginWidth + pixelMetric( PM_SmallIconSize, option, widget );
399 
400  // sliders
401  case PM_SliderThickness: return Metrics::Slider_ControlThickness;
402  case PM_SliderControlThickness: return Metrics::Slider_ControlThickness;
403  case PM_SliderLength: return Metrics::Slider_ControlThickness;
404 
405  // checkboxes and radio buttons
406  case PM_IndicatorWidth: return Metrics::CheckBox_Size;
407  case PM_IndicatorHeight: return Metrics::CheckBox_Size;
408  case PM_ExclusiveIndicatorWidth: return Metrics::CheckBox_Size;
409  case PM_ExclusiveIndicatorHeight: return Metrics::CheckBox_Size;
410 
411  // list heaaders
412  case PM_HeaderMarkSize: return Metrics::Header_ArrowSize;
413  case PM_HeaderMargin: return Metrics::Header_MarginWidth;
414 
415  // dock widget
416  // return 0 here, since frame is handled directly in polish
417  case PM_DockWidgetFrameWidth: return 0;
418  case PM_DockWidgetTitleMargin: return Metrics::Frame_FrameWidth;
419  case PM_DockWidgetTitleBarButtonMargin: return Metrics::ToolButton_MarginWidth;
420 
421  case PM_SplitterWidth: return Metrics::Splitter_SplitterWidth;
422  case PM_DockWidgetSeparatorExtent: return Metrics::Splitter_SplitterWidth;
423 #ifdef DTKWIDGET_CLASS_DTabBar
424  case PM_TabBarScrollButtonWidth: {
425  if (qobject_cast<DTabBar*>(widget->parent())) {
426  return 0;
427  }
428  break;
429  }
430 #endif
431  // fallback
432  default:
433  break;
434  }
435 
436  return QCommonStyle::pixelMetric(metric, option, widget);
437 }

710 QSize Style::sizeFromContents(QStyle::ContentsType type, const QStyleOption *option, const QSize &size, const QWidget *widget) const
711 {
712  QSize newSize = QCommonStyle::sizeFromContents(type, option, size, widget);
713  switch (type) {
714  case CT_PushButton:
715  if (const QStyleOptionButton *btn = qstyleoption_cast<const QStyleOptionButton *>(option)) {
716  const bool flat = btn->features & QStyleOptionButton::Flat;
717  const bool semiTransparent = m_type == StyleSemiDark || m_type == StyleSemiLight;
718  if (!btn->text.isEmpty() && !flat) {
719  if (newSize.width() < 80)
720  newSize.setWidth(80);
721  // QPushButton should be set a minimum height 36 in d-semi-transparent style.
722  if (semiTransparent && newSize.height() < 36)
723  newSize.setHeight(36);
724  }
725  if (!btn->icon.isNull() && btn->iconSize.height() > 16)
726  newSize -= QSize(0, 2);
727  }
728  break;
729  case CT_GroupBox:
730  if (option) {
731  int topMargin = qMax(pixelMetric(PM_ExclusiveIndicatorHeight), option->fontMetrics.height()) + GroupBox_TopMargin;
732  newSize += QSize(10, topMargin); // Add some space below the groupbox
733  }
734  break;
735  case CT_RadioButton:
736  case CT_CheckBox:
737  newSize += QSize(0, 1);
738  break;
739  case CT_ToolButton:
740  newSize += QSize(2, 2);
741  break;
742  case CT_SpinBox:
743  newSize += QSize(0, -3);
744  break;
745  case CT_ComboBox:
746  newSize += QSize(0, -6);
747  break;
748  case CT_LineEdit:
749  newSize += QSize(0, 4);
750  break;
751  case CT_MenuBarItem:
752  newSize += QSize(8, 5);
753  break;
754  case CT_MenuItem:
755  if (const QStyleOptionMenuItem *menuItem = qstyleoption_cast<const QStyleOptionMenuItem *>(option)) {
756  int w = newSize.width();
757  bool hideShortcutText = !isVisibleMenuShortText();
758 
759  if (hideShortcutText) {
760  w -= menuItem->tabWidth;
761 
762  int t = menuItem->text.indexOf(QLatin1Char('\t'));
763  if (t != -1) {
764  int textWidth = option->fontMetrics.width(menuItem->text.mid(t + 1));
765 
766  if (menuItem->tabWidth == 0)
767  w -= textWidth;
768  }
769  }
770 
771  int maxpmw = menuItem->maxIconWidth;
772  int tabSpacing = 20;
773  if (menuItem->text.contains(QLatin1Char('\t'))) {
774  if (!hideShortcutText)
775  w += tabSpacing;
776  } else {
777  if (menuItem->menuItemType == QStyleOptionMenuItem::SubMenu) {
778  w += 2 * Menu_ArrowHMargin;
779  } else if (menuItem->menuItemType == QStyleOptionMenuItem::DefaultItem) {
780  QFontMetrics fm(menuItem->font);
781  QFont fontBold = menuItem->font;
782  fontBold.setBold(true);
783  QFontMetrics fmBold(fontBold);
784  w += fmBold.width(menuItem->text) - fm.width(menuItem->text);
785  }
786  }
787  int checkcol = qMax<int>(maxpmw, Menu_CheckMarkWidth); // Windows always shows a check column
788  w += checkcol;
789  w += Menu_RightBorder + 10;
790  newSize.setWidth(w);
791  if (menuItem->menuItemType == QStyleOptionMenuItem::Separator) {
792  if (!menuItem->text.isEmpty()) {
793  newSize.setHeight(menuItem->fontMetrics.height());
794  } else {
795  newSize.setHeight(newSize.height() + 4);
796  }
797  }
798  else if (!menuItem->icon.isNull()) {
799  if (const QComboBox *combo = qobject_cast<const QComboBox*>(widget)) {
800  newSize.setHeight(qMax(combo->iconSize().height() + 2, newSize.height()));
801  }
802  }
803  newSize.setWidth(newSize.width() + 12);
804  newSize.setWidth(qMax(newSize.width(), 100));
805  }
806 
807  newSize.setWidth(newSize.width() + Menu_ItemHMargin * 2);
808  newSize.setHeight(newSize.height() + Menu_ItemVMargin * 2 + 1);
809  break;
810  case CT_SizeGrip:
811  newSize += QSize(4, 4);
812  break;
813  case CT_MdiControls:
814  if (const QStyleOptionComplex *styleOpt = qstyleoption_cast<const QStyleOptionComplex *>(option)) {
815  int width = 0;
816  if (styleOpt->subControls & SC_MdiMinButton)
817  width += 19 + 1;
818  if (styleOpt->subControls & SC_MdiNormalButton)
819  width += 19 + 1;
820  if (styleOpt->subControls & SC_MdiCloseButton)
821  width += 19 + 1;
822  newSize = QSize(width, 19);
823  } else {
824  newSize = QSize(60, 19);
825  }
826  break;
827  default:
828  break;
829  }
830  return newSize;
831 }
