81 {
82  D_QC(DBlurEffectWidget);
83 
84  QWidget *oldTopLevelWidget = windowOfBlurEffectHash.value(q);

182 bool DBlurEffectWidgetPrivate::updateWindowBlurArea(QWidget *topLevelWidget)
183 {
184  if (!topLevelWidget->isVisible()) {
185  return false;
186  }
187 
188  QList<const DBlurEffectWidget *> blurEffectWidgetList = blurEffectWidgetHash.values(topLevelWidget);
189 
190  bool isExistMaskPath = false;
191 
192  Q_FOREACH (const DBlurEffectWidget *w, blurEffectWidgetList) {
193  if (!w->d_func()->blurEnabled) {
194  continue;
195  }
196 
197  // 当存在一个模糊区域填充整个窗口时，不需要再考虑其它的控件，直接对此窗口开启全模糊即可
198  if (w->d_func()->isFull() && w->isVisible()) {
199  DPlatformWindowHandle handle(topLevelWidget);
200 
201  if (!handle.enableBlurWindow()) {
202  handle.setEnableBlurWindow(true);
203  }
204 
205  return true;
206  }
207 
208  if (!w->d_func()->maskPath.isEmpty() && w->isVisible()) {
209  isExistMaskPath = true;
210  break;
211  }
212  }
213 
214  // 清理窗口背景模糊是否完全开启的状态
215  DPlatformWindowHandle handle(topLevelWidget);
216 
217  if (handle.enableBlurWindow()) {
218  handle.setEnableBlurWindow(false);
219  }
220 
221  bool ok = false;
222 
223  if (isExistMaskPath) {
224  QList<QPainterPath> pathList;
225 
226  Q_FOREACH (const DBlurEffectWidget *w, blurEffectWidgetList) {
227  if (!w->d_func()->blurEnabled) {
228  continue;
229  }
230 
231  if (!w->isVisible()) {
232  continue;
233  }
234 
235  QPainterPath p;
236  QRect r = w->rect();
237 
238  r.moveTopLeft(w->mapTo(topLevelWidget, r.topLeft()));
239  p.addRoundedRect(r, w->blurRectXRadius(), w->blurRectYRadius());
240 
241  if (!w->d_func()->maskPath.isEmpty()) {
242  p &= w->d_func()->maskPath.translated(r.topLeft());
243  }
244 
245  pathList << p;
246  }
247 
248  ok = handle.setWindowBlurAreaByWM(pathList);
249  } else {
250  QVector<DPlatformWindowHandle::WMBlurArea> areaList;
251 
252  areaList.reserve(blurEffectWidgetList.size());
253 
254  Q_FOREACH (const DBlurEffectWidget *w, blurEffectWidgetList) {
255  if (!w->d_func()->blurEnabled) {
256  continue;
257  }
258 
259  if (!w->isVisible()) {
260  continue;
261  }
262 
263  QRect r = w->rect();
264 
265  r.moveTopLeft(w->mapTo(topLevelWidget, r.topLeft()));
266 
267  areaList << dMakeWMBlurArea(r.x(), r.y(), r.width(), r.height(), w->blurRectXRadius(), w->blurRectYRadius());
268  }
269 
270  ok = handle.setWindowBlurAreaByWM(areaList);
271  }
272 
273  if (blurEffectWidgetList.isEmpty()) {
274  blurEffectWidgetHash.remove(topLevelWidget);
275  }
276 
277  return ok;
278 }

893 void DBlurEffectWidget::updateBlurSourceImage(const QRegion &ren)
894 {
895  D_D(DBlurEffectWidget);
896 
897  // 自定义模式，或者属于某个组时不需要以下操作
898  if (d->customSourceImage || d->group)
899  return;
900 
901  const qreal device_pixel_ratio = devicePixelRatioF();
902  const QPoint point_offset = mapTo(window(), QPoint(0, 0));
903 
904  if (d->sourceImage.isNull()) {
905  const QRect &tmp_rect = rect().translated(point_offset).adjusted(-d->radius, -d->radius, d->radius, d->radius);
906 
907  d->sourceImage = window()->backingStore()->handle()->toImage().copy(tmp_rect * device_pixel_ratio);
908  d->sourceImage = d->sourceImage.scaledToWidth(d->sourceImage.width() / device_pixel_ratio);
909  } else {
910  QPainter pa_image(&d->sourceImage);
911 
912  pa_image.setCompositionMode(QPainter::CompositionMode_Source);
913 
914  if (device_pixel_ratio > 1) {
915  const QRect &tmp_rect = this->rect().translated(point_offset);
916  QImage area = window()->backingStore()->handle()->toImage().copy(tmp_rect * device_pixel_ratio);
917  area = area.scaledToWidth(area.width() / device_pixel_ratio);
918 
919  for (const QRect &rect : ren.rects()) {
920  pa_image.drawImage(rect.topLeft() + QPoint(d->radius, d->radius), rect == area.rect() ? area : area.copy(rect));
921  }
922  } else {
923  for (const QRect &rect : ren.rects()) {
924  pa_image.drawImage(rect.topLeft() + QPoint(d->radius, d->radius),
925  window()->backingStore()->handle()->toImage().copy(rect.translated(point_offset)));
926  }
927  }
928 
929  pa_image.end();
930  }

940 void DBlurEffectWidget::paintEvent(QPaintEvent *event)
941 {
942  D_D(DBlurEffectWidget);
943 
944  if (!d->blurEnabled)
945  return;
946 
947  QPainter pa(this);
948 
949  if (d->blurRectXRadius > 0 || d->blurRectYRadius > 0) {
950  QPainterPath path;
951 
952  path.addRoundedRect(rect(), d->blurRectXRadius, d->blurRectYRadius);
953  pa.setRenderHint(QPainter::Antialiasing);
954  pa.setClipPath(path);
955  }
956 
957  if (!d->maskPath.isEmpty()) {
958  QPainterPath path = pa.clipPath();
959 
960  if (path.isEmpty()) {
961  path = d->maskPath;
962  } else {
963  path &= d->maskPath;
964  }
965 
966  pa.setClipPath(path);
967  }
968 
969  if (d->isBehindWindowBlendMode()) {
970  pa.setCompositionMode(QPainter::CompositionMode_Source);
971  } else {
972  // 此模式下是自行控制sourceImage的更新
973  if (d->blendMode != InWidgetBlend) {
974  updateBlurSourceImage(event->region());
975  }
976 
977  if (d->customSourceImage || !d->sourceImage.isNull()) {
978  int radius = d->radius;
979  qreal device_pixel_ratio = devicePixelRatioF();
980  QImage image;
981  const QRect &paintRect = event->rect();
982 
983  if (d->customSourceImage) {
984  image = d->sourceImage.copy(paintRect.adjusted(0, 0, 2 * radius, 2 * radius) * device_pixel_ratio);
985  image.setDevicePixelRatio(device_pixel_ratio);
986  pa.setOpacity(0.2);
987  } else {// 非customSourceImage不考虑缩放产生的影响
988  image = d->sourceImage.copy(paintRect.adjusted(0, 0, 2 * radius, 2 * radius));
989  }
990 
991  QTransform old_transform = pa.transform();
992  pa.translate(paintRect.topLeft() - QPoint(radius, radius));
993  qt_blurImage(&pa, image, radius, false, false);
994  pa.setTransform(old_transform);
995  pa.setOpacity(1);
996  } else if (d->group) { // 组模式
997  d->group->paint(&pa, this);
998  }
999  }
1000 
1001  pa.fillRect(rect(), maskColor());
1002 }
