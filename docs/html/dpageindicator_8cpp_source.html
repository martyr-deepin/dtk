249 void DPageIndicator::paintEvent(QPaintEvent *e)
250 {
251  QWidget::paintEvent(e);
252 
253  const int w = width();
254  const int h = height();
255 
256  D_DC(DPageIndicator);
257 
258  const int total_w = d->pageCount * d->pointDistance;
259  const QPoint offset = QPoint((w - total_w) / 2, h / 2);
260 
261  QColor currentPtColor = d->pointColor;
262  QColor nonCurrentPtColor = d->secondaryPointColor;
263 
264  if (!d->pointColor.isValid())
265  currentPtColor = this->palette().highlight().color();
266 
267  if (!d->secondaryPointColor.isValid()) {
268  nonCurrentPtColor = this->palette().button().color();
269  }
270 
271  QPainter painter(this);
272  painter.setRenderHint(QPainter::Antialiasing);
273  painter.setPen(Qt::transparent);
274  for (int i(0); i != d->pageCount; ++i) {
275  if (d->currentPage == i) {
276  painter.setBrush(currentPtColor);
277  painter.drawEllipse(offset + QPoint(d->pointDistance / 2 + d->pointDistance * i, 0),
278  d->pointRadius, d->pointRadius);
279  } else {
280  painter.setBrush(nonCurrentPtColor);
281  painter.drawEllipse(offset + QPoint(d->pointDistance / 2 + d->pointDistance * i, 0),
282  d->secondaryPointRadius, d->secondaryPointRadius);
283  }
284  }
285 }
