79 bool DFileWatcherPrivate::start()
80 {
81  Q_Q(DFileWatcher);
82 
83  started = true;
84 
85  Q_FOREACH (const QString &path, parentPathList(this->path)) {
86  if (watchFileList.contains(path))
87  continue;
88 
89  if (filePathToWatcherCount.value(path, -1) <= 0) {
90  if (!watcher_file_private->addPath(path)) {
91  qWarning() << Q_FUNC_INFO << "start watch failed, file path =" << path;
92  q->stopWatcher();
93  started = false;
94  return false;
95  }
96  }
97 
98  watchFileList << path;
99  filePathToWatcherCount[path] = filePathToWatcherCount.value(path, 0) + 1;
100  }
101 
102  q->connect(watcher_file_private, &DFileSystemWatcher::fileDeleted,
103  q, &DFileWatcher::onFileDeleted);
104  q->connect(watcher_file_private, &DFileSystemWatcher::fileAttributeChanged,
105  q, &DFileWatcher::onFileAttributeChanged);
106  q->connect(watcher_file_private, &DFileSystemWatcher::fileMoved,
107  q, &DFileWatcher::onFileMoved);
108  q->connect(watcher_file_private, &DFileSystemWatcher::fileCreated,
109  q, &DFileWatcher::onFileCreated);
110  q->connect(watcher_file_private, &DFileSystemWatcher::fileModified,
111  q, &DFileWatcher::onFileModified);
112  q->connect(watcher_file_private, &DFileSystemWatcher::fileClosed,
113  q, &DFileWatcher::onFileClosed);
114 
115  return true;
116 }

118 bool DFileWatcherPrivate::stop()
119 {
120  Q_Q(DFileWatcher);
121 
122  q->disconnect(watcher_file_private, 0, q, 0);
123 
124  bool ok = true;
125 
126  Q_FOREACH (const QString &path, watchFileList) {
127  int count = filePathToWatcherCount.value(path, 0);
128 
129  --count;
130 
131  if (count <= 0) {
132  filePathToWatcherCount.remove(path);
133  watchFileList.removeOne(path);
134  ok = ok && watcher_file_private->removePath(path);
135  } else {
136  filePathToWatcherCount[path] = count;
137  }
138  }
139 
140  return ok;
141 }
