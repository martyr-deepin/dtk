37 void DScrollBarPrivate::init()
38 {
39  D_Q(DScrollBar);
40 
41  timer = new QTimer(q);
42  opacityTimer = new QTimer(q);
43 
44  timer->setInterval(1000);
45  timer->setSingleShot(true);
46 
47  q->connect(timer, SIGNAL(timeout()), q, SLOT(_q_hidden()));
48  q->connect(opacityTimer, SIGNAL(timeout()), q, SLOT(_q_updateOpacity()));
49  q->connect(q, &DScrollBar::valueChanged, q, [this] {
50  timer->start();
51  opacityTimer->stop();
52 
53  setOpacity(1);
54  });
55 }

57 void DScrollBarPrivate::setOpacity(qreal opacity)
58 {
59  D_Q(DScrollBar);

62  return;
63  }
64 
65  this->opacity = opacity;
66  this->canPaint = false;
67 
68  q->update();

77 void DScrollBarPrivate::_q_updateOpacity()

135 void DScrollBar::paintEvent(QPaintEvent *event)
136 {
137  D_D(DScrollBar);
138 
139  if (d->canPaint || !d->autoHide) {
140  return QScrollBar::paintEvent(event);
141  }
142 
143  QPixmap pixmap;
144  QPainter painter(this);
145 
146  d->canPaint = true;
147 
148  setAttribute(Qt::WA_WState_InPaintEvent, false);
149  pixmap = grab();
150 
151  painter.setOpacity(d->opacity);
