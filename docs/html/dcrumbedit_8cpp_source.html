352  void _q_onTextChanged()
353  {
354  D_Q(DCrumbEdit);
355 
356  bool formatsChanged = false;
357 
358  QTextCursor cursor(q->document());
359  QSet<QString> crumbList;
360  int last_pos = 0;
361 
362  cursor.setPosition(0);
363  formatList.clear();
364 
365  while (!cursor.atEnd()) {
366  cursor.setPosition(last_pos + 1);
367 
368  if (last_pos == cursor.position())
369  break;
370 
371  last_pos = cursor.position();
372 
373  DCrumbTextFormat format(cursor.charFormat());
374 
375  if (format.objectType() == objectType) {
376  const QString &text = format.text();
377 
378  if (!text.isEmpty()) {
379  crumbList << text;
380  formatList << text;
381 
382  if (!formats.contains(text)) {
383  formats[text] = format;
384  formatsChanged = true;
385 
386  Q_EMIT q->crumbAdded(text);
387  }
388  }
389  }
390  }
391 
392  Q_FOREACH(const DCrumbTextFormat &f, formats) {
393  if (!crumbList.contains(f.text())) {
394  formats.remove(f.text());
395  formatsChanged = true;
396 
397  Q_EMIT q->crumbRemoved(f.text());
398  }
399  }
400 
401  if (formatsChanged)
402  Q_EMIT q->crumbListChanged();
403  }

431 void CrumbObjectInterface::drawObject(QPainter *painter, const QRectF &rect,
432  QTextDocument *doc, int posInDocument, const QTextFormat &format)
433 {
434  Q_UNUSED(doc)
435  Q_UNUSED(posInDocument)
436 
437  const QRectF new_rect = rect.adjusted(1, 1, -1, -1);
438  const DCrumbTextFormat crumb_format(format);
439  const QFontMetricsF font_metrics(crumb_format.font());
440  const int radius = crumb_format.backgroundRadius();
441 
442  QPainterPath background_path;
443  QPainterPath tag_path;
444  const QRectF tag_rect(new_rect.x() + 2, new_rect.y() + 2, font_metrics.height() - 4, font_metrics.height() - 4);
445 
446  tag_path.addEllipse(tag_rect);
447  background_path.addRoundedRect(new_rect, radius, crumb_format.backgroundRadius());
448 
449  painter->setRenderHint(QPainter::Antialiasing);
450  painter->fillPath(background_path, crumb_format.background());
451 
452  if (crumb_format.tagColor().isValid()) {
453  painter->fillPath(tag_path, crumb_format.tagColor());
454 
455  painter->setPen(crumb_format.textColor());
456  painter->drawText(new_rect.adjusted(tag_rect.width() + 2, 0, -radius, 0),
457  crumb_format.text(), Qt::AlignVCenter | Qt::AlignRight);
458  } else {
459  painter->setPen(crumb_format.textColor());
460  painter->drawText(new_rect.adjusted(radius, 0, -radius, 0), crumb_format.text());
461  }
462 }

1000 void DCrumbEdit::insertFromMimeData(const QMimeData *source)
1001 {
1002  if (!source->hasFormat("deepin/dtkwidget-DCrumbTextFormat-data"))
1003  QTextEdit::insertFromMimeData(source);
1004 
1005  const QString &plain_text = QString::fromUtf8(source->data("deepin/dtkwidget-DCrumbTextFormat-data"));
1006 
1007  if (plain_text.isEmpty())
1008  return;
1009 
1010  D_DC(DCrumbEdit);
1011 
1012  const QList<DCrumbTextFormat> &list = d->crumbListFromData(source->data("deepin/dtkwidget-DCrumbTextFormat-list"));
1013  QList<DCrumbTextFormat>::const_iterator current_format = list.constBegin();
1014  QString text;
1015 
1016  textCursor().beginEditBlock();
1017 
1018  for (const QChar &ch : plain_text) {
1019  if (ch == QChar::ObjectReplacementCharacter) {
1020  if (!text.isEmpty()) {
1021  textCursor().insertText(text);
1022  text.clear();
1023  }
1024 
1025  insertCrumb(*current_format);
1026  ++current_format;
1027  } else {
1028  text.append(ch);
1029  }
1030  }
1031 
1032  if (!text.isEmpty())
1033  textCursor().insertText(text);
1034 
1035  textCursor().endEditBlock();
