169 QSize DButtonBoxButton::sizeHint() const
170 {
171  QAbstractButtonPrivate *d = static_cast<QAbstractButtonPrivate*>(QAbstractButton::d_ptr.data());
172 
173  if (d->sizeHint.isValid())
174  return d->sizeHint;
175 
176  ensurePolished();
177 
178  int w = 0, h = 0;
179 
180  DStyleOptionButtonBoxButton opt;
181  initStyleOption(&opt);
182 
183  if (!icon().isNull()) {
184  int ih = opt.iconSize.height();
185  int iw = opt.iconSize.width() + 4;
186  w += iw;
187  h = qMax(h, ih);
188  }
189 
190  QString s(text());
191  bool empty = s.isEmpty();
192  if (empty)
193  s = QStringLiteral("XXXX");
194  QFontMetrics fm = fontMetrics();
195  QSize sz = fm.size(Qt::TextShowMnemonic, s);
196  if(!empty || !w)
197  w += sz.width();
198  if(!empty || !h)
199  h = qMax(h, sz.height());
200  opt.rect.setSize(QSize(w, h)); // PM_MenuButtonIndicator depends on the height
201 
202  DStyleHelper dstyle(style());
203 
204  d->sizeHint = (dstyle.sizeFromContents(DStyle::CT_ButtonBoxButton, &opt, QSize(w, h), this).
205  expandedTo(QApplication::globalStrut()));
206  return d->sizeHint;
207 }

249 void DButtonBoxButton::keyPressEvent(QKeyEvent *event)
250 {
251  switch (event->key()) {
252  case Qt::Key_Enter:
253  case Qt::Key_Return:
254  if (hasFocus()) {
255  click();
256  break;
257  }
258  Q_FALLTHROUGH();
259  case Qt::Key_Left:
260  case Qt::Key_Right:
261  case Qt::Key_Up:
262  case Qt::Key_Down:
263  if (DButtonBox *p = qobject_cast<DButtonBox *>(parent())) {
264  if (p->focusWidget() == this) {
265  int index = p->buttonList().indexOf(this);
266 
267  if (p->orientation() == Qt::Horizontal) {
268  if (event->key() == Qt::Key_Right)
269  p->buttonList().at((index + 1) >= p->buttonList().length() ? 0 : index + 1)->setFocus();
270  else if (event->key() == Qt::Key_Left)
271  p->buttonList().at((index - 1) < 0 ? p->buttonList().length() - 1 : index - 1)->setFocus();
272  } else {
273  if (event->key() == Qt::Key_Down)
274  p->buttonList().at((index + 1) >= p->buttonList().length() ? 0 : index + 1)->setFocus();
275  else if ((event->key() == Qt::Key_Up))
276  p->buttonList().at((index - 1) < 0 ? p->buttonList().length() - 1 : index - 1)->setFocus();
277  }
278  }
279  }
280  break;
281  default:
282  QAbstractButton::keyPressEvent(event);
283  }
284 }

327 DStyleOptionButtonBoxButton::ButtonPosition DButtonBoxPrivate::getButtonPosition(const DButtonBoxButton *button) const
328 {
329  D_QC(DButtonBox);
330 
331  DStyleOptionButtonBoxButton::ButtonPosition position;
332  auto list = q->buttonList();
333  bool reverse = q->layoutDirection() == Qt::RightToLeft && q->orientation() == Qt::Horizontal;
334 
335  if (list.count() == 1) {
336  position = DStyleOptionButtonBoxButton::OnlyOne;
337  } else if (button == list.first()) {
338  position = reverse ? DStyleOptionButtonBoxButton::End : DStyleOptionButtonBoxButton::Beginning;
339  } else if (button == list.last()) {
340  position = reverse ? DStyleOptionButtonBoxButton::Beginning : DStyleOptionButtonBoxButton::End;
341  } else {
342  position = DStyleOptionButtonBoxButton::Middle;
343  }
344 
345  return position;
346 }
