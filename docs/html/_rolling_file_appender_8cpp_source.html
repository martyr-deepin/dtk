87 void RollingFileAppender::computeFrequency()
88 {
89  QMutexLocker locker(&m_rollingMutex);
90 
91  const QDateTime startTime(QDate(1999, 1, 1), QTime(0, 0));
92  const QString startString = startTime.toString(m_datePatternString);
93 
94  if (startString != startTime.addSecs(60).toString(m_datePatternString))
95  m_frequency = MinutelyRollover;
96  else if (startString != startTime.addSecs(60 * 60).toString(m_datePatternString))
97  m_frequency = HourlyRollover;
98  else if (startString != startTime.addSecs(60 * 60 * 12).toString(m_datePatternString))
99  m_frequency = HalfDailyRollover;
100  else if (startString != startTime.addDays(1).toString(m_datePatternString))
101  m_frequency = DailyRollover;
102  else if (startString != startTime.addDays(7).toString(m_datePatternString))
103  m_frequency = WeeklyRollover;
104  else if (startString != startTime.addMonths(1).toString(m_datePatternString))
105  m_frequency = MonthlyRollover;
106  else
107  {
108  Q_ASSERT_X(false, "DailyRollingFileAppender::computeFrequency", "The pattern '%1' does not specify a frequency");
109  return;
110  }
111 }

114 void RollingFileAppender::removeOldFiles()
115 {
116  if (m_logFilesLimit <= 1)
117  return;
118 
119  QFileInfo fileInfo(fileName());
120  QDir logDirectory(fileInfo.absoluteDir());
121  logDirectory.setFilter(QDir::Files);
122  logDirectory.setNameFilters(QStringList() << fileInfo.fileName() + "*");
123  QFileInfoList logFiles = logDirectory.entryInfoList();
124 
125  QMap<QDateTime, QString> fileDates;
126  for (int i = 0; i < logFiles.length(); ++i)
127  {
128  QString name = logFiles[i].fileName();
129  QString suffix = name.mid(name.indexOf(fileInfo.fileName()) + fileInfo.fileName().length());
130  QDateTime fileDateTime = QDateTime::fromString(suffix, datePatternString());
131 
132  if (fileDateTime.isValid())
133  fileDates.insert(fileDateTime, logFiles[i].absoluteFilePath());
134  }
135 
136  QList<QString> fileDateNames = fileDates.values();
137  for (int i = 0; i < fileDateNames.length() - m_logFilesLimit + 1; ++i)
138  QFile::remove(fileDateNames[i]);
139 }

142 void RollingFileAppender::computeRollOverTime()
143 {
144  Q_ASSERT_X(!m_datePatternString.isEmpty(), "DailyRollingFileAppender::computeRollOverTime()", "No active date pattern");
145 
146  QDateTime now = QDateTime::currentDateTime();
147  QDate nowDate = now.date();
148  QTime nowTime = now.time();
149  QDateTime start;
150 
151  switch (m_frequency)
152  {
153  case MinutelyRollover:
154  {
155  start = QDateTime(nowDate, nowTime);
156  m_rollOverTime = start.addSecs(60);
157  }
158  break;
159  case HourlyRollover:
160  {
161  start = QDateTime(nowDate, nowTime);
162  m_rollOverTime = start.addSecs(60*60);
163  }
164  break;
165  case HalfDailyRollover:
166  {
167  int hour = nowTime.hour();
168  if (hour >= 12)
169  hour = 12;
170  else
171  hour = 0;
172  start = QDateTime(nowDate, nowTime);
173  m_rollOverTime = start.addSecs(60*60*12);
174  }
175  break;
176  case DailyRollover:
177  {
178  start = QDateTime(nowDate, nowTime);
179  m_rollOverTime = start.addDays(1);
180  }
181  break;
182  case WeeklyRollover:
183  {
184  // Qt numbers the week days 1..7. The week starts on Monday.
185  // Change it to being numbered 0..6, starting with Sunday.
186  int day = nowDate.dayOfWeek();
187  if (day == Qt::Sunday)
188  day = 0;
189  start = QDateTime(nowDate, nowTime).addDays(-1 * day);
190  m_rollOverTime = start.addDays(7);
191  }
192  break;
193  case MonthlyRollover:
194  {
195  start = QDateTime(QDate(nowDate.year(), nowDate.month(), 1), nowTime);
196  m_rollOverTime = start.addMonths(1);
197  }
198  break;
199  default:
200  Q_ASSERT_X(false, "DailyRollingFileAppender::computeInterval()", "Invalid datePattern constant");
201  m_rollOverTime = QDateTime::fromTime_t(0);
202  }
203 
204  m_rollOverSuffix = start.toString(m_datePatternString);
205  Q_ASSERT_X(now.toString(m_datePatternString) == m_rollOverSuffix,
206  "DailyRollingFileAppender::computeRollOverTime()", "File name changes within interval");
207  Q_ASSERT_X(m_rollOverSuffix != m_rollOverTime.toString(m_datePatternString),
208  "DailyRollingFileAppender::computeRollOverTime()", "File name does not change with rollover");
209 }

212 void RollingFileAppender::rollOver()
213 {
214  Q_ASSERT_X(!m_datePatternString.isEmpty(), "DailyRollingFileAppender::rollOver()", "No active date pattern");
215 
216  QString rollOverSuffix = m_rollOverSuffix;
217  computeRollOverTime();
218  if (rollOverSuffix == m_rollOverSuffix)
219  return;
220 
221  closeFile();
222 
223  QString targetFileName = fileName() + rollOverSuffix;
224  QFile f(targetFileName);
225  if (f.exists() && !f.remove())
226  return;
227  f.setFileName(fileName());
228  if (!f.rename(targetFileName))
229  return;
230 
231  openFile();
232  removeOldFiles();
233 }
