268 const QPoint DRegionMonitorPrivate::deviceScaledCoordinate(const QPoint &p, const double ratio) const
269 {
270  D_QC(DRegionMonitor);
271 
272  if (type == q->Original) {
273  return p;
274  }
275 
276  for (const auto *s : qApp->screens())
277  {
278  const QRect &g(s->geometry());
279  const QRect realRect(g.topLeft(), g.size() * ratio);
280

283  }
