42 static void tabLayout(const QStyle *proxyStyle, const QStyleOptionTab *opt, const QWidget *widget, QRect *textRect, QRect *iconRect)
43 {
44  Q_ASSERT(textRect);
45  Q_ASSERT(iconRect);
46  QRect tr = opt->rect;
47  bool verticalTabs = opt->shape == QTabBar::RoundedEast
48  || opt->shape == QTabBar::RoundedWest
49  || opt->shape == QTabBar::TriangularEast
50  || opt->shape == QTabBar::TriangularWest;
51  if (verticalTabs)
52  tr.setRect(0, 0, tr.height(), tr.width()); //0, 0 as we will have a translate transform
53 
54  int verticalShift = proxyStyle->pixelMetric(QStyle::PM_TabBarTabShiftVertical, opt, widget);
55  int horizontalShift = proxyStyle->pixelMetric(QStyle::PM_TabBarTabShiftHorizontal, opt, widget);
56  int hpadding = proxyStyle->pixelMetric(QStyle::PM_TabBarTabHSpace, opt, widget) / 2;
57  int vpadding = proxyStyle->pixelMetric(QStyle::PM_TabBarTabVSpace, opt, widget) / 2;
58  if (opt->shape == QTabBar::RoundedSouth || opt->shape == QTabBar::TriangularSouth)
59  verticalShift = -verticalShift;
60  tr.adjust(hpadding, verticalShift - vpadding, horizontalShift - hpadding, vpadding);
61  bool selected = opt->state & QStyle::State_Selected;
62  if (selected) {
63  tr.setTop(tr.top() - verticalShift);
64  tr.setRight(tr.right() - horizontalShift);
65  }
66 
67  // left widget
68  if (!opt->leftButtonSize.isEmpty()) {
69  tr.setLeft(tr.left() + 4 +
70  (verticalTabs ? opt->leftButtonSize.height() : opt->leftButtonSize.width()));
71  }
72  // right widget
73  if (!opt->rightButtonSize.isEmpty()) {
74  tr.setRight(tr.right() - 4 -
75  (verticalTabs ? opt->rightButtonSize.height() : opt->rightButtonSize.width()));
76  }
77 
78  // icon
79  if (!opt->icon.isNull()) {
80  QSize iconSize = opt->iconSize;
81  if (!iconSize.isValid()) {
82  int iconExtent = proxyStyle->pixelMetric(QStyle::PM_SmallIconSize);
83  iconSize = QSize(iconExtent, iconExtent);
84  }
85  QSize tabIconSize = opt->icon.actualSize(iconSize,
86  (opt->state & QStyle::State_Enabled) ? QIcon::Normal : QIcon::Disabled,
87  (opt->state & QStyle::State_Selected) ? QIcon::On : QIcon::Off );
88  // High-dpi icons do not need adjustmet; make sure tabIconSize is not larger than iconSize
89  tabIconSize = QSize(qMin(tabIconSize.width(), iconSize.width()), qMin(tabIconSize.height(), iconSize.height()));
90 
91  *iconRect = QRect(tr.left(), tr.center().y() - tabIconSize.height() / 2,
92  tabIconSize.width(), tabIconSize .height());
93  if (!verticalTabs)
94  *iconRect = proxyStyle->visualRect(opt->direction, opt->rect, *iconRect);
95  tr.setLeft(tr.left() + tabIconSize.width() + 4);
96  }
97 
98  if (!verticalTabs)
99  tr = proxyStyle->visualRect(opt->direction, opt->rect, tr);
100 
101  *textRect = tr;
102 }

130 bool Style::drawTabBarTabLabelControl(const QStyleOption *opt, QPainter *p, const QWidget *widget) const
131 {
132  auto tab = static_cast<const QStyleOptionTab*>(opt);
133 
134  QRect tr = tab->rect;
135  bool verticalTabs = tab->shape == QTabBar::RoundedEast
136  || tab->shape == QTabBar::RoundedWest
137  || tab->shape == QTabBar::TriangularEast
138  || tab->shape == QTabBar::TriangularWest;
139 
140  int alignment = Qt::AlignCenter | Qt::TextShowMnemonic;
141  if (!proxy()->styleHint(SH_UnderlineShortcut, opt, widget))
142  alignment |= Qt::TextHideMnemonic;
143 
144  if (verticalTabs) {
145  p->save();
146  int newX, newY, newRot;
147  if (tab->shape == QTabBar::RoundedEast || tab->shape == QTabBar::TriangularEast) {
148  newX = tr.width() + tr.x();
149  newY = tr.y();
150  newRot = 90;
151  } else {
152  newX = tr.x();
153  newY = tr.y() + tr.height();
154  newRot = -90;
155  }
156  QTransform m = QTransform::fromTranslate(newX, newY);
157  m.rotate(newRot);
158  p->setTransform(m, true);
159  }
160  QRect iconRect;
161  tabLayout(proxy(), tab, widget, &tr, &iconRect);
162  tr = proxy()->subElementRect(SE_TabBarTabText, opt, widget); //we compute tr twice because the style may override subElementRect
163 
164  if (!tab->icon.isNull()) {
165  QPixmap tabIcon = tab->icon.pixmap(qt_getWindow(widget), tab->iconSize,
166  (tab->state & State_Enabled) ? QIcon::Normal
167  : QIcon::Disabled,
168  (tab->state & State_Selected) ? QIcon::On
169  : QIcon::Off);
170  p->drawPixmap(iconRect.x(), iconRect.y(), tabIcon);
171  }
172 
173  bool visible_close_button = false;
174 
175  if (const QTabBar *tb = qobject_cast<const QTabBar*>(widget)) {
176  visible_close_button = tb->tabsClosable();
177  }
178 
179 #ifdef DTKWIDGET_CLASS_DTabBar
180  if (const DTabBar *tb = qobject_cast<const DTabBar*>(widget)) {
181  visible_close_button = tb->tabsClosable();
182  }
183 
184  visible_close_button = visible_close_button && (opt->state & QStyle::State_MouseOver);
185 #endif
186 
187  QBrush text_brush;
188  // TODO(zccrs): 临时解决方案，用于支持应用程序中自定义DTabBar的被选中Tab的文本颜色
189  const QWidget *tab_bar = findDTabBar(widget);
190  if (opt->state.testFlag(QStyle::State_Selected) && tab_bar) {
191  text_brush = tab_bar->palette().brush(QPalette::Active, QPalette::Text);
192  } else {
193  text_brush = m_palette->brush(PaletteExtended::TabBarTab_TextColor, opt);
194  }
195 
196  if (visible_close_button) {
197  QRect text_rect = opt->fontMetrics.boundingRect(tr, alignment, tab->text);
198 
199  const QBrush &brush = text_brush;
200  QLinearGradient lg(0, 0, 1, 0);
201  QGradientStops stops;
202  qreal stop = qreal(tr.right() - 35 - text_rect.x()) / text_rect.width();
203 
204  stops << QGradientStop{0, brush.color()};
205  stops << QGradientStop{stop, brush.color()};
206  stops << QGradientStop{stop, QColor(brush.color().red(), brush.color().green(), brush.color().blue(), 200)};
207  stops << QGradientStop{1, Qt::transparent};
208 
209  lg.setCoordinateMode(QLinearGradient::ObjectBoundingMode);
210  lg.setStops(stops);
211  p->setPen(QPen(QBrush(lg), 1));
212  } else {
213  p->setPen(QPen(text_brush, 1));
214  }
215 
216  p->drawText(tr, alignment, tab->text);
217 
218  if (verticalTabs)
219  p->restore();
220 
221  if (tab->state & State_HasFocus) {
222  const int OFFSET = 1 + pixelMetric(PM_DefaultFrameWidth);
223 
224  int x1, x2;
225  x1 = tab->rect.left();
226  x2 = tab->rect.right() - 1;
227 
228  QStyleOptionFocusRect fropt;
229  fropt.QStyleOption::operator=(*tab);
230  fropt.rect.setRect(x1 + 1 + OFFSET, tab->rect.y() + OFFSET,
231  x2 - x1 - 2*OFFSET, tab->rect.height() - 2*OFFSET);
232  drawPrimitive(PE_FrameFocusRect, &fropt, p, widget);
233  }
234 
235  return true;
236 }

238 bool Style::drawTabBarTabShapeControl(const QStyleOption *opt, QPainter *p, const QWidget *widget) const
239 {
240  auto tab = static_cast<const QStyleOptionTab*>(opt);
241 
242  p->save();
243 
244  QRect rect(tab->rect);
245  bool selected = tab->state & State_Selected;
246  bool onlyOne = tab->position == QStyleOptionTab::OnlyOneTab;
247  int tabOverlap = onlyOne ? 0 : proxy()->pixelMetric(PM_TabBarTabOverlap, opt, widget);
248 
249  if (!selected) {
250  switch (tab->shape) {
251  case QTabBar::TriangularNorth:
252  rect.adjust(1, 1, -1, -tabOverlap);
253  break;
254  case QTabBar::TriangularSouth:
255  rect.adjust(1, tabOverlap, -1, -1);
256  break;
257  case QTabBar::TriangularEast:
258  rect.adjust(tabOverlap, 1, -1, -1);
259  break;
260  case QTabBar::TriangularWest:
261  rect.adjust(1, 1, -tabOverlap, -1);
262  break;
263  default:
264  break;
265  }
266  }
267 
268  p->setPen(QPen(m_palette->brush(PaletteExtended::TabBarTab_BorderBrush, opt), 0));
269  p->setBrush(m_palette->brush(PaletteExtended::TabBarTab_BackgroundBrush, opt));
270 // if (selected) {
271 // p->setBrush(tab->palette.base());
272 // } else {
273 // if (widget && widget->parentWidget())
274 // p->setBrush(widget->parentWidget()->palette().background());
275 // else
276 // p->setBrush(tab->palette.background());
277 // }
278 
279  int y;
280  int x;
281  QPolygon a(10);
282  switch (tab->shape) {
283  case QTabBar::TriangularNorth:
284  case QTabBar::TriangularSouth: {
285  a.setPoint(0, 0, -1);
286  a.setPoint(1, 0, 0);
287  y = rect.height() - 2;
288  x = y / 3;
289  a.setPoint(2, x++, y - 1);
290  ++x;
291  a.setPoint(3, x++, y++);
292  a.setPoint(4, x, y);
293 
294  int i;
295  int right = rect.width() - 1;
296  for (i = 0; i < 5; ++i)
297  a.setPoint(9 - i, right - a.point(i).x(), a.point(i).y());
298  if (tab->shape == QTabBar::TriangularNorth)
299  for (i = 0; i < 10; ++i)
300  a.setPoint(i, a.point(i).x(), rect.height() - 1 - a.point(i).y());
301 
302  a.translate(rect.left(), rect.top());
303  p->setRenderHint(QPainter::Antialiasing);
304  p->translate(0, 0.5);
305 
306  QPainterPath path;
307  path.addPolygon(a);
308  p->drawPath(path);
309  break; }
310  case QTabBar::TriangularEast:
311  case QTabBar::TriangularWest: {
312  a.setPoint(0, -1, 0);
313  a.setPoint(1, 0, 0);
314  x = rect.width() - 2;
315  y = x / 3;
316  a.setPoint(2, x - 1, y++);
317  ++y;
318  a.setPoint(3, x++, y++);
319  a.setPoint(4, x, y);
320  int i;
321  int bottom = rect.height() - 1;
322  for (i = 0; i < 5; ++i)
323  a.setPoint(9 - i, a.point(i).x(), bottom - a.point(i).y());
324  if (tab->shape == QTabBar::TriangularWest)
325  for (i = 0; i < 10; ++i)
326  a.setPoint(i, rect.width() - 1 - a.point(i).x(), a.point(i).y());
327  a.translate(rect.left(), rect.top());
328  p->setRenderHint(QPainter::Antialiasing);
329  p->translate(0.5, 0);
330  QPainterPath path;
331  path.addPolygon(a);
332  p->drawPath(path);
333  break; }
334  case QTabBar::RoundedNorth:
335  case QTabBar::RoundedSouth: {
336  const QRect &border_rect = tab->rect.adjusted(0, 0, 1, 1);
337 
338  p->fillRect(tab->rect, p->brush());
339 
340  if (tab->position != QStyleOptionTab::Beginning
341  && tab->position != QStyleOptionTab::OnlyOneTab) {
342  p->drawLine(border_rect.topLeft(), border_rect.bottomLeft());
343  }
344 
345  if (Q_UNLIKELY(selected)) {
346  QRect active_rect = tab->rect;
347 
348  if (Q_LIKELY(tab->shape == QTabBar::RoundedNorth)) {
349  active_rect.setTop(active_rect.bottom() - qMin(3, active_rect.height() / 10) + 1);
350  } else {
351  active_rect.setBottom(active_rect.top() + qMin(3, active_rect.height() / 10) + 1);
352  }
353 
354  // TODO(zccrs): 临时解决方案，用于支持应用程序中自定义DTabBar的被选中Tab的文本颜色
355  if (const QWidget *tab_bar = findDTabBar(widget))
356  p->fillRect(active_rect, tab_bar->palette().brush(QPalette::Active, QPalette::Text));
357  else
358  p->fillRect(active_rect, m_palette->brush(PaletteExtended::TabBarTab_ActiveColor, opt));
359  }
360 
361  break;
362  }
363  case QTabBar::RoundedEast:
364  case QTabBar::RoundedWest: {
365  const QRect &border_rect = tab->rect.adjusted(0, 0, 1, 1);
366 
367  p->fillRect(tab->rect, p->brush());
368 
369  if (tab->position != QStyleOptionTab::Beginning
370  && tab->position != QStyleOptionTab::OnlyOneTab) {
371  p->drawLine(border_rect.topLeft(), border_rect.topRight());
372  }
373 
374  if (Q_UNLIKELY(selected)) {
375  QRect active_rect = tab->rect;
376 
377  if (Q_LIKELY(tab->shape == QTabBar::RoundedWest)) {
378  active_rect.setLeft(active_rect.right() - qMin(3, active_rect.width() / 10) + 1);
379  } else {
380  active_rect.setRight(active_rect.left() + qMin(3, active_rect.width() / 10) + 1);
381  }
382 
383  // TODO(zccrs): 临时解决方案，用于支持应用程序中自定义DTabBar的被选中Tab的文本颜色
384  if (const QWidget *tab_bar = findDTabBar(widget))
385  p->fillRect(active_rect, tab_bar->palette().brush(QPalette::Active, QPalette::Text));
386  else
387  p->fillRect(active_rect, m_palette->brush(PaletteExtended::TabBarTab_ActiveColor, opt));
388  }
389 
390  break;
391  }
392  }
393  p->restore();
394 
395  return true;
396 }

431 bool Style::drawIndicatorTabClosePrimitive(const QStyleOption *opt, QPainter *p, const QWidget *widget) const
432 {
433  const QTabBar *tb = qobject_cast<QTabBar*>(widget->parent());
434 
435  if (!tb) {
436  if (const QWidget *w = dynamic_cast<const QWidget*>(p->device()))
437  widget = w;
438 
439  tb = qobject_cast<QTabBar*>(widget->parent());
440  }
441 
442  if (Q_UNLIKELY(!tb))
443  return false;
444 
445  int index = -1;
446 
447  for (int i = 0; i < tb->count(); ++i) {
448  if (Q_LIKELY(tb->tabButton(i, QTabBar::LeftSide) != widget
449  && tb->tabButton(i, QTabBar::RightSide) != widget)) {
450  continue;
451  }
452 
453  index = i;
454  break;
455  }
456 
457  if (Q_UNLIKELY(index < 0))
458  return true;
459 
460  QStyleOptionTab tab;
461 
462  static_cast<const DQTabBar*>(tb)->initStyleOption(&tab, index);
463 
464  if (Q_LIKELY((tab.state | QStyle::State_MouseOver) != tab.state
465  // 关闭标签时标签栏会重新布局，此时鼠标没有移动，但是标签item相对于鼠标的位置发生了变化
466  // 如果鼠标落在了标签页关闭按钮右边位置，则这个标签页的state中不会包含MouseOver状态
467  && !tb->tabRect(index).contains(tb->mapFromGlobal(QCursor::pos())))) {
468  return true;
469  }
470 
471  fillBrush(p, opt->rect, m_palette->brush(PaletteExtended::TabBarTab_CloseIcon, opt));
472 
473  return true;
474 }
