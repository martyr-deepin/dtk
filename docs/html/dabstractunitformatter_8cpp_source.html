73 qreal DAbstractUnitFormatter::formatAs(qreal value, int currentUnit, const int targetUnit) const

92 QPair<qreal, int> DAbstractUnitFormatter::format(const qreal value, const int unit) const
93 {
94  // can convert to smaller unit
95  if (unit > unitMin() && value < unitValueMin(unit))
96  return format(value * unitConvertRate(unit - 1), unit - 1);
97 
98  // can convert to bigger unit
99  if (unit < unitMax() && value > unitValueMax(unit))
100  return format(value / unitConvertRate(unit), unit + 1);
101 
102  return QPair<qreal, int>(value, unit);
103 }

112 QList<QPair<qreal, int> > DAbstractUnitFormatter::formatAsUnitList(const qreal value, int unit) const
113 {
114  if (qFuzzyIsNull(value))
115  return QList<QPair<qreal, int>>();
116 
117  if (value < unitValueMin(unit) || unit == unitMin())
118  {
119  if (unit != unitMin())
120  return formatAsUnitList(value * unitConvertRate(unit - 1), unit - 1);
121  else
122  return std::move(QList<QPair<qreal, int>>() << QPair<qreal, int>(value, unit));
123  }
124 
125  ulong _value = ulong(value);
126  QList<QPair<qreal, int>> ret = formatAsUnitList(value - _value, unit);
127 
128  while (_value && unit != unitMax())
129  {
130  const ulong rate = unitConvertRate(unit);
131  const ulong r = _value % rate;
132  if (r)
133  ret.push_front(QPair<qreal, int>(r, unit));
134 
135  unit += 1;
136  _value /= rate;
137  }
138 
139  if (_value)
140  ret.push_front(QPair<qreal, int>(_value, unit));
141 
142  return ret;
143 }
