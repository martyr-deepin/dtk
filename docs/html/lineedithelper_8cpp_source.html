35 PaletteExtended::PseudoClassType lineEditStateToPseudoClassType(QStyle::State state)
36 {
37  if (!state.testFlag(QStyle::State_Enabled)) {
38  return PaletteExtended::PseudoClass_Disabled;

52 bool Style::drawFrameLineEditPrimitive(const QStyleOption *option, QPainter *painter, const QWidget *widget) const
53 {
54  Q_UNUSED(widget)
55 
56  const QRect& rect( option->rect );
57 
58  // render
59  const QBrush background(m_palette->brush(PaletteExtended::LineEdit_BackgroundBrush, lineEditStateToPseudoClassType(option->state)) );
60  QBrush outline(m_palette->brush(PaletteExtended::LineEdit_BorderBrush, lineEditStateToPseudoClassType(option->state)) );
61 
62 #ifdef DTKWIDGET_CLASS_DLineEdit
63  if (const DLineEdit *edit = qobject_cast<const DLineEdit*>(widget)) {
64  DStyleOptionLineEdit edit_option;
65  edit_option.init(const_cast<DLineEdit*>(edit));
66 
67  if (edit_option.features.testFlag(DStyleOptionLineEdit::Alert))
68  outline = m_palette->brush(PaletteExtended::LineEdit_AlertBorderBrush, option, PaletteExtended::PseudoClass_Unknown, outline);
69 
70  CommonHelper::renderFrame( painter, rect, background, outline );
71 
72  if (edit_option.features.testFlag(DStyleOptionLineEdit::IconButton)) {
73  painter->save();
74  painter->setPen(QPen(outline, Metrics::Painter_PenWidth));
75  painter->drawLine(QPointF(edit_option.iconButtonRect.topLeft()) - QPointF(0.5, 0), QPointF(edit_option.iconButtonRect.bottomLeft()) + QPoint(-0.5, 1));
76  painter->restore();
77  }
78  } else
79 #endif
80  {
81  CommonHelper::renderFrame( painter, rect, background, outline );
82  }
83 
84  return true;
85 }
