51 QStringList DFileSystemWatcherPrivate::addPaths(const QStringList &paths, QStringList *files, QStringList *directories)
52 {
53  QStringList p = paths;
54  QMutableListIterator<QString> it(p);
55  while (it.hasNext()) {
56  QString path = it.next();
57  QFileInfo fi(path);
58  bool isDir = fi.isDir();
59  if (isDir) {
60  if (directories->contains(path))
61  continue;
62  } else {
63  if (files->contains(path))
64  continue;
65  }
66 
67  int wd = inotify_add_watch(inotifyFd,
68  QFile::encodeName(path),
69  (isDir
70  ? (0
71  | IN_ATTRIB
72  | IN_MOVE
73  | IN_MOVE_SELF
74  | IN_CREATE
75  | IN_DELETE
76  | IN_DELETE_SELF
77  | IN_MODIFY
78  )
79  : (0
80  | IN_ATTRIB
81  | IN_CLOSE_WRITE
82  | IN_MODIFY
83  | IN_MOVE
84  | IN_MOVE_SELF
85  | IN_DELETE_SELF
86  )));
87  if (wd < 0) {
88  perror("DFileSystemWatcherPrivate::addPaths: inotify_add_watch failed");
89  continue;
90  }
91 
92  it.remove();
93 
94  int id = isDir ? -wd : wd;
95  if (id < 0) {
96  directories->append(path);
97  } else {
98  files->append(path);
99  }
100 
101  pathToID.insert(path, id);
102  idToPath.insert(id, path);
103  }
104 
105  return p;
106 }

108 QStringList DFileSystemWatcherPrivate::removePaths(const QStringList &paths, QStringList *files, QStringList *directories)
109 {
110  QStringList p = paths;
111  QMutableListIterator<QString> it(p);
112  while (it.hasNext()) {
113  QString path = it.next();
114  int id = pathToID.take(path);
115  for (auto hit = idToPath.find(id); hit != idToPath.end() && hit.key() == id; ++hit) {
116  if (hit.value() == path) {
117  idToPath.erase(hit);
118  break;
119  }
120  }
121 
122  it.remove();
123 
124  if (!idToPath.contains(id)) {
125  int wd = id < 0 ? -id : id;
126  //qDebug() << "removing watch for path" << path << "wd" << wd;
127  inotify_rm_watch(inotifyFd, wd);
128  }
129 
130  if (id < 0) {
131  directories->removeAll(path);
132  } else {
133  files->removeAll(path);
134  }
135  }
136 
137  return p;
138 }

140 void DFileSystemWatcherPrivate::_q_readFromInotify()
141 {
142  Q_Q(DFileSystemWatcher);
143 // qDebug() << "QInotifyFileSystemWatcherEngine::readFromInotify";
144 
145  int buffSize = 0;
146  ioctl(inotifyFd, FIONREAD, (char *) &buffSize);
147  QVarLengthArray<char, 4096> buffer(buffSize);
148  buffSize = read(inotifyFd, buffer.data(), buffSize);
149  char *at = buffer.data();
150  char * const end = at + buffSize;
151 
152  QList<inotify_event *> eventList;
153  QMultiHash<int, QString> batch_pathmap;
155  QMultiMap<int, QString> cookieToFilePath;
156  QMultiMap<int, QString> cookieToFileName;
157  QSet<int> hasMoveFromByCookie;
158 #ifdef QT_DEBUG
159  int exist_count = 0;
160 #endif
161  while (at < end) {
162  inotify_event *event = reinterpret_cast<inotify_event *>(at);
163  QStringList paths;
164 
165  at += sizeof(inotify_event) + event->len;
166 
167  int id = event->wd;
168  paths = idToPath.values(id);
169  if (paths.empty()) {
170  // perhaps a directory?
171  id = -id;
172  paths = idToPath.values(id);
173  if (paths.empty())
174  continue;
175  }
176 
177  if (!(event->mask & IN_MOVED_TO) || !hasMoveFromByCookie.contains(event->cookie)) {
178  auto it = std::find_if(eventList.begin(), eventList.end(), [event](inotify_event *e){
179  return event->wd == e->wd && event->mask == e->mask &&
180  event->cookie == e->cookie &&
181  event->len == e->len &&
182  !strcmp(event->name, e->name);
183  });
184 
185  if (it==eventList.end()) {
186  eventList.append(event);
187  }
188 #ifdef QT_DEBUG
189  else {
190  qDebug() << "exist event:" << "event->wd" << event->wd <<
191  "event->mask" << event->mask <<
192  "event->cookie" << event->cookie << "exist counts " << ++exist_count;
193  }
194 #endif
195  const QList<QString> bps = batch_pathmap.values(id);
196  for (auto &path : paths) {
197  if (!bps.contains(path)) {
198  batch_pathmap.insert(id, path);
199  }
200  }
201  }
202 
203  if (event->mask & IN_MOVED_TO) {
204  for (auto &path : paths) {
205  cookieToFilePath.insert(event->cookie, path);
206  }
207  cookieToFileName.insert(event->cookie, QString::fromUtf8(event->name));
208  }
209 
210  if (event->mask & IN_MOVED_FROM)
211  hasMoveFromByCookie << event->cookie;
212  }
213 
214 // qDebug() << "event count:" << eventList.count();
215 
216  QList<inotify_event *>::const_iterator it = eventList.constBegin();
217  while (it != eventList.constEnd()) {
218  const inotify_event &event = **it;
219  ++it;
220 
221 // qDebug() << "inotify event, wd" << event.wd << "cookie" << event.cookie << "mask" << hex << event.mask;
222 
223  int id = event.wd;
224  QStringList paths = batch_pathmap.values(id);
225 
226  if (paths.empty()) {
227  id = -id;
228  paths = batch_pathmap.values(id);
229 
230  if (paths.empty())
231  continue;
232  }
233  const QString &name = QString::fromUtf8(event.name);
234 
235  for (auto &path : paths) {
236 // qDebug() << "event for path" << path;
237 
238 // /// TODO: Existence of invalid utf8 characters QFile can not read the file information
239 // if (event.name != QString::fromLocal8Bit(event.name).toLocal8Bit()) {
240 // if (event.mask & (IN_CREATE | IN_MOVED_TO)) {
241 // DFMGlobal::fileNameCorrection(path);
242 // }
243 // }
244 
245  if ((event.mask & (IN_DELETE_SELF | IN_MOVE_SELF | IN_UNMOUNT)) != 0) {
246  do {
247  if (event.mask & IN_MOVE_SELF) {
248  QMap<int, QString>::const_iterator iterator = cookieToFilePath.constBegin();
249 
250  bool isMove = false;
251 
252  while (iterator != cookieToFilePath.constEnd()) {
253  const QString &_path = iterator.value();
254  const QString &_name = cookieToFileName.value(iterator.key());
255 
256  if (QFileInfo(_path + QDir::separator() + _name) == QFileInfo(path)) {
257  isMove = true;
258  break;
259  }
260 
261  ++iterator;
262  }
263 
264  if (isMove)
265  break;
266  }
267 
269 // pathToID.remove(path);
270 // idToPath.remove(id, getPathFromID(id));
271 // if (!idToPath.contains(id))
272 // inotify_rm_watch(inotifyFd, event.wd);
273 
274 // if (id < 0)
275 // onDirectoryChanged(path, true);
276 // else
277 // onFileChanged(path, true);
278 
279  Q_EMIT q->fileDeleted(path, QString(), DFileSystemWatcher::QPrivateSignal());
280  } while (false);
281  } else {
282  if (id < 0)
283  onDirectoryChanged(path, false);
284  else
285  onFileChanged(path, false);
286  }
287 
288  QString filePath = path;
289 
290  if (id < 0) {
291  if (path.endsWith(QDir::separator()))
292  filePath = path + name;
293  else
294  filePath = path + QDir::separator() + name;
295  }
296 
297  if (event.mask & IN_CREATE) {
298 // qDebug() << "IN_CREATE" << filePath << name;
299 
300  if (name.isEmpty()) {
301  if (pathToID.contains(path)) {
302  q->removePath(path);
303  q->addPath(path);
304  }
305  } else if (pathToID.contains(filePath)) {
306  q->removePath(filePath);
307  q->addPath(filePath);
308  }
309 
310  Q_EMIT q->fileCreated(path, name, DFileSystemWatcher::QPrivateSignal());
311  }
312 
313  if (event.mask & IN_DELETE) {
314 // qDebug() << "IN_DELETE" << filePath;
315 
316  Q_EMIT q->fileDeleted(path, name, DFileSystemWatcher::QPrivateSignal());
317  }
318 
319  if (event.mask & IN_MOVED_FROM) {
320  const QString toName = cookieToFileName.value(event.cookie);
321 
322  if (cookieToFilePath.values(event.cookie).empty()) {
323  Q_EMIT q->fileMoved(path, name, QString(), QString(), DFileSystemWatcher::QPrivateSignal());
324  } else {
325  for (QString &toPath : cookieToFilePath.values(event.cookie)) {
326 // qDebug() << "IN_MOVED_FROM" << filePath << "to path:" << toPath << "to name:" << toName;
327 
328  Q_EMIT q->fileMoved(path, name, toPath, toName, DFileSystemWatcher::QPrivateSignal());
329  }
330  }
331  }
332 
333  if (event.mask & IN_MOVED_TO) {
334 // qDebug() << "IN_MOVED_TO" << filePath;
335 
336  if (!hasMoveFromByCookie.contains(event.cookie))
337  Q_EMIT q->fileMoved(QString(), QString(), path, name, DFileSystemWatcher::QPrivateSignal());
338  }
339 
340  if (event.mask & IN_ATTRIB) {
341 // qDebug() << "IN_ATTRIB" << event.mask << filePath;
342 
343  Q_EMIT q->fileAttributeChanged(path, name, DFileSystemWatcher::QPrivateSignal());
344  }
345 
346  /*only monitor file close event which is opend by write mode*/
347  if (event.mask & IN_CLOSE_WRITE) {
348 // qDebug() << "IN_CLOSE_WRITE" << event.mask << filePath;
349 
350  Q_EMIT q->fileClosed(path, id < 0 ? name : QString(), DFileSystemWatcher::QPrivateSignal());
351  }
352 
353  if (event.mask & IN_MODIFY) {
354 // qDebug() << "IN_MODIFY" << event.mask << filePath << name;
355 
356  Q_EMIT q->fileModified(path, name, DFileSystemWatcher::QPrivateSignal());
357  }
358  }
359  }
