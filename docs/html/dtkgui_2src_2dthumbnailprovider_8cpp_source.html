127 bool DThumbnailProvider::hasThumbnail(const QFileInfo &info) const
128 {
129  Q_D(const DThumbnailProvider);
130 
131  if (!info.isReadable() || !info.isFile())
132  {
133  return false;
134  }
135 
136  qint64 fileSize = info.size();
137 
138  if (fileSize <= 0)
139  {
140  return false;
141  }
142 
143  const QMimeType &mime = d->mimeDatabase.mimeTypeForFile(info);
144 
145  if (fileSize > sizeLimit(mime))
146  {
147  return false;
148  }
149 
150  return hasThumbnail(mime);
151 }

153 bool DThumbnailProvider::hasThumbnail(const QMimeType &mimeType) const
154 {
155  const QString &mime = mimeType.name();
156 
157  if (DThumbnailProviderPrivate::hasThumbnailMimeHash.isEmpty())
158  {
159  const QList<QByteArray> &mimeTypes = QImageReader::supportedMimeTypes();
160 
161  if (mimeTypes.isEmpty())
162  {
163  DThumbnailProviderPrivate::hasThumbnailMimeHash.insert("");
164 
165  return false;
166  }
167 
168  DThumbnailProviderPrivate::hasThumbnailMimeHash.reserve(mimeTypes.size());
169 
170  for (const QByteArray &t : mimeTypes)
171  {
172  DThumbnailProviderPrivate::hasThumbnailMimeHash.insert(QString::fromLocal8Bit(t));
173  }
174  }
175 
176  return DThumbnailProviderPrivate::hasThumbnailMimeHash.contains(mime);
177 }

185 QString DThumbnailProvider::thumbnailFilePath(const QFileInfo &info, Size size) const
186 {
187  Q_D(const DThumbnailProvider);
188 
189  const QString &absolutePath = info.absolutePath();
190  const QString &absoluteFilePath = info.absoluteFilePath();
191 
192  if (absolutePath == d->sizeToFilePath(Small)
193  || absolutePath == d->sizeToFilePath(Normal)
194  || absolutePath == d->sizeToFilePath(Large)
195  || absolutePath == THUMBNAIL_FAIL_PATH)
196  {
197  return absoluteFilePath;
198  }
199 
200  const QString thumbnailName = dataToMd5Hex(QUrl::fromLocalFile(absoluteFilePath).toString(QUrl::FullyEncoded).toLocal8Bit()) + FORMAT;
201  QString thumbnail = d->sizeToFilePath(size) + QDir::separator() + thumbnailName;
202 
203  if (!QFile::exists(thumbnail))
204  {
205  return QString();
206  }
207 
208  QImage image(thumbnail);
209 
210  if (image.text(QT_STRINGIFY(Thumb::MTime)).toInt() != (int)info.lastModified().toTime_t())
211  {
212  QFile::remove(thumbnail);
213 
214  Q_EMIT thumbnailChanged(absoluteFilePath, QString());
215 
216  return QString();
217  }
218 
219  return thumbnail;
220 }

228 QString DThumbnailProvider::createThumbnail(const QFileInfo &info, DThumbnailProvider::Size size)
229 {
230  Q_D(DThumbnailProvider);
231 
232  d->errorString.clear();
233 
234  const QString &absolutePath = info.absolutePath();
235  const QString &absoluteFilePath = info.absoluteFilePath();
236 
237  if (absolutePath == d->sizeToFilePath(Small)
238  || absolutePath == d->sizeToFilePath(Normal)
239  || absolutePath == d->sizeToFilePath(Large)
240  || absolutePath == THUMBNAIL_FAIL_PATH)
241  {
242  return absoluteFilePath;
243  }
244 
245  if (!hasThumbnail(info))
246  {
247  d->errorString = QStringLiteral("This file has not support thumbnail: ") + absoluteFilePath;
248 
250  return QString();
251  }
252 
253  const QString fileUrl = QUrl::fromLocalFile(absoluteFilePath).toString(QUrl::FullyEncoded);
254  const QString thumbnailName = dataToMd5Hex(fileUrl.toLocal8Bit()) + FORMAT;
255 
256  // the file is in fail path
257  QString thumbnail = THUMBNAIL_FAIL_PATH + QDir::separator() + thumbnailName;
258 
259  if (QFile::exists(thumbnail))
260  {
261  QImage image(thumbnail);
262 
263  if (image.text(QT_STRINGIFY(Thumb::MTime)).toInt() != (int)info.lastModified().toTime_t())
264  {
265  QFile::remove(thumbnail);
266  }
267  else
268  {
269  return QString();
270  }
271  }// end
272 
273  QScopedPointer<QImage> image(new QImage(QSize(size, size), QImage::Format_ARGB32_Premultiplied));
274  QImageReader reader(absoluteFilePath);
275 
276  if (!reader.canRead())
277  {
278  reader.setFormat(d->mimeDatabase.mimeTypeForFile(info).name().toLocal8Bit());
279 
280  if (!reader.canRead())
281  {
282  d->errorString = reader.errorString();
283  }
284  }
285 
286  if (d->errorString.isEmpty())
287  {
288  const QSize &imageSize = reader.size();
289 
290  if (imageSize.isValid())
291  {
292  if (imageSize.width() >= size || imageSize.height() >= size)
293  {
294  reader.setScaledSize(reader.size().scaled(size, size, Qt::KeepAspectRatio));
295  }
296 
297  if (!reader.read(image.data()))
298  {
299  d->errorString = reader.errorString();
300  }
301  }
302  else
303  {
304  d->errorString = "Fail to read image file attribute data:" + info.absoluteFilePath();
305  }
306  }
307 
308  // successful
309  if (d->errorString.isEmpty())
310  {
311  thumbnail = d->sizeToFilePath(size) + QDir::separator() + thumbnailName;
312  }
313  else
314  {
315  //fail
316  image.reset(new QImage(1, 1, QImage::Format_Mono));
317  }
318 
319  image->setText(QT_STRINGIFY(Thumb::URL), fileUrl);
320  image->setText(QT_STRINGIFY(Thumb::MTime), QString::number(info.lastModified().toTime_t()));
321 
322  // create path
323  QFileInfo(thumbnail).absoluteDir().mkpath(".");
324 
325  if (!image->save(thumbnail, Q_NULLPTR, 80))
326  {
327  d->errorString = QStringLiteral("Can not save image to ") + thumbnail;
328  }
329 
330  if (d->errorString.isEmpty())
331  {
332  Q_EMIT createThumbnailFinished(absoluteFilePath, thumbnail);
333  Q_EMIT thumbnailChanged(absoluteFilePath, thumbnail);
334 
335  return thumbnail;
336  }
337 
338  // fail
339  Q_EMIT createThumbnailFailed(absoluteFilePath);
340 
341  return QString();
342 }

468 void DThumbnailProvider::run()
469 {
470  Q_D(DThumbnailProvider);
471 
472  Q_FOREVER
473  {
474  QWriteLocker locker(&d->dataReadWriteLock);
475 
476  if (d->produceQueue.isEmpty())
477  {
478  d->waitCondition.wait(&d->dataReadWriteLock);
479  }
480 
481  if (!d->running)
482  {
483  return;
484  }
485 
486  const DThumbnailProviderPrivate::ProduceInfo &task = d->produceQueue.dequeue();
487  const QPair<QString, DThumbnailProvider::Size> &tmpKey = qMakePair(task.fileInfo.absoluteFilePath(), task.size);
488 
489  if (d->discardedProduceInfos.contains(tmpKey))
490  {
491  d->discardedProduceInfos.remove(tmpKey);
492  locker.unlock();
493  continue;
494  }
495 
496  locker.unlock();
497 
498  const QString &thumbnail = createThumbnail(task.fileInfo, task.size);
499 
500  if (task.callback)
501  {
502  task.callback(thumbnail);
503  }
504  }
505 }
