83 void DApplicationSettingsPrivate::_q_onChanged(const QString &key)
84 {
85 #ifdef Q_OS_LINUX
86  if (key != PALETTE_TYPE_KEY)
87  return;
88 
89  const QString &palette_type = genericSettings->get(PALETTE_TYPE_KEY).toString();
90 
91  if (palette_type == "LightType") {
92  DGuiApplicationHelper::instance()->setPaletteType(DGuiApplicationHelper::LightType);
93  } else if (palette_type == "DarkType") {
94  DGuiApplicationHelper::instance()->setPaletteType(DGuiApplicationHelper::DarkType);
95  } else if (palette_type == "UnknownType") {
96  DGuiApplicationHelper::instance()->setPaletteType(DGuiApplicationHelper::UnknownType);
97  }
98 #endif
99 }

101 void DApplicationSettingsPrivate::_q_onPaletteTypeChanged()
102 {
103 #ifdef Q_OS_LINUX
104  switch (DGuiApplicationHelper::instance()->paletteType()) {
105  case DGuiApplicationHelper::LightType:
106  genericSettings->set(PALETTE_TYPE_KEY, "LightType");
107  break;
108  case DGuiApplicationHelper::DarkType:
109  genericSettings->set(PALETTE_TYPE_KEY, "DarkType");
110  break;
111  default:
112  genericSettings->set(PALETTE_TYPE_KEY, "UnknownType");
113  break;
114  }
115 #endif
116 }
