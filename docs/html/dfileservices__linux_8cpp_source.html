39 static QStringList urls2uris(const QList<QUrl> &urls)
40 {
41  QStringList list;
42 
43  list.reserve(urls.size());
44 
45  for (const QUrl &url : urls) {
46  list << url.toString();
47  }
48 
49  return list;
50 }

52 static QList<QUrl> path2urls(const QList<QString> &paths)
53 {
54  QList<QUrl> list;
55 
56  list.reserve(paths.size());
57 
58  for (const QString &path : paths) {
59  list << QUrl::fromLocalFile(path);
60  }
61 
62  return list;
63 }
