25 static QWindow *ensureWindowHandle(QWidget *widget)
26 {
27  QWidget *window = widget->window();
28  QWindow *handle = window->windowHandle();
29 
30  if (!handle) {
31  bool save_flag = qApp->testAttribute(Qt::AA_DontCreateNativeWidgetSiblings);
32  qApp->setAttribute(Qt::AA_DontCreateNativeWidgetSiblings, true);
33 
34  if (window->testAttribute(Qt::WA_NativeWindow)) {
35  // FIX null handle return on app kdevelop
36  window->setAttribute(Qt::WA_NativeWindow, false);
37  }
38 
39  window->setAttribute(Qt::WA_NativeWindow);
40  handle = window->windowHandle();
41  window->setAttribute(Qt::WA_NativeWindow, false);
42 
43  // dxcb version >= 1.1.6
44  if (!DPlatformWindowHandle::pluginVersion().isEmpty()) {
46  qApp->setAttribute(Qt::AA_DontCreateNativeWidgetSiblings, save_flag);
47  }
48  }
49 
50  return handle;
51 }
