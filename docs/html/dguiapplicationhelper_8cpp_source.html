73 void DGuiApplicationHelperPrivate::initApplication(QGuiApplication *app)
74 {
75  D_Q(DGuiApplicationHelper);
76 
77  q->connect(app, &QGuiApplication::paletteChanged, q, [q, this, app] {
78  if (themeType == DGuiApplicationHelper::UnknownType)
79  Q_EMIT q->themeTypeChanged(q->toColorType(app->palette()));
80  });
81 
82 
83  app->connect(systemTheme, &DPlatformTheme::themeNameChanged, app, [this, app] {
84  if (appTheme == systemTheme)
85  return;
86 
87  appTheme->setFallbackProperty(false);
88  const QByteArray &theme_name = appTheme->themeName();
89  appTheme->setFallbackProperty(true);
90 
91  if (!theme_name.isEmpty())
92  notifyAppThemeChanged(app);
93  });
94  app->connect(systemTheme, &DPlatformTheme::activeColorChanged, app, [this, app] {
95  if (appTheme == systemTheme)
96  return;
97 
98  appTheme->setFallbackProperty(false);
99  const QColor &active_color = appTheme->activeColor();
100  appTheme->setFallbackProperty(true);
101 
102  if (!active_color.isValid())
103  notifyAppThemeChanged(app, true);
104  });
105  app->connect(systemTheme, &DPlatformTheme::paletteChanged, app, [this, app] {
106  if (appTheme == systemTheme)
107  return;
108 
109  if (!appTheme->isValid())
110  notifyAppThemeChanged(app);
111  });
112 
113  if (QGuiApplicationPrivate::is_app_running) {
114  _q_initApplicationTheme();
115  } else {
116  // 延后初始化数据，因为在调用 clientLeader 前必须要保证QGuiApplication已经完全构造完成
117  q->metaObject()->invokeMethod(q, "_q_initApplicationTheme", Qt::QueuedConnection, Q_ARG(bool, true));
118  }
119 }

256 inline static int adjustColorValue(int base, qint8 increment, int max = 255)

259  : base * (1 + increment / 100.0);

294 QColor DGuiApplicationHelper::adjustColor(const QColor &base,
295  qint8 hueFloat, qint8 saturationFloat, qint8 lightnessFloat,
296  qint8 redFloat, qint8 greenFloat, qint8 blueFloat, qint8 alphaFloat)
297 {
298  // 按HSL格式调整
299  int H, S, L, A;
300  base.getHsl(&H, &S, &L, &A);
301 
302  H = H > 0 ? adjustColorValue(H, hueFloat, 359) : H;
303  S = adjustColorValue(S, saturationFloat);
304  L = adjustColorValue(L, lightnessFloat);
305  A = adjustColorValue(A, alphaFloat);
306 
307  QColor new_color = QColor::fromHsl(H, S, L, A);
308 
309  // 按RGB格式调整
310  int R, G, B;
311  new_color.getRgb(&R, &G, &B);
312 
313  R = adjustColorValue(R, redFloat);
314  G = adjustColorValue(G, greenFloat);
315  B = adjustColorValue(B, blueFloat);
316 
317  new_color.setRgb(R, G, B, A);
318 
319  return new_color;
320 }

328 QColor DGuiApplicationHelper::blendColor(const QColor &substrate, const QColor &superstratum)
329 {
330  QColor c2 = superstratum.toRgb();
331 
332  if (c2.alpha() >= 255)
333  return c2;
334 
335  QColor c1 = substrate.toRgb();
336  qreal c1_weight = 1 - c2.alphaF();
337 
338  int r = c1_weight * c1.red() + c2.alphaF() * c2.red();
339  int g = c1_weight * c1.green() + c2.alphaF() * c2.green();
340  int b = c1_weight * c1.blue() + c2.alphaF() * c2.blue();
341 
342  return QColor(r, g, b, c1.alpha());
343 }

426 DPalette DGuiApplicationHelper::standardPalette(DGuiApplicationHelper::ColorType type)
427 {
428  static const DPalette *light_palette = nullptr, *dark_palette = nullptr;
429  static const DPalette *alpha_light_palette = nullptr, *alpha_dark_palette = nullptr;
430 
431  if (type == LightType) {
432  if (Q_UNLIKELY(DGuiApplicationHelperPrivate::compositingColor)) {
433  if (Q_LIKELY(alpha_light_palette)) {
434  return *alpha_light_palette;
435  }
436  }
437 
438  if (Q_LIKELY(light_palette)) {
439  return *light_palette;
440  }
441  } else if (type == DarkType) {
442  if (Q_UNLIKELY(DGuiApplicationHelperPrivate::compositingColor)) {
443  if (Q_LIKELY(alpha_dark_palette)) {
444  return *alpha_dark_palette;
445  }
446  }
447 
448  if (Q_LIKELY(dark_palette)) {
449  return *dark_palette;
450  }
451  } else {
452  return DPalette();
453  }
454 
455  DPalette *pa;
456  const QColor *qcolor_list, *dcolor_list;
457 
458  if (type == DarkType) {
459  pa = new DPalette();
460 
461  if (DGuiApplicationHelperPrivate::compositingColor)
462  alpha_dark_palette = pa;
463  else
464  dark_palette = pa;
465 
466  qcolor_list = dark_qpalette;
467  dcolor_list = dark_dpalette;
468  } else {
469  pa = new DPalette();
470 
471  if (DGuiApplicationHelperPrivate::compositingColor)
472  alpha_light_palette = pa;
473  else
474  light_palette = pa;
475 
476  qcolor_list = light_qpalette;
477  dcolor_list = light_dpalette;
478  }
479 
480  for (int i = 0; i < DPalette::NColorRoles; ++i) {
481  QPalette::ColorRole role = static_cast<QPalette::ColorRole>(i);
482  QColor color = qcolor_list[i];
483 
484  // 处理半透明色
485  if (DGuiApplicationHelperPrivate::compositingColor) {
486  switch (role) {
487  case QPalette::Window:
488  color = type == LightType ? adjustColor(color, 0, 0, 0, 0, 0, 0, -20) : adjustColor(color, 0, 0, -10, 0, 0, 0, -20);
489  break;
490  case QPalette::Base:
491  color = adjustColor(color, 0, 0, 0, 0, 0, 0, -20);
492  break;
493  case QPalette::WindowText:
494  case QPalette::Text:
495  color = adjustColor(color, 0, 0, type == LightType ? -20 : +20, 0, 0, 0, -20);
496  break;
497  case QPalette::ButtonText:
498  color = type == LightType ? adjustColor(color, 0, 0, -20, 0, 0, 0, -20) : adjustColor(color, 0, 0, +20, 0, 0, 0, 0);
499  break;
500  case QPalette::Button:
501  case QPalette::Light:
502  case QPalette::Mid:
503  case QPalette::Midlight:
504  case QPalette::Dark:
505  color = adjustColor(color, 0, 0, -20, 0, 0, 0, -40);
506  break;
507  default:
508  break;
509  }
510  }
511 
512  pa->setColor(DPalette::Active, role, color);
513  generatePaletteColor(*pa, role, type);
514  }
515 
516  for (int i = 0; i < DPalette::NColorTypes; ++i) {
517  DPalette::ColorType role = static_cast<DPalette::ColorType>(i);
518  QColor color = dcolor_list[i];
519 
520  // 处理半透明色
521  if (DGuiApplicationHelperPrivate::compositingColor) {
522  switch (role) {
523  case DPalette::ItemBackground:
524  color = adjustColor(color, 0, 0, 100, 0, 0, 0, type == LightType ? -80 : -90);
525  break;
526  case DPalette::TextTitle:
527  color = adjustColor(color, 0, 0, -20, 0, 0, 0, -20);
528  break;
529  case DPalette::TextTips:
530  color = type == LightType ? adjustColor(color, 0, 0, -40, 0, 0, 0, -40) : adjustColor(color, 0, 0, +40, 0, 0, 0, -50);
531  break;
532  default:
533  break;
534  }
535  }
536 
537  pa->setColor(DPalette::Active, role, color);
538  generatePaletteColor(*pa, role, type);
539  }
540 
541  return *const_cast<const DPalette*>(pa);
542 }

544 template<typename M>
545 static void generatePaletteColor_helper(DPalette &base, M role, DGuiApplicationHelper::ColorType type)
546 {
547  if (type == DGuiApplicationHelper::UnknownType) {
548  type = DGuiApplicationHelper::toColorType(base);
549  }
550 
551  QColor disable_mask_color, inactive_mask_color;
552 
553  if (type == DGuiApplicationHelper::DarkType) {
554  disable_mask_color = dark_qpalette[QPalette::Window];
555  inactive_mask_color = dark_qpalette[QPalette::Window];
556  disable_mask_color.setAlphaF(0.7);
557  inactive_mask_color.setAlphaF(0.6);
558  } else {
559  disable_mask_color = light_qpalette[QPalette::Window];
560  inactive_mask_color = light_qpalette[QPalette::Window];
561  disable_mask_color.setAlphaF(0.6);
562  inactive_mask_color.setAlphaF(0.4);
563  }
564 
565  const QColor &color = base.color(QPalette::Normal, role);
566  base.setColor(QPalette::Disabled, role, DGuiApplicationHelper::blendColor(color, disable_mask_color));
567 
568  QPalette::ColorRole qr = static_cast<QPalette::ColorRole>(role);
569  if (qr == QPalette::Text) {
570  // disable text color olny -60% alpha
571  base.setColor(QPalette::Disabled, role, DGuiApplicationHelper::adjustColor(color, 0, 0, 0, 0, 0, 0, -60));
572  }
573 
574  if (DGuiApplicationHelperPrivate::useInactiveColor)
575  base.setColor(QPalette::Inactive, role, DGuiApplicationHelper::blendColor(color, inactive_mask_color));
576  else
577  base.setColor(QPalette::Inactive, role, color);
578 }

586 void DGuiApplicationHelper::generatePaletteColor(DPalette &base, QPalette::ColorRole role, DGuiApplicationHelper::ColorType type)
587 {
588  if (role == QPalette::Window) {
589  const QBrush &window = base.brush(QPalette::Normal, role);
590  base.setBrush(QPalette::Disabled, role, window);
591  base.setBrush(QPalette::Inactive, role, window);
592  return;
593  } else if (role == QPalette::Highlight && toColorType(base) == DarkType) {
594  // 暗色模式下的高亮色亮度要降低10%，避免太突兀
595  QColor highlight = base.highlight().color();
596 
597  if (highlight.isValid()) {
598  base.setColor(QPalette::Highlight, adjustColor(highlight, 0, 0, -20, 0, 0, 0, 0));
599  }
600  }
601 
602  generatePaletteColor_helper(base, role, type);
603 }

621 void DGuiApplicationHelper::generatePalette(DPalette &base, ColorType type)
622 {
623  // 先判断调色板的色调
624  if (type == UnknownType) {
625  type = toColorType(base);
626  }
627 
628  for (int i = 0; i < QPalette::NColorRoles; ++i) {
629  QPalette::ColorRole role = static_cast<QPalette::ColorRole>(i);
630  generatePaletteColor(base, role, type);
631  }
632 
633  for (int i = 0; i < QPalette::NColorRoles; ++i) {
634  DPalette::ColorType role = static_cast<DPalette::ColorType>(i);
635  generatePaletteColor(base, role, type);
636  }
637 }

644 DPalette DGuiApplicationHelper::fetchPalette(const DPlatformTheme *theme)
645 {
646  DPalette base_palette;
647  const QByteArray theme_name = theme->themeName();
648  ColorType type = LightType;
649 
650  // 深色主题
651  if (theme_name.endsWith("dark")) {
652  type = DarkType;
653  }
654 
655  bool ok = false;
656  base_palette = theme->fetchPalette(standardPalette(type), &ok);
657  const QColor &active_color = theme->activeColor();
658 
659  if (active_color.isValid()) {
660  base_palette.setColor(QPalette::Normal, QPalette::Highlight, active_color);
661 
662  // ok为true时会整体处理palette的所有颜色
663  if (!ok) {
664  generatePaletteColor(base_palette, QPalette::Highlight, type);
665  }
666  }
667 
668  if (ok) {
669  generatePalette(base_palette, type);
670  }
671 
672  return base_palette;
673 }

746 DPalette DGuiApplicationHelper::applicationPalette() const
747 {
748  D_DC(DGuiApplicationHelper);
749

752  }
753 
754  ColorType type = UnknownType;
755

757  if (qGuiApp && qGuiApp->testAttribute(Qt::AA_SetPalette)) {

883 bool DGuiApplicationHelper::setSingleInstance(const QString &key, DGuiApplicationHelper::SingleScope singleScope)
884 {
885  bool new_server = !_d_singleServer.exists();
886 
887  if (_d_singleServer->isListening()) {
888  _d_singleServer->close();
889  }
890 
891  QString socket_key = "_d_dtk_single_instance_";
892 
893  switch (singleScope) {
894  case GroupScope:
895  _d_singleServer->setSocketOptions(QLocalServer::GroupAccessOption);
896 #ifdef Q_OS_LINUX
897  socket_key += QString("%1_").arg(getgid());
898 #endif
899  break;
900  case WorldScope:
901  _d_singleServer->setSocketOptions(QLocalServer::WorldAccessOption);
902  break;
903  default:
904  _d_singleServer->setSocketOptions(QLocalServer::UserAccessOption);
905 #ifdef Q_OS_LINUX
906  socket_key += QString("%1_").arg(getuid());
907 #endif
908  break;
909  }
910 
911  socket_key += key;
912 
913  // 通知别的实例
914  QLocalSocket socket;
915  socket.connectToServer(socket_key);
916 
917  // 等待到有效数据时认为server实例有效
918  if (socket.waitForConnected(DGuiApplicationHelperPrivate::waitTime) && socket.waitForReadyRead(DGuiApplicationHelperPrivate::waitTime)) {
919  // 读取数据
920  qint8 version;
921  qint64 pid;
922  QStringList arguments;
923 
924  QDataStream ds(&socket);
925  ds >> version >> pid >> arguments;
926  qInfo() << "Process is started: pid=" << pid << "arguments=" << arguments;
927 
928  // 把自己的信息告诉第一个实例
929  ds << _d_singleServerVersion << qApp->applicationPid() << qApp->arguments();
930  socket.flush();
931 
932  return false;
933  }
934 
935  if (!_d_singleServer->listen(socket_key)) {
936  return false;
937  }
938 
939  if (new_server) {
940  QObject::connect(_d_singleServer, &QLocalServer::newConnection, qApp, [] {
941  QLocalSocket *instance = _d_singleServer->nextPendingConnection();
942  // 先发送数据告诉新的实例自己收到了它的请求
943  QDataStream ds(instance);
944  ds << _d_singleServerVersion // 协议版本
945  << qApp->applicationPid() // 进程id
946  << qApp->arguments(); // 启动时的参数
947 
948  QObject::connect(instance, &QLocalSocket::readyRead, qApp, [instance] {
949  // 读取数据
950  QDataStream ds(instance);
951 
952  qint8 version;
953  qint64 pid;
954  QStringList arguments;
955 
956  ds >> version >> pid >> arguments;
957  instance->close();
958 
959  qInfo() << "New instance: pid=" << pid << "arguments=" << arguments;
960 
961  // 通知新进程的信息
962  if (_globalHelper.exists())
963  Q_EMIT _globalHelper->helper->newProcessInstance(pid, arguments);
964  });
965 
966  instance->flush(); //发送数据给新的实例
967  });
968  }
969 
970  return true;
971 }
