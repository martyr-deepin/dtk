110 DFileDragClient::DFileDragClient(const QMimeData *data, QObject *parent)
111  : QObject(parent)
112  , DCORE_NAMESPACE::DObject(*new DFileDragClientPrivate(this))
113 {
114  D_D(DFileDragClient);
115  Q_ASSERT(checkMimeData(data));
116 
117  d->service = data->data(DND_MIME_SERVICE);
118  d->uuid = QUuid(data->data(DND_MIME_UUID));
119  d->connectionmap[d->uuid.toString()] = this;
120 
121  if (DDndClientSignalRelay::relayref.isNull()) {
122  d->relay = QSharedPointer<DDndClientSignalRelay>(new DDndClientSignalRelay);
123  DDndClientSignalRelay::relayref = d->relay.toWeakRef();
124  } else {
125  d->relay = DDndClientSignalRelay::relayref.toStrongRef();
126  }
127 
128  if (!DFileDragClientPrivate::ifacemap.contains(d->service)) {
129  QDBusConnection sysbus(QDBusConnection::systemBus());
130  d->iface = QSharedPointer<QDBusInterface>(new QDBusInterface(d->service, DND_OBJPATH, DND_INTERFACE, sysbus), [d](QDBusInterface* intf){
131  QDBusConnection sysbus(QDBusConnection::systemBus());
132  sysbus.disconnect(d->service, DND_OBJPATH, DND_INTERFACE, "progressChanged", "si", d->relay.data(), SLOT(progressChanged(QString, int)));
133  sysbus.disconnect(d->service, DND_OBJPATH, DND_INTERFACE, "stateChanged", "si", d->relay.data(), SLOT(stateChanged(QString, int)));
134  sysbus.disconnect(d->service, DND_OBJPATH, DND_INTERFACE, "serverDestroyed", "s", d->relay.data(), SLOT(serverDestroyed(QString)));
135  intf->deleteLater();
136  DFileDragClientPrivate::ifacemap.remove(d->service);
137  });
138  DFileDragClientPrivate::ifacemap[d->service] = d->iface.toWeakRef();
139  sysbus.connect(d->service, DND_OBJPATH, DND_INTERFACE, "progressChanged", "si", d->relay.data(), SLOT(progressChanged(QString, int)));
140  sysbus.connect(d->service, DND_OBJPATH, DND_INTERFACE, "stateChanged", "si", d->relay.data(), SLOT(stateChanged(QString, int)));
141  sysbus.connect(d->service, DND_OBJPATH, DND_INTERFACE, "serverDestroyed", "s", d->relay.data(), SLOT(serverDestroyed(QString)));
142  } else {
143  d->iface = DFileDragClientPrivate::ifacemap[d->service].toStrongRef();
144  }
145 
146 }

188 void DFileDragClient::setTargetData(const QMimeData *data, QString key, QVariant value)
189 {
190  Q_ASSERT(checkMimeData(data));
191  QString service(data->data(DND_MIME_SERVICE));
192  QString uuid(data->data(DND_MIME_UUID));
193  QDBusInterface iface(service, DND_OBJPATH, DND_INTERFACE, QDBusConnection::systemBus());
194 
195  QDBusReply<uint> pid = QDBusConnection::systemBus().interface()->servicePid(service);
196  if (QString::number(pid).toUtf8() != data->data(DND_MIME_PID)) {
197  return;
198  }
199  iface.call("setData", uuid, key, value.toString());
200 }
