140 void DLabel::paintEvent(QPaintEvent *event)
141 {
142  Q_UNUSED(event)
143  QLabelPrivate *d = static_cast<QLabelPrivate*>(d_ptr.data());
144  QStyle *style = QWidget::style();
145  QPainter painter(this);
146  drawFrame(&painter);
147  QRect cr = contentsRect();
148  cr.adjust(d->margin, d->margin, -d->margin, -d->margin);
149  int align = QStyle::visualAlignment(d->isTextLabel ? DLabelPrivate::textDirection(d)
150  : layoutDirection(), QFlag(d->align));
151 
152 #if QT_CONFIG(movie)
153  if (d->movie && !d->movie->currentPixmap().isNull()) {
154  if (d->scaledcontents)
155  style->drawItemPixmap(&painter, cr, align, d->movie->currentPixmap().scaled(cr.size()));
156  else
157  style->drawItemPixmap(&painter, cr, align, d->movie->currentPixmap());
158  }
159  else
160 #endif
161  if (d->isTextLabel) {
162  QRectF lr = DLabelPrivate::layoutRect(d).toAlignedRect();
163  QStyleOption opt;
164  opt.initFrom(this);
165 
166  if (d->control) {
167 #ifndef QT_NO_SHORTCUT
168  const bool underline = (bool)style->styleHint(QStyle::SH_UnderlineShortcut, 0, this, 0);
169  if (d->shortcutId != 0
170  && underline != d->shortcutCursor.charFormat().fontUnderline()) {
171  QTextCharFormat fmt;
172  fmt.setFontUnderline(underline);
173  d->shortcutCursor.mergeCharFormat(fmt);
174  }
175 #endif
176  DLabelPrivate::ensureTextLayouted(d);
177  QAbstractTextDocumentLayout::PaintContext context;
178  // Adjust the palette
179  context.palette = opt.palette;
180 
181  if (d_func()->color != DPalette::NoType) {
182  context.palette.setBrush(QPalette::Text, DApplicationHelper::instance()->palette(this).brush(d_func()->color));
183  } else if (foregroundRole() != QPalette::Text && isEnabled()) {
184  context.palette.setColor(QPalette::Text, context.palette.color(foregroundRole()));
185  }
186 
187  painter.save();
188  painter.translate(lr.topLeft());
189  painter.setClipRect(lr.translated(-lr.x(), -lr.y()));
190  d->control->setPalette(context.palette);
191  d->control->drawContents(&painter, QRectF(), this);
192  painter.restore();
193  } else {
194  int flags = align | (DLabelPrivate::textDirection(d) == Qt::LeftToRight ? Qt::TextForceLeftToRight
195  : Qt::TextForceRightToLeft);
196  if (d->hasShortcut) {
197  flags |= Qt::TextShowMnemonic;
198  if (!style->styleHint(QStyle::SH_UnderlineShortcut, &opt, this))
199  flags |= Qt::TextHideMnemonic;
200  }
201 
202  QPalette palette = opt.palette;
203 
204  if (d_func()->color != DPalette::NoType) {
205  palette.setBrush(foregroundRole(), DApplicationHelper::instance()->palette(this).brush(d_func()->color));
206  }
207 
208  QString text = d->text;
209  if (elideMode() != Qt::ElideNone) {
210  const QFontMetrics fm(fontMetrics());
211  text = fm.elidedText(text, elideMode(), width(), Qt::TextShowMnemonic);
212  }
213  style->drawItemText(&painter, lr.toRect(), flags, palette, isEnabled(), text, foregroundRole());
214  }
215  } else
216 #ifndef QT_NO_PICTURE
217  if (d->picture) {
218  QRect br = d->picture->boundingRect();
219  int rw = br.width();
220  int rh = br.height();
221  if (d->scaledcontents) {
222  painter.save();
223  painter.translate(cr.x(), cr.y());
224  painter.scale((double)cr.width()/rw, (double)cr.height()/rh);
225  painter.drawPicture(-br.x(), -br.y(), *d->picture);
226  painter.restore();
227  } else {
228  int xo = 0;
229  int yo = 0;
230  if (align & Qt::AlignVCenter)
231  yo = (cr.height()-rh)/2;
232  else if (align & Qt::AlignBottom)
233  yo = cr.height()-rh;
234  if (align & Qt::AlignRight)
235  xo = cr.width()-rw;
236  else if (align & Qt::AlignHCenter)
237  xo = (cr.width()-rw)/2;
238  painter.drawPicture(cr.x()+xo-br.x(), cr.y()+yo-br.y(), *d->picture);
239  }
240  } else
241 #endif
242  if (d->pixmap && !d->pixmap->isNull()) {
243  QPixmap pix;
244  if (d->scaledcontents) {
245  QSize scaledSize = cr.size() * devicePixelRatioF();
246  if (!d->scaledpixmap || d->scaledpixmap->size() != scaledSize) {
247  if (!d->cachedimage)
248  d->cachedimage = new QImage(d->pixmap->toImage());
249  delete d->scaledpixmap;
250  QImage scaledImage =
251  d->cachedimage->scaled(scaledSize,
252  Qt::IgnoreAspectRatio, Qt::SmoothTransformation);
253  d->scaledpixmap = new QPixmap(QPixmap::fromImage(scaledImage));
254  d->scaledpixmap->setDevicePixelRatio(devicePixelRatioF());
255  }
256  pix = *d->scaledpixmap;
257  } else
258  pix = *d->pixmap;
259  QStyleOption opt;
260  opt.initFrom(this);
261  if (!isEnabled())
262  pix = style->generatedIconPixmap(QIcon::Disabled, pix, &opt);
263  style->drawItemPixmap(&painter, cr, align, pix);
264  }
265 }

287 QRectF DLabelPrivate::documentRect(QLabelPrivate *d)
288 {
289  QLabel *q = qobject_cast<QLabel*>(d->q_ptr);
290  Q_ASSERT_X(d->isTextLabel, "documentRect", "document rect called for label that is not a text label!");
291  QRect cr = q->contentsRect();
292  cr.adjust(d->margin, d->margin, -d->margin, -d->margin);
293  const int align = QStyle::visualAlignment(d->isTextLabel ? textDirection(d)
294  : q->layoutDirection(), QFlag(d->align));
295  int m = d->indent;
296  if (m < 0 && q->frameWidth()) // no indent, but we do have a frame
297  m = q->fontMetrics().horizontalAdvance(QLatin1Char('x')) / 2 - d->margin;
298  if (m > 0) {
299  if (align & Qt::AlignLeft)
300  cr.setLeft(cr.left() + m);
301  if (align & Qt::AlignRight)
302  cr.setRight(cr.right() - m);
303  if (align & Qt::AlignTop)
304  cr.setTop(cr.top() + m);
305  if (align & Qt::AlignBottom)
306  cr.setBottom(cr.bottom() - m);
307  }
308  return cr;
309 }

311 QRectF DLabelPrivate::layoutRect(QLabelPrivate *d)
312 {
313  QRectF cr = documentRect(d);
314  if (!d->control)
315  return cr;
316  ensureTextLayouted(d);
317  // Caculate y position manually
318  qreal rh = d->control->document()->documentLayout()->documentSize().height();
319  qreal yo = 0;
320  if (d->align & Qt::AlignVCenter)
321  yo = qMax((cr.height()-rh)/2, qreal(0));
322  else if (d->align & Qt::AlignBottom)
323  yo = qMax(cr.height()-rh, qreal(0));
324  return QRectF(cr.x(), yo + cr.y(), cr.width(), cr.height());
325 }

327 void DLabelPrivate::ensureTextLayouted(QLabelPrivate *d)
328 {
329  if (d->textLayoutDirty) {
330  if (d->textDirty) {
331  if (d->control) {
332  QTextDocument *doc = d->control->document();
333  if (d->textDirty) {
334  #ifndef QT_NO_TEXTHTMLPARSER
335  #if (QT_VERSION >= QT_VERSION_CHECK(5, 14, 0))
336  if (d->textformat == Qt::TextFormat::RichText)
337  #else
338  if (d->isRichText)
339  #endif
340  doc->setHtml(d->text);
341  else
342  doc->setPlainText(d->text);
343  #else
344  doc->setPlainText(d->text);
345  #endif
346  doc->setUndoRedoEnabled(false);
347 
348  #ifndef QT_NO_SHORTCUT
349  if (d->hasShortcut) {
350  // Underline the first character that follows an ampersand (and remove the others ampersands)
351  int from = 0;
352  bool found = false;
353  QTextCursor cursor;
354  while (!(cursor = d->control->document()->find((QLatin1String("&")), from)).isNull()) {
355  cursor.deleteChar(); // remove the ampersand
356  cursor.movePosition(QTextCursor::NextCharacter, QTextCursor::KeepAnchor);
357  from = cursor.position();
358  if (!found && cursor.selectedText() != QLatin1String("&")) { //not a second &
359  found = true;
360  d->shortcutCursor = cursor;
361  }
362  }
363  }
364  #endif
365  }
366  }
367  d->textDirty = false;
368  }
369 
370  if (d->control) {
371  QTextDocument *doc = d->control->document();
372  QTextOption opt = doc->defaultTextOption();
373 
374  opt.setAlignment(QFlag(d->align));
375 
376  if (d->align & Qt::TextWordWrap)
377  opt.setWrapMode(QTextOption::WordWrap);
378  else
379  opt.setWrapMode(QTextOption::ManualWrap);
380 
381  doc->setDefaultTextOption(opt);
382 
383  QTextFrameFormat fmt = doc->rootFrame()->frameFormat();
384  fmt.setMargin(0);
385  doc->rootFrame()->setFrameFormat(fmt);
386  doc->setTextWidth(DLabelPrivate::documentRect(d).width());
387  }
388  d->textLayoutDirty = false;
389  }
390 }
