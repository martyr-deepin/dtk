74  static QSize actionSize(const DViewItemAction *action, const QSize &max, const QSize &fallbackIconSize, int spacing)
75  {
76  if (action->widget()) {
77  return action->widget()->size();
78  }
79 
80  const QString &text = action->text();
81  QSize icon_size = action->iconSize();
82 
83  if (!icon_size.isValid() && !action->icon().isNull()) {
84  icon_size = fallbackIconSize;
85  }
86 
87  QSize max_size;
88 
89  max_size.setWidth(max.width() >= 0 ? max.width() : QWIDGETSIZE_MAX);
90  max_size.setHeight(max.height() >= 0 ? max.height() : QWIDGETSIZE_MAX);
91 
92  if (text.isEmpty()) {
93  return icon_size.boundedTo(max_size);
94  }
95 
96  QFontMetrics fm(action->font());
97  const QSize &text_size = fm.size(0, text);
98 
99  if (!icon_size.isValid())
100  spacing = 0;
101 
102  return QSize(icon_size.width() + text_size.width() + spacing, qMax(icon_size.height(), text_size.height()));
103  }

119  static QList<QRect> doActionsLayout(QRect base, const QList<DViewItemAction*> &list, Qt::Orientation orientation,
120  Qt::LayoutDirection direction, const QSize &fallbackIconSize, QSize *bounding)
121  {
122  if (list.isEmpty()) {
123  if (bounding) {
124  bounding->setWidth(0);
125  bounding->setHeight(0);
126  }
127 
128  return {};
129  }
130 
131  QList<QRect> result_list;
132  int max_width = 0;
133  int max_height = 0;
134  QList<QSize> size_list;
135  size_list.reserve(list.size());
136  int spacing = DStyleHelper(qApp->style()).pixelMetric(DStyle::PM_ContentsSpacing);
137 
138  if (orientation == Qt::Horizontal) {
139  // 记录上中下三列的宽度
140  int top_width = 0;
141  int center_width = 0;
142  int bottom_width = 0;
143 
144  // 先找到最大宽度
145  for (const DViewItemAction *action : list) {
146  size_list.append(actionSize(action, action->maximumSize(), fallbackIconSize, spacing));
147 
148  if (action->alignment().testFlag(Qt::AlignVCenter)) {
149  center_width += size_list.last().width();
150  } else if (action->alignment().testFlag(Qt::AlignBottom)) {
151  bottom_width += size_list.last().width();
152  } else {
153  top_width += size_list.last().width();
154  }
155  }
156 
157  max_width = qBound(qMax(top_width, center_width), bottom_width, base.width()) + (list.size() - 1) * spacing;
158  base.setWidth(max_width);
159 
160  QRect top_rect = base;
161  QRect center_rect = base;
162  QRect bottom_rect = base;
163 
164  // 开始布局
165  for (int i = 0; i < list.size(); ++i) {
166  const DViewItemAction *action = list.at(i);
167 
168  if (action->alignment().testFlag(Qt::AlignVCenter)) {
169  result_list << QStyle::alignedRect(direction, action->alignment(), size_list.at(i), center_rect);
170  center_rect.setLeft(result_list.last().right() + spacing);
171  } else if (action->alignment().testFlag(Qt::AlignBottom)) {
172  result_list << QStyle::alignedRect(direction, action->alignment(), size_list.at(i), bottom_rect);
173  bottom_rect.setLeft(result_list.last().right() + spacing);
174  } else {
175  result_list << QStyle::alignedRect(direction, action->alignment(), size_list.at(i), top_rect);
176  top_rect.setLeft(result_list.last().right() + spacing);
177  }
178 
179  if (bounding)
180  max_height = qMax(max_height, result_list.last().bottom() - base.top() + 1);
181  }
182  } else {
183  // 记录左中右三列的宽度
184  int left_height = 0;
185  int center_height = 0;
186  int right_height = 0;
187 
188  // 先找到最大高度
189  for (const DViewItemAction *action : list) {
190  size_list.append(actionSize(action, action->maximumSize(), fallbackIconSize, spacing));
191 
192  if (action->alignment().testFlag(Qt::AlignHCenter)) {
193  center_height += size_list.last().height();
194  } else if (action->alignment().testFlag(Qt::AlignRight)) {
195  right_height += size_list.last().height();
196  } else {
197  left_height += size_list.last().height();
198  }
199  }
200 
201  max_height = qBound(qMax(left_height, center_height), right_height, base.height()) + (list.size() - 1) * spacing;
202  base.setHeight(max_height);
203 
204  QRect left_rect = base;
205  QRect center_rect = base;
206  QRect right_rect = base;
207 
208  // 开始布局
209  for (int i = 0; i < list.size(); ++i) {
210  const DViewItemAction *action = list.at(i);
211 
212  if (action->alignment().testFlag(Qt::AlignHCenter)) {
213  result_list << QStyle::alignedRect(direction, action->alignment(), size_list.at(i), center_rect);
214  center_rect.setTop(result_list.last().bottom() + spacing);
215  } else if (action->alignment().testFlag(Qt::AlignRight)) {
216  result_list << QStyle::alignedRect(direction, action->alignment(), size_list.at(i), right_rect);
217  right_rect.setTop(result_list.last().bottom() + spacing);
218  } else {
219  result_list << QStyle::alignedRect(direction, action->alignment(), size_list.at(i), left_rect);
220  left_rect.setTop(result_list.last().bottom() + spacing);
221  }
222 
223  if (bounding)
224  max_height = qMax(max_height, result_list.last().bottom() - base.top() + 1);
225  }
226  }
227 
228  if (bounding) {
229  bounding->setWidth(max_width);
230  bounding->setHeight(max_height);
231  }
232 
233  return result_list;
234  }

236  static void drawAction(QPainter *pa, const QStyleOptionViewItem &option, const QRect &rect, const DViewItemAction *action, int spacing)
237  {
238  if (!action->isVisible()) {
239  return;
240  }
241 
242  if (action->widget()) {
243  if (action->widget()->geometry() != rect) {
244  action->widget()->setGeometry(rect);
245  }
246 
247  return;
248  }
249 
250  QSize icon_size(action->iconSize());
251 
252  if (!icon_size.isValid() && !action->icon().isNull()) {
253  icon_size = option.decorationSize;
254  }
255 
256  // 图标的绘制用也可能会使用这些颜色
257  QPalette::ColorGroup cg = option.state & QStyle::State_Enabled
258  ? QPalette::Normal : QPalette::Disabled;
259  if (cg == QPalette::Normal && !(option.state & QStyle::State_Active))
260  cg = QPalette::Inactive;
261 
262  if (option.state & QStyle::State_Selected) {
263  pa->setPen(option.palette.color(cg, QPalette::HighlightedText));
264  } else {
265  if (action->textColorType() > 0) {
266  pa->setPen(QPen(DApplicationHelper::instance()->palette(option.widget).brush(cg, action->textColorType()), 1));
267  } else {
268  QPalette::ColorRole role = action->textColorRole() > 0 ? action->textColorRole() : QPalette::Text;
269  pa->setPen(QPen(option.palette.brush(cg, role), 1));
270  }
271  }
272 
273  // draw icon
274  if (icon_size.isValid()) {
275  const QIcon &icon = action->icon();
276  QRect icon_rect(QPoint(0, 0), icon_size);
277 
278  icon_rect.moveCenter(rect.center());
279  icon_rect.moveLeft(rect.left());
280  icon.paint(pa, icon_rect);
281  }
282 
283  // draw text
284  if (!action->text().isEmpty()) {
285  QRect text_rect = rect;
286 
287  if (!icon_size.isValid()) {
288  spacing = 0;
289  }
290 
291  text_rect.setLeft(text_rect.left() + icon_size.width() + spacing);
292  pa->setFont(action->font());
293  pa->drawText(text_rect, Qt::AlignVCenter, action->text());
294  }
295  }

297  static QSize drawActions(QPainter *pa, const QStyleOptionViewItem &option, const QVariant &value, Qt::Edge edge, QList<QPair<QAction*, QRect>> *clickableActionRect)
298  {
299  const DViewItemActionList &actionList = qvariant_cast<DViewItemActionList>(value);
300  DViewItemActionList visiable_actionList;
301  for (auto action : actionList) {
302  if (action->isVisible()) {
303  visiable_actionList.append(action);
304  if (action->widget())
305  action->widget()->setVisible(true);
306  } else {
307  // action为隐藏状态需要将widget隐藏
308  if (action->widget())
309  action->widget()->setVisible(false);
310  }
311  }
312 
313  const Qt::Orientation orientation = (edge == Qt::TopEdge || edge == Qt::BottomEdge) ? Qt::Vertical : Qt::Horizontal;
314  QSize bounding;
315  const QList<QRect> &list = doActionsLayout(option.rect, visiable_actionList, orientation, option.direction, option.decorationSize, &bounding);
316  QPoint origin(0, 0);
317  int spacing = DStyleHelper(qApp->style()).pixelMetric(DStyle::PM_ContentsSpacing);
318 
319  switch (edge) {
320  case Qt::BottomEdge:
321  origin = QPoint(0, option.rect.height() - bounding.height());
322  break;
323  case Qt::RightEdge:
324  origin = QPoint(option.rect.width() - bounding.width(), 0);
325  break;
326  default:
327  break;
328  }
329 
330  for (int i = 0; i < list.count(); ++i) {
331  DViewItemAction *action = visiable_actionList.at(i);
332  const QRect &rect = list.at(i).translated(origin);
333 
334  drawAction(pa, option, rect, action, spacing);
335 
336  if (action->isClickable()) {
337  clickableActionRect->append(qMakePair(action, rect.marginsAdded(action->clickAreaMargins())));
338  }
339  }
340 
341  return bounding;
342  }

701 void DStyledItemDelegate::paint(QPainter *painter, const QStyleOptionViewItem &option, const QModelIndex &index) const
702 {
703  D_DC(DStyledItemDelegate);
704 
705  QStyleOptionViewItem opt(option);
706  initStyleOption(&opt, index);
707 
708  QRect backup_opt_rect = opt.rect;
709 
710  const QWidget *widget = option.widget;
711  QStyle *style = widget ? widget->style() : QApplication::style();
712  bool editing = false;
713 
714  if (auto view = qobject_cast<const QListView*>(widget)) {
715  editing = !qobject_cast<const QListWidget*>(view) && view->isPersistentEditorOpen(index);
716  }
717 
718  if (!editing)
719  style->drawPrimitive(QStyle::PE_PanelItemViewItem, &opt, painter, widget);
720 
721  // draw the background
722  if (!editing && backgroundType() != NoBackground && !(opt.state & QStyle::State_Selected)
723  && !(d->backgroundType & NoNormalState && DStyle::getState(&opt) == DStyle::SS_NormalState)) {
724  DStyleOptionBackgroundGroup boption;
725  boption.init(widget);
726  boption.QStyleOption::operator =(opt);
727  boption.position = DStyleOptionBackgroundGroup::ItemBackgroundPosition(opt.viewItemPosition);
728 
729  if (opt.backgroundBrush.style() != Qt::NoBrush) {
730  boption.dpalette.setBrush(DPalette::ItemBackground, opt.backgroundBrush);
731  }
732 
733  boption.rect = opt.rect;
734 
735  if (backgroundType() != RoundedBackground) {
736  boption.directions = Qt::Vertical;
737  }
738 
739  style->drawPrimitive(static_cast<QStyle::PrimitiveElement>(DStyle::PE_ItemBackground), &boption, painter, widget);
740  }
741 
742  // 设置内容区域
743  QMargins margins = d->margins;
744  const QVariant &margins_varinat = index.data(Dtk::MarginsRole);
745 
746  if (margins_varinat.isValid()) {
747  margins = qvariant_cast<QMargins>(margins_varinat);
748  }
749 
750  opt.rect = opt.rect.marginsRemoved(margins);
751  QRect itemContentRect = opt.rect;
752  QSize action_area_size(0, 0);
753  QList<QPair<QAction*, QRect>> clickActionList;
754  int spacing = DStyleHelper(qApp->style()).pixelMetric(DStyle::PM_ContentsSpacing);
755 
756  action_area_size = d->drawActions(painter, opt, index.data(Dtk::LeftActionListRole), Qt::LeftEdge, &clickActionList);
757  itemContentRect.setLeft(itemContentRect.left() + action_area_size.width() + (action_area_size.isNull() ? 0 : spacing));
758 
759  action_area_size = d->drawActions(painter, opt, index.data(Dtk::RightActionListRole), Qt::RightEdge, &clickActionList);
760  itemContentRect.setRight(itemContentRect.right() - action_area_size.width() - (action_area_size.isNull() ? 0 : spacing));
761 
762  action_area_size = d->drawActions(painter, opt, index.data(Dtk::TopActionListRole), Qt::TopEdge, &clickActionList);
763  itemContentRect.setTop(itemContentRect.top() + action_area_size.height() + (action_area_size.isNull() ? 0 : spacing));
764 
765  action_area_size = d->drawActions(painter, opt, index.data(Dtk::BottomActionListRole), Qt::BottomEdge, &clickActionList);
766  itemContentRect.setBottom(itemContentRect.bottom() - action_area_size.height() - (action_area_size.isNull() ? 0 : spacing));
767 
768  if (!clickActionList.isEmpty()) {
769  const_cast<DStyledItemDelegatePrivate*>(d)->clickableActionMap[index] = clickActionList;
770  } else {
771  const_cast<DStyledItemDelegatePrivate*>(d)->clickableActionMap.remove(index);
772  }
773 
774  const DViewItemActionList &text_action_list = qvariant_cast<DViewItemActionList>(index.data(Dtk::TextActionListRole));
775 
776  opt.rect = itemContentRect;
777  QRect iconRect, textRect, checkRect;
778  DStyle::viewItemLayout(style, &opt, &iconRect, &textRect, &checkRect, false);
779 
780  // 图标的绘制用也可能会使用这些颜色
781  QPalette::ColorGroup cg = opt.state & QStyle::State_Enabled
782  ? QPalette::Normal : QPalette::Disabled;
783  if (opt.state & QStyle::State_Selected) {
784  painter->setPen(opt.palette.color(cg, QPalette::HighlightedText));
785  } else {
786  painter->setPen(opt.palette.color(cg, QPalette::Text));
787  }
788 
789  // draw icon
790  if (opt.features & QStyleOptionViewItem::HasDecoration) {
791  QIcon::Mode mode = QIcon::Normal;
792  if (!(opt.state & QStyle::State_Enabled))
793  mode = QIcon::Disabled;
794  else if (opt.state & QStyle::State_Selected)
795  mode = QIcon::Selected;
796  QIcon::State state = opt.state & QStyle::State_Open ? QIcon::On : QIcon::Off;
797  opt.icon.paint(painter, iconRect, opt.decorationAlignment, mode, state);
798  }
799 
800  // draw the text
801  if (opt.features & QStyleOptionViewItem::HasDisplay) {
802  if (opt.state & QStyle::State_Editing) {
803  painter->setPen(opt.palette.color(cg, QPalette::Text));
804  painter->drawRect(textRect.adjusted(0, 0, -1, -1));
805  }
806 
807  if (!text_action_list.isEmpty()) {
808  opt.displayAlignment &= ~Qt::AlignBottom;
809  opt.displayAlignment &= ~Qt::AlignVCenter;
810  opt.displayAlignment |= Qt::AlignTop;
811  }
812 
813  if (text_action_list.isEmpty()) {
814  DStyle::viewItemDrawText(style, painter, &opt, textRect);
815  } else {
816  opt.displayAlignment &= ~Qt::AlignVCenter;
817  opt.displayAlignment &= ~Qt::AlignBottom;
818 
819  QRect bounding = DStyle::viewItemDrawText(style, painter, &opt, textRect);
820 
821  // draw action text
822  for (const DViewItemAction *action : text_action_list) {
823  textRect.setTop(bounding.bottom());
824  opt.font = action->font();
