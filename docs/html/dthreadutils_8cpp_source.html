27 FunctionCallProxy::FunctionCallProxy(QThread *thread)
28 {
29  qRegisterMetaType<QPointer<QObject>>();
30 
31  connect(this, &FunctionCallProxy::callInLiveThread, this, [] (QSemaphore *s, QPointer<QObject> target, FunctionType *func) {
32  if (Q_LIKELY(target)) {
33  (*func)();
34  } else {
35  qWarning() << "DThreadUtils::runInThread:" << "The target object is destoryed";
36  }
37 
38  s->release();
39  }, Qt::QueuedConnection);
40  connect(thread, &QThread::finished, this, [this] {
41  qWarning() << "DThreadUtils::runInThread:" << sender() << "the thread finished";
42  }, Qt::DirectConnection);
43 }

45 void FunctionCallProxy::proxyCall(QSemaphore *s, QThread *thread, QObject *target, FunctionType fun)
46 {
47  if (QThread::currentThread() == thread)
48  return fun();
49 
50  FunctionCallProxy proxy(thread);
51  proxy.moveToThread(thread);
52 
53  // 如果线程未开启事件循环，且不是主线程，则需要给出严重警告信息，因为可能会导致死锁
54  if (thread->loopLevel() <= 0 && (!QCoreApplication::instance() || thread != QCoreApplication::instance()->thread())) {
55  qCritical() << Q_FUNC_INFO << thread << ", the thread no event loop";
56  }
57 
58  proxy.callInLiveThread(s, target ? target : &proxy, &fun);
59  s->acquire();
60 }
61 } // end namespace DThreadUtil
