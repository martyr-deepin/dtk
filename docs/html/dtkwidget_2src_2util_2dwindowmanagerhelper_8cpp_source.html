605 QList<DForeignWindow *> DWindowManagerHelper::currentWorkspaceWindows() const
606 {
607  D_DC(DWindowManagerHelper);
608 
609  for (QWindow *w : d->windowList) {
610  w->deleteLater();
611  }
612 
613  d->windowList.clear();
614 
615  QList<WId> currentApplicationWindowList;
616  const QWindowList &list = qApp->allWindows();
617 
618  currentApplicationWindowList.reserve(list.size());
619 
620  for (auto window : list) {
621  if (window->property("_q_foreignWinId").isValid()) continue;
622 
623  currentApplicationWindowList.append(window->winId());
624  }
625 
626  QVector<quint32> wmClientList = currentWorkspaceWindowIdList();
627 
628  for (WId wid : wmClientList) {
629  if (currentApplicationWindowList.contains(wid))
630  continue;
631 
632  if (DForeignWindow *w = DForeignWindow::fromWinId(wid)) {
633  d->windowList << w;
634  }
635  }
636 
637  return d->windowList;
638 }
