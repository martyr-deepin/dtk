73 public Q_SLOTS:
74  void setData(QString uuid, QString key, QString value) {
75  DFileDragServer *srv = DFileDragServerPrivate::servermap.value(uuid);
76  Q_ASSERT(srv);
77 
78  DFileDragServerPrivate *d = srv->d_func();
79 
80  if (d->data.value(key) != value) {
81  d->data[key] = value;
82  Q_EMIT srv->targetDataChanged(key);
83  }
84  }
85  int state(QString uuid) const {return m_statemap.value(uuid);}
86  int progress(QString uuid) const {return m_progressmap.value(uuid);}
87 
88 private:

90  QHash<QString, int> m_progressmap;
