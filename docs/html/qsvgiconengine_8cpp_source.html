79  static inline int hashKey(QIcon::Mode mode, QIcon::State state)

137 void QSvgIconEnginePrivate::loadDataForModeAndState(DSvgRenderer *renderer, QIcon::Mode mode, QIcon::State state)
138 {
139  QByteArray buf;
140  const QIcon::State oppositeState = state == QIcon::Off ? QIcon::On : QIcon::Off;
141  if (svgBuffers) {
142  buf = svgBuffers->value(hashKey(mode, state));
143  if (buf.isEmpty())
144  buf = svgBuffers->value(hashKey(QIcon::Normal, state));
145  if (buf.isEmpty())
146  buf = svgBuffers->value(hashKey(QIcon::Normal, oppositeState));
147  }
148  if (!buf.isEmpty()) {
149 #ifndef QT_NO_COMPRESS
150  buf = qUncompress(buf);
151 #endif
152  renderer->load(buf);
153  } else {
154  QString svgFile = svgFiles.value(hashKey(mode, state));
155  if (svgFile.isEmpty())
156  svgFile = svgFiles.value(hashKey(QIcon::Normal, state));
157  if (svgFile.isEmpty())
158  svgFile = svgFiles.value(hashKey(QIcon::Normal, oppositeState));
159  if (!svgFile.isEmpty())
160  renderer->load(svgFile);
161  }
162 }

186 QPixmap QSvgIconEngine::pixmap(const QSize &size, QIcon::Mode mode,
187  QIcon::State state)
188 {
189  QPixmap pm;
190 
191  QString pmckey(d->pmcKey(size, mode, state));
192  if (Q_LIKELY(QPixmapCache::find(pmckey, &pm)))
193  return pm;
194 
195  if (d->addedPixmaps) {
196  pm = d->addedPixmaps->value(d->hashKey(mode, state));
197  if (!pm.isNull() && pm.size() == size)
198  return pm;
199  }
200 
201  const QIcon::State oppositeState = state == QIcon::Off ? QIcon::On : QIcon::Off;
202  QString svgFile = d->svgFiles.value(d->hashKey(mode, state));
203  if (svgFile.isEmpty())
204  svgFile = d->svgFiles.value(d->hashKey(QIcon::Normal, state));
205  if (svgFile.isEmpty())
206  svgFile = d->svgFiles.value(d->hashKey(QIcon::Normal, oppositeState));
207 
208  QString cacheFile;
209 
210  if (Q_LIKELY(!svgFile.startsWith(":/") && QFile::exists(svgFile))) {
211  static const QString &cachePath = getIconCachePath();
212 
213  if (Q_LIKELY(!cachePath.isEmpty())) {
214  QByteArray mdst;
215  if (d->svgBuffers && !d->svgBuffers->isEmpty())
216  mdst = QByteArray::number(d->hashKey(mode, state)).prepend("_");
217  // svgFiles[-1] 是为了取到额外的cache key,例如会跟随颜色变化的svg图标，将会添加此特殊的标记
218  const QByteArray cacheKey = svgFile.toLocal8Bit() + d->svgFiles[-1].toLocal8Bit() + mdst; // add mode state to cachekey
219  const QString &svgFileSha1 = QString::fromLatin1(QCryptographicHash::hash(cacheKey, QCryptographicHash::Sha1).toHex());
220  cacheFile = QStringLiteral("%1/%2.png").arg(cachePath).arg(svgFileSha1);
221  }
222  }
223 
224  const QFileInfo cacheFileInfo(cacheFile);
225  if (Q_LIKELY(cacheFileInfo.exists())) {
226  const QFileInfo svgFileInfo(svgFile);
227 
228  if (Q_UNLIKELY(svgFileInfo.lastModified() != cacheFileInfo.lastModified())) {
229  // clear invalid cache file
230  QFile::remove(cacheFile);
231  } else {
232  qCDebug(lcDSvg()) << "found cache file:" << cacheFile << ", for:" << svgFile;
233 
234  QImageReader ir(cacheFile);
235  const QSize &cacheSize = ir.size();
236 
237  if (Q_LIKELY(cacheSize.isValid() && cacheSize.width() >= size.width())) {
238  pm = QPixmap::fromImage(ir.read().scaled(size, Qt::KeepAspectRatio, Qt::SmoothTransformation));
239  } else { // 当对同一个图标文件有更大的size要求时，应当继续从源svg文件重新渲染图标并更新缓存文件
240  qCDebug(lcDSvg()) << "cache image size less then target size, cache size:" << cacheSize << ", request size:" << size;
241 
242  if (Q_UNLIKELY(!ir.canRead())) {
243  qCWarning(lcDSvg()) << "can't read the cache image:" << cacheFile << "file size:" << cacheFileInfo.size() << ", permissions:" << cacheFileInfo.permissions();
244  QFile::remove(cacheFile);
245  }
246  }
247  }
248  }
249 
250  if (Q_UNLIKELY(pm.isNull())) {
251  DSvgRenderer renderer;
252  d->loadDataForModeAndState(&renderer, mode, state);
253  if (Q_UNLIKELY(!renderer.isValid()))
254  return pm;
255 
256  QSize actualSize = renderer.defaultSize();
257  if (actualSize.width() < size.width())
258  actualSize.scale(size, Qt::KeepAspectRatio);
259 
260  if (Q_UNLIKELY(actualSize.isEmpty()))
261  return pm;
262 
263  const QImage image = renderer.toImage(actualSize);
264 
265  if (Q_LIKELY(!image.isNull() && !cacheFile.isEmpty())) {
266  QtConcurrent::run(QThreadPool::globalInstance(), [image, cacheFile, svgFile] {
267  QSaveFile file(cacheFile);
268  // 增加cache文件能被成功保存的概率
269  file.setDirectWriteFallback(true);
270  if (file.open(QFile::WriteOnly)) {
271  if (image.save(&file, "png", 80) && file.commit()) {
272  QFileInfo svgFileInfo(svgFile);
273  // 不能直接使用QSaveFile改写文件的时间
274  QFile file(cacheFile);
275  if (!file.open(QFile::ReadWrite) || !file.setFileTime(svgFileInfo.lastModified(), QFileDevice::FileModificationTime)) {
276  qCWarning(lcDSvg()) << "set cache file modified date time failed, error message:" << file.errorString()
277  << ", cache file:" << cacheFile << ", svg file:" << svgFile;
278  }
279  } else {
280  file.cancelWriting();
281  qCWarning(lcDSvg()) << "save cache image failed, cache file:" << cacheFile << ", svg file:" << svgFile;
282  }
283  } else {
284  qCWarning(lcDSvg()) << "open cache file failed, error message:" << file.errorString()
285  << ", cache file:" << cacheFile << ", svg file:" << svgFile;
286  }
287  });
288  }
289 
290  // 缩放图标到目标大小
291  pm = QPixmap::fromImage(image.scaled(size, Qt::KeepAspectRatio, Qt::SmoothTransformation));
292  }
293 
294  if (Q_LIKELY(qobject_cast<QGuiApplication *>(QCoreApplication::instance()))) {
295  const QPixmap generated = QGuiApplicationPrivate::instance()->applyQIconStyleHelper(mode, pm);
296  if (Q_LIKELY(!generated.isNull()))
297  pm = generated;
298  }
299 
300  if (Q_LIKELY(!pm.isNull()))
301  QPixmapCache::insert(pmckey, pm);
302 
303  return pm;
304 }

318 static FileType fileType(const QFileInfo &fi)
319 {
320  const QString &abs = fi.absoluteFilePath();
321  if (abs.endsWith(QLatin1String(".svg"), Qt::CaseInsensitive))
322  return SvgFile;
323  if (abs.endsWith(QLatin1String(".svgz"), Qt::CaseInsensitive)
324  || abs.endsWith(QLatin1String(".svg.gz"), Qt::CaseInsensitive)) {
325  return CompressedSvgFile;
326  }
327 #ifndef QT_NO_MIMETYPE
328  const QString &mimeTypeName = QMimeDatabase().mimeTypeForFile(fi).name();
329  if (mimeTypeName == QLatin1String("image/svg+xml"))
330  return SvgFile;
331  if (mimeTypeName == QLatin1String("image/svg+xml-compressed"))
332  return CompressedSvgFile;
333 #endif // !QT_NO_MIMETYPE
334  return OtherFile;
335 }

337 void QSvgIconEngine::addFile(const QString &fileName, const QSize &,
338  QIcon::Mode mode, QIcon::State state)
339 {
340  if (!fileName.isEmpty()) {
341  const QFileInfo fi(fileName);
342  const QString abs = fi.absoluteFilePath();
343  const FileType type = fileType(fi);
344 #ifndef QT_NO_COMPRESS
345  if (type == SvgFile || type == CompressedSvgFile) {
346 #else
347  if (type == SvgFile) {
348 #endif
349  DSvgRenderer renderer(abs);
350  if (renderer.isValid()) {
351  d->stepSerialNum();
352  d->svgFiles.insert(d->hashKey(mode, state), abs);
353  }
354  } else if (type == OtherFile) {
355  QPixmap pm(abs);
356  if (!pm.isNull())
357  addPixmap(pm, mode, state);
358  }
359  }

382 bool QSvgIconEngine::read(QDataStream &in)
383 {
384  d = new QSvgIconEnginePrivate;
385  d->svgBuffers = new QHash<int, QByteArray>;
386 
387  if (in.version() >= QDataStream::Qt_4_4) {
388  int isCompressed;
389  QHash<int, QString> fileNames; // For memoryoptimization later
390  in >> fileNames >> isCompressed >> *d->svgBuffers;
391 #ifndef QT_NO_COMPRESS
392  if (!isCompressed) {
393  for (auto it = d->svgBuffers->begin(), end = d->svgBuffers->end(); it != end; ++it)
394  it.value() = qCompress(it.value());
395  }
396 #else
397  if (isCompressed) {
398  qWarning("QSvgIconEngine: Can not decompress SVG data");
399  d->svgBuffers->clear();
400  }
401 #endif
402  int hasAddedPixmaps;
403  in >> hasAddedPixmaps;
404  if (hasAddedPixmaps) {
405  d->addedPixmaps = new QHash<int, QPixmap>;
406  in >> *d->addedPixmaps;
407  }
408 
409  d->svgFiles = fileNames;
410  }
411  else {
412  QPixmap pixmap;
413  QByteArray data;
414  uint mode;
415  uint state;
416  int num_entries;
417 
418  in >> data;
419  if (!data.isEmpty()) {
420 #ifndef QT_NO_COMPRESS
421  data = qUncompress(data);
422 #endif
423  if (!data.isEmpty())
424  d->svgBuffers->insert(d->hashKey(QIcon::Normal, QIcon::Off), data);
425  }
426  in >> num_entries;
427  for (int i=0; i<num_entries; ++i) {
428  if (in.atEnd())
429  return false;
430  in >> pixmap;
431  in >> mode;
432  in >> state;
433  // The pm list written by 4.3 is buggy and/or useless, so ignore.
434  //addPixmap(pixmap, QIcon::Mode(mode), QIcon::State(state));
435  }
436  }
437 
438  return true;
439 }

442 bool QSvgIconEngine::write(QDataStream &out) const
443 {
444  if (out.version() >= QDataStream::Qt_4_4) {
445  int isCompressed = 0;
446 #ifndef QT_NO_COMPRESS
447  isCompressed = 1;
448 #endif
449  QHash<int, QByteArray> svgBuffers;
450  if (d->svgBuffers)
451  svgBuffers = *d->svgBuffers;
452  for (auto it = d->svgFiles.cbegin(), end = d->svgFiles.cend(); it != end; ++it) {
453  QByteArray buf;
454  QFile f(it.value());
455  if (f.open(QIODevice::ReadOnly))
456  buf = f.readAll();
457 #ifndef QT_NO_COMPRESS
458  buf = qCompress(buf);
459 #endif
460  svgBuffers.insert(it.key(), buf);
461  }
462  out << d->svgFiles << isCompressed << svgBuffers;
463  if (d->addedPixmaps)
464  out << (int)1 << *d->addedPixmaps;
465  else
466  out << (int)0;
467  }
468  else {
469  QByteArray buf;
470  if (d->svgBuffers)
471  buf = d->svgBuffers->value(d->hashKey(QIcon::Normal, QIcon::Off));
472  if (buf.isEmpty()) {
473  QString svgFile = d->svgFiles.value(d->hashKey(QIcon::Normal, QIcon::Off));
474  if (!svgFile.isEmpty()) {
475  QFile f(svgFile);
476  if (f.open(QIODevice::ReadOnly))
477  buf = f.readAll();
478  }
479  }
480 #ifndef QT_NO_COMPRESS
481  buf = qCompress(buf);
482 #endif
483  out << buf;
484  // 4.3 has buggy handling of added pixmaps, so don't write any
485  out << (int)0;
486  }
487  return true;
488 }
