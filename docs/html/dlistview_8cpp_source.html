489 int DListView::addHeaderWidget(QWidget *widget)
490 {
491  D_D(DListView);
492 
493  int index = d->headerList.indexOf(widget);
494 
495  if (index >= 0)
496  return index;
497 
498  if (!d->headerLayout) {
499  bool isVerticalLayout = (orientation() == Qt::Vertical);
500 
501  d->headerLayout = new DBoxWidget(isVerticalLayout
502  ? QBoxLayout::TopToBottom
503  : QBoxLayout::LeftToRight, this);
504 
505  if (isVerticalLayout)
506  d->headerLayout->resize(width(), d->headerLayout->height());
507  else
508  d->headerLayout->resize(d->headerLayout->width(), height());
509 
510  connect(d->headerLayout, &DBoxWidget::sizeChanged,
511  this, [this](const QSize & size) {
512  bool isVerticalLayout = (orientation() == Qt::Vertical);
513 
514  QMargins margins = this->viewportMargins();
515 
516  if (isVerticalLayout)
517  margins.setTop(size.height());
518  else
519  margins.setLeft(size.width());
520 
521  setViewportMargins(margins);
522  });
523 
524  d->headerLayout->show();
525  }
526 
527  d->headerLayout->addWidget(widget);
528  d->headerList << widget;
529 
530  return d->headerList.count() - 1;
531 }

862 QModelIndex DListView::moveCursor(QAbstractItemView::CursorAction cursorAction, Qt::KeyboardModifiers modifiers)
863 {
864  switch (cursorAction) {
865  case MoveNext: {
866  auto curIdx = currentIndex();
867  auto curCol = modelColumn();
868 
869  auto nextRow = 0;
870  if (curIdx.isValid()) {
871  nextRow = (curIdx.row() + 1) % model()->rowCount();
872  }
873 
874  int hiddenCount = 0;
875  while (isRowHidden(nextRow)) {
876  nextRow = (nextRow + 1) % model()->rowCount();
877 
878  //防止所有列都是隐藏的死循环
879  ++hiddenCount;
880  if (hiddenCount >= count()) {
881  return currentIndex();
882  }
883  }
884 
885  return model()->index(nextRow, curCol);
886  }
887  default:
888  break;
889  }
890 
891  return QListView::moveCursor(cursorAction, modifiers);
892 }
