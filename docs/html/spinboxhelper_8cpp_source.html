129 QRect Style::spinboxSubControlRect(const QStyleOptionComplex *opt, QStyle::SubControl sc, const QWidget *widget) const
130 {
131  Q_UNUSED(widget)
132 
133  QRect ret;
134 
135  if (const QStyleOptionSpinBox *spinbox = qstyleoption_cast<const QStyleOptionSpinBox *>(opt)) {
136  QSize bs;
137  int fw = 0;
138  bs.setHeight(spinbox->rect.height() - 2 * fw);
139  // 1.6 -approximate golden mean
140  bs.setWidth(bs.height());
141  bs = bs.expandedTo(QApplication::globalStrut());
142  int y = fw;
143  int x, lx, rx;
144  x = spinbox->rect.right() - fw - bs.width();
145  lx = fw;
146  rx = x - fw - bs.width();
147  switch (sc) {
148  case SC_SpinBoxUp:
149  if (spinbox->buttonSymbols == QAbstractSpinBox::NoButtons)
150  return QRect();
151  ret = QRect(x, y, bs.width(), bs.height());
152  break;
153  case SC_SpinBoxDown:
154  if (spinbox->buttonSymbols == QAbstractSpinBox::NoButtons)
155  return QRect();
156 
157  ret = QRect(x - bs.width(), y, bs.width(), bs.height());
158  break;
159  case SC_SpinBoxEditField:
160  if (spinbox->buttonSymbols == QAbstractSpinBox::NoButtons) {
161  ret = QRect(lx, fw, spinbox->rect.width() - 2*fw, spinbox->rect.height() - 2*fw);
162  } else {
163  ret = QRect(lx, fw, rx, spinbox->rect.height() - 2*fw);
164  }
165  break;
166  case SC_SpinBoxFrame:
167  ret = spinbox->rect;
168  default:
169  break;
170  }
171  ret = visualRect(spinbox->direction, spinbox->rect, ret);
172  }
173 
174  return ret;
175 }
