121 const QBrush &DPalette::brush(QPalette::ColorGroup cg, DPalette::ColorType cr) const
122 {
123  if (cr >= NColorTypes) {
124  return QPalette::brush(cg, QPalette::NoRole);
125  }
126 
127  if (cg == Current) {
128  cg = currentColorGroup();
129  } else if (cg >= NColorGroups) {
130  cg = Active;
131  }
132 
133  return d->data->br[cg][cr];
