393 void DTabBarPrivate::updateMoveingTabPosition(const QPoint &mouse)
394 {
395  QTabBarPrivate *d = reinterpret_cast<QTabBarPrivate *>(qGetPtrHelper(d_ptr));
396 
397  bool vertical = verticalTabs(d->shape);
398  int dragDistance;
399  if (vertical) {
400  dragDistance = (mouse.y() - d->dragStartPosition.y());
401  } else {
402  dragDistance = (mouse.x() - d->dragStartPosition.x());
403  }
404 
405  if (!(d->pressedIndex == 0 && dragDistance < 0)
406  && !(d->pressedIndex == d->tabList.count() - 1 && dragDistance > 0)) {
407  d->tabList[d->pressedIndex].dragOffset = dragDistance;
408  } else {
409  d->tabList[d->pressedIndex].dragOffset = 0;
410  }
411 
412  QRect startingRect = tabRect(d->pressedIndex);
413  if (vertical)
414  startingRect.moveTop(startingRect.y() + dragDistance);
415  else
416  startingRect.moveLeft(startingRect.x() + dragDistance);
417 
418  int overIndex;
419  if (dragDistance < 0)
420  overIndex = tabAt(startingRect.topLeft());
421  else
422  overIndex = tabAt(startingRect.topRight());
423 
424  if (overIndex != d->pressedIndex && overIndex != -1) {
425  int offset = 1;
426  if (isRightToLeft() && !vertical)
427  offset *= -1;
428  if (dragDistance < 0) {
429  dragDistance *= -1;
430  offset *= -1;
431  }
432  for (int i = d->pressedIndex;
433  offset > 0 ? i < overIndex : i > overIndex;
434  i += offset) {
435  QRect overIndexRect = tabRect(overIndex);
436  int needsToBeOver = (vertical ? overIndexRect.height() : overIndexRect.width()) / 2;
437  if (dragDistance > needsToBeOver) {
438  slide(i + offset, d->pressedIndex);
439  }
440  }
441  }
442 
443  // Buttons needs to follow the dragged tab
444  layoutTab(d->pressedIndex);
445  update();
446 }

491 void DTabBarPrivate::slide(int from, int to)
492 {
493  QTabBarPrivate *d = reinterpret_cast<QTabBarPrivate *>(qGetPtrHelper(d_ptr));
494 
495  if (from == to
496  || !d->validIndex(from)
497  || !d->validIndex(to))
498  return;
499  bool vertical = verticalTabs(d->shape);
500  int preLocation = vertical ? tabRect(from).y() : tabRect(from).x();
501  setUpdatesEnabled(false);
502  moveTab(from, to);
503  setUpdatesEnabled(true);
504  int postLocation = vertical ? tabRect(to).y() : tabRect(to).x();
505  int length = postLocation - preLocation;
506 
507  d->tabList[to].dragOffset -= length;
508 
509  if (!d->isAnimated()) {
510  moveTabFinished(to);
511  return;
512  }
513 
514  QTabBarPrivate::Tab *tab = &d->tabList[to];
515 
516  if (!tab->animation)
517  tab->animation = reinterpret_cast<QTabBarPrivate::Tab::TabBarAnimation*>(new TabBarAnimation(tab, d, this));
518  tab->animation->setStartValue(tab->dragOffset);
519  tab->animation->setEndValue(0);
520  tab->animation->setEasingCurve(QEasingCurve::OutCubic);
521  tab->animation->setDuration(ANIMATION_DURATION);
522  tab->animation->start();
523 }

561 void DTabBarPrivate::moveTabFinished(int index)
562 {
563  QTabBarPrivate *d = reinterpret_cast<QTabBarPrivate *>(qGetPtrHelper(d_ptr));
564  bool cleanup = (d->pressedIndex == index) || (d->pressedIndex == -1) || !d->validIndex(index);
565  bool allAnimationsFinished = true;
566 #ifndef QT_NO_ANIMATION
567  for(int i = 0; allAnimationsFinished && i < d->tabList.count(); ++i) {
568  const QTabBarPrivate::Tab &t = d->tabList.at(i);
569  if (t.animation && t.animation->state() == QAbstractAnimation::Running)
570  allAnimationsFinished = false;
571  }
572 #endif //QT_NO_ANIMATION
573  if (allAnimationsFinished && cleanup) {
574  if(d->movingTab)
575  d->movingTab->setVisible(false); // We might not get a mouse release
576  for (int i = 0; i < d->tabList.count(); ++i) {
577  d->tabList[i].dragOffset = 0;
578  }
579  if (d->pressedIndex != -1 && d->movable) {
580  d->pressedIndex = -1;
581  d->dragInProgress = false;
582  d->dragStartPosition = QPoint();
583  }
584  layoutWidgets();
585  } else {
586  if (!d->validIndex(index))
587  return;
588  d->tabList[index].dragOffset = 0;
589  }
590  update();
591 }

600 void DTabBarPrivate::makeVisible(int index)
601 {
602  QTabBarPrivate *d = reinterpret_cast<QTabBarPrivate *>(qGetPtrHelper(d_ptr));
603 
604  if (!d->validIndex(index))
605  return;
606 
607  const QRect tabRect = d->tabList.at(index).rect;
608  const int oldScrollOffset = d->scrollOffset;
609  const bool horiz = !verticalTabs(d->shape);
610  const int extra_width = 2 * qMax(style()->pixelMetric(QStyle::PM_TabBarScrollButtonWidth, 0, this),
611  QApplication::globalStrut().width());
612  const int available = (horiz ? width() : height()) - extra_width;
613  const int start = horiz ? tabRect.left() : tabRect.top();
614  const int end = horiz ? tabRect.right() : tabRect.bottom();
615  if (start < d->scrollOffset) // too far left
616  d->scrollOffset = start - (index ? 8 : 0);
617  else if (end > d->scrollOffset + available) // too far right
618  d->scrollOffset = end - available + 1;
619 
620  d->leftB->setEnabled(d->scrollOffset > 0);
621  const int last = horiz ? d->tabList.last().rect.right() : d->tabList.last().rect.bottom();
622  d->rightB->setEnabled(last - d->scrollOffset >= available);
623  if (oldScrollOffset != d->scrollOffset) {
624  layoutWidgets();
625  }

628 void DTabBarPrivate::autoScrollTabs(const QPoint &mouse)
629 {
630  const QRect &rect = this->rect();
631 
632  if (!rect.contains(mouse)) {
633  return stopAutoScrollTabs();
634  }
635 
636  QTabBarPrivate *d = reinterpret_cast<QTabBarPrivate *>(qGetPtrHelper(d_ptr));
637  bool vertical = verticalTabs(d->shape);
638  const QSize &size_hint = this->size();
639  int scroll_distance = qMin(50, vertical ? rect.height() / 3 : rect.width() / 3);
640 
641  int start = 0;
642  int end = 0;
643  int scroll_end = 0;
644  int scroll_speed = 0;
645 
646  if (vertical) {
647  end = size_hint.height() - rect.height();
648 
649  if (mouse.y() < scroll_distance) {
650  if (d->scrollOffset <= start)
651  return;
652 
653  // to up
654  scroll_end = start;
655  scroll_speed = mouse.y();
656  } else if (mouse.y() > rect.height() - scroll_distance) {
657  if (d->scrollOffset >= end)
658  return;
659
