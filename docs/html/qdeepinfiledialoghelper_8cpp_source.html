105 bool QDeepinFileDialogHelper::show(Qt::WindowFlags flags, Qt::WindowModality modality, QWindow *parent)
106 {
107  ensureDialog();
108  applyOptions();
109 
110  if (nativeDialog) {
111  if (parent)
112  activeWindow = parent;
113  else
114  activeWindow = QGuiApplication::focusWindow();
115 
116  nativeDialog->setParent(parent);
117 
118  Qt::WindowFlags nd_flags = static_cast<Qt::WindowFlags>(nativeDialog->windowFlags());
119  Qt::WindowFlags need_flags = Qt::WindowTitleHint | Qt::WindowSystemMenuHint
120  | Qt::WindowMinMaxButtonsHint | Qt::WindowContextHelpButtonHint
121  | Qt::WindowStaysOnTopHint | Qt::WindowTransparentForInput
122  | Qt::WindowDoesNotAcceptFocus | Qt::WindowStaysOnBottomHint
123  | Qt::WindowCloseButtonHint | Qt::BypassWindowManagerHint;
124 
125  if (flags & need_flags)
126  nativeDialog->setWindowFlags(nd_flags | (flags & need_flags));
127 
128  static bool i_am_dbus_server = iAmFileDialogDBusServer();
129 
130  if (i_am_dbus_server) {
131  WId native_dialog_winId = nativeDialog->winId();
132  QWindow *real_native_dialog = nullptr;
133 
134  for (QWindow *window : qGuiApp->topLevelWindows()) {
135  if (window->winId() == native_dialog_winId) {
136  real_native_dialog = window;
137  break;
138  }
139  }
140 
141  if (real_native_dialog) {
142  real_native_dialog->setTransientParent(parent);
143  real_native_dialog->setModality(modality);
144  // call later
145  QMetaObject::invokeMethod(nativeDialog.data(), "show", Qt::QueuedConnection);
146 
147  return true;
148  }
149  }
150 
151  auxiliaryWindow->setParent(parent);
152  auxiliaryWindow->setFlags(flags);
153  auxiliaryWindow->setModality(modality);
154 
155  if (modality != Qt::NonModal) {
156  QGuiApplicationPrivate::showModalWindow(auxiliaryWindow);
157 
158  if (modality == Qt::ApplicationModal) {
159  connect(qApp, &QGuiApplication::applicationStateChanged,
160  this, [this] (Qt::ApplicationState state) {
161  if (state == Qt::ApplicationActive)
162  nativeDialog->activateWindow();
163  });
164  connect(nativeDialog, &DFileDialogHandle::windowActiveChanged, this, [this] {
165  if (qApp->platformName() != "dxcb" && !qApp->property("_d_isDxcb").toBool())
166  return;
167 
168  QWindow *focus_window = qApp->focusWindow();
169 
170  if (!focus_window)
171  return;
172 
173  if (focus_window->type() != Qt::Widget
174  && focus_window->type() != Qt::Window
175  && focus_window->type() != Qt::Dialog) {
176  return;
177  }
178 
179  if (!nativeDialog->windowActive() && qApp->applicationState() == Qt::ApplicationActive) {
180  nativeDialog->activateWindow();
181  }
182  });
183  }
184  }
185  } else {
186  qtDialog->setAttribute(Qt::WA_NativeWindow);
187 
188  if (parent) {
189  if (parent->inherits("QWidgetWindow")) {
190  qtDialog->setParent(static_cast<QWidgetWindow*>(parent)->widget());
191  } else {
192  qtDialog->windowHandle()->setParent(parent);
193  }
194  }
195 
196  qtDialog->setWindowModality(modality);
197  qtDialog->setWindowFlags(flags | qtDialog->windowFlags());
198  }
199 
200  DIALOG_CALL(show());
201 
202  if (nativeDialog && parent) {
203  // 如果能获取到wayland下的display 应该 XSetTransientForHint(wayland_dispaly
204  // 此处暂时返回，至少不会导致崩溃。。。task-view-31919.
205  if (qApp->platformName() != "dxcb" && !qApp->property("_d_isDxcb").toBool())
206  return true;
207 
208  XSetTransientForHint(QX11Info::display(), nativeDialog->winId(), parent->winId());
209  }
210 
211  return true;
212 }

357 void QDeepinFileDialogHelper::ensureDialog() const
358 {
359  if (nativeDialog || qtDialog)
360  return;
361 
362  if (manager) {
363  QDBusPendingReply<QDBusObjectPath> reply = manager->createDialog(QString());
364  reply.waitForFinished();
365  const QString &path = reply.value().path();
366 
367  if (path.isEmpty()) {
368  qWarning("Can not create native dialog, Will be use QFileDialog");
369  } else {
370  nativeDialog = new DFileDialogHandle(DIALOG_SERVICE, path, QDBusConnection::sessionBus());
371  auxiliaryWindow = new QWindow();
372 
373  connect(nativeDialog, &QObject::destroyed, auxiliaryWindow, &QWindow::deleteLater);
374  connect(nativeDialog, &QObject::destroyed, nativeDialog, &DFileDialogHandle::deleteLater);
375  connect(nativeDialog, &DFileDialogHandle::destroyed, nativeDialog, &QObject::deleteLater);
376  connect(nativeDialog, &DFileDialogHandle::accepted, this, &QDeepinFileDialogHelper::accept);
377  connect(nativeDialog, &DFileDialogHandle::rejected, this, &QDeepinFileDialogHelper::reject);
378  connect(nativeDialog, &DFileDialogHandle::destroyed, this, &QDeepinFileDialogHelper::reject);
379 
380  QTimer *heartbeatTimer = new QTimer(nativeDialog);
381 
382  connect(heartbeatTimer, &QTimer::timeout, this, [this, heartbeatTimer] {
383  QDBusPendingReply<> reply = nativeDialog->makeHeartbeat();
384 
385  reply.waitForFinished();
386 
387  if (reply.isError()) {
388  qWarning() << "Make heartbeat is failed:" << reply.error();
389 
390  if (reply.error().type() == QDBusError::UnknownMethod) {
391  qWarning() << "Make heartbeat is't support for current dbus file dialog, Will be stop heartbeat timer.";
392 
393  heartbeatTimer->stop();
394  return;
395  }
396 
397  nativeDialog->QObject::deleteLater();
398  const_cast<QDeepinFileDialogHelper*>(this)->reject();
399  }
400  });
401  int heartbeatInterval = nativeDialog->heartbeatInterval();
402  heartbeatTimer->setInterval(qMax(1 * 1000, qMin(int(heartbeatInterval / 1.5), heartbeatInterval - 5 * 1000)));
403  heartbeatTimer->start();
404  }
405  }
406 
407  if (!nativeDialog && qobject_cast<QApplication*>(qGuiApp)) {
408  QDeepinTheme::m_usePlatformNativeDialog = false;
409  qtDialog = new QFileDialog();
410  QDeepinTheme::m_usePlatformNativeDialog = true;
411 
412  connect(qtDialog, &QFileDialog::accepted, this, &QDeepinFileDialogHelper::accept);
413  connect(qtDialog, &QFileDialog::rejected, this, &QDeepinFileDialogHelper::reject);
414  }
415 }

417 void QDeepinFileDialogHelper::applyOptions()
418 {
419  QFileDialogOptions *options = this->options().data();
420 
421  for (int i = 0; i < QFileDialogOptions::DialogLabelCount; ++i) {
422  if (options->isLabelExplicitlySet((QFileDialogOptions::DialogLabel)i)) {
423  if (nativeDialog)
424  nativeDialog->setLabelText(i, options->labelText((QFileDialogOptions::DialogLabel)i));
425  else
426  qtDialog->setLabelText((QFileDialog::DialogLabel)i, options->labelText((QFileDialogOptions::DialogLabel)i));
427  }
428  }
429 
430  DIALOG_CALL(setOptions((QFileDialog::Options)(int)options->options()));
431  DIALOG_CALL(setFilter(options->filter()));
432  DIALOG_CALL(setWindowTitle(options->windowTitle()));
433  DIALOG_CALL(setViewMode((QFileDialog::ViewMode)options->viewMode()));
434  DIALOG_CALL(setFileMode((QFileDialog::FileMode)options->fileMode()));
435  DIALOG_CALL(setAcceptMode((QFileDialog::AcceptMode)options->acceptMode()));
436  DIALOG_CALL(setNameFilters(options->nameFilters()));
437 
438  setDirectory(options->initialDirectory());
439 
440  foreach (const QUrl &filename, options->initiallySelectedFiles())
441  selectFile(filename);
442 
443  selectNameFilter(options->initiallySelectedNameFilter());
444 
445  if (!sourceDialog) {
446  sourceDialog = reinterpret_cast<QFileDialog*>(qvariant_cast<quintptr>(property("_dtk_widget_QFileDialog")));
447 
448  if (sourceDialog && nativeDialog) {
449  const QStringList lineedit_list = sourceDialog->property("_dtk_widget_custom_lineedit_list").toStringList();
450  const QStringList combobox_list = sourceDialog->property("_dtk_widget_custom_combobox_list").toStringList();
451 
452  nativeDialog->beginAddCustomWidget();
453 
454  for (const QString &i : lineedit_list)
455  nativeDialog->addCustomWidget(LineEditType, i);
456 
457  for (const QString &i : combobox_list)
458  nativeDialog->addCustomWidget(ComboBoxType, i);
459 
460  nativeDialog->endAddCustomWidget();
461 
462  const QVariant & mixedSelection = sourceDialog->property("_dtk_widget_filedialog_mixed_selection");
463  if (mixedSelection.isValid()) {
464  nativeDialog->setAllowMixedSelection(mixedSelection.toBool());
465  }
466  }
467  }
