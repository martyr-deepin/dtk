340 void DAbstractDialog::mouseMoveEvent(QMouseEvent *event)
341 {
342  D_D(DAbstractDialog);
343 
344  QWidget *compoment = childAt(event->pos());
345  if (qobject_cast<QComboBox *>(compoment)) {
346  return;
347  }
348

350  d->handle->setEnableSystemMove(true);
351 
352  return QDialog::mouseMoveEvent(event);

382 void DAbstractDialog::showEvent(QShowEvent *event)
383 {
384  D_DC(DAbstractDialog);
385 
386  // 不要在resizeEvent中改变窗口geometry，可能会导致以下问题。在双屏开启不同缩放比时：
387  // 屏幕A缩放为1，屏幕B缩放为2。窗口初始在屏幕B上，此时大小为 100x100，则其缩放
388  // 后的真实大小为200x200，由于窗口管理器会对未设置过位置的窗口自动布局，接下来
389  // 窗口可能会因为位置改变而被移动到屏幕A上，则此对话框的大小将变化为 200x200，
390  // 触发了resize事件，紧接着又会尝试将窗口位置更新到鼠标所在屏幕的中心，如果鼠标
391  // 所在屏幕为B，则在移动窗口位置时又会根据B的缩放比重新设置窗口大小，窗口将变为：
392  // 400x400，将形成一个循环。。。
393  if (!d->mouseMoved) {
394  setDisplayPosition(displayPosition());
395  }
396 
397  QDialog::showEvent(event);
398 }
