192 void DWaterProgressPrivate::resizePixmap(QSize sz)
193 {
194  // resize water;
195  auto waterWidth = 500 * sz.width() / 100;
196  auto waterHeight = 110 * sz.height() / 100;
197  auto waterSize = QSizeF(waterWidth, waterHeight).toSize();
198 
199  if (waterFrontImage.size() != waterSize) {
200  QIcon renderer = QIcon::fromTheme("water_front");
201  QImage image(waterWidth, waterHeight, QImage::Format_ARGB32);
202  image.fill(Qt::transparent); // partly transparent red-ish background
203  QPainter waterPainter(&image);
204  renderer.paint(&waterPainter, image.rect());
205  waterFrontImage = image;
206  }
207  if (waterBackImage.size() != waterSize) {
208  QIcon renderer = QIcon::fromTheme("water_back");
209  QImage image(waterWidth, waterHeight, QImage::Format_ARGB32);
210  image.fill(Qt::transparent); // partly transparent red-ish background
211  QPainter waterPainter(&image);
212  renderer.paint(&waterPainter, image.rect());
213  waterBackImage = image;
214  }

217 void DWaterProgressPrivate::initUI()
218 {
219  D_Q(DWaterProgress);
220  q->setMinimumSize(100, 100);
221 
222  value = 0;
223 
224  timer = new QTimer(q);
225  timer->setInterval(interval);
226  resizePixmap(q->size());
227  frontXOffset = q->width();
228  backXOffset = 0;
229 
230  q->connect(timer, &QTimer::timeout, q, [ = ] {
231  // interval can not be zero, and limit to 1
232  interval = (interval < 1) ? 1 : interval;
233 
234  // move 60% per second
235  double frontXDeta = 40.0 / (1000.0 / interval);
236  // move 90% per second
237  double backXDeta = 60.0 / (1000.0 / interval);
238 
239  int canvasWidth = static_cast<int>(q->width() * q->devicePixelRatioF());
240  frontXOffset -= frontXDeta *canvasWidth / 100;
241  backXOffset += backXDeta *canvasWidth / 100;
242 
243  if (frontXOffset > canvasWidth)
244  {
245  frontXOffset = canvasWidth;
246  }
247  if (frontXOffset < - (waterFrontImage.width() - canvasWidth))
248  {
249  frontXOffset = canvasWidth;
250  }
251 
252  if (backXOffset > waterBackImage.width())
253  {
254  backXOffset = 0;
255  }
256 
257  // update pop
258  // move 25% per second default
259  double speed = 25 / (1000.0 / interval) ;
260  for (auto &pop : pops)
261  {
262  // yOffset 0 ~ 100;
263  pop.yOffset += speed * pop.ySpeed;
264  if (pop.yOffset < 0) {
265  }
266  if (pop.yOffset > value) {
267  pop.yOffset = 0;
268  }
269  pop.xOffset = qSin((pop.yOffset / 100) * 2 * 3.14) * 18 * pop.xSpeed + 50;
270  }
271  q->update();
272  });
273 }

281 void DWaterProgressPrivate::paint(QPainter *p)
282 {
283  D_Q(DWaterProgress);
284  p->setRenderHint(QPainter::Antialiasing);
285 
286  qreal pixelRatio = q->devicePixelRatioF();
287  QRectF rect = QRectF(0, 0, q->width() * pixelRatio, q->height() * pixelRatio);
288  QSize sz = QSizeF(q->width() * pixelRatio, q->height() * pixelRatio).toSize();
289 
290  resizePixmap(sz);
291 
292  int yOffset = rect.toRect().topLeft().y() + (100 - value - 10) * sz.height() / 100;
293 
294  // draw water
295  QImage waterImage = QImage(sz, QImage::Format_ARGB32_Premultiplied);
296  QPainter waterPinter(&waterImage);
297  waterPinter.setRenderHint(QPainter::Antialiasing);
298  waterPinter.setCompositionMode(QPainter::CompositionMode_Source);
299 
300  QPointF pointStart(sz.width() / 2, 0);
301  QPointF pointEnd(sz.width() / 2, sz.height());
302  QLinearGradient linear(pointStart, pointEnd);
303  QColor startColor("#1F08FF");
304  startColor.setAlphaF(1);
305  QColor endColor("#50FFF7");
306  endColor.setAlphaF(0.28);
307  linear.setColorAt(0, startColor);
308  linear.setColorAt(1, endColor);
309  linear.setSpread(QGradient::PadSpread);
310  waterPinter.setPen(Qt::NoPen);
311  waterPinter.setBrush(linear);
312  waterPinter.drawEllipse(waterImage.rect().center(), sz.width() / 2 + 1, sz.height() / 2 + 1);
313 
314  waterPinter.setCompositionMode(QPainter::CompositionMode_SourceOver);
315  waterPinter.drawImage(static_cast<int>(backXOffset), yOffset, waterBackImage);
316  waterPinter.drawImage(static_cast<int>(backXOffset) - waterBackImage.width(), yOffset, waterBackImage);
317  waterPinter.drawImage(static_cast<int>(frontXOffset), yOffset, waterFrontImage);
318  waterPinter.drawImage(static_cast<int>(frontXOffset) - waterFrontImage.width(), yOffset, waterFrontImage);
319 
320  //drwa pop
321  if (value > 30) {
322  for (auto &pop : pops) {
323  QPainterPath popPath;
324  popPath.addEllipse(pop.xOffset * sz.width() / 100, (100 - pop.yOffset) * sz.height() / 100,
325  pop.size * sz.width() / 100, pop.size * sz.height() / 100);
326  QColor color(255, 255, 255, 255 * 0.3);
327  waterPinter.fillPath(popPath, color);
328  }
329  }
330 
331  if (textVisible) {
332  auto font = waterPinter.font();
333 
334  QRect rectValue;
335  if (progressText == "100") {
336  font.setPixelSize(sz.height() * 35 / 100);
337  waterPinter.setFont(font);
338 
339  rectValue.setWidth(sz.width() * 60 / 100);
340  rectValue.setHeight(sz.height() * 35 / 100);
341  rectValue.moveCenter(rect.center().toPoint());
342  waterPinter.setPen(Qt::white);
343  waterPinter.drawText(rectValue, Qt::AlignCenter, progressText);
344 
345  } else {
346  font.setPixelSize(sz.height() * 40 / 100);
347  waterPinter.setFont(font);
348 
349  rectValue.setWidth(sz.width() * 45 / 100);
350  rectValue.setHeight(sz.height() * 40 / 100);
351  rectValue.moveCenter(rect.center().toPoint());
352  rectValue.moveLeft(rect.left() + rect.width() * 0.45 * 0.5);
353 
354  waterPinter.setPen(Qt::white);
355  waterPinter.drawText(rectValue, Qt::AlignCenter, progressText);
356  font.setPixelSize(font.pixelSize() / 2);
357  waterPinter.setFont(font);
358  QRect rectPerent(QPoint(rectValue.right(), rectValue.bottom() - rect.height() * 20 / 100),
359  QPoint(rectValue.right() + rect.width() * 20 / 100, rectValue.bottom()));
360 
361  waterPinter.drawText(rectPerent, Qt::AlignCenter, "%");
362  }
363  }
364  waterPinter.end();
365 
366  QPixmap maskPixmap(sz);
367  maskPixmap.fill(Qt::transparent);
368  QPainterPath path;
369  path.addEllipse(QRectF(0, 0, sz.width(), sz.height()));
370  QPainter maskPainter(&maskPixmap);
371  maskPainter.setRenderHint(QPainter::Antialiasing);
372  maskPainter.setPen(QPen(Qt::white, 1));
373  maskPainter.fillPath(path, QBrush(Qt::white));
374 
375  QPainter::CompositionMode mode = QPainter::CompositionMode_SourceIn;
376  QImage contentImage = QImage(sz, QImage::Format_ARGB32_Premultiplied);
377  QPainter contentPainter(&contentImage);
378  contentPainter.setCompositionMode(QPainter::CompositionMode_Source);
379  contentPainter.fillRect(contentImage.rect(), Qt::transparent);
380  contentPainter.setCompositionMode(QPainter::CompositionMode_SourceOver);
381  contentPainter.drawImage(0, 0, maskPixmap.toImage());
382  contentPainter.setCompositionMode(mode);
383  contentPainter.drawImage(0, 0, waterImage);
384  contentPainter.setCompositionMode(QPainter::CompositionMode_DestinationOver);
385  contentPainter.end();
386 
387  contentImage.setDevicePixelRatio(pixelRatio);
388  p->drawImage(q->rect(), contentImage);
389 }
