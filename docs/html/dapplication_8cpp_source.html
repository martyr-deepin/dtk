90  q->restoreOverrideCursor();

104  QTimer::singleShot(5 * 1000, q, [id, cancelNotification](){

107  });

138 static bool tryAcquireSystemSemaphore(QSystemSemaphore *ss, qint64 timeout = 10)
139 {
140  if (ss->error() != QSystemSemaphore::NoError) {
141  return false;
142  }
143 
144  QSystemSemaphore _tmp_ss(QString("%1-%2").arg("DTK::tryAcquireSystemSemaphore").arg(ss->key()), 1, QSystemSemaphore::Open);
145 
146  _tmp_ss.acquire();
147 
148  QElapsedTimer t;
149  QFuture<bool> request = QtConcurrent::run(ss, &QSystemSemaphore::acquire);
150 
151  t.start();
152 
153  while (Q_LIKELY(t.elapsed() < timeout && !request.isFinished()));
154 
155  if (request.isFinished()) {
156  return true;
157  }
158 
159  if (Q_LIKELY(request.isRunning())) {
160  if (Q_LIKELY(ss->release(1))) {
161  request.waitForFinished();
162  }
163  }
164 
165  return false;
166 }

168 bool DApplicationPrivate::setSingleInstanceBySemaphore(const QString &key)
169 {
170  static QSystemSemaphore ss(key, 1, QSystemSemaphore::Open);
171  static bool singleInstance = false;
172 
173  if (singleInstance) {
174  return true;
175  }
176 
177  Q_ASSERT_X(ss.error() == QSystemSemaphore::NoError, "DApplicationPrivate::setSingleInstanceBySemaphore:", ss.errorString().toLocal8Bit().constData());
178 
179  singleInstance = tryAcquireSystemSemaphore(&ss);
180 
181  if (singleInstance) {
182  QtConcurrent::run([this] {
183  QPointer<DApplication> that = q_func();
184 
185  while (ss.acquire() && singleInstance)
186  {
187  if (!that) {
188  return;
189  }
190 
191  if (that->startingUp() || that->closingDown()) {
192  break;
193  }
194 
195  ss.release(1);
196 
197  if (that) {
198  Q_EMIT that->newInstanceStarted();
199  }
200  }
201  });
202 
203  auto clean_fun = [] {
204  ss.release(1);
205  singleInstance = false;
206  };
207 
208  qAddPostRoutine(clean_fun);
209  atexit(clean_fun);
210  }
211 
212  return singleInstance;
213 }

271 bool DApplicationPrivate::loadTranslator(QList<DPathBuf> translateDirs, const QString &name, QList<QLocale> localeFallback)
272 {
273  D_Q(DApplication);
274 
275  QStringList missingQmfiles;
276  for (auto &locale : localeFallback) {
277  QString translateFilename = QString("%1_%2").arg(name).arg(locale.name());
278  for (auto &path : translateDirs) {
279  QString translatePath = (path / translateFilename).toString();
280  if (QFile::exists(translatePath + ".qm")) {
281  qDebug() << "load translate" << translatePath;
282  auto translator = new QTranslator(q);
283  translator->load(translatePath);
284  q->installTranslator(translator);
285  return true;
286  }
287  }
288 
289  // fix english does not need to translation..
290  if (locale.language() != QLocale::English) {
291  missingQmfiles << translateFilename + ".qm";
292  }
293 
294  QStringList parseLocalNameList = locale.name().split("_", QString::SkipEmptyParts);
295  if (parseLocalNameList.length() > 0) {
296  translateFilename = QString("%1_%2").arg(name)
297  .arg(parseLocalNameList.at(0));
298  for (auto &path : translateDirs) {
299  QString translatePath = (path / translateFilename).toString();
300  if (QFile::exists(translatePath + ".qm")) {
301  qDebug() << "translatePath after feedback:" << translatePath;
302  auto translator = new QTranslator(q);
303  translator->load(translatePath);
304  q->installTranslator(translator);
305  return true;
306  }
307  }
308  }
309 
310  // fix english does not need to translation..
311  if (locale.language() != QLocale::English) {
312  missingQmfiles << translateFilename + ".qm";
313  }
314  }
315 
316  if (missingQmfiles.size() > 0) {
317  qWarning() << name << "can not find qm files" << missingQmfiles;
318  }
319  return false;
320 }

322 // 自动激活DMainWindow类型的窗口
323 void DApplicationPrivate::_q_onNewInstanceStarted()
324 {
325  if (!autoActivateWindows)
326  return;
327 
328  for (QWidget *window : qApp->topLevelWidgets()) {
329  if (qobject_cast<DMainWindow*>(window)) {
330  // 如果窗口最小化或者隐藏了，應當先將其show出來
331  if (window->isMinimized() || window->isHidden())
332  window->showNormal();
333 
334  window->activateWindow();
335  break; // 只激活找到的第一个窗口
336  }
337  }

434 #else

446  if (gsettings.keys().contains("qpixmapCacheLimit"))
447  QPixmapCache::setCacheLimit(gsettings.get("qpixmap-cache-limit").toInt());
448  }
449 
450  // set QTapAndHoldGesture::timeout
451  if (QGSettings::isSchemaInstalled("com.deepin.dde.touchscreen")) {
452  QGSettings gsettings("com.deepin.dde.touchscreen");
453  if (gsettings.keys().contains("longpressDuration"))
454  // NOTE(hualet): -100 is a workaround against the situation that that sometimes longpress
455  // and release on Dock cause App launches which should be avoided.
456  //
457  // I guess it happens like this: longpress happens on Dock,
458  // Dock menu shows(doesn't grab the mouse which is a bug can't be fixed easily),
459  // user ends the longpress, DDE Dock recevies mouseReleaseEvent and checks for
460  // QTapAndHoldGesture which is still not happening (maybe because the timer used is
461  // a CoarseTimer?), so Dock treats the event as a normal mouseReleaseEvent, launches the
462  // App or triggers the action.
463  //
464  // see: https://github.com/linuxdeepin/internal-discussion/issues/430
465  //
466  // This workaround hopefully can fix most of this situations.
467  QTapAndHoldGesture::setTimeout(gsettings.get("longpress-duration").toInt() - 100);
468  }
469 #endif
470 }
