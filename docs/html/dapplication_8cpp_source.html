141 static bool tryAcquireSystemSemaphore(QSystemSemaphore *ss, qint64 timeout = 10)
142 {
143  if (ss->error() != QSystemSemaphore::NoError) {
144  return false;
145  }
146 
147  QSystemSemaphore _tmp_ss(QString("%1-%2").arg("DTK::tryAcquireSystemSemaphore").arg(ss->key()), 1, QSystemSemaphore::Open);
148 
149  _tmp_ss.acquire();
150 
151  QElapsedTimer t;
152  QFuture<bool> request = QtConcurrent::run(ss, &QSystemSemaphore::acquire);
153 
154  t.start();
155 
156  while (Q_LIKELY(t.elapsed() < timeout && !request.isFinished()));
157 
158  if (request.isFinished()) {
159  return true;
160  }
161 
162  if (Q_LIKELY(request.isRunning())) {
163  if (Q_LIKELY(ss->release(1))) {
164  request.waitForFinished();
165  }
166  }
167 
168  return false;
169 }

171 bool DApplicationPrivate::setSingleInstanceBySemaphore(const QString &key)
172 {
173  static QSystemSemaphore ss(key, 1, QSystemSemaphore::Open);
174  static bool singleInstance = false;
175 
176  if (singleInstance) {
177  return true;
178  }
179 
180  Q_ASSERT_X(ss.error() == QSystemSemaphore::NoError, "DApplicationPrivate::setSingleInstanceBySemaphore:", ss.errorString().toLocal8Bit().constData());
181 
182  singleInstance = tryAcquireSystemSemaphore(&ss);
183 
184  if (singleInstance) {
185  QtConcurrent::run([this] {
186  QPointer<DApplication> that = q_func();
187 
188  while (ss.acquire() && singleInstance)
189  {
190  if (!that) {
191  return;
192  }
193 
194  if (that->startingUp() || that->closingDown()) {
195  break;
196  }
197 
198  ss.release(1);
199 
200  if (that) {
201  Q_EMIT that->newInstanceStarted();
202  }
203  }
204  });
205 
206  auto clean_fun = [] {
207  ss.release(1);
208  singleInstance = false;
209  };
210 
211  qAddPostRoutine(clean_fun);
212  atexit(clean_fun);
213  }
214 
215  return singleInstance;
216 }

274 bool DApplicationPrivate::loadTranslator(QList<DPathBuf> translateDirs, const QString &name, QList<QLocale> localeFallback)
275 {
276  D_Q(DApplication);
277 
278  QStringList missingQmfiles;
279  for (auto &locale : localeFallback) {
280  QString translateFilename = QString("%1_%2").arg(name).arg(locale.name());
281  for (auto &path : translateDirs) {
282  QString translatePath = (path / translateFilename).toString();
283  if (QFile::exists(translatePath + ".qm")) {
284  qDebug() << "load translate" << translatePath;
285  auto translator = new QTranslator(q);
286  translator->load(translatePath);
287  q->installTranslator(translator);
288  return true;
289  }
290  }
291 
292  // fix english does not need to translation..
293  if (locale.language() != QLocale::English) {
294  missingQmfiles << translateFilename + ".qm";
295  }
296 
297  QStringList parseLocalNameList = locale.name().split("_", QString::SkipEmptyParts);
298  if (parseLocalNameList.length() > 0) {
299  translateFilename = QString("%1_%2").arg(name)
300  .arg(parseLocalNameList.at(0));
301  for (auto &path : translateDirs) {
302  QString translatePath = (path / translateFilename).toString();
303  if (QFile::exists(translatePath + ".qm")) {
304  qDebug() << "translatePath after feedback:" << translatePath;
305  auto translator = new QTranslator(q);
306  translator->load(translatePath);
307  q->installTranslator(translator);
308  return true;
309  }
310  }
311  }
312 
313  // fix english does not need to translation..
314  if (locale.language() != QLocale::English) {
315  missingQmfiles << translateFilename + ".qm";
316  }
317  }
318 
319  if (missingQmfiles.size() > 0) {
320  qWarning() << name << "can not find qm files" << missingQmfiles;
321  }
322  return false;
323 }

343 void DApplicationPrivate::doAcclimatizeVirtualKeyboard(QWidget *window, QWidget *widget, bool allowResizeContentsMargins)
344 {
345  // 如果新激活的输入窗口跟已经在处理中的窗口不一致，则恢复旧窗口的状态
346  if (activeInputWindow && activeInputWindow != window) {
347  activeInputWindow->setContentsMargins(activeInputWindowContentsMargins);
348  activeInputWindow = nullptr;
349  }
350 
351  auto platform_context = QGuiApplicationPrivate::platformIntegration()->inputContext();
352  auto input_method = QGuiApplication::inputMethod();
353  // 先检查输入面板的状态
354  if (!platform_context->inputMethodAccepted() || !input_method->isVisible()) {
355  if (activeInputWindow) {
356  activeInputWindow->setContentsMargins(activeInputWindowContentsMargins);
357  activeInputWindow = nullptr;
358  }
359 
360  return;
361  }
362 
363  // 如果输入控件的主窗口处于未激活状态则忽略
364  if (!window->isActiveWindow())
365  return;
366 
367  // 虚拟键盘相对于当前窗口的geometry
368  const QRectF &kRect = input_method->keyboardRectangle().translated(-window->mapToGlobal(QPoint(0, 0)));
369  if (kRect.isEmpty() || !kRect.isValid())
370  return;
371 
372  // 记录待处理窗口的数据
373  if (!activeInputWindow) {
374  activeInputWindow = window;
375  activeInputWindowContentsMargins = window->contentsMargins();
376  lastContentsMargins = qMakePair(0, 0);
377  }
378 
379  // 可以被压缩的高度
380  int resizeableHeight = lastContentsMargins.first;
381  // 将要平移的距离
382  int panValue = 0;
383 
384  const QRectF &cRect = input_method->cursorRectangle();
385  const QRectF &iRect = input_method->inputItemClipRectangle();
386  const QRectF &wRect = window->rect().marginsRemoved(activeInputWindowContentsMargins);
387 
388  if (allowResizeContentsMargins) {
389  // 判断输入控件是否处于一个可滚动区域中
390  QWidget *scrollWidget = widget;
391  while (scrollWidget && !qobject_cast<QAbstractScrollArea*>(scrollWidget)) {
392  scrollWidget = scrollWidget->parentWidget();
393  }
394 
395  if (QAbstractScrollArea *scrollArea = qobject_cast<QAbstractScrollArea*>(scrollWidget)) {
396  resizeableHeight = scrollArea->maximumViewportSize().height()
397  // 至少要保证可滚动区域的最小高度，以及光标的可显示区域
398  - qMax(scrollArea->minimumHeight(), qRound(cRect.height()));
399  resizeableHeight = qMax(0, resizeableHeight);
400  }
401  }
402 
403  // 优先保证输入框下面的内容都能正常显示
404  int shadowHeight = wRect.bottom() - kRect.top();
405  if (shadowHeight > cRect.y()) {
406  // 如果窗口底部的内容必须要被遮挡，则优先保证能完整显示整个输入区域，且最低限度是要显示输入光标
407  shadowHeight = qMin(cRect.y(), iRect.bottom() - kRect.top());
408  }
409  // 如果输入区域没有被盖住则忽略
410  if (shadowHeight <= 0)
411  return;
412 
413  // 更新需要平移的区域
414  int resizeHeight = qMin(resizeableHeight, shadowHeight);
415  panValue = shadowHeight - resizeHeight;
416 
417  if (lastContentsMargins.first == resizeableHeight
418  && lastContentsMargins.second == panValue) {
419  return;
420  }
421 
422  // 记录本次的计算结果
423  lastContentsMargins.first = resizeableHeight;
424  lastContentsMargins.second = panValue;
425 
426  // 更新窗口内容显示区域以确保虚拟键盘能正常显示
427  window->setContentsMargins(0, -panValue, 0, resizeHeight + panValue);
428 }

550 DApplication *DApplication::globalApplication(int &argc, char **argv)
551 {
552  if (instance()) {
553  auto dp = static_cast<QCoreApplicationPrivate*>(qApp->QCoreApplication::d_ptr.data());
554  Q_ASSERT(dp);
555 
556  // 清理QGuiApplication库的命令行参数
557  int j = argc ? 1 : 0;
558  const QByteArrayList qt_command = {
559  "-platformpluginpath",
560  "-platform",
561  "-platformtheme",
562  "-qwindowgeometry",
563  "-geometry",
564  "-qwindowtitle",
565  "-title",
566  "-qwindowicon",
567  "-icon",
568  "-plugin",
569  "-reverse",
570  "-session",
571  "-style"
572  };
573  for (int i = 1; i < argc; i++) {
574  if (!argv[i])
575  continue;
576  if (*argv[i] != '-') {
577  argv[j++] = argv[i];
578  continue;
579  }
580  const char *arg = argv[i];
581  if (arg[1] == '-') // startsWith("--")
582  ++arg;
583  if (qt_command.indexOf(arg) >= 0) {
584  // 跳过这个参数的值
585  ++i;
586  } else if (strcmp(arg, "-testability") != 0
587  && strncmp(arg, "-style=", 7) != 0) {
588  argv[j++] = argv[i];
589  }
590  }
591 
592  if (j < argc) {
593  argv[j] = nullptr;
594  argc = j;
595  }
596 
597  dp->argc = argc;
598  dp->argv = argv;
599  dp->processCommandLineArguments();
600  static_cast<QApplicationPrivate*>(dp)->process_cmdline();
601  return qApp;
602  }
603 
604  return new DApplication(argc, argv);
605 }
