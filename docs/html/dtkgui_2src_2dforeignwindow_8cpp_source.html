181 {
182  if (e->type() == QEvent::DynamicPropertyChange) {
183  QDynamicPropertyChangeEvent *event = static_cast<QDynamicPropertyChangeEvent*>(e);
184 
185  if (event->propertyName() == _WmClass) {
186  Q_EMIT wmClassChanged();

188  return true;
189  } else if (event->propertyName() == _ProcessId) {
190  Q_EMIT pidChanged();
191

194  }
195 
196  return false;
