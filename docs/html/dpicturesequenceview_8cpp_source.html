66 QPixmap DPictureSequenceViewPrivate::loadPixmap(const QString &path)
67 {
68  D_Q(DPictureSequenceView);
69 
70  qreal ratio = 1.0;
71 
72  const qreal devicePixelRatio = q->devicePixelRatioF();
73 
74  QPixmap pixmap;
75 
76  if (!qFuzzyCompare(ratio, devicePixelRatio)) {
77  QImageReader reader;
78  reader.setFileName(qt_findAtNxFile(path, devicePixelRatio, &ratio));
79  if (reader.canRead()) {
80  reader.setScaledSize(reader.size() * (devicePixelRatio / ratio));
81  pixmap = QPixmap::fromImage(reader.read());
82  pixmap.setDevicePixelRatio(devicePixelRatio);
83  }
84  } else {
85  pixmap.load(path);
86  }
87 
88  return pixmap;
89 }

91 void DPictureSequenceViewPrivate::_q_refreshPicture()
92 {
93  QGraphicsPixmapItem *item = pictureItemList.value(lastItemPos++);
94 
95  if (item)
96  item->hide();
97 
98  if (lastItemPos == pictureItemList.count()) {
99  lastItemPos = 0;
100 
101  if (singleShot)
102  refreshTimer->stop();
103 
104  D_QC(DPictureSequenceView);
105 
106  Q_EMIT q->playEnd();
107  }
108 
109  item = pictureItemList.value(lastItemPos);
110 
111  if (item)
112  item->show();
113 }

210 void DPictureSequenceView::setPictureSequence(const QList<QPixmap> &sequence, const bool autoScale)
211 {
212  D_D(DPictureSequenceView);
213 
214  stop();
215  d->scene->clear();
216  d->pictureItemList.clear();
217 
218  for (QPixmap pixmap : sequence) {
219  if (autoScale) {
220  pixmap = pixmap.scaled(size(), Qt::KeepAspectRatio, Qt::SmoothTransformation);
221  }
222 
223  d->pictureItemList << d->scene->addPixmap(pixmap);
224  d->pictureItemList.last()->hide();
225  }
226 
227  if (!d->pictureItemList.isEmpty()) {
228  d->pictureItemList.first()->show();
229  }
230 
231  setStyleSheet("background-color:transparent;");
232 }
