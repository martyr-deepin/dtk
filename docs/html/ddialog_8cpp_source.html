184 QMap<int, QString> DDialogPrivate::scanTags(QString origin) const
185 {
186  QMap<int, QString> result;
187 
188  QRegExp re("<.*?>");
189  int index = origin.indexOf(re, 0);
190  int matchLength = 0;
191  while (index >= 0) {
192  result[index] = re.cap();
193  matchLength = re.matchedLength();
194  index = origin.indexOf(re, index + matchLength);
195  }
196 
197  return result;
198 }

200 // It's not a perfect solution to elide rich text, but it should be sufficient
201 // for working with rich text only change text token color.
202 QString DDialogPrivate::elideString(QString str, const QFontMetrics &fm, int width) const
203 {
204  QString trimmed = trimTag(str);
205  if (fm.width(trimmed) > width) {
206  QMap<int, QString> info = scanTags(str);
207  QString elided = fm.elidedText(trimmed, Qt::ElideMiddle, width);
208  int elideStart = elided.indexOf("…");
209  int elideLength = trimmed.length() - elided.length();
210  QList<int> indexes = info.keys();
211  std::sort(indexes.begin(), indexes.end());
212  for (int index : indexes) {
213  if (index <= elideStart) {
214  elided.insert(index, info[index]);
215  } else if (elideStart < index && index <= elideStart + elideLength) {
216  elided.insert(elideStart, info[index]);
217  elideStart += 1;
218  } else {
219  elided.insert(index - elideLength + 1, info[index]);
220  }
221  }
222  return elided;
223  } else {
224  return str;
225  }
226 }

537 void DDialog::insertButton(int index, QAbstractButton *button, bool isDefault)
538 {
539  D_D(DDialog);
540 
541  QLabel* label = new QLabel;
542  label->setObjectName("VLine");
543  label->setFixedWidth(1);
544  label->hide();
545 
546  if (index > 0 && index >= buttonCount()) {
547  QLabel *label = qobject_cast<QLabel*>(d->buttonLayout->itemAt(d->buttonLayout->count() - 1)->widget());
548  if(label)
549  label->show();
550  }
551 
552  d->buttonLayout->insertWidget(index * 2, button);
553  d->buttonList << button;
554  d->buttonLayout->insertWidget(index * 2 + 1, label);
555 
556  connect(button, SIGNAL(clicked(bool)), this, SLOT(_q_onButtonClicked()));
557 
558  if (isDefault) {
559  setDefaultButton(button);
560  }
561 
562  const QString &text = button->text();
563 
564  if (text.count() == 2) {
565  for (const QChar &ch : text) {
566  switch (ch.script()) {
567  case QChar::Script_Han:
568  case QChar::Script_Katakana:
569  case QChar::Script_Hiragana:
570  case QChar::Script_Hangul:
571  break;
572  default:
573  return;
574  }
575  }
576 
577  button->setText(QString().append(text.at(0)).append(QChar::Nbsp).append(text.at(1)));
578  }
579 }
