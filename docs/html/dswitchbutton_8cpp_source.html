172 void DSwitchButton::paintEvent(QPaintEvent *e)
173 {
174  Q_UNUSED(e);
175 
176  D_D(DSwitchButton);
177 
178  QColor frontground = Qt::white;
179  QColor background = d->enabledBackground;
180 
181  if (d->checked) {
182  background = d->checkedBackground;
183  }
184 
185  if (!isEnabled()) {
186  background.setAlpha(255 * 55 / 100);
187  frontground.setAlpha(255 * 55 / 100);
188  }
189 
190  QPainter p(this);
191  p.setRenderHints(QPainter::Antialiasing);
192 
193  QPainterPath path;
194  path.addRoundedRect(rect(), height() / 2.0, height() / 2.0);
195  path.closeSubpath();
196 
197  p.setClipPath(path);
198 
199  double indicatorX = 0;
200 
201  if (d->animation->state() == QVariantAnimation::Stopped) {
202  if (!d->checked) {
203  indicatorX = d->animationStartValue;
204  } else {
205  indicatorX = d->animationEndValue;
206  }
207  } else {
208  indicatorX = d->animation->currentValue().toDouble();
209  }
210 
211  int moveWidth = width() - height();
212  int border = 1 * 1;
213  int btSize = height() - border * 2;
214  QRectF btRect(moveWidth * indicatorX + border, border, btSize, btSize);
215  QPainterPath btPath;
216  btPath.addRoundedRect(btRect, btSize / 2.0, btSize / 2.0);
217  btPath.closeSubpath();
218 
219  p.fillPath(path, background);
220  p.fillPath(btPath, frontground);
221 }
