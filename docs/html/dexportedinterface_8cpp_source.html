122 QString DExportedInterfaceDBusInterface::help(const QString &action)
123 {
124  if (action.length()) {
125  return p->actionHelp(action, 0).join('\n');
126  } else {
127  QString ret = "Available actions:";
128  QStringList actions = p->actions.keys();
129  actions.sort();
130  for (auto action : actions) {
131  ret += QString("\n\n") + p->actionHelp(action, 1).join('\n');
132  }
133  return ret;
134  }
135 }

137 QDBusVariant DExportedInterfaceDBusInterface::invoke(QString action, QString parameters)
138 {
139  QDBusVariant ret;
140  if (!p->actions.contains(action)) {
141  sendErrorReply(QDBusError::ErrorType::InvalidArgs, QString("Action \"%1\" is not registered").arg(action));
142  } else {
143  ret.setVariant(p->q_func()->invoke(action, parameters));
144  }
145  return ret;
146 }
