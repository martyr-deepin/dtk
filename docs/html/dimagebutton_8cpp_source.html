729 void DImageButtonPrivate::updateIcon()
730 {
731  D_Q(DImageButton);
732 
733  switch (m_state) {
734  case DImageButton::Hover:
735  q->setProperty("state", "hover");
736  if (!m_hoverPic.isEmpty()) { q->setPixmap(loadPixmap(m_hoverPic)); }
737  break;
738  case DImageButton::Press:
739  q->setProperty("state", "press");
740  if (!m_pressPic.isEmpty()) { q->setPixmap(loadPixmap(m_pressPic)); }
741  break;
742  case DImageButton::Checked:
743  q->setProperty("state", "checked");
744  if (!m_checkedPic.isEmpty()) { q->setPixmap(loadPixmap(m_checkedPic)); }
745  break;
746  case DImageButton::Disabled:
747  q->setProperty("state", "disabled");
748  if (!m_disabledPic.isEmpty()) { q->setPixmap(loadPixmap(m_disabledPic)); }
749  break;
750  default:
751  q->setProperty("state", "");
752  if (!m_normalPic.isEmpty()) { q->setPixmap(loadPixmap(m_normalPic)); }
753  break;
754  }
755 
756  q->setAlignment(Qt::AlignCenter);
757 
758  Q_EMIT q->stateChanged();
759 }

772 QPixmap DImageButtonPrivate::loadPixmap(const QString &path)
773 {
774  D_Q(DImageButton);
775 
776  qreal ratio = 1.0;
777 
778  const qreal devicePixelRatio = q->devicePixelRatioF();
779 
780  QPixmap pixmap;
781 
782  if (!qFuzzyCompare(ratio, devicePixelRatio)) {
783  QImageReader reader;
784  reader.setFileName(qt_findAtNxFile(path, devicePixelRatio, &ratio));
785  if (reader.canRead()) {
786  reader.setScaledSize(reader.size() * (devicePixelRatio / ratio));
787  pixmap = QPixmap::fromImage(reader.read());
788  pixmap.setDevicePixelRatio(devicePixelRatio);
789  }
790  } else {
791  pixmap.load(path);
792  }
793 
794  return pixmap;
795 }
