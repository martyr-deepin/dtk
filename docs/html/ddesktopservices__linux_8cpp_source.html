105 static bool systemSoundEffectEnabled(const QString &name)
106 {
107  QGSettings settings("com.deepin.dde.sound-effect");
108 
109  const bool effEnabled = settings.get("enabled").toBool();
110  QString newName = GSettingsKeyToCamelCase(name);
111 
112  if (effEnabled) {
113  const QStringList list = settings.keys();
114  if (!list.contains(newName)) {
115  return false;
116  }
117 
118  return settings.get(newName).toBool();
119  }
120 
121  return effEnabled;
122 }
