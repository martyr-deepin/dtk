64 void DMainWindowPrivate::init()
65 {
66  D_Q(DMainWindow);
67 
68  // 默认开启标题栏阴影
69  q->setTitlebarShadowEnabled(true);
70 
71  DApplication *dapp = qobject_cast<DApplication *>(qApp);
72  if (dapp) {
73  q->setWindowTitle(dapp->productName());
74  q->setAttribute(Qt::WA_LayoutOnEntireRect, false);
75  q->setAttribute(Qt::WA_ContentsMarginsRespectsSafeArea, false);
76  dapp->acclimatizeVirtualKeyboard(q);
77  } else {
78  q->setWindowTitle(qApp->applicationDisplayName());
79  }
80 
81  if (handle) {
82  q->connect(handle, &DPlatformWindowHandle::borderColorChanged, q, &DMainWindow::borderColorChanged);
83  q->connect(handle, &DPlatformWindowHandle::borderWidthChanged, q, &DMainWindow::borderWidthChanged);
84  q->connect(handle, &DPlatformWindowHandle::clipPathChanged, q, &DMainWindow::clipPathChanged);
85  q->connect(handle, &DPlatformWindowHandle::frameMarginsChanged, q, &DMainWindow::frameMarginsChanged);
86  q->connect(handle, &DPlatformWindowHandle::frameMaskChanged, q, &DMainWindow::frameMaskChanged);
87  q->connect(handle, &DPlatformWindowHandle::shadowColorChanged, q, &DMainWindow::shadowColorChanged);
88  q->connect(handle, &DPlatformWindowHandle::shadowOffsetChanged, q, &DMainWindow::shadowOffsetChanged);
89  q->connect(handle, &DPlatformWindowHandle::shadowRadiusChanged, q, &DMainWindow::shadowRadiusChanged);
90  q->connect(handle, &DPlatformWindowHandle::windowRadiusChanged, q, &DMainWindow::windowRadiusChanged);
91  q->connect(handle, &DPlatformWindowHandle::translucentBackgroundChanged, q, &DMainWindow::translucentBackgroundChanged);
92  q->connect(handle, &DPlatformWindowHandle::enableSystemMoveChanged, q, &DMainWindow::enableSystemMoveChanged);
93  q->connect(handle, &DPlatformWindowHandle::enableSystemResizeChanged, q, &DMainWindow::enableSystemResizeChanged);
94  q->connect(handle, &DPlatformWindowHandle::enableBlurWindowChanged, q, &DMainWindow::enableBlurWindowChanged);
95  q->connect(handle, &DPlatformWindowHandle::autoInputMaskByClipPathChanged, q, &DMainWindow::autoInputMaskByClipPathChanged);
96 
97  if (!handle->isEnabledNoTitlebar(q->windowHandle())) {
98  q->connect(qApp, &QGuiApplication::focusWindowChanged, q, [q] {
99  if (q->isActiveWindow())
100  {
101  q->setShadowColor(SHADOW_COLOR_ACTIVE);
102  } else
103  {
104  q->setShadowColor(SHADOW_COLOR_NORMAL);
105  }
106  });
107  }
108  }
109 
110  if (!help && DApplicationPrivate::isUserManualExists()) {
111  help = new QShortcut(QKeySequence(Qt::Key_F1), q);
112  QObject::connect(help, &QShortcut::activated,
113  q, [ = ]() {
114  DApplication *dapp = qobject_cast<DApplication *>(qApp);
115  if (dapp) {
116  dapp->handleHelpAction();
117  }
118  });
119  }
120 }

122 void DMainWindowPrivate::updateTitleShadowGeometry()
123 {
124  D_QC(DMainWindow);
125 
126  if (!titleShadow)
127  return;
128 
129  QRect rect(0, titlebar->geometry().bottom() + 1, q->width(), titleShadow->sizeHint().height());
130  titleShadow->setGeometry(rect);
131  // 全凭时会隐藏窗口标题栏，因此不应该显示标题栏的阴影
132  titleShadow->setVisible(!q->isFullScreen());
133  titleShadow->raise();
134 }

646 void DMainWindow::setTitlebarShadowEnabled(bool titlebarShadowEnabled)
647 {
