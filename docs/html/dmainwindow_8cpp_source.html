41 DMainWindowPrivate::DMainWindowPrivate(DMainWindow *qq)
42  : DObjectPrivate(qq)
43 {
44  titlebar = new DTitlebar(qq);
45  if (DApplication::isDXcbPlatform()) {

58 void DMainWindowPrivate::init()
59 {
60  D_Q(DMainWindow);
61 
62  const DApplication *dapp = qobject_cast<DApplication *>(qApp);
63  if (dapp) {
64  q->setWindowTitle(dapp->productName());
65  } else {
66  q->setWindowTitle(qApp->applicationDisplayName());
67  }
68 
69  if (handle) {
70  q->connect(handle, &DPlatformWindowHandle::borderColorChanged, q, &DMainWindow::borderColorChanged);
71  q->connect(handle, &DPlatformWindowHandle::borderWidthChanged, q, &DMainWindow::borderWidthChanged);
72  q->connect(handle, &DPlatformWindowHandle::clipPathChanged, q, &DMainWindow::clipPathChanged);
73  q->connect(handle, &DPlatformWindowHandle::frameMarginsChanged, q, &DMainWindow::frameMarginsChanged);
74  q->connect(handle, &DPlatformWindowHandle::frameMaskChanged, q, &DMainWindow::frameMaskChanged);
75  q->connect(handle, &DPlatformWindowHandle::shadowColorChanged, q, &DMainWindow::shadowColorChanged);
76  q->connect(handle, &DPlatformWindowHandle::shadowOffsetChanged, q, &DMainWindow::shadowOffsetChanged);
77  q->connect(handle, &DPlatformWindowHandle::shadowRadiusChanged, q, &DMainWindow::shadowRadiusChanged);
78  q->connect(handle, &DPlatformWindowHandle::windowRadiusChanged, q, &DMainWindow::windowRadiusChanged);
79  q->connect(handle, &DPlatformWindowHandle::translucentBackgroundChanged, q, &DMainWindow::translucentBackgroundChanged);
80  q->connect(handle, &DPlatformWindowHandle::enableSystemMoveChanged, q, &DMainWindow::enableSystemMoveChanged);
81  q->connect(handle, &DPlatformWindowHandle::enableSystemResizeChanged, q, &DMainWindow::enableSystemResizeChanged);
82  q->connect(handle, &DPlatformWindowHandle::enableBlurWindowChanged, q, &DMainWindow::enableBlurWindowChanged);
83  q->connect(handle, &DPlatformWindowHandle::autoInputMaskByClipPathChanged, q, &DMainWindow::autoInputMaskByClipPathChanged);
84 
85  if (!handle->isEnableNoTitlebar(q->windowHandle())) {
86  q->connect(qApp, &QGuiApplication::focusWindowChanged, q, [q] {
87  if (q->isActiveWindow())
88  {
89  q->setShadowColor(SHADOW_COLOR_ACTIVE);
90  } else
91  {
92  q->setShadowColor(SHADOW_COLOR_NORMAL);
93  }
94  });
95  }
96  }
97 
98  if (!help && DApplicationPrivate::isUserManualExists()) {
99  help = new QShortcut(QKeySequence(Qt::Key_F1), q);
100  help->setContext(Qt::ApplicationShortcut);
101  QObject::connect(help, &QShortcut::activated,
102  q, [ = ]() {
103  DApplication *dapp = qobject_cast<DApplication *>(qApp);
104  if (dapp) {
105  dapp->handleHelpAction();
106  }
107  });
108  }
109 
110 
111 }
