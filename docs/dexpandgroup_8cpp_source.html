150 void DExpandGroup::onExpandChanged(bool v)
151 {
152  DBaseExpand *dSender = qobject_cast<DBaseExpand *>(sender());
153  if (!dSender || dSender == checkedExpand() || !v)
154  return;
155 
156  m_checkedMap.clear();
157  int checkedIndex = m_expandMap.values().indexOf(dSender);
158  m_checkedMap.insert(m_expandMap.keys().at(checkedIndex), dSender);
159 
160  QList<DBaseExpand *> expandList = m_expandMap.values();
161  for (int i = 0; i < expandList.count(); i ++){
162  DBaseExpand *target = expandList.at(i);
163  if (target != dSender && target->expand())
164  target->setExpand(false);
165  }
166 }
