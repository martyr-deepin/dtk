31 bool Style::drawProgressBarControl(const QStyleOption *option, QPainter *painter, const QWidget *widget) const
32 {
33  const QStyleOptionProgressBar* progressBarOption( qstyleoption_cast<const QStyleOptionProgressBar*>( option ) );
34  if( !progressBarOption ) return false;
35 
36  // render groove
37  QStyleOptionProgressBar progressBarOption2 = *progressBarOption;
38  progressBarOption2.rect = subElementRect( SE_ProgressBarGroove, progressBarOption, widget );
39  drawControl( CE_ProgressBarGroove, &progressBarOption2, painter, widget );
40 
41  // render contents
42  progressBarOption2.rect = subElementRect( SE_ProgressBarContents, progressBarOption, widget );
43  drawControl( CE_ProgressBarContents, &progressBarOption2, painter, widget );
44 
45  // render text
46  const bool textVisible( progressBarOption->textVisible );
47  const bool busy( progressBarOption->minimum == 0 && progressBarOption->maximum == 0 );
48  if( textVisible && !busy )

50  progressBarOption2.rect = subElementRect( SE_ProgressBarLabel, progressBarOption, widget );
51  drawControl( CE_ProgressBarLabel, &progressBarOption2, painter, widget );
52  }
53 
54  return true;

57 bool Style::drawProgressBarContentsControl(const QStyleOption *option, QPainter *painter, const QWidget *widget) const
58 {
59  // QStyleSheetStyle won't draw progress bar groove which is necessary for us.
60  if (widget &&
61  widget->style() &&
62  widget->style()->metaObject() &&
63  widget->style()->metaObject()->className() == QLatin1String("QStyleSheetStyle")) {
64  drawProgressBarGrooveControl(option, painter, widget);
65  }
66 
67  const QStyleOptionProgressBar* progressBarOption( qstyleoption_cast<const QStyleOptionProgressBar*>( option ) );
68  if( !progressBarOption ) return false;
69 
70  const bool horizontal( !progressBarOption || progressBarOption->orientation == Qt::Horizontal );
71 
72  // copy rect and palette
73  const qreal radius( ProgressBar_Radius );
74 
75  // get progress and steps
76  const qreal progress( progressBarOption->progress - progressBarOption->minimum );
77  const int steps( qMax( progressBarOption->maximum - progressBarOption->minimum, 1 ) );
78 
79  //Calculate width fraction
80  const qreal widthFrac = qMin( qreal(1), progress/steps );
81 
82  QRect rect( option->rect);
83 
84  // convert the pixel width
85  const int indicatorSize( widthFrac*( horizontal ? rect.width():rect.height() ) );
86 
87  rect.setWidth(indicatorSize);
88 
89  QPainterPath path;
90  path.addRoundedRect(rect, radius, radius);
91 
92  painter->save();
93  painter->setRenderHints(painter->renderHints() | QPainter::Antialiasing);
94  painter->setClipPath(path);
95  painter->fillRect(rect, m_palette->brush(PaletteExtended::ProgressBar_ContentBackground, option));
96  painter->strokePath(path, QPen(m_palette->brush(PaletteExtended::ProgressBar_ContentBorder), ProgressBar_BorderWidth));
97  painter->restore();
98 
99  return true;
100 }

149 QRect Style::progressBarGrooveRect(const QStyleOption *option, const QWidget *widget) const
150 {
151  // cast option and check
152  const QStyleOptionProgressBar* progressBarOption( qstyleoption_cast<const QStyleOptionProgressBar*>( option ) );
153  if( !progressBarOption ) return option->rect;
154 
155  // get flags and orientation
156  const bool textVisible( progressBarOption->textVisible );
157  const bool busy( progressBarOption->minimum == 0 && progressBarOption->maximum == 0 );
158 
159  const QStyleOptionProgressBar *progressBarOption2(qstyleoption_cast<const QStyleOptionProgressBar *>(option));
160  const bool horizontal( !progressBarOption2 || progressBarOption2->orientation == Qt::Horizontal );
161 
162  // copy rectangle and adjust
163  QRect rect( option->rect );
164  const int frameWidth( pixelMetric( PM_DefaultFrameWidth, option, widget ) );
165  if( horizontal ) rect = GeometryUtils::insideMargin( rect, frameWidth, 0 );
166  else rect = GeometryUtils::insideMargin( rect, 0, frameWidth );
167 
168  if( textVisible && !busy && horizontal )
169  {
170  QRect textRect( subElementRect( SE_ProgressBarLabel, option, widget ) );
171  textRect = visualRect( option->direction, option->rect, textRect );
172  rect = visualRect( option->direction, option->rect, rect );
173  }
174 
175  return rect;

178 QRect Style::progressBarContentsRect(const QStyleOption *option, const QWidget *widget) const
179 {
180  // cast option and check
181  const QStyleOptionProgressBar* progressBarOption( qstyleoption_cast<const QStyleOptionProgressBar*>( option ) );
182  if( !progressBarOption ) return QRect();
183 
184  // get groove rect
185  const QRect rect( progressBarGrooveRect( option, widget ) );
186 
187  // in busy mode, grooveRect is used
188  const bool busy( progressBarOption->minimum == 0 && progressBarOption->maximum == 0 );
189  if( busy ) return rect;
190 
191  // get orientation
192  const QStyleOptionProgressBar *progressBarOption2(qstyleoption_cast<const QStyleOptionProgressBar *>(option));
193  const bool horizontal( !progressBarOption2 || progressBarOption2->orientation == Qt::Horizontal );
194 
195  // check inverted appearance
196  const bool inverted( progressBarOption2 ? progressBarOption2->invertedAppearance : false );
197 
198  // get progress and steps
199  const qreal progress( progressBarOption->progress - progressBarOption->minimum );
200  const int steps( qMax( progressBarOption->maximum - progressBarOption->minimum, 1 ) );
201 
202  //Calculate width fraction
203  const qreal widthFrac = qMin( qreal(1), progress/steps );
204 
205  // convert the pixel width
206  const int indicatorSize( widthFrac*( horizontal ? rect.width():rect.height() ) );
207 
208  QRect indicatorRect;
209  if( horizontal )
210  {
211 
212  indicatorRect = QRect( inverted ? ( rect.right() - indicatorSize + 1):rect.left(), rect.y(), indicatorSize, rect.height() );
213  indicatorRect = visualRect( option->direction, rect, indicatorRect );
214 
215  } else indicatorRect = QRect( rect.x(), inverted ? rect.top() : (rect.bottom() - indicatorSize + 1), rect.width(), indicatorSize );
216 
217  return indicatorRect;
218 }

240

242
