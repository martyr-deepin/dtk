162 QMap<int, QString> DDialogPrivate::scanTags(QString origin) const
163 {
164  QMap<int, QString> result;
165 
166  QRegExp re("<.*?>");
167  int index = origin.indexOf(re, 0);
168  int matchLength = 0;
169  while (index >= 0) {
170  result[index] = re.cap();
171  matchLength = re.matchedLength();
172  index = origin.indexOf(re, index + matchLength);
173  }
174 
175  return result;
176 }

178 // It's not a perfect solution to elide rich text, but it should be sufficient
179 // for working with rich text only change text token color.
180 QString DDialogPrivate::elideString(QString str, const QFontMetrics &fm, int width) const
181 {
182  QString trimmed = trimTag(str);
183  if (fm.width(trimmed) > width) {
184  QMap<int, QString> info = scanTags(str);
185  QString elided = fm.elidedText(trimmed, Qt::ElideMiddle, width);
186  int elideStart = elided.indexOf("…");
187  int elideLength = trimmed.length() - elided.length();
188  QList<int> indexes = info.keys();
189  std::sort(indexes.begin(), indexes.end());
190  for (int index : indexes) {
191  if (index <= elideStart) {
192  elided.insert(index, info[index]);
193  } else if (elideStart < index && index <= elideStart + elideLength) {
194  elided.insert(elideStart, info[index]);
195  elideStart += 1;
196  } else {
197  elided.insert(index - elideLength + 1, info[index]);
198  }
199  }
200  return elided;
201  } else {
202  return str;
203  }
204 }

575 void DDialog::insertButton(int index, QAbstractButton *button, bool isDefault)
576 {
577  D_D(DDialog);
578 
579  DVerticalLine *line = new DVerticalLine;
580  line->setObjectName("VLine");
581  line->setFixedHeight(30);
582 
583  d->buttonLayout->insertWidget(index * 2 , line);
584  d->buttonLayout->insertWidget(index * 2 + 1, button);
585  d->buttonList << button;
586  if (d->buttonList.count() == 1) {
587  int spacing = DStyleHelper(style()).pixelMetric(DStyle::PM_ContentsSpacing);
588  d->setSpacer(spacing);
589  }
590  line->show();
591  d->buttonLayout->itemAt(0)->widget()->hide();
592 
593  connect(button, SIGNAL(clicked(bool)), this, SLOT(_q_onButtonClicked()));
594 
595  if (isDefault) {
596  setDefaultButton(button);
597  }
598 
599  const QString &text = button->text();
600 
601  if (text.count() == 2) {
602  for (const QChar &ch : text) {
603  switch (ch.script()) {
604  case QChar::Script_Han:
605  case QChar::Script_Katakana:
606  case QChar::Script_Hiragana:
607  case QChar::Script_Hangul:
608  break;
609  default:
610  return;
611  }
612  }
613 
614  button->setText(QString().append(text.at(0)).append(QChar::Nbsp).append(text.at(1)));
615  }
616 }
