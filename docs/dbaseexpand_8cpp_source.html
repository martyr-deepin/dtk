47 DBaseExpandPrivate::DBaseExpandPrivate(DBaseExpand *qq)
48  : q_ptr(qq)
49 {
50  qq->setSizePolicy(QSizePolicy::Preferred, QSizePolicy::Maximum);
51  QVBoxLayout *mainLayout = new QVBoxLayout(qq);
52  mainLayout->setContentsMargins(0, 0, 0, 0);
53  mainLayout->setSpacing(0);
54 
55  m_headerLayout = new QVBoxLayout();
56  m_headerLayout->setContentsMargins(0, 0, 0, 0);
57  m_headerLayout->setAlignment(Qt::AlignCenter);
58 
59  m_hSeparator = new DHorizontalLine();
60  m_bottom_separator = new DHorizontalLine;
61  m_bottom_separator->hide();
62  m_hSeparator->setAccessibleName("DBaseExpandHorizontalSeparator");
63  m_bottom_separator->setAccessibleName("DBaseExpandBottomSeparator");
64 
65  qq->connect(qq, &DBaseExpand::expandChange, m_bottom_separator, &QWidget::setVisible);
66 
67  m_contentLoader = new ContentLoader();
68  m_contentLoader->setFixedHeight(0); // default to not expanded.
69  m_contentLoader->setAccessibleName("DBaseExpandContentLoader");
70 
71  m_boxWidget = new DVBoxWidget;
72  m_contentLayout = m_boxWidget->layout();
73  m_boxWidget->setAccessibleName("DBaseExpandVBoxWidget");
74 
75  QVBoxLayout *layout_contentLoader = new QVBoxLayout(m_contentLoader);
76 
77  layout_contentLoader->setMargin(0);
78  layout_contentLoader->setSpacing(0);
79  layout_contentLoader->addWidget(m_boxWidget);
80  layout_contentLoader->addStretch();
81 
82  m_animation = new QPropertyAnimation(m_contentLoader, "height", qq);
83  m_animation->setDuration(200);
84  m_animation->setEasingCurve(QEasingCurve::InSine);
85  qq->connect(m_animation, &QPropertyAnimation::valueChanged, qq, [qq] {
86  qq->setFixedHeight(qq->sizeHint().height());
87  });
88 
89  mainLayout->addLayout(m_headerLayout);
90  mainLayout->addWidget(m_hSeparator);
91  mainLayout->addWidget(m_contentLoader);
92  mainLayout->addWidget(m_bottom_separator);
93 
94  qq->setLayout(mainLayout);
95 
96  qq->connect(m_boxWidget, &DBoxWidget::sizeChanged, qq, [this] {
97  if (m_expand) {
98  int endHeight = 0;
99  endHeight = m_boxWidget->height();
100 
101  m_animation->setStartValue(m_contentLoader->height());
102  m_animation->setEndValue(endHeight);
103  m_animation->stop();
104  m_animation->start();
105  }
106  });
107 }

223 void DBaseExpand::setExpand(bool value)
224 {
225  Q_D(DBaseExpand);
226 
227  if (d->m_expand == value) {
228  return;
229  }
230 
231  d->m_expand = value;
232  Q_EMIT expandChange(value);
233 
234  if (value) {
235  d->m_animation->setStartValue(0);
236  d->m_animation->setEndValue(d->m_boxWidget->height());
237  } else {
238  d->m_animation->setStartValue(d->m_boxWidget->height());
239  d->m_animation->setEndValue(0);
240  }
241 
242  d->m_animation->stop();
243  d->m_animation->start();
244 }
