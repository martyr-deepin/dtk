115  QPainter p(this);
116  QStyleOptionSlider opt;
117  initStyleOption(&opt);
118 
119  DSlider* dSlider = qobject_cast<DSlider *>(this->parent());

121  if (!dSlider)
122  return;
123 
124  if (dSlider->handleVisible())
125  opt.subControls = QStyle::SC_SliderGroove | QStyle::SC_SliderHandle;

128 
129  style()->drawComplexControl(QStyle::CC_Slider, &opt, &p, parentWidget());
