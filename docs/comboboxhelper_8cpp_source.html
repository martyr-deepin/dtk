110 QRect Style::comboBoxSubControlRect(const QStyleOptionComplex *opt, QStyle::SubControl sc, const QWidget *widget) const
111 {
112  Q_UNUSED(widget);
113 
114  if (const QStyleOptionComboBox *cb = qstyleoption_cast<const QStyleOptionComboBox *>(opt)) {
115  QRect ret;
116 
117  int x = cb->rect.x(),
118  y = cb->rect.y(),
119  wi = cb->rect.width(),
120  he = cb->rect.height();
121  int xpos = x;
122  int margin = cb->frame ? 3 : 0;
123  int bmarg = cb->frame ? 2 : 0;
124  xpos += wi - bmarg - 16;
125 
126  switch (sc) {
127  case SC_ComboBoxFrame:
128  ret = cb->rect;
129  break;
130  case SC_ComboBoxArrow:
131  ret.setRect(xpos, y + bmarg, 16, he - 2*bmarg);
132  break;
133  case SC_ComboBoxEditField:
134  ret.setRect(x + margin, y + margin, wi - 2 * margin - 16, he - 2 * margin);
135  break;
136  case SC_ComboBoxListBoxPopup:
137  ret = cb->rect;
138  break;
139  default:
140  break;
141  }
142 
143  return visualRect(cb->direction, cb->rect, ret);
144  }
145 
146  return opt->rect;

149 bool Style::drawComboBoxLabelControl(const QStyleOption *option, QPainter *painter, const QWidget *widget) const
150 {
151  // cast option and check
152  const QStyleOptionComboBox* cb( qstyleoption_cast<const QStyleOptionComboBox*>( option ) );
153  if( !cb ) return false;
154 
155  // copy rect and palette
156  const QRect& rect( option->rect.adjusted(0, 0, -20, 0) );
157 
158  // state
159  const QStyle::State& state( option->state );
160  const bool enabled( state & QStyle::State_Enabled );
161  const bool sunken( state & (QStyle::State_On | QStyle::State_Sunken) );
162  const bool mouseOver(option->state & QStyle::State_MouseOver);
163  const bool hasFocus(option->state & QStyle::State_HasFocus);
164  const bool flat(!cb->frame);
165  const bool editable(cb->editable);
166 
167  // content
168  const bool hasText( !cb->currentText.isEmpty() );
169  const bool hasIcon( !cb->currentIcon.isNull() );
170 
171  // contents
172  QRect contentsRect( rect );
173  if( sunken && !flat ) contentsRect.translate( 1, 1 );
174  contentsRect.adjust(Metrics::Layout_ChildMarginWidth, 0, -Metrics::Layout_ChildMarginWidth, 0);
175 
176  // icon size
177  QSize iconSize;
178  if( hasIcon )
179  {
180  iconSize = cb->iconSize;
181  if( !iconSize.isValid() )
182  {
183  const int metric( widget->style()->pixelMetric( QStyle::PM_SmallIconSize, option, widget ) );
184  iconSize = QSize( metric, metric );
185  }
186  }
187 
188  // text size
189  int textFlags( Qt::AlignVCenter | Qt::AlignLeft );
190  const QSize textSize( option->fontMetrics.size( textFlags, cb->currentText ) );
191 
192  if (styleHint(SH_UnderlineShortcut, cb, widget))
193  textFlags |= Qt::TextShowMnemonic;
194  else
195  textFlags |= Qt::TextHideMnemonic;
196 
197  // adjust text and icon rect based on options
198  QRect iconRect;
199  QRect textRect;
200 
201  if( hasText && !hasIcon ) textRect = contentsRect;
202  else if( hasIcon && !hasText ) iconRect = contentsRect;
203  else {
204  const int contentsWidth( iconSize.width() + textSize.width() + Metrics::Button_ItemSpacing );
205  const int contentLeftPadding = flat ? (contentsRect.width() - contentsWidth )/2 : 0;
206  iconRect = QRect( QPoint( contentsRect.left() + contentLeftPadding,
207  contentsRect.top() + (contentsRect.height() - iconSize.height())/2 ), iconSize );
208  textRect = QRect( QPoint( iconRect.right() + Metrics::Button_ItemSpacing + 1,
209  contentsRect.top() + (contentsRect.height() - textSize.height())/2 ), textSize );
210 
211  }
212 
213  // handle right to left
214  if( iconRect.isValid() ) iconRect = widget->style()->visualRect( option->direction, option->rect, iconRect );
215  if( textRect.isValid() ) textRect = widget->style()->visualRect( option->direction, option->rect, textRect );
216 
217  // make sure there is enough room for icon
218  if( iconRect.isValid() ) iconRect = GeometryUtils::centerRect( iconRect, iconSize );
219 
220  // render icon
221  if( hasIcon && iconRect.isValid() ) {
222  // icon state and mode
223  const QIcon::State iconState( sunken ? QIcon::On : QIcon::Off );
224  QIcon::Mode iconMode;
225  if( !enabled ) iconMode = QIcon::Disabled;
226  else if( !flat && hasFocus ) iconMode = QIcon::Selected;
227  else if( mouseOver && flat ) iconMode = QIcon::Active;
228  else iconMode = QIcon::Normal;
229 
230  const QPixmap pixmap = cb->currentIcon.pixmap( iconSize, iconMode, iconState );
231  widget->style()->drawItemPixmap( painter, iconRect, Qt::AlignCenter, pixmap );
232 
233  }
234 
235  // render text
236  if( hasText && textRect.isValid() && !editable) {
237  painter->setPen(m_palette->brush(PaletteExtended::PushButton_TextColor, option).color());
238  painter->drawText(textRect, textFlags, cb->currentText);
239  }
240 
241  return true;
242 }
