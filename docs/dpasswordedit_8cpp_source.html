136  q->lineEdit()->setEchoMode(QLineEdit::Password);

151 void DPasswordEditPrivate::_q_toggleEchoMode()
152 {
153  D_Q(DPasswordEdit);
154 
155  if (q->isEchoMode()) {
156  togglePasswordVisibleButton->setIcon(DStyle::standardIcon(q->style(),DStyle::SP_ShowPassword));
157  q->setEchoMode(q->lineEdit()->Password);
158  } else {
159  togglePasswordVisibleButton->setIcon(DStyle::standardIcon(q->style(), DStyle::SP_HidePassword));
160  q->setEchoMode(q->lineEdit()->Normal);
161  }
162 }
