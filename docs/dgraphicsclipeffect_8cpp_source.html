156 void DGraphicsClipEffect::draw(QPainter *painter)
157 {
158  if (!painter->isActive()) {
159  qWarning() << "DGraphicsClipEffect::draw: The painter is not active!";
160 
161  drawSource(painter);
162  return;
163  }
164 
165  D_D(DGraphicsClipEffect);
166 
167  QPoint offset;
168  Qt::CoordinateSystem system = sourceIsPixmap() ? Qt::LogicalCoordinates : Qt::DeviceCoordinates;
169  QPixmap pixmap = sourcePixmap(system, &offset, QGraphicsEffect::PadToEffectiveBoundingRect);
170 
171  if (pixmap.isNull())
172  return;
173 
174  QPainter pixmapPainter(&pixmap);
175  QPainterPath newPath;
176 
177  newPath.addRect(pixmap.rect().marginsRemoved(d->margins));
178  newPath -= d->clipPath;
179  pixmapPainter.setRenderHints(painter->renderHints() | QPainter::Antialiasing);
180  pixmapPainter.setCompositionMode(QPainter::CompositionMode_Clear);
181  pixmapPainter.fillPath(newPath, Qt::transparent);
182 
183  painter->save();
184 
185  if (system == Qt::DeviceCoordinates)
186  painter->setWorldTransform(QTransform());
187 
188  painter->drawPixmap(offset, pixmap);
189  painter->restore();
190 }
