104 
105  if (bp->sizeHint.isValid()) {

210 void DIconButton::initStyleOption(DStyleOptionButton *option) const
211 {
212  D_DC(DIconButton);
213 
214  option->initFrom(this);
215  option->init(this);
216 
217  if (d->flat)
218  option->features |= QStyleOptionButton::Flat;
219 
220  if (isChecked())
221  option->state |= QStyle::State_On;
222 
223  if (isDown())
224  option->state |= QStyle::State_Sunken;
225 
226  if (!d->flat && !isDown())
227  option->state |= QStyle::State_Raised;
228 
229  if (enabledCircle()) {
230  option->features = QStyleOptionButton::ButtonFeature(DStyleOptionButton::CircleButton);
231  }
232 
233  option->text = text();
234  option->icon = icon();
235  option->iconSize = iconSize();
236 
237  if (backgroundRole() == QPalette::Highlight) {
238  option->palette.setBrush(QPalette::Light, option->dpalette.lightLively());
239  option->palette.setBrush(QPalette::Button, option->palette.highlight());
240  option->palette.setBrush(QPalette::Dark, option->dpalette.darkLively());
241  option->palette.setBrush(QPalette::ButtonText, option->palette.highlightedText());

270 bool DIconButton::event(QEvent *e)
271 {
272  if (e->type() == QEvent::Polish) {
273  D_DC(DIconButton);
274 
275  if (d->iconType >= 0) {
276  if (d->iconType > static_cast<qint64>(QStyle::SP_CustomBase)) {
277  DStyleHelper dstyle(style());
278  setIcon(dstyle.standardIcon(static_cast<DStyle::StandardPixmap>(d->iconType), nullptr, this));
279  } else {
280  setIcon(style()->standardIcon(static_cast<QStyle::StandardPixmap>(d->iconType), nullptr, this));
281  }
282  }
283  } else if (e->type() == QEvent::ShowToParent) {
284  // 默认调整大小，否则可能会导致按钮显示后为 QWidget 的默认大小
285  resize(sizeHint());
286  }
287 
288  return QAbstractButton::event(e);
289 }
