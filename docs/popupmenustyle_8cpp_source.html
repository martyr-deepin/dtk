111 void PopupMenuStyle::drawPrimitive(QStyle::PrimitiveElement element,
112  const QStyleOption *option,
113  QPainter *painter,
114  const QWidget *widget) const
115 {
116  if (element == QStyle::PE_PanelMenu) {
117  painter->setRenderHint(QPainter::Antialiasing);
118  QPainterPath path;
119  // Skill
120  // Make path's corner draw smooth
121  // http://www.cnblogs.com/smileEvday/p/iOS_PixelVsPoint.html
122  QRectF rect = option->rect;
123  rect.setTopLeft(QPointF(0.5, 0.5));
124  rect.setBottomRight(QPointF(rect.width() - 1, rect.height() - 1));
125 
126  path.addRoundedRect(rect,
127  FRAME_BORDER_RADIUS, FRAME_BORDER_RADIUS);
128  painter->fillPath(path, QBrush(ITEM_BG_NORMAL_COLOR));
129 
130  // Do not draw outside border if it's sub-menu
131  if (widget->parentWidget()) {
132  // Skill
133  // Draw outside border
134  QPainterPathStroker stroker;
135  stroker.setWidth(FRAME_BORDER_WIDTH);
136  stroker.setJoinStyle(Qt::RoundJoin);
137  QPainterPath borderPath = stroker.createStroke(path);
138  QPen pen(FRAME_BORDER_COLOR, FRAME_BORDER_WIDTH);
139  painter->setPen(pen);
140  painter->drawPath(borderPath);
141  }
142  }
143  else {
144  QProxyStyle::drawPrimitive(element, option, painter, widget);
145  }
146 }

217 void PopupMenuStyle::drawMenuItemIcon(const QStyleOptionMenuItem* menuItem,
218  QPainter* painter) const
219 {

221  menuItem->icon,

233 void PopupMenuStyle::drawMenuItemText(const QStyleOptionMenuItem* menuItem,
234  QPainter* painter) const
235 {
236  int width = menuItem->rect.width();
237  int height = menuItem->rect.height();
238  int y = menuItem->rect.y();
239 
240  // Draw shortcut
241  QFont shortcutFont;
242  shortcutFont.setPixelSize(SHORTCUT_FONT_PIXEL_SIZE);
243  QFontMetrics shortcutFM(shortcutFont);
244  const QString shortcutContent = menuItem->text.split(SHORTCUT_SPLIT_FLAG).last();
245  const int shortcutWidth = shortcutFM.width(shortcutContent.trimmed());
246  const int shortcutX = width - SEPARATOR_RIGHT_MARGIN - shortcutWidth;
247  QPen shortcutPen((menuItem->state & QStyle::State_Selected) ?
248  SHORTCUT_HOVER_COLOR : SHORTCUT_COLOR);
249  painter->setPen(shortcutPen);
250  painter->setFont(shortcutFont);
251  painter->drawText(shortcutX, y, shortcutWidth, height,
252  Qt::AlignRight | Qt::AlignVCenter,
253  shortcutContent);
254 
255  // Draw title
256  QFont titleFont;
257  titleFont.setPixelSize(TEXT_FONT_PIXEL_SIZE);
258  QFontMetrics titleFM(titleFont);
259  const QString titleContent = menuItem->text.split(SHORTCUT_SPLIT_FLAG).first();
260 
261  int leftSpacing = ICON_LEFT_MARGIN + ICON_PIXEL_SIZE + TEXT_LEFT_MARGIN;
262  const int titleWidth = width - leftSpacing - shortcutWidth - SEPARATOR_LEFT_MARGIN;
263  int titleX = leftSpacing;
264 
265  QPen titlePen(TEXT_COLOR);
266  painter->setPen(titlePen);
267  painter->setFont(titleFont);
268  painter->drawText(titleX, y, titleWidth, height,
269  Qt::AlignLeft | Qt::AlignVCenter,
270  titleFM.elidedText(titleContent, Qt::ElideRight, titleWidth));
271 }

303 QSize PopupMenuStyle::getItemSize(const QStyleOption* option,
304  const QSize& size) const
305 {
306  const int m = SEPARATOR_LEFT_MARGIN + SEPARATOR_RIGHT_MARGIN;
307  if (const QStyleOptionMenuItem* menuItem =
308  qstyleoption_cast<const QStyleOptionMenuItem* >(option)) {
309  switch (menuItem->menuItemType) {
310  case QStyleOptionMenuItem::Separator:
311  return QSize(size.width() / 2 + m, SEPARATOR_HEIGHT);
312  case QStyleOptionMenuItem::Normal:
313  case QStyleOptionMenuItem::SubMenu:
314  return QSize(size.width() / 2 + m, ITEM_HEIGHT);
315  default:
316  return size;
317  }
318  }
319  else {
320  return size;
321  }
322 }

343 QPixmap PopupMenuStyle::getIconPixmap(const QStyle::State state,
344  const QIcon& icon,
345  const QSize& size) const
346 {
347  QIcon::Mode iconMode = QIcon::Normal;
348  QIcon::State iconState = QIcon::Off;
349  if (state & QStyle::State_Active) {
350  iconMode = QIcon::Active;
351  } else if (state & QStyle::State_Selected) {
352  iconMode = QIcon::Selected;
353  } else if (!(state & QStyle::State_Enabled)) {
354  iconMode = QIcon::Disabled;
355  }
356 
357  if (state & QStyle::State_On) {
358  iconState = QIcon::On;
359  }
360 
361  return icon.pixmap(size, iconMode, iconState);
362 }
