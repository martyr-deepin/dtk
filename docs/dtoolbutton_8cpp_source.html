43 void DToolButton::initStyleOption(QStyleOptionToolButton *option) const
44 {
45  QToolButton::initStyleOption(option);
46  //判断条件不用Qt::ToolButtonTextBesideIcon原因
47  //会强制居中，大小不受sizeHint（）控制
48  if (!option->icon.isNull() && !option->text.isEmpty()) {
49  option->toolButtonStyle = Qt::ToolButtonTextBesideIcon;
50  }
51 }

53 QSize DToolButton::sizeHint() const
54 {
55  QSize size = QToolButton::sizeHint();
56  //计算图标文字同时存在时的宽度
57  if (!icon().isNull() && !text().isEmpty()) {
58  QFontMetrics fm = fontMetrics();
59  int textWidth = fm.size(Qt::TextShowMnemonic, text()).width();
60  size.setWidth(size.width() + textWidth);
61  }
62  return size;
63 }
