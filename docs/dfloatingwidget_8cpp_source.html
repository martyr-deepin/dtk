162 bool DFloatingWidget::event(QEvent *event)
163 {
164  D_D(DFloatingWidget);
165 
166  switch (event->type()) {
167  case QEvent::Polish: {
168  DStyleHelper dstyle(style());
169  int margins = dstyle.pixelMetric(DStyle::PM_FloatingWidgetShadowMargins, nullptr, this);
170  setContentsMargins(margins, margins, margins, margins);
171 
172  if (d->background) {
173  int radius = DStyleHelper(style()).pixelMetric(DStyle::PM_TopLevelWindowRadius);
174 
175  d->background->setBlurRectXRadius(radius);
176  d->background->setBlurRectYRadius(radius);
177  d->background->setGeometry(contentsRect());
178 
179  d->adjustPalette();
180  }
181  break;
182  }
183  case QEvent::PaletteChange: {
184  d->adjustPalette();
185  break;
186  }
187  case QEvent::Resize: {
188  if (d->background) {
189  d->background->setGeometry(contentsRect());
190  }
191  break;
192  }
193  default:
194  break;
195  }
196 
197  return QWidget::event(event);
198 }

238 void DFloatingWidget::setBlurBackgroundEnabled(bool blurBackgroundEnabled)
239 {
240  D_D(DFloatingWidget);
241 
242  if (bool(d->background) == blurBackgroundEnabled)
243  return;
244 
245  if (blurBackgroundEnabled) {
246  d->background = new DBlurEffectWidget(this);
247  d->background->setGeometry(contentsRect());
248  d->background->lower();
249  d->background->setFocusPolicy(Qt::NoFocus);
250  d->background->setAttribute(Qt::WA_TransparentForMouseEvents);
251  d->background->show();
252 
253  int radius = DStyleHelper(style()).pixelMetric(DStyle::PM_TopLevelWindowRadius);
254 
255  d->background->setBlurRectXRadius(radius);
256  d->background->setBlurRectYRadius(radius);
257  d->background->setBlendMode(DBlurEffectWidget::InWidgetBlend);
258 
259  connect(d->background, &DBlurEffectWidget::blurSourceImageDirtied,
260  this, static_cast<void(DFloatingWidget::*)()>(&DFloatingWidget::update));
261  } else {
262  d->background->hide();
263  d->background->deleteLater();
264  }
265 
266  d->adjustPalette();
267 }
