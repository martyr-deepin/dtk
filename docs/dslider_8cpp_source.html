469 void DSliderPrivate::updtateTool(int value)
470 {
471  Q_Q(DSlider);
472 
473  if (!tipvalue)
474  return;
475 
476  int min = q->minimum();
477  int max = q->maximum();
478  QStyleOptionSlider opt;
479  const QRectF rectHandle = q->style()->subControlRect(QStyle::CC_Slider, &opt, QStyle::SC_SliderHandle, slider); //滑块
480 
481  double x = ((value - min) * 1.0 / (max -min)) * (slider->rect().width() - rectHandle.width());
482  double y = slider->y() + slider->height();
483 
484  //x是以实际滑槽为起始坐标，而非DSlider为参考坐标（实际滑槽长度 == Slider长度 - 滑块长度）
485  //气泡位置
486  if (slider->invertedAppearance()) {
487  tipvalue->move(slider->width() - x - (tipvalue->width() + rectHandle.width()) / 2.0 , y);
488  } else {
489  tipvalue->move(x + (rectHandle.width() - tipvalue->width()) / 2.0, y);
490  }
491 
492  int shadowMarge = DStyle::pixelMetric(q->style(), DStyle::PM_FloatingWidgetShadowMargins) - 1;
493 
494  //气泡贴边
495  if (tipvalue->x() < 0) {
496  tipvalue->move(0 - shadowMarge, y);
497  }
498 
499  if (tipvalue->x() + tipvalue->width() > slider->width()) {
500  tipvalue->move(slider->width() - tipvalue->width() + shadowMarge, y);
501  }
502 
503  tipvalue->raise();
504  tipvalue->adjustSize();
505 }

717 void SliderStrip::paintEvent(QPaintEvent *event)
718 {
719  Q_UNUSED(event)
720  QPainter pa(this);
721  auto dpa = DApplicationHelper::instance()->palette(this);
722  QPen penLine = QPen(dpa.color(DPalette::ObviousBackground), 1);
723  QPen penNumber = QPen(dpa.textTips(), 1);
724 
725  int tickSize = fontMetrics().height() / 2;
726  int offsetSize = style()->pixelMetric(QStyle::PM_SliderLength, nullptr, this) / 2;
727  int margin = DStyle::pixelMetric(style(), DStyle::PM_FocusBorderSpacing) + \
728  DStyle::pixelMetric(style(), DStyle::PM_FocusBorderSpacing);
729  qreal startX = 0, startY = 0, endX = 0, endY = 0;
730  int width = this->width(), height = this->height(); //slider宽高
731  int paragraph = scaleInfo.count(); //刻度个数
732  qreal average = 0; //每一段的距离
733  qreal textPos = 0; //字体位置
734 
735  QSlider* slider = static_cast<DSlider *>(this->parent())->slider();
736  for (int i = 0; i < list.count(); i++) {
737  qreal percentage = (list.at(i) - slider->minimum()) * 1.0 / (slider->maximum() - slider->minimum());
738  pa.setPen(penLine);
739 
740  if (orient == Qt::Horizontal) {
741  qreal sliderX = percentage * width;
742  if (tick == QSlider::TicksAbove)
743  pa.drawLine(QPointF(sliderX + offsetSize, height), QPointF(sliderX + offsetSize, height - tickSize));
744  if (tick == QSlider::TicksBelow)
745  pa.drawLine(QPointF(sliderX + offsetSize, 0), QPointF(sliderX + offsetSize, tickSize));
746  } else {
747  qreal sliderX = percentage * height;
748  if (tick == QSlider::TicksLeft)
749  pa.drawLine(QPointF(width - tickSize, height - (sliderX + offsetSize)), QPointF(width, height - (sliderX + offsetSize)));
750  if (tick == QSlider::TicksRight)
751  pa.drawLine(QPointF(0, height - (sliderX + offsetSize)), QPointF(tickSize, height - (sliderX + offsetSize)));
752  }
753  }
754 
755  if (scaleInfo.isEmpty())
756  return;
757 
758  if (orient == Qt::Horizontal) {
759  width -= 2 * offsetSize + margin * 2;
760  startX += offsetSize + margin;
761  endX = startX;
762  average = width / static_cast<qreal>(paragraph - 1);
763 
764  if (tick == QSlider::TicksBelow) {
765  endY = tickSize;
766  textPos = endY;
767  } else if (tick == QSlider::TicksAbove) {
768  startY = height;
769  endY = startY - tickSize;
770  }
771 
772  pa.setPen(penLine);
773  pa.drawLine(QPointF(startX, startY), QPointF(endX, endY));
774  pa.setPen(penNumber);
775  pa.drawText(QRectF(endX, textPos, width, height - tickSize), Qt::AlignLeft, scaleInfo[0]);
776 
777  for (int i = 1; i < paragraph - 1; i++) {
778  startX += average;
779  endX = startX;
780  pa.setPen(penLine);
781  pa.drawLine(QPointF(startX, startY), QPointF(endX, endY));
782  pa.setPen(penNumber);
783  pa.drawText(QRectF(endX - width / 2, textPos, width, height - tickSize), Qt::AlignHCenter, scaleInfo[i]);
784  }
785 
786  if (paragraph > 1) {
787  startX += average;
788  endX = startX;
789  pa.setPen(penLine);
790  pa.drawLine(QPointF(startX, startY), QPointF(endX, endY));
791  pa.setPen(penNumber);
792  pa.drawText(QRectF(endX - width, textPos, width, height - tickSize), Qt::AlignRight, scaleInfo[paragraph - 1]);
793  }
794  } else {
795  startY = offsetSize + margin;
796  height -= offsetSize * 2 + margin * 2;
797  endY = startY;
798  average = height / static_cast<qreal>(paragraph - 1);
799  Qt::Alignment text_flags = Qt::AlignVCenter;
800 
801  if (tick == QSlider::TicksLeft) {
802  startX = width;
803  endX = startX - tickSize;
804  textPos = 0;
805  text_flags |= Qt::AlignRight;
806  } else if (tick == QSlider::TicksRight) {
807  endX = tickSize;
808  textPos = endX;
809  text_flags |= Qt::AlignLeft;
810  }
811 
812  pa.setPen(penLine);
813  pa.drawLine(QPointF(startX, startY), QPointF(endX, endY));
814  pa.setPen(penNumber);
815  pa.drawText(QRectF(textPos, endY - average / 2 + offsetSize / 2, width - tickSize, average), text_flags, scaleInfo[0]);
816 
817  for (int i = 1; i < paragraph - 1; i++) {
818  startY += average;
819  endY = startY;
820  pa.setPen(penLine);
821  pa.drawLine(QPointF(startX, startY), QPointF(endX, endY));
822  pa.setPen(penNumber);
823  pa.drawText(QRectF(textPos, endY - average / 2, width - tickSize, average), text_flags, scaleInfo[i]);
824  }
825 
826  if (paragraph > 1) {
827  startY += average;
828  endY = startY;
829  pa.setPen(penLine);
830  pa.drawLine(QPointF(startX, startY), QPointF(endX, endY));
831  pa.setPen(penNumber);
832  pa.drawText(QRectF(textPos, endY - average / 2 - offsetSize / 2, width - tickSize, average), text_flags, scaleInfo[paragraph - 1]);
833  }
834  }
835 }
