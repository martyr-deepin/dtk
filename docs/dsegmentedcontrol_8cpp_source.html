383 bool DSegmentedControl::setCurrentIndex(int currentIndex)
384 {
385  D_D(DSegmentedControl);
386 
387  if(currentIndex == d->currentIndex)
388  return true;
389 
390  if(currentIndex<0||currentIndex>count()-1){
391  qErrnoWarning("index range over!");
392  return false;
393  }
394 
395  d->currentIndex = currentIndex;
396 
397  Q_FOREACH (QToolButton *button, d->tabList) {
398  button->setEnabled(true);
399  }
400 
401  at(currentIndex)->setFocus();
402  at(currentIndex)->setEnabled(false);
403 
404  updateHighlightGeometry();
405 
406  Q_EMIT currentChanged(currentIndex);
407  Q_EMIT currentTitleChanged(at(currentIndex)->text());
408 
409  return true;
410 }

487 void DSegmentedControl::updateHighlightGeometry(bool animation)
488 {
489  D_D(DSegmentedControl);
490 
491  if (d->currentIndex < 0)
492  return;
493 
494  QRect tmp = at(d->currentIndex)->geometry();
495 
496  d->highlight->resize(tmp.size());
497 
498  if (d->currentIndex == 0) {
499  tmp.setX(0);
500  }
501 
502  tmp.setY(0);
503 
504  if (d->highlight->pos() == tmp.topLeft())
505  return;
506 
507  if (animation) {
508  d->highlightMoveAnimation->setStartValue(d->highlight->pos());
509  d->highlightMoveAnimation->setEndValue(tmp.topLeft());
510  d->highlightMoveAnimation->start();
511  } else {
512  d->highlight->move(tmp.topLeft());
513  }
514 }
