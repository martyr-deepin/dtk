312 void DTitlebarPrivate::updateFullscreen()
313 {
314  D_Q(DTitlebar);
315 
316  if (!autoHideOnFullscreen) {
317  return;
318  }
319 
320  bool isFullscreen = targetWindow()->windowState().testFlag(Qt::WindowFullScreen);
321  auto mainWindow = qobject_cast<DMainWindow *>(targetWindow());
322  if (!isFullscreen) {
323  quitFullButton->hide();
324  mainWindow->setMenuWidget(q);
325  showOnFullscreen();
326  } else {
327  // must set to empty
328  quitFullButton->show();
329  if (mainWindow->menuWidget()) {
330  mainWindow->menuWidget()->setParent(nullptr);
331  mainWindow->setMenuWidget(Q_NULLPTR);
332  }
333 
334  q->setParent(mainWindow);
335  q->show();
336  hideOnFullscreen();
337  }

340 void DTitlebarPrivate::updateButtonsState(Qt::WindowFlags type)
341 {
342  D_Q(DTitlebar);
343  bool useDXcb = DPlatformWindowHandle::isEnabledDXcb(targetWindow());
344  bool isFullscreen = targetWindow()->windowState().testFlag(Qt::WindowFullScreen);
345 
346 // bool forceShow = !useDXcb;
347  bool forceShow = false;
348 #ifndef Q_OS_LINUX
349  forceShow = false;
350 #endif
351 
352  bool showTitle = (type.testFlag(Qt::WindowTitleHint) || forceShow) && !embedMode && !DGuiApplicationHelper::isTabletEnvironment();
353  if (titleLabel) {
354  if (showTitle) {
355  titleLabel->setText(q->property("_dtk_title").toString());
356  } else {
357  q->setProperty("_dtk_title", titleLabel->text());
358  titleLabel->clear();
359  }
360  }
361 
362  // Never show in embed/fullscreen
363  bool forceHide = (!useDXcb) || embedMode || isFullscreen;
364 
365  bool showMin = (type.testFlag(Qt::WindowMinimizeButtonHint) || forceShow) && !forceHide;
366  minButton->setVisible(showMin);
367 
368  bool allowResize = true;
369 
370  if (q->window() && q->window()->windowHandle()) {
371  auto functions_hints = DWindowManagerHelper::getMotifFunctions(q->window()->windowHandle());
372  allowResize = functions_hints.testFlag(DWindowManagerHelper::FUNC_RESIZE);
373  }
374 
375  bool showMax = (type.testFlag(Qt::WindowMaximizeButtonHint) || forceShow) && !forceHide && allowResize;
376  bool showClose = type.testFlag(Qt::WindowCloseButtonHint) && useDXcb;
377  bool showQuit = isFullscreen && useDXcb && fullScreenButtonVisible;
378  maxButton->setVisible(showMax);
379  closeButton->setVisible(showClose);
380  quitFullButton->setVisible(showQuit);
381  // qDebug() << "max:"
382  // << "allowResize" << allowResize
383  // << "useDXcb" << useDXcb
384  // << "forceHide" << forceHide
385  // << "type.testFlag(Qt::WindowMaximizeButtonHint)" << type.testFlag(Qt::WindowMaximizeButtonHint);
386 }

388 void DTitlebarPrivate::updateButtonsFunc()
389 {
390  // TASK-18145 (bug-17474) do not setMotifFunctions on wayland
391  if (!targetWindowHandle || !qgetenv("WAYLAND_DISPLAY").isEmpty()) {
392  return;
393  }
394 
395  // 根据 disableFlags 更新窗口标志，而标题栏上具体按钮的开启/禁用状态只根据
396  // 窗口标志改变后做更新，且窗口标志改变后也会同步更新 disableFlags 的值，
397  // 也就是说，实际上窗口按钮的状态只受窗口标志影响，而 setDisableFlags
398  // 只是单纯的记录新的 disableFlags 的值，由于调用 setDisableFlags
399  // 是应用中的主动行为，所以同时也会根据这个值更新窗口标志。
400  DWindowManagerHelper::setMotifFunctions(
401  targetWindowHandle,
402  DWindowManagerHelper::FUNC_MAXIMIZE,
403  !disableFlags.testFlag(Qt::WindowMaximizeButtonHint));
404  DWindowManagerHelper::setMotifFunctions(
405  targetWindowHandle,
406  DWindowManagerHelper::FUNC_MINIMIZE,
407  !disableFlags.testFlag(Qt::WindowMinimizeButtonHint));
408  DWindowManagerHelper::setMotifFunctions(
409  targetWindowHandle,
410  DWindowManagerHelper::FUNC_CLOSE,
411  !disableFlags.testFlag(Qt::WindowCloseButtonHint));
412 }

479 void DTitlebarPrivate::_q_onTopWindowMotifHintsChanged(quint32 winId)
480 {
481  D_QC(DTitlebar);
482 
483  if (!DPlatformWindowHandle::isEnabledDXcb(targetWindow())) {
484  q->disconnect(DWindowManagerHelper::instance(), SIGNAL(windowMotifWMHintsChanged(quint32)),
485  q, SLOT(_q_onTopWindowMotifHintsChanged(quint32)));
486  return;
487  }
488 
489  if (winId != q->window()->internalWinId()) {
490  return;
491  }
492 
493  DWindowManagerHelper::MotifDecorations decorations_hints = DWindowManagerHelper::getMotifDecorations(q->window()->windowHandle());
494  DWindowManagerHelper::MotifFunctions functions_hints = DWindowManagerHelper::getMotifFunctions(q->window()->windowHandle());
495 
496  if (titleLabel) {
497  titleLabel->setVisible(decorations_hints.testFlag(DWindowManagerHelper::DECOR_TITLE));
498  }
499 
500  updateButtonsState(targetWindow()->windowFlags());
501 
502  minButton->setEnabled(functions_hints.testFlag(DWindowManagerHelper::FUNC_MINIMIZE));
503  maxButton->setEnabled(functions_hints.testFlag(DWindowManagerHelper::FUNC_MAXIMIZE)
504  && functions_hints.testFlag(DWindowManagerHelper::FUNC_RESIZE));
505  closeButton->setEnabled(functions_hints.testFlag(DWindowManagerHelper::FUNC_CLOSE));
506  // sync button state
507 #if QT_VERSION >= QT_VERSION_CHECK(5, 7, 0)
508  disableFlags.setFlag(Qt::WindowMinimizeButtonHint, !minButton->isEnabled());
509  disableFlags.setFlag(Qt::WindowMaximizeButtonHint, !maxButton->isEnabled());
510  disableFlags.setFlag(Qt::WindowCloseButtonHint, !closeButton->isEnabled());
511 #else
512  setWindowFlag(disableFlags, Qt::WindowMinimizeButtonHint, !minButton->isEnabled());
513  setWindowFlag(disableFlags, Qt::WindowMaximizeButtonHint, !maxButton->isEnabled());
514  setWindowFlag(disableFlags, Qt::WindowCloseButtonHint, !closeButton->isEnabled());
515 #endif
516 }

536  autoThemeAction = switchThemeMenu->addAction(qApp->translate("TitleBarMenu", "System Theme"));
537 
538  autoThemeAction->setCheckable(true);
539  lightThemeAction->setCheckable(true);
540  darkThemeAction->setCheckable(true);
541 
542  QActionGroup *group = new QActionGroup(switchThemeMenu);
543  group->addAction(autoThemeAction);
544  group->addAction(lightThemeAction);
545  group->addAction(darkThemeAction);
546 
547  QObject::connect(group, SIGNAL(triggered(QAction*)),
548  q, SLOT(_q_switchThemeActionTriggered(QAction*)));
549 
550  menu->addMenu(switchThemeMenu);
551  themeSeparator = menu->addSeparator();
552 
553  switchThemeMenu->menuAction()->setVisible(canSwitchTheme);
554  themeSeparator->setVisible(canSwitchTheme);
555  }
556  }
557 
558  // add help menu item.
559  if (!helpAction && DApplicationPrivate::isUserManualExists()) {
560  helpAction = new QAction(qApp->translate("TitleBarMenu", "Help"), menu);
561  QObject::connect(helpAction, SIGNAL(triggered(bool)), q, SLOT(_q_helpActionTriggered()));

570  }
571

574  quitAction = new QAction(qApp->translate("TitleBarMenu", "Exit"), menu);
575  QObject::connect(quitAction, SIGNAL(triggered(bool)), q, SLOT(_q_quitActionTriggered()));
576  menu->addAction(quitAction);
577  }

857 bool DTitlebar::eventFilter(QObject *obj, QEvent *event)
858 {
859  D_D(DTitlebar);
860 
861  if (event->type() == QEvent::MouseButtonPress &&
862  static_cast<QMouseEvent *>(event)->button() == Qt::RightButton &&
863  (obj ==d->minButton || obj == d->maxButton ||
864  obj == d->closeButton || obj == d->optionButton ||
865  obj == d->quitFullButton))
866  {
867  event->accept(); // button on titlebar should not show kwin menu
868  return true;
869  }
870 
871  if (obj == d->targetWindow()) {
872  switch (event->type()) {
873  case QEvent::ShowToParent:
874  d->handleParentWindowIdChange();
875  d->updateButtonsState(d->targetWindow()->windowFlags());
876  break;
877  case QEvent::Resize:
878  if (d->autoHideOnFullscreen) {
879  setFixedWidth(d->targetWindow()->width());
880  }
881  break;
882  case QEvent::HoverMove: {
883  auto mouseEvent = reinterpret_cast<QMouseEvent *>(event);
884  bool isFullscreen = d->targetWindow()->windowState().testFlag(Qt::WindowFullScreen);
885  if (isFullscreen && d->autoHideOnFullscreen) {
886  if (mouseEvent->pos().y() > height() && d->isVisableOnFullscreen()) {
887  d->hideOnFullscreen();
888  }
889  if (mouseEvent->pos().y() < 2) {
890  d->showOnFullscreen();
891  }
892  }
893  break;
894  }
895  case QEvent::WindowStateChange: {
896  d->handleParentWindowStateChange();
897  break;
898  }
899  default:
900  break;
901  }
902  }
903  return QWidget::eventFilter(obj, event);
904 }
