34 QImage dropShadow(const QPixmap &px, qreal radius, const QColor &color)
35 {
36  if (px.isNull()) {
37  return QImage();
38  }
39 
40  QSize size = px.size();
41 
42  QImage tmp(size + QSize(radius * 2, radius * 2), QImage::Format_ARGB32_Premultiplied);
43  tmp.fill(0);
44  QPainter tmpPainter(&tmp);
45  tmpPainter.setCompositionMode(QPainter::CompositionMode_Source);
46  tmpPainter.drawPixmap(QPoint(radius, radius), px);
47  tmpPainter.end();
48 
49  // blur the alpha channel
50  QImage blurred(tmp.size(), QImage::Format_ARGB32_Premultiplied);
51  blurred.fill(0);
52  QPainter blurPainter(&blurred);
53  qt_blurImage(&blurPainter, tmp, radius, false, true);
54  blurPainter.end();
55 
56  if (color == QColor(Qt::black)) {
57  return blurred;
58  }
59 
60  tmp = blurred;
61 
62  // blacken the image...
63  tmpPainter.begin(&tmp);
64  tmpPainter.setCompositionMode(QPainter::CompositionMode_SourceIn);
65  tmpPainter.fillRect(tmp.rect(), color);
66  tmpPainter.end();
67 
68  return tmp;
69 }

85 QIcon getCircleIcon(const QPixmap &pixmap, int diameter)
86 {
87  QPixmap output(QSize(diameter, diameter));
88  output.fill(Qt::transparent);
89  QPainter painter(&output);
90  // 抗锯齿 + 平滑边缘处理
91  painter.setRenderHints(QPainter::Antialiasing, true);
92  painter.setRenderHints(QPainter::SmoothPixmapTransform, true);
93  painter.setPen(Qt::NoPen);
94 
95  QPainterPath path;
96  path.addEllipse(0, 0, diameter, diameter);
97  painter.setClipPath(path);
98  painter.drawPixmap(0, 0, diameter, diameter, pixmap);
99  return QIcon(output);
100 }
