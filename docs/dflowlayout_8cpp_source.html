32 QSize DFlowLayoutPrivate::doLayout(const QRect &rect, bool testOnly) const
33 {
34  D_QC(DFlowLayout);
35 
36  if(!testOnly) {
37  if(!q->parentWidget() || q->parentWidget()->isHidden())
38  return QSize();
39  } else if(rect.width() <= 0){
40  return QSize();
41  }
42 
43  int left, top, right, bottom;
44 
45  q->getContentsMargins(&left, &top, &right, &bottom);
46  QRect effectiveRect = rect.adjusted(+left, +top, -right, -bottom);
47 
48  int x = q->parentWidget()->layoutDirection() == Qt::RightToLeft
49  ? effectiveRect.right()
50  :effectiveRect.x();
51  int y = effectiveRect.y();
52 
53  QSize size_hint;
54 
55  if(flow == DFlowLayout::Flow::LeftToRight) {
56  int maxWidth = 0;
57  int lineHeight = 0;
58 
59  if(q->parentWidget()->layoutDirection() == Qt::RightToLeft) {
60  for (QLayoutItem *item : itemList) {
61  int nextX = x - item->sizeHint().width() - horizontalSpacing;
62 
63  if (nextX + horizontalSpacing < effectiveRect.x() && lineHeight > 0) {
64  maxWidth = qMax(effectiveRect.right() - x, maxWidth);
65  x = effectiveRect.right();
66  y = y + lineHeight + verticalSpacing;
67  nextX = x - item->sizeHint().width() - horizontalSpacing;
68  lineHeight = 0;
69  }
70 
71  if (!testOnly) {
72  QRect item_geometry;
73 
74  item_geometry.setSize(item->sizeHint());
75  item_geometry.moveTopRight(QPoint(x, y));
76  item->setGeometry(item_geometry);
77  }
78 
79  x = nextX;
80  lineHeight = qMax(lineHeight, item->sizeHint().height());
81  }
82 
83  size_hint = QSize(maxWidth, y + lineHeight - rect.y() + bottom);
84  } else {
85  for (QLayoutItem *item : itemList) {
86  int nextX = x + item->sizeHint().width() + horizontalSpacing;
87 
88  if (nextX - horizontalSpacing > effectiveRect.right() && lineHeight > 0) {
89  maxWidth = qMax(x, maxWidth);
90  x = effectiveRect.x();
91  y = y + lineHeight + verticalSpacing;
92  nextX = x + item->sizeHint().width() + horizontalSpacing;
93  lineHeight = 0;
94  }
95 
96  if (!testOnly)
97  item->setGeometry(QRect(QPoint(x, y), item->sizeHint()));
98 
99  x = nextX;
100  lineHeight = qMax(lineHeight, item->sizeHint().height());
101  }
102 
103  size_hint = QSize(maxWidth, y + lineHeight - rect.y() + bottom);
104  }
105  } else {
106  int maxHeight = 0;
107  int lineWidth = 0;
108 
109  if(q->parentWidget()->layoutDirection() == Qt::RightToLeft) {
110  for (QLayoutItem *item : itemList) {
111  int nextY = y + item->sizeHint().height() + verticalSpacing;
112 
113  if(nextY - verticalSpacing > effectiveRect.bottom() && lineWidth > 0) {
114  maxHeight = qMax(y, maxHeight);
115  y = effectiveRect.y();
116  x = x - lineWidth - horizontalSpacing;
117  nextY = y + item->sizeHint().height() + verticalSpacing;
118  lineWidth = 0;
119  }
120 
121  if (!testOnly)
122  item->setGeometry(QRect(QPoint(x - item->sizeHint().width(), y), item->sizeHint()));
123 
124  y = nextY;
125  lineWidth = qMax(lineWidth, item->sizeHint().width());
126  }
127 
128  size_hint = QSize(rect.right() - x + lineWidth + right + 1, maxHeight);
129  } else {
130  for (QLayoutItem *item : itemList) {
131  int nextY = y + item->sizeHint().height() + verticalSpacing;
132 
133  if(nextY - verticalSpacing > effectiveRect.bottom() && lineWidth > 0) {
134  maxHeight = qMax(y, maxHeight);
135  y = effectiveRect.y();
136  x = x + lineWidth + horizontalSpacing;
137  nextY = y + item->sizeHint().height() + verticalSpacing;
138  lineWidth = 0;
139  }
140 
141  if (!testOnly)
142  item->setGeometry(QRect(QPoint(x, y), item->sizeHint()));
143 
144  y = nextY;
145  lineWidth = qMax(lineWidth, item->sizeHint().width());
146  }
147 
148  size_hint = QSize(x + lineWidth - rect.x() + right, maxHeight);
149  }
150  }
151 
152  if(!testOnly) {
153  if(sizeHint != size_hint) {
154  sizeHint = size_hint;
155  Q_EMIT q->sizeHintChanged(sizeHint);
156  }
157  }
158 
159  return size_hint;
160 }

454 QLayoutItem *DFlowLayout::takeAt(int index)
455 {
456  D_D(DFlowLayout);
457 
458  if (index < 0 || index >= d->itemList.count()) {
459  return 0;
460  }
461 
462  QLayoutItem *item = d->itemList.takeAt(index);
463 
464  if (QLayout *l = item->layout()) {
465  // sanity check in case the user passed something weird to QObject::setParent()
466  if (l->parent() == this)
467  l->setParent(0);
468  }
469 
470  Q_EMIT countChanged(count());
471 
472  return item;
473 }
