170 void DToast::pop()
171 {
172  Q_D(DToast);
173 
174  adjustSize();
175  show();
176 
177  if (d->animation) {
178  return;
179  }
180 
181  int _duration = d->duration < 0 ? 2000 : d->duration;
182 
183  d->animation = new QPropertyAnimation(this, "opacity");
184  d->animation->setDuration(_duration);
185  d->animation->setStartValue(0);
186  d->animation->setKeyValueAt(0.4, 1.0);
187  d->animation->setKeyValueAt(0.8, 1.0);
188  d->animation->setEndValue(0);
189  d->animation->start();
190  d->animation->connect(d->animation, &QPropertyAnimation::finished,
191  this, [ = ]() {
192  hide();
193  d->animation->deleteLater();
194  d->animation = Q_NULLPTR;
195  });
196 }

234  q->setWindowFlags(q->windowFlags() | Qt::WindowStaysOnTopHint);
235 
236  auto layout = new QHBoxLayout(q);
237  layout->setContentsMargins(10, 10, 10, 10);

243  textLabel->setVisible(false);
244 
245  layout->addWidget(iconLabel);

251  effect->setOffset(0, 0);
