403 void DSimpleListView::refreshItems(QList<DSimpleListItem*> items)
404 {
405  D_D(DSimpleListView);
406 
407  // Init.
408  QList<DSimpleListItem*> *newSelectionItems = new QList<DSimpleListItem*>();
409  DSimpleListItem *newLastSelectionItem = NULL;
410  DSimpleListItem *newLastHoverItem = NULL;
411 
412  // Save selection items and last selection item.
413  for (DSimpleListItem *item:items) {
414  for (DSimpleListItem *selectionItem:*d->selectionItems) {
415  if (item->sameAs(selectionItem)) {
416  newSelectionItems->append(item);
417  break;
418  }
419  }
420  }
421 
422  if (d->lastSelectItem != NULL) {
423  for (DSimpleListItem *item:items) {
424  if (item->sameAs(d->lastSelectItem)) {
425  newLastSelectionItem = item;
426  break;
427  }
428  }
429  }
430 
431  if (d->lastHoverItem != NULL) {
432  for (DSimpleListItem *item:items) {
433  if (item->sameAs(d->lastHoverItem)) {
434  newLastHoverItem = item;
435  break;
436  }
437  }
438  }
439  d->lastHoverItem = NULL;
440 
441  // Update items.
442  clearItems();
443  d->listItems->append(items);
444  QList<DSimpleListItem*> searchItems = d->getSearchItems(items);
445  d->renderItems->append(searchItems);
446 
447  // Sort once if default sort column hasn't init.
448  if (d->defaultSortingColumn != -1) {
449  d->sortItemsByColumn(d->defaultSortingColumn, d->defaultSortingOrder);
450  }
451 
452  // Restore selection items and last selection item.
453  clearSelections();
454  addSelections(*newSelectionItems, false);
455  d->lastSelectItem = newLastSelectionItem;
456  d->lastHoverItem = newLastHoverItem;
457 
458  // Keep scroll position.
459  d->renderOffset = adjustRenderOffset(d->renderOffset);
460 
461  // Render.
462  repaint();
463 }

813  selectNextItem();
814  }
815  } else if (keyEvent->key() == Qt::Key_PageUp) {
816  if (keyEvent->modifiers() == Qt::ControlModifier) {

838 void DSimpleListView::mouseMoveEvent(QMouseEvent *mouseEvent)
839 {
840  D_D(DSimpleListView);
841 
842  // Scroll if mouse drag at scrollbar.
843  if (d->mouseDragScrollbar) {
844  int barHeight = getScrollbarHeight();
845  d->renderOffset = adjustRenderOffset((mouseEvent->y() - barHeight / 2 - d->titleHeight) / (getScrollAreaHeight() * 1.0) * d->getItemsTotalHeight());
846 
847  repaint();
848  }
849  // Update scrollbar status with mouse position.
850  else if (isMouseAtScrollArea(mouseEvent->x()) != d->mouseAtScrollArea) {
851  d->mouseAtScrollArea = isMouseAtScrollArea(mouseEvent->x());
852  repaint();
853  }
854  // Otherwise to check titlebar arrow status.
855  else {
856  bool atTitleArea = isMouseAtTitleArea(mouseEvent->y());
857 
858  if (atTitleArea) {
859  int hoverColumn = -1;
860
