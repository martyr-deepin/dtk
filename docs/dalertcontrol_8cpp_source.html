39 void DAlertControlPrivate::updateTooltipPos()
40 {
41  if (!target || !target->parentWidget()) {
42  return;
43  }
44  QWidget *p = target->parentWidget();
45 
46  int w = DStyle::pixelMetric(p->style(), DStyle::PM_FloatingWidgetShadowMargins) / 2;
47  QPoint point = QPoint(target->x() - w, target->y() + target->height() - w);
48  frame->move(p->mapTo(qobject_cast<QWidget *>(frame->parentWidget()), point));
49  int tipWidth= frame->parentWidget()->width() - 40;
50  tooltip->setMaximumWidth(tipWidth);
51  frame->setMinimumHeight(tooltip->heightForWidth(tipWidth) + frame->layout()->spacing() *2);
52  frame->adjustSize();
53 
54  int tw = target->width();
55  int fw = frame->width();
56  int leftPending = tw - fw;
57  switch (alignment) {
58  case Qt::AlignRight:
59  point += QPoint(leftPending, 0);
60  break;
61  case Qt::AlignHCenter:
62  case Qt::AlignCenter:
63  point += QPoint(leftPending/2, 0);
64  break;
65  default:
66  return;
67  }
68 
69  frame->move(p->mapTo(qobject_cast<QWidget *>(frame->parentWidget()), point));
70 }

199 void DAlertControl::showAlertMessage(const QString &text, QWidget *follower, int duration)
200 {
201  D_D(DAlertControl);
202 
203  if (!d->tooltip) {
204  d->tooltip = new DToolTip(text);
205  d->tooltip->setObjectName("AlertTooltip");
206  d->tooltip->setAccessibleName("DAlertControlAlertToolTip");
207  d->tooltip->setForegroundRole(DPalette::TextWarning);
208  d->tooltip->setWordWrap(true);
209 
210  d->frame = new DFloatingWidget;
211  d->frame->setAccessibleName("DAlertControlFloatingWidget");
212  d->frame->setFramRadius(DStyle::pixelMetric(d->target->style(), DStyle::PM_FrameRadius));
213  d->frame->setBackgroundRole(QPalette::ToolTipBase);
214  d->frame->setWidget(d->tooltip);
215  }
216 
217  if (follower) {
218  d->frame->setParent(follower->parentWidget());
219  d->follower = follower;
220  //installEventFilter(d->follower);
221  d->follower->installEventFilter(this);
222  } else {
223  d->frame->setParent(d->target->parentWidget());
224  d->follower = d->target;
225  d->follower->installEventFilter(this);
226  }
227 
228  d->tooltip->setText(text);
229  if (d->frame->parent()) {
230  d->updateTooltipPos();
231  d->frame->show();
232  d->frame->adjustSize();
233  d->frame->raise();
234  }
235  if (duration < 0)
236  return;
237 
238  d->timer.start(duration);
239 }
