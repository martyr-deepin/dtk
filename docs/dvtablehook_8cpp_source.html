40 bool DVtableHook::copyVtable(quintptr **obj)
41 {
42  int vtable_size = getVtableSize(obj);
43 
44  if (vtable_size == 0)
45  return false;
46 
47  // 多开辟一个元素, 新的虚表结构如下:
48  // 假设obj对象原虚表长度为2, 表结构为:
49  // ┏━━┳━━┳━━┓其中v1 v2为虚函数地址, 0为数组结尾
50  // ┃v1┃v2┃\0┃
51  // ┗━━┻━━┻━━┛
52  // 则新的表结构为:
53  // ┏━━┳━━┳━━┳━━┓其中前三个元素为原虚表的复制, sv为原虚表入口地址
54  // ┃v1┃v2┃\0┃sv┃
55  // ┗━━┻━━┻━━┻━━┛
56  vtable_size += 2;
57 
58  quintptr *new_vtable = new quintptr[vtable_size];
59 
60  memcpy(new_vtable, *obj, (vtable_size - 1) * sizeof(quintptr));
61 
63  objToOriginalVfptr[obj] = *obj;
64  // 存储对象原虚表入口地址
65  new_vtable[vtable_size - 1] = quintptr(*obj);
66 
67  *obj = new_vtable;
69  objToGhostVfptr[obj] = new_vtable;
70 
71  return true;
72 }

74 bool DVtableHook::clearGhostVtable(const void *obj)
75 {
76  objToOriginalVfptr.remove((quintptr**)obj);
77  objDestructFun.remove(obj);
78 
79  quintptr *vtable = objToGhostVfptr.take(obj);
80 
81  if (vtable) {
82  delete[] vtable;
83 
84  return true;
85  }
86 
87  return false;
88 }

96 int DVtableHook::getDestructFunIndex(quintptr **obj, std::function<void(void)> destoryObjFun)
97 {
98  class _DestoryProbe
99  {
100  public:
101  static quintptr probe(quintptr obj) {
102  static quintptr _obj = 0;
103 
104  if (obj == 0) {
105  obj = _obj;
106  _obj = 0;
107  } else {
108  _obj = obj;
109  }
110 
111  return obj;
112  }
113 
114  static void nothing() {
115 
116  }
117  };
118 
119  quintptr *vtable = *obj;
120  int vtable_size = getVtableSize(obj);
121 
122  if (vtable_size == 0)
123  return -1;
124 
125  quintptr *new_vtable = new quintptr[vtable_size];
126  std::fill(new_vtable, new_vtable + vtable_size, quintptr(&_DestoryProbe::nothing));
127 
128  // 给对象设置新的虚表
129  *obj = new_vtable;
130 
131  int index = -1;
132 
133  for (int i = 0; i < vtable_size; ++i) {
134  new_vtable[i] = quintptr(&_DestoryProbe::probe);
135 
136  // 尝试销毁此对象, 且观察_DestoryProbe::probe是否被调用
137  // 如果被调用, 则证明覆盖此虚函数能达到监控对象被销毁的目的
138  destoryObjFun();
139 
140  if (_DestoryProbe::probe(0) == quintptr(obj)) {
141  index = i;
142  break;
143  }
144  }
145 
146  // 恢复旧的虚表
147  *obj = vtable;
148  // 销毁临时虚表
149  delete[] new_vtable;
150 
151  return index;
152 }

154 void DVtableHook::autoCleanVtable(const void *obj)
155 {
156  quintptr fun = objDestructFun.value(obj);
157 
158  if (!fun)
159  return;
160 
161  typedef void(*Destruct)(const void*);
162  Destruct destruct = *reinterpret_cast<Destruct*>(&fun);
163  // call origin destruct function
164  destruct(obj);
165 
166  if (hasVtable(obj)) {// 需要判断一下，有可能在执行析构函数时虚表已经被删除
167  // clean
168  clearGhostVtable(obj);
169  }
170 }

172 bool DVtableHook::ensureVtable(const void *obj, std::function<void ()> destoryObjFun)
173 {
174  quintptr **_obj = (quintptr**)(obj);
175 
176  if (objToOriginalVfptr.contains(_obj)) {
177  // 不知道什么原因, 此时obj对象的虚表已经被还原
178  if (objToGhostVfptr.value((void*)obj) != *_obj) {
179  clearGhostVtable((void*)obj);
180  } else {
181  return true;
182  }
183  }
184 
185  if (!copyVtable(_obj))
186  return false;
187 
188  // 查找对象的析构函数
189  int index = getDestructFunIndex(_obj, destoryObjFun);
190 
191  // 虚析构函数查找失败
192  if (index < 0) {
193  qWarning("Failed do override destruct function");
194  qDebug() << "object:" << obj;
195  abort();
196  }
197 
198  quintptr *new_vtable = *_obj;
199  // 保存对象真实的析构函数
200  objDestructFun[(void*)obj] = new_vtable[index];
201 
202  // 覆盖析构函数, 用于在对象析构时自动清理虚表
203  new_vtable[index] = reinterpret_cast<quintptr>(&autoCleanVtable);
204 
205  return true;
206 }

220 void DVtableHook::resetVtable(const void *obj)
221 {
222  quintptr **_obj = (quintptr**)obj;
223  int vtable_size = getVtableSize(_obj);
224  // 获取obj对象原本虚表的入口
225  quintptr *vfptr_t2 = (quintptr*)(*_obj)[vtable_size + 1];
226 
227  if (!vfptr_t2)
228  return;
229 
230  if (!clearGhostVtable(obj))
231  return;
232 
233  // 还原虚表
234  *_obj = vfptr_t2;
235 }

243 quintptr DVtableHook::resetVfptrFun(const void *obj, quintptr functionOffset)
244 {
245  quintptr *vfptr_t1 = *(quintptr**)obj;
246  quintptr current_fun = *(vfptr_t1 + functionOffset / sizeof(quintptr));
247  quintptr origin_fun = originalFun(obj, functionOffset);
248 
249  if (!origin_fun) {
250  return 0;
251  }
252 
253  // reset to original fun
254  *(vfptr_t1 + functionOffset / sizeof(quintptr)) = origin_fun;
255 
256  return current_fun;
257 }

265 quintptr DVtableHook::originalFun(const void *obj, quintptr functionOffset)
266 {
267  quintptr **_obj = (quintptr**)obj;
268  int vtable_size = getVtableSize(_obj);
269  // 获取obj对象原本虚表的入口
270  quintptr *vfptr_t2 = (quintptr*)(*_obj)[vtable_size + 1];
271 
272  if (!vfptr_t2) {
273  qWarning() << "Not override the object virtual table" << obj;
274 
275  return 0;
276  }
277 
278  if (functionOffset > UINT_LEAST16_MAX) {
279  qWarning() << "Is not a virtual function, function address: 0x" << hex << functionOffset;
280 
281  return 0;
282  }
283 
284  return *(vfptr_t2 + functionOffset / sizeof(quintptr));
285 }

287 bool DVtableHook::forceWriteMemory(void *adr, const void *data, size_t length)
288 {
289 #ifdef Q_OS_LINUX
290  //int page_size = sysconf(_SC_PAGESIZE);
291  int page_size = 4096; // fix 64k crashed
292  quintptr x = reinterpret_cast<quintptr>(adr);
293  void *new_adr = reinterpret_cast<void*>((x - page_size - 1) & ~(page_size -1));
294  size_t override_data_length = length + x - reinterpret_cast<quintptr>(new_adr);
295 
296  // 失败时直接放弃
297  if (mprotect(new_adr, override_data_length, PROT_READ | PROT_WRITE)) {
298  return false;
299  }
300 #endif
301  // 复制数据
302  memcpy(adr, data, length);
303 #ifdef Q_OS_LINUX
304  // 恢复内存标志位
305  mprotect(new_adr, override_data_length, PROT_READ);
306 #endif
307 
308  return true;
309 }
