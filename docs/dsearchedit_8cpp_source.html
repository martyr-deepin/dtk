176 bool DSearchEdit::eventFilter(QObject *o, QEvent *e)
177 {
178  if (o == m_edt && e->type() == QEvent::FocusOut && m_edt->text().isEmpty()) {
179  auto fe = dynamic_cast<QFocusEvent *>(e);
180  if (fe && fe->reason() != Qt::PopupFocusReason) {
181  m_animation->stop();
182  m_animation->setStartValue(m_edt->width());
183  m_animation->setEndValue(0);
184  m_animation->setEasingCurve(m_hideCurve);
185  m_animation->start();
186 
187  connect(m_animation, &QPropertyAnimation::finished, m_placeHolder, &QLabel::show);
188  }
189  }
190 
191  if (o == m_edt) {
192  if (e->type() == QEvent::FocusOut) {
193  Q_EMIT focusOut();
194  }
195 
196  if (e->type() == QEvent::FocusIn) {
197  Q_EMIT focusIn();
198  }
199  }
200 
201  return QFrame::eventFilter(o, e);
202 }

214 void DSearchEdit::toEditMode()
215 {
216  // already in edit mode
217  if (!m_placeHolder->isVisible()) {
218  return;
219  }
220 
221  disconnect(m_animation, &QPropertyAnimation::finished, m_placeHolder, &QLabel::show);
222 
223  m_animation->stop();
224  m_animation->setStartValue(0);
225  m_animation->setEndValue(m_size.width() - m_searchBtn->width() - 6); // left + right margins = 6
226  m_animation->setEasingCurve(m_showCurve);
227  m_animation->start();
228 
229  m_placeHolder->hide();
230  m_edt->setFocus();
231 }

253 {

255  const QFocusEvent *event = static_cast<QFocusEvent*>(e);
