51 class VoiceDevice : public QIODevice
52 {
53  Q_OBJECT
54 public:
55  using QIODevice::QIODevice;
56  ~VoiceDevice()
57  {
58  if (isOpen())

62  bool open(OpenMode mode) override
63  {
64  if (mode != WriteOnly)
65  return false;
66 
67  com::iflytek::aiservice::session session("com.iflytek.aiservice", "/", QDBusConnection::sessionBus());
68  QDBusObjectPath path = session.createSession(qApp->arguments().first().split("/").last(), "iat");
69 
70  if (session.lastError().type() != QDBusError::NoError)
71  return false;
72 
73  m_iat = new com::iflytek::aiservice::iat("com.iflytek.aiservice", path.path(), QDBusConnection::sessionBus(), this);
74 
75  QVariantMap param_dict;
76  param_dict[QString("accent")] = QVariant(QString("mandarin"));
77  param_dict[QString("ptt")] = QVariant(QString("0"));
78  const QString &json_data = m_iat->startIat(param_dict);
79  QJsonDocument document = QJsonDocument::fromJson(json_data.toLocal8Bit());
80 
81  if (document.object().value("status").toInt(-1) != 0) {
82  m_iat->deleteLater();
83  m_iat = nullptr;
84  return false;
85  }
86 
87  // 清理旧的数据
88  m_message.clear();
89 
90  connect(m_iat, &ComIflytekAiserviceIatInterface::onEnd, this, &VoiceDevice::onEnd);
91  connect(m_iat, &ComIflytekAiserviceIatInterface::onResult, this, &VoiceDevice::onResult);
92 
93  return QIODevice::open(mode);
94  }
95 
96  void close() override
97  {
98  if (m_iat) {
99  m_iat->stopIat();
100  m_iat->deleteLater();
101  m_iat = nullptr;
102  }
103 
104  QIODevice::close();
105  }

119  Q_SLOT void onResult(const QString &json)
120  {
121  QJsonDocument document = QJsonDocument::fromJson(json.toLocal8Bit());
122 
123  if (!document.isObject()) {
124  return;
125  }
126 
127  document = QJsonDocument::fromJson(document["text"].toString().toLocal8Bit());
128  const QJsonArray &words = document["ws"].toArray();
129  bool replace = document["pgs"].toString() == "rpl";
130 
131  if (replace) {
132  m_message.clear();
133  }
134 
135  for (const QJsonValue &v : words) {
136  const QJsonArray &cw = v["cw"].toArray();
137 
138  for (const QJsonValue &v : cw) {
139  m_message.append(v["w"].toString());
140  }
141  }
142 
143  Q_EMIT voiceReply(m_message);
144  }
145 
146  Q_SLOT void onError(const QString &error)
147  {
148  qDebug() << error;
149  Q_EMIT voiceReply(QString());
150  }

153  {
154  close();
155  Q_EMIT finished();
156  }

393 void DSearchEditPrivate::_q_onVoiceActionTrigger(bool checked)
394 {
395 #ifdef ENABLE_AI
396  if (checked) {
397  voiceAction->setIcon(QIcon::fromTheme("button_voice_active"));
398 
399  if (!voiceInput) {
400  QAudioFormat format;
401  format.setSampleRate(16000);
402  format.setChannelCount(1);
403  format.setSampleSize(16);
404  format.setSampleType(QAudioFormat::SignedInt);
405  format.setByteOrder(QAudioFormat::LittleEndian);
406  format.setCodec("audio/pcm");
407 
408  D_Q(DSearchEdit);
409  voiceInput = new QAudioInput(format, q);
410  voiceIODevice = new VoiceDevice(voiceInput);
411 
412  q->connect(voiceIODevice, &VoiceDevice::voiceReply, q, [q, this](const QString & text) {
413  q->setText(text);
414  }, Qt::QueuedConnection);
415 
416  q->connect(voiceIODevice, &VoiceDevice::finished, q, [q, this] {
417  // 自动结束录制
418  voiceAction->setChecked(false);
419  _q_onVoiceActionTrigger(false);
420  Q_EMIT q->voiceInputFinished();
421  });
422 
423  q->connect(voiceInput, &QAudioInput::stateChanged, q, &DSearchEdit::voiceChanged);
424  }
425 
426  if (voiceIODevice->open(QIODevice::WriteOnly))
427  voiceInput->start(voiceIODevice);
428  } else {
429  voiceAction->setIcon(QIcon::fromTheme("button_voice"));
430 
431  if (voiceInput) {
432  voiceInput->stop();
433  voiceIODevice->close();
434  voiceInput->deleteLater();
435  voiceInput = nullptr;
436  }
437  }
438 #endif
439 }
