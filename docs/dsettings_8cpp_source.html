432 void DSettings::parseJson(const QByteArray &json)
433 {
434  Q_D(DSettings);
435 
436  auto jsonDoc = QJsonDocument::fromJson(json);
437  d->meta = jsonDoc.object();
438  auto mainGroups = d->meta.value("groups");
439  for (auto groupJson : mainGroups.toArray()) {
440  auto group = DSettingsGroup::fromJson("", groupJson.toObject());
441  group->setParent(this);
442  for (auto option : group->options()) {
443  d->options.insert(option->key(), option);
444  }
445  d->childGroupKeys << group->key();
446  d->childGroups.insert(group->key(), group);
447  }
448 
449  for (auto option : d->options.values()) {
450  d->options.insert(option->key(), option);
451  connect(option.data(), &DSettingsOption::valueChanged,
452  this, [ = ](QVariant value) {
453  if (d->backend) {
454  Q_EMIT d->backend->setOption(option->key(), value);
455  } else {
456  qWarning() << "backend was not setted..!";
457  }
458  Q_EMIT valueChanged(option->key(), value);
459  });
460  }
461 }

463 void DSettings::loadValue()
464 {
465  Q_D(DSettings);
466  if (!d->backend) {
467  qWarning() << "backend was not setted..!";
468  return;
469  }
470 
471  for (auto key : d->backend->keys()) {
472  auto value = d->backend->getOption(key);
473  auto opt = option(key);
474  if (!value.isValid() || opt.isNull()) {
475  continue;
476  }
477 
478  opt->blockSignals(true);
479  opt->setValue(value);
480  opt->blockSignals(false);
481  }
482 }
