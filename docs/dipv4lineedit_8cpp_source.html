124 void DIpv4LineEditPrivate::_q_updateLineEditText()
125 {
126  D_Q(DIpv4LineEdit);
127  QString text;
128 
129  for (const QLineEdit *edit : editList) {
130  text.append(".").append(edit->text());
131  }
132 
133  QObject::disconnect(q, SIGNAL(textChanged(QString)), q, SLOT(_q_setIpLineEditText(QString)));
134 
135  if (text == "....") {
136  if (!q->text().isEmpty()) {
137  q->setText("");
138  Q_EMIT q->textEdited(q->text());
139  }
140  } else {
141  text = text.mid(1);
142 
143  if (text != q->text()) {
144  q->setText(text);
145  Q_EMIT q->textEdited(q->text());
146  }
147  }
148 
149  QObject::connect(q, SIGNAL(textChanged(QString)), q, SLOT(_q_setIpLineEditText(QString)), Qt::QueuedConnection);
150 
151  q->QLineEdit::setCursorPosition(q->cursorPosition());
152 }

154 void DIpv4LineEditPrivate::_q_setIpLineEditText(const QString &text)
155 {
156  D_Q(DIpv4LineEdit);
157 
158  int pos = 0;
159 
160  if (q->validator()->validate(const_cast<QString &>(text), pos) != QValidator::Acceptable) {
161  _q_updateLineEditText();
162  return;
163  }
164 
165  const QStringList text_list = text.split('.');
166 
167  int min_count = qMin(editList.count(), text_list.count());
168 
169  for (int i = 0; i < min_count; ++i) {
170  QLineEdit *edit = editList[i];
171  bool edit_blocked = edit->blockSignals(true);
172  edit->setText(text_list[i]);
173  edit->blockSignals(edit_blocked);
174  }
175 
176  for (int i = min_count; i < editList.count(); ++i) {
177  QLineEdit *edit = editList[i];
178  bool edit_blocked = edit->blockSignals(true);
179  edit->clear();
180  edit->blockSignals(edit_blocked);
181  }
182 }

417 bool DIpv4LineEdit::eventFilter(QObject *obj, QEvent *e)
418 {
419  if (e->type() == QEvent::KeyPress) {
420  QLineEdit *edit = qobject_cast<QLineEdit *>(obj);
421 
422  if (edit) {
423  QKeyEvent *event = static_cast<QKeyEvent *>(e);
424 
425  if (event) {
426  // 按住shlft+left/right 时选中当前的edit文字，而非移动光标
427  if (event->modifiers() == Qt::ShiftModifier &&
428  (event->key() == Qt::Key_Left ||
429  event->key() == Qt::Key_Right)) {
430  return QLineEdit::eventFilter(obj, e);
431  }
432 
433  D_D(DIpv4LineEdit);
434  if (event->key() <= Qt::Key_9 && event->key() >= Qt::Key_0) {
435  if (edit->cursorPosition() == edit->text().count()) {
436  QRegularExpression rx(RX_PATTERN_IP);
437 
438  const QString number = QString::number(event->key() - Qt::Key_0);
439 
440  if (!rx.match(edit->text().append(number)).hasMatch()) {
441  int index = d->editList.indexOf(edit) + 1;
442 
443  if (index < d->editList.count()) {
444  d->editList[index]->setFocus();
445 
446  if (d->editList[index]->text().isEmpty()) {
447  d->editList[index]->setText(number);
448  d->_q_updateLineEditText();
449  }
450  }
451 
452  return true;
453  }
454  }
455  } else {
456  if (event->key() == Qt::Key_Backspace) {
457  bool accept = false;
458 
459  for (QLineEdit *edit : d->editList) {
460  if (!edit->selectedText().isEmpty()) {
461  edit->setText(edit->text().remove(edit->selectedText()));
462  d->_q_updateLineEditText();
463  accept = true;
464  }
465  }
466 
467  if (accept)
468  return true;
469  }
470 
471  if (event->key() == Qt::Key_Left
472  || (event->key() == Qt::Key_Backspace
473  && edit->cursorPosition() == 0)) {
474  setCursorPosition(cursorPosition() - 1);
475 
476  return true;
477  }
478 
479  if (event->key() == Qt::Key_Right) {
480  setCursorPosition(cursorPosition() + 1);
481 
482  return true;
483  }
484 
485  if (event->key() == Qt::Key_Period || event->key() == Qt::Key_Space) {
486  int index = d->editList.indexOf(edit) + 1;
487 
488  if (index < d->editList.count()) {
489  d->editList[index]->setFocus();
490  }
491 
492  return true;
493  }
494 
495  if (event->modifiers() == Qt::ControlModifier) {
496  if (event->key() == Qt::Key_V) {
497  QString clipboarg_text = qApp->clipboard()->text();
498  QString text = edit->text().insert(edit->cursorPosition(), clipboarg_text);
499 
500  QRegularExpression rx(RX_PATTERN_IP);
501 
502  if (rx.match(text).hasMatch()) {
503  edit->setText(text);
504  d->_q_updateLineEditText();
505  } else {
506  int pos = 0;
507 
508  if (validator()->validate(clipboarg_text, pos) == QValidator::Acceptable)
509  d->_q_setIpLineEditText(clipboarg_text);
510  }
511 
512  return true;
513  } else if (event->key() == Qt::Key_A) {
514  selectAll();
515  } else if (!selectedText().isEmpty() && event->matches(QKeySequence::Cut)) {
516  cut();
517  } else if (!selectedText().isEmpty() && event->matches(QKeySequence::Copy)) {
518  copy();
519  } else {
520  return QLineEdit::eventFilter(obj, e);
521  }
522 
523  return true;
524  }
525  }
526  }
527  }
528  } else if (e->type() == QEvent::FocusIn) {
529  QLineEdit *edit = qobject_cast<QLineEdit *>(obj);
530 
531  if (edit) {
532  // focusProxy 只能设置一个， 所以每次focusIn时设置当前这个edit为FocusProxy吧
533  setFocusProxy(edit);
534  QLineEdit::setCursorPosition(cursorPosition());
535  }
536 
537  d_func()->setFocus(true);
538 
539  } else if (e->type() == QEvent::FocusOut || e->type() == QEvent::MouseButtonPress) {
540  D_D(DIpv4LineEdit);
541 
542  bool focus = false;
543 
544  for (QLineEdit *edit : d->editList) {
545  edit->setSelection(edit->cursorPosition(), 0);
546 
547  focus = edit->hasFocus() | focus;
548  }
549 
550  if (!focus)
551  d_func()->setFocus(false);
552  }
553 
554  return QLineEdit::eventFilter(obj, e);
555 }
