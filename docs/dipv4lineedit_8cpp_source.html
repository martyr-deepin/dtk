108 void DIpv4LineEditPrivate::_q_updateLineEditText()
109 {
110  D_Q(DIpv4LineEdit);
111 
112  QString text;
113 
114  for(const QLineEdit *edit : editList) {
115  text.append(".").append(edit->text());
116  }
117 
118  QObject::disconnect(q, SIGNAL(textChanged(QString)), q, SLOT(_q_setIpLineEditText(QString)));
119 
120  if(text == "....") {
121  if(!q->text().isEmpty()) {
122  q->setText("");
123  Q_EMIT q->textEdited(q->text());
124  }
125  } else {
126  text = text.mid(1);
127 
128  if(text != q->text()) {
129  q->setText(text);
130  Q_EMIT q->textEdited(q->text());
131  }
132  }

139 void DIpv4LineEditPrivate::_q_setIpLineEditText(const QString &text)
140 {
141  D_QC(DIpv4LineEdit);
142 
143  int pos = 0;
144 
145  if(q->validator()->validate(const_cast<QString&>(text), pos) != QValidator::Acceptable) {
146  _q_updateLineEditText();
147  return;
148  }
149 
150  const QStringList text_list = text.split('.');
151 
152  int min_count = qMin(editList.count(), text_list.count());
153 
154  for(int i = 0; i < min_count; ++i ) {
155  QLineEdit *edit = editList[i];
156  bool edit_blocked = edit->blockSignals(true);
157  edit->setText(text_list[i]);
158  edit->blockSignals(edit_blocked);
159  }
160 
161  for(int i = min_count; i < editList.count(); ++i) {
162  QLineEdit *edit = editList[i];
163  bool edit_blocked = edit->blockSignals(true);
164  edit->clear();
165  edit->blockSignals(edit_blocked);
166  }

404 bool DIpv4LineEdit::eventFilter(QObject *obj, QEvent *e)
405 {
406  if(e->type() == QEvent::KeyPress) {
407  QLineEdit *edit = qobject_cast<QLineEdit*>(obj);
408 
409  if(edit) {
410  QKeyEvent *event = static_cast<QKeyEvent*>(e);
411 
412  if(event) {
413  D_D(DIpv4LineEdit);
414 
415  if(event->key() <= Qt::Key_9 && event->key() >= Qt::Key_0) {
416  if(edit->cursorPosition() == edit->text().count()) {
417  QRegularExpression rx(RX_PATTERN_IP);
418 
419  const QString number = QString::number(event->key() - Qt::Key_0);
420 
421  if(!rx.match(edit->text().append(number)).hasMatch()) {
422  int index = d->editList.indexOf(edit) + 1;
423 
424  if(index < d->editList.count()) {
425  d->editList[index]->setFocus();
426 
427  if(d->editList[index]->text().isEmpty()) {
428  d->editList[index]->setText(number);
429  d->_q_updateLineEditText();
430  }
431  }
432 
433  return true;
434  }
435  }
436  } else {
437  if(event->key() == Qt::Key_Backspace) {
438  bool accept = false;
439 
440  for(QLineEdit *edit : d->editList) {
441  if(!edit->selectedText().isEmpty()) {
442  edit->setText(edit->text().remove(edit->selectedText()));
443  d->_q_updateLineEditText();
444  accept = true;
445  }
446  }
447 
448  if(accept)
449  return true;
450  }
451 
452  if(event->key() == Qt::Key_Left
453  || (event->key() == Qt::Key_Backspace
454  && edit->cursorPosition() == 0)) {
455  setCursorPosition(cursorPosition() - 1);
456 
457  return true;
458  }
459 
460  if(event->key() == Qt::Key_Right) {
461  setCursorPosition(cursorPosition() + 1);
462 
463  return true;
464  }
465 
466  if(event->key() == Qt::Key_Period || event->key() == Qt::Key_Space) {
467  int index = d->editList.indexOf(edit) + 1;
468 
469  if(index < d->editList.count()) {
470  d->editList[index]->setFocus();
471  }
472 
473  return true;
474  }
475 
476  if(event->modifiers() == Qt::ControlModifier) {
477  if(event->key() == Qt::Key_V) {
478  QString clipboarg_text = qApp->clipboard()->text();
479  QString text = edit->text().insert(edit->cursorPosition(), clipboarg_text);
480 
481  QRegularExpression rx(RX_PATTERN_IP);
482 
483  if(rx.match(text).hasMatch()) {
484  edit->setText(text);
485  d->_q_updateLineEditText();
486  } else {
487  int pos = 0;
488 
489  if(this->validator()->validate(clipboarg_text, pos) == QValidator::Acceptable)
490  d->_q_setIpLineEditText(clipboarg_text);
491  }
492 
493  return true;
494  } else if(event->key() == Qt::Key_A) {
495  selectAll();
496  } else if(event->key() == Qt::Key_X) {
497  cut();
498  } else {
499  DLineEdit::keyPressEvent(event);
500  }
501 
502  return true;
503  }
504  }
505  }
506  }
507  } else if(e->type() == QEvent::FocusIn) {
508  QLineEdit *edit = qobject_cast<QLineEdit*>(obj);
509 
510  if(edit) {
511  DLineEdit::setCursorPosition(cursorPosition());
512  }
513 
514  d_func()->setFocus(true);
515  } else if(e->type() == QEvent::FocusOut || e->type() == QEvent::MouseButtonPress) {
516  D_D(DIpv4LineEdit);
517 
518  bool focus = false;
519 
520  for(QLineEdit *edit : d->editList) {
521  edit->setSelection(edit->cursorPosition(), 0);
522 
523  focus = edit->hasFocus() | focus;
524  }
525 
526  if(!focus)
527  d_func()->setFocus(false);
528  }
529 
530  return DLineEdit::eventFilter(obj, e);
531 }
