78 bool QSvgIOHandlerPrivate::load(QIODevice *device)
79 {
80  if (loaded)
81  return true;
82  if (q->format().isEmpty())
83  q->canRead();
84 
85  // # The SVG renderer doesn't handle trailing, unrelated data, so we must
86  // assume that all available data in the device is to be read.
87  bool res = false;
88  QBuffer *buf = qobject_cast<QBuffer *>(device);
89  if (buf) {
90  const QByteArray &ba = buf->data();
91  res = r.load(QByteArray::fromRawData(ba.constData() + buf->pos(), ba.size() - buf->pos()));
92  buf->seek(ba.size());
93  } else if (q->format() == "svgz") {
94  res = r.load(device->readAll());
95  } else {
96  res = r.load(device->readAll());
97  }
98 
99  if (res) {
100  defaultSize = QSize(r.viewBox().width(), r.viewBox().height());
101  loaded = true;
102  }
103 
104  return loaded;
105 }

121 bool QSvgIOHandler::canRead() const
122 {
123  if (!device())
124  return false;
125  if (d->loaded && !d->readDone)
126  return true; // Will happen if we have been asked for the size
127 
128  QByteArray buf = device()->peek(8);
129  if (buf.startsWith("\x1f\x8b")) {
130  setFormat("svgz");
131  return true;
132  } else if (buf.contains("<?xml") || buf.contains("<svg") || buf.contains("<!--")) {
133  setFormat("svg");
134  return true;
135  }
136  return false;
137 }

146 bool QSvgIOHandler::read(QImage *image)
147 {
148  if (d->readDone || d->load(device())) {
149  bool xform = (d->clipRect.isValid() || d->scaledSize.isValid() || d->scaledClipRect.isValid());
150  QSize finalSize = d->defaultSize;
151  QRectF bounds;
152  if (xform && !d->defaultSize.isEmpty()) {
153  bounds = QRectF(QPointF(0,0), QSizeF(d->defaultSize));
154  QPoint tr1, tr2;
155  QSizeF sc(1, 1);
156  if (d->clipRect.isValid()) {
157  tr1 = -d->clipRect.topLeft();
158  finalSize = d->clipRect.size();
159  }
160  if (d->scaledSize.isValid()) {
161  sc = QSizeF(qreal(d->scaledSize.width()) / finalSize.width(),
162  qreal(d->scaledSize.height()) / finalSize.height());
163  finalSize = d->scaledSize;
164  }
165  if (d->scaledClipRect.isValid()) {
166  tr2 = -d->scaledClipRect.topLeft();
167  finalSize = d->scaledClipRect.size();
168  }
169  QTransform t;
170  t.translate(tr2.x(), tr2.y());
171  t.scale(sc.width(), sc.height());
172  t.translate(tr1.x(), tr1.y());
173  bounds = t.mapRect(bounds);
174  }
175  if (!finalSize.isEmpty()) {
176  if (bounds.isEmpty() && d->backColor.alpha() == 0) {
177  *image = d->r.toImage(finalSize);
178  } else {
179  *image = QImage(finalSize, QImage::Format_ARGB32_Premultiplied);
180  image->fill(d->backColor.rgba());
181  QPainter p(image);
182  p.setRenderHints(QPainter::SmoothPixmapTransform);
183  d->r.render(&p, bounds);
184  p.end();
185  }
186  }
187  d->readDone = true;
188  return true;
189  }
190 
191  return false;
192 }
