29 void CommonHelper::renderFrame(QPainter *painter, const QRect &rect, const QBrush &brush, const QBrush &outline)
30 {
31  painter->setRenderHint( QPainter::Antialiasing );
32 
33  QRectF frameRect( rect );
34  qreal radius( GeometryUtils::frameRadius() );
35 
36  // set pen
37  if( outline.style() != Qt::NoBrush )
38  {
39  painter->setPen( QPen(outline, Metrics::Painter_PenWidth) );
40  } else {
41  painter->setPen( Qt::NoPen );
42  }
43 
44  // set brush
45  painter->setBrush( brush );
46 
47  // render
48  painter->drawRoundedRect( frameRect.adjusted(0.5, 0.5, -0.5, -0.5), radius, radius );
49 }

51 Style *CommonHelper::widgetStyle(const QWidget *widget)
52 {
53  if (!widget)
54  return Q_NULLPTR;
55 
56  Style *style (Q_NULLPTR);
57  style = qobject_cast<Style*>(widget->style());
58  if (!style) {
59  // FIXME(hualet): why qobject_cast won't work and cause the whole plugin
60  // to stop loading.
61  if (widget->style()->metaObject()->className() != QLatin1String("QStyleSheetStyle"))
62  return Q_NULLPTR;
63 
64  QStyleSheetStyle * sstyle = static_cast<QStyleSheetStyle*>(widget->style());
65 
66  // FIXME(hualet): don't know why QStyleSheetStyle::base can be empty sometime,
67  // and also don't know why QStyleSheetStyle::baseStyle is not exported or something,
68  // causing the program to crash if I use baseStyle() instead of base directly here.
69  // Basically below 9 lines are copied from QStyleSheetStyle::baseStyle implementation.
70  QStyle *base ( Q_NULLPTR );
71  if (sstyle->base) {
72  base = sstyle->base;
73  } else if (qApp->style()->metaObject()->className() == QLatin1String("QStyleSheetStyle")) {
74  base = static_cast<QStyleSheetStyle*>(qApp->style())->base;
75  } else {
76  base = qApp->style();
77  }
78 
79  style = qobject_cast<Style*>(base);
80  }
81  return style;
82 }
