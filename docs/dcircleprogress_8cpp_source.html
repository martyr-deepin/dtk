52 void DCircleProgressPrivate::paint(QPainter *painter)
53 {
54  Q_Q(const DCircleProgress);
55 
56  painter->setRenderHints(QPainter::Antialiasing);
57 
58  QPen pen(m_chunkColor);
59  pen.setWidth(m_lineWidth);
60  painter->setPen(pen);
61  const QRect widgetRect = q->rect();
62  QRect outerCircleRect = widgetRect;
63  outerCircleRect.setWidth(outerCircleRect.width() - (m_lineWidth - 1) * 2);
64  outerCircleRect.setHeight(outerCircleRect.height() - (m_lineWidth - 1) * 2);
65  if (outerCircleRect.width() < outerCircleRect.height())
66  outerCircleRect.setHeight(outerCircleRect.width());
67  else
68  outerCircleRect.setWidth(outerCircleRect.height());
69  outerCircleRect.setTop((widgetRect.height() - outerCircleRect.height()) / 2);
70  outerCircleRect.setLeft((widgetRect.width() - outerCircleRect.width()) / 2);
71 
72  const double percent = double(m_currentValue) / (m_maximumValue - m_minmumValue);
73  const int splitPos = -percent * 16 * 360;
74  painter->drawArc(outerCircleRect, 90 * 16, splitPos);
75  pen.setColor(m_backgroundColor);
76  painter->setPen(pen);
77  painter->drawArc(outerCircleRect, 90 * 16, 16 * 360 + splitPos);
78 }
