310  {

327  bool makeCrumb()
328  {
329  D_Q(DCrumbEdit);
330 
331  QString text = q->toPlainText().remove(QChar::ObjectReplacementCharacter);
332 
333  if (text.isEmpty())
334  return false;
335 
336  QTextCursor cursor = q->document()->find(text);
337 
338  if (cursor.isNull())
339  return false;
340 
341  if (splitter.isEmpty()) {
342  makeCrumb(cursor, text);
343 
344  return true;
345  }
346 
347  for (const QString &tag_text : text.split(splitter)) {
348  makeCrumb(cursor, tag_text);
349  }
350 
351  return true;
352  }

355  {

376

458  void _q_onTextChanged()
459  {
460  D_Q(DCrumbEdit);
461 
462  bool formatsChanged = false;
463 
464  QTextCursor cursor(q->document());
465  QSet<QString> crumbList;
466  int last_pos = 0;
467 
468  cursor.setPosition(0);
469  formatList.clear();
470 
471  while (!cursor.atEnd()) {
472  cursor.setPosition(last_pos + 1);
473 
474  if (last_pos == cursor.position())
475  break;
476 
477  last_pos = cursor.position();
478 
479  DCrumbTextFormat format(cursor.charFormat());
480 
481  if (format.objectType() == objectType) {
482  const QString &text = format.text();
483 
484  if (!text.isEmpty()) {
485  crumbList << text;
486  formatList << text;
487 
488  if (!formats.contains(text)) {
489  formats[text] = format;
490  formatsChanged = true;
491 
492  Q_EMIT q->crumbAdded(text);
493  }
494  }
495  }
496  }
497 
498  Q_FOREACH(const DCrumbTextFormat &f, formats) {
499  if (!crumbList.contains(f.text())) {
500  formats.remove(f.text());
501  formatsChanged = true;
502 
503  Q_EMIT q->crumbRemoved(f.text());
504  }
505  }
506 
507  if (formatsChanged)
508  Q_EMIT q->crumbListChanged();
509  }

543 void CrumbObjectInterface::drawObject(QPainter *painter, const QRectF &rect,
544  QTextDocument *doc, int posInDocument, const QTextFormat &format)
545 {
546  Q_UNUSED(doc)
547  Q_UNUSED(posInDocument)
548 
549  const QRect new_rect = rect.adjusted(LeftMargin, TopMargin, 0, -TopMargin).toRect();
550  const DCrumbTextFormat crumb_format(format);
551  const QFontMetricsF font_metrics(crumb_format.font());
552  const int radius = crumb_format.backgroundRadius();
553 
554  QPainterPath background_path;
555  QPainterPath tag_path;
556  const QRectF tag_rect(new_rect.x() + 2, new_rect.y() + 2, font_metrics.height() - 4, font_metrics.height() - 4);
557 
558  tag_path.addEllipse(tag_rect);
559  background_path.addRoundedRect(new_rect, radius, crumb_format.backgroundRadius());
560 
561  painter->setRenderHint(QPainter::Antialiasing);
562  painter->fillPath(background_path, backgroundBrush(new_rect, crumb_format.background()));
563 
564  if (crumb_format.tagColor().isValid()) {
565  painter->fillPath(tag_path, crumb_format.tagColor());
566 
567  painter->setPen(crumb_format.textColor());
568  painter->drawText(new_rect.adjusted(tag_rect.width() + 2, 0, -radius, 0),
569  crumb_format.text(), Qt::AlignVCenter | Qt::AlignRight);
570  } else {
571  painter->setPen(crumb_format.textColor());
572  painter->drawText(new_rect, Qt::AlignCenter, crumb_format.text());
573  }
574 }
