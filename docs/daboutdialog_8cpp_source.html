215 QPixmap DAboutDialogPrivate::loadPixmap(const QString &file)
216 {
217  D_Q(DAboutDialog);
218 
219  qreal ratio = 1.0;
220 
221  const qreal devicePixelRatio = q->devicePixelRatioF();
222 
223  QPixmap pixmap;
224 
225  if (!qFuzzyCompare(ratio, devicePixelRatio)) {
226  QImageReader reader;
227  reader.setFileName(qt_findAtNxFile(file, devicePixelRatio, &ratio));
228  if (reader.canRead()) {
229  reader.setScaledSize(reader.size() * (devicePixelRatio / ratio));
230  pixmap = QPixmap::fromImage(reader.read());
231  pixmap.setDevicePixelRatio(devicePixelRatio);
232  }
233  } else {
234  pixmap.load(file);
235  }
236 
237  return pixmap;
238 }
