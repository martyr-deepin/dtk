33 static QString getNotExistsFileName(const QString &fileName, const QString &targetPath)
34 {
35  QByteArray name = fileName.toUtf8();
36 
37  int index = name.lastIndexOf('.');
38  QByteArray suffix;
39 
40  if (index >= 0)
41  {
42  suffix = name.mid(index);
43  }
44 
45  if (suffix.size() > 200)
46  {
47  suffix = suffix.left(200);
48  }
49 
50  name.chop(suffix.size());
51  name = name.left(200 - suffix.size());
52 
53  while (QFile::exists(targetPath + "/" + name + suffix))
54  {
55  name = QCryptographicHash::hash(name, QCryptographicHash::Md5).toHex();
56  }
57 
58  return QString::fromUtf8(name + suffix);
59 }

61 static bool renameFile(const QFileInfo &fileInfo, const QString &target, QString *errorString = NULL)
62 {
63  if (fileInfo.isFile() || fileInfo.isSymLink())
64  {
65  QFile file(fileInfo.filePath());
66 
67  if (!file.rename(target))
68  {
69  if (errorString)
70  {
71  *errorString = file.errorString();
72  }
73 
74  return false;
75  }
76 
77  return true;
78  }
79  else
80  {
81  QDirIterator iterator(fileInfo.filePath(),
82  QDir::AllEntries | QDir::NoDotAndDotDot | QDir::Hidden | QDir::System);
83 
84  while (iterator.hasNext())
85  {
86  iterator.next();
87 
88  const QString newFile = iterator.filePath().replace(0, fileInfo.filePath().length(), target);
89 
90  if (!QDir().mkpath(QFileInfo(newFile).path()))
91  {
92  if (errorString)
93  {
94  *errorString = QString("Make the %1 path is failed").arg(QFileInfo(newFile).path());
95  }
96 
97  return false;
98  }
99 
100  if (!renameFile(iterator.fileInfo(), newFile, errorString))
101  {
102  return false;
103  }
104  }
105 
106  if (!QDir().rmdir(fileInfo.filePath()))
107  {
108  if (errorString)
109  {
110  *errorString = QString("Cannot remove the %1 dir").arg(fileInfo.filePath());
111  }
112 
113  return false;
114  }
115  }
116 
117  return true;
118 }
