184 bool DBoxWidget::event(QEvent *ee)
185 {
186  Q_D(const DBoxWidget);
187 
188  if(ee->type() == QEvent::LayoutRequest) {
189  if(size() != d->layout->sizeHint()) {
190  updateSize(d->layout->sizeHint());
191  updateGeometry();
192  }
193  } else if(ee->type() == QEvent::Resize) {
194  Q_EMIT sizeChanged(size());
195  } else if(ee->type() == QEvent::ChildAdded) {
196  updateSize(d->layout->sizeHint());
197  } else if(ee->type() == QEvent::ChildRemoved) {
198  updateSize(d->layout->sizeHint());
199  } else if(ee->type() == QEvent::Show) {
200  updateSize(d->layout->sizeHint());
201  }
202 
203  return QWidget::event(ee);
204 }
