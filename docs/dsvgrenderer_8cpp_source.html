117 QImage DSvgRendererPrivate::getImage(const QSize &size, const QString &elementId) const
118 {
119  if (!__rsvg->isValid())
120  return QImage();
121 
122  QImage image(size, QImage::Format_ARGB32_Premultiplied);
123 
124  image.fill(Qt::transparent);
125 
126  cairo_surface_t *surface = __rsvg->cairo_image_surface_create_for_data(image.bits(), CAIRO_FORMAT_ARGB32, image.width(), image.height(), image.bytesPerLine());
127  cairo_t *cairo = __rsvg->cairo_create(surface);
128  __rsvg->cairo_scale(cairo, image.width() / viewBox.width(), image.height() / viewBox.height());
129  __rsvg->cairo_translate(cairo, -viewBox.x(), -viewBox.y());
130 
131  if (elementId.isEmpty())
132  __rsvg->rsvg_handle_render_cairo(handle, cairo);
133  else
134  __rsvg->rsvg_handle_render_cairo_sub(handle, cairo, elementId.toUtf8().constData());
135 
136  __rsvg->cairo_destroy(cairo);
137  __rsvg->cairo_surface_destroy(surface);
138 
139  return image;
140 }

272 bool DSvgRenderer::load(const QByteArray &contents)
273 {
274  D_D(DSvgRenderer);
275 
276  if (!__rsvg->isValid())
277  return false;
278 
279  if (d->handle) {
280  __rsvg->g_object_unref(d->handle);
281  d->handle = nullptr;
282  }
283 
284  GError *error = nullptr;
285  d->handle = __rsvg->rsvg_handle_new_from_data((const guint8*)contents.constData(), contents.length(), &error);
286 
287  if (error) {
288  qWarning("DSvgRenderer::load: %s", error->message);
289  g_error_free(error);
290 
291  return false;
292  }
293 
294  RsvgDimensionData rsvg_data;
295 
296  __rsvg->rsvg_handle_get_dimensions(d->handle, &rsvg_data);
297 
298  d->defaultSize.setWidth(rsvg_data.width);
299  d->defaultSize.setHeight(rsvg_data.height);
300  d->viewBox = QRectF(QPointF(0, 0), d->defaultSize);
301 
302  return true;
303 }

315 void DSvgRenderer::render(QPainter *p, const QString &elementId, const QRectF &bounds)
316 {
317  D_D(DSvgRenderer);
318 
319  if (!d->handle)
320  return;
321 
322  p->save();
323 
324  const QImage image = d->getImage(QSize(p->device()->width(), p->device()->height()), elementId);
325 
326  if (bounds.isEmpty())
327  p->drawImage(0, 0, image);
328  else
329  p->drawImage(bounds, image);
330 
331  p->restore();
332 }
