73 class ChangeDDialog : public DDialog
74 {
75 public:
76  ChangeDDialog(QString key, KeySequenceEdit *edit, QString text = QString())
77  {
78  QPushButton *cancel = new QPushButton(qApp->translate("DSettingsDialog", "Cancel"));
79  DSuggestButton *replace = new DSuggestButton(qApp->translate("DSettingsDialog", "Replace"));
80 
81  QString str = qApp->translate("DSettingsDialog", "This shortcut conflicts with %1, click on Add to make this shortcut effective immediately")
82  .arg(QString("<span style=\"color: rgba(255, 90, 90, 1);\">%1 %2</span>").arg(text).arg(QString("[%1]").arg(edit->getKeySequence(key))));
83  setMessage(str);
84  cancel->setAccessibleName("ChangeDDialogCancelButton");
85  replace->setAccessibleName("ChangeDDialogReplaceButton");
86  insertButton(1, cancel);
87  insertButton(1, replace);
88  connect(replace, &DSuggestButton::clicked, [ = ] { //替换
89  auto value = shortcutMap.value(key);
90  value->option()->setValue(SHORTCUT_VALUE);
91  shortcutMap.remove(key);
92 
93  edit->option()->setValue(key);
94  });
95  connect(cancel, &QPushButton::clicked, [ = ] { //取消
96  cancelSettings(edit);
97  });
98  connect(this, &DDialog::closed, [ = ] {
99  cancelSettings(edit);
100  });
101  connect(this, &DDialog::rejected, [ = ] {
102  cancelSettings(edit);
103  });
104  }
105 private:
106  void cancelSettings(KeySequenceEdit *edit)
107  {
108  if (shortcutMap.key(edit).isEmpty()) { //第一次被设置
109  edit->clear();
110  } else {
111  edit->setKeySequence(edit->option()->value().toString());
112  }
113  }
114 };

194 QPair<QWidget *, QWidget *> createShortcutEditOptionHandle(DSettingsWidgetFactoryPrivate *p, QObject *opt)
195 {
196  static DSettingsWidgetFactoryPrivate *pFlg = nullptr;
197  if (pFlg != p) {
198  shortcutMap.clear();
199  pFlg = p;
200  }
201 
202  auto option = qobject_cast<DTK_CORE_NAMESPACE::DSettingsOption *>(opt);
203  auto rightWidget = new KeySequenceEdit(option);
204 
205  rightWidget->setObjectName("OptionShortcutEdit");
206  rightWidget->setAccessibleName("OptionShortcutEdit");
207  rightWidget->ShortcutDirection(Qt::AlignLeft);
208 
209  auto optionValue = option->value();
210  auto translateContext = opt->property(PRIVATE_PROPERTY_translateContext).toByteArray();
211 
212  option->connect(rightWidget, &KeySequenceEdit::editingFinished, [ = ](const QKeySequence & sequence) {
213 
214  QString keyseq = sequence.toString();
215  if (shortcutMap.value(sequence.toString()) == rightWidget) //键位于自己相同
216  return;
217 
218  if (shortcutMap.value(keyseq)) {
219  ChangeDDialog frame(keyseq, rightWidget, rightWidget->text());
220  frame.setAccessibleName("ChangeDDialog");
221  frame.exec();
222  } else {
223  shortcutMap.remove(shortcutMap.key(rightWidget));
224  shortcutMap.insert(keyseq, rightWidget);
225  option->setValue(keyseq);
226  }
227  });
228 
229  auto updateWidgetValue = [ = ](const QVariant & optionValue, DTK_CORE_NAMESPACE::DSettingsOption * opt) {
230  QKeySequence sequence(optionValue.toString());
231  QString keyseq = sequence.toString();
232 
233  if (shortcutMap.value(keyseq)) {
234  return;
235  }
236 
237  if (rightWidget->setKeySequence(sequence)) {
238  shortcutMap.insert(keyseq, rightWidget);
239  opt->setValue(keyseq);
240  }
241  };
242  updateWidgetValue(optionValue, option);
243 
244  option->connect(option, &DTK_CORE_NAMESPACE::DSettingsOption::valueChanged, rightWidget, [ = ](const QVariant & value) {
245 
246  if (value.toString() == SHORTCUT_VALUE) {
247  rightWidget->clear();
248  return;
249  }
250  QKeySequence sequence(value.toString());
251  QString keyseq = sequence.toString();
252 
253  shortcutMap.remove(shortcutMap.key(rightWidget));
254 
255  if (rightWidget->setKeySequence(sequence)) { //设置快捷键
256 
257  shortcutMap.insert(keyseq, rightWidget);
258  option->setValue(keyseq);
259  }
260  });
261 
262  return DSettingsWidgetFactory::createStandardItem(translateContext, option, rightWidget);
263 }

315 QPair<QWidget *, QWidget *> createComboBoxOptionHandle(QObject *opt)
316 {
317  auto translateContext = opt->property(PRIVATE_PROPERTY_translateContext).toByteArray();
318  auto option = qobject_cast<DTK_CORE_NAMESPACE::DSettingsOption *>(opt);
319  auto rightWidget = new ComboBox();
320  rightWidget->setFocusPolicy(Qt::StrongFocus);
321  rightWidget->setObjectName("OptionLineEdit");
322  rightWidget->setAccessibleName("OptionComboBox");
323 
324  auto initComboxList = [ = ](const QStringList & data) {
325  for (auto item : data) {
326  auto trName = DWIDGET_NAMESPACE::tr(translateContext, item.toStdString().c_str());
327  rightWidget->addItem(trName);
328  }
329  auto current = option->value().toInt();
330  rightWidget->setCurrentIndex(current);
331  option->connect(rightWidget, static_cast<void (QComboBox::*)(int index)>(&QComboBox::currentIndexChanged),
332  option, [ = ](int index) {
333  option->setValue(index);
334  });
335 
336  option->connect(option, &DTK_CORE_NAMESPACE::DSettingsOption::valueChanged,
337  rightWidget, [ = ](const QVariant & value) {
338  rightWidget->setCurrentIndex(value.toInt());
339  });
340  };
341 
342  auto initComboxMap = [ = ](const QMap<QString, QVariant> &data) {
343  auto keys = data.value("keys").toStringList();
344  auto values = data.value("values").toStringList();
345 
346  for (int i = 0; i < keys.length(); ++i) {
347  auto trName = DWIDGET_NAMESPACE::tr(translateContext, values.value(i).toStdString().c_str());
348  rightWidget->addItem(trName, keys.value(i));
349  }
350 
351  auto currentData = option->value().toString();
352  auto currentIndex = rightWidget->findData(currentData);
353  rightWidget->setCurrentIndex(currentIndex);
354 
355  option->connect(rightWidget, static_cast<void (QComboBox::*)(int index)>(&QComboBox::currentIndexChanged),
356  option, [ = ](int index) {
357  option->setValue(keys.value(index));
358  });
359 
360  option->connect(option, &DTK_CORE_NAMESPACE::DSettingsOption::valueChanged,
361  rightWidget, [ = ](const QVariant & value) {
362  auto currentIndex = rightWidget->findData(value.toString());
363  rightWidget->setCurrentIndex(currentIndex);
364  });
365  };
366 
367  auto updateData = [ = ](const QString & dataType, const QVariant & data) {
368  if ("items" != dataType) {
369  return;
370  }
371 
372  rightWidget->clear();
373  if (data.type() == QVariant::StringList) {
374  initComboxList(data.toStringList());
375  }
376  if (data.type() == QVariant::Map) {
377  initComboxMap(data.toMap());
378  }
379  rightWidget->update();
380  };
381 
382  auto initData = option->data("items");
383  updateData("items", initData);
384 
385  option->connect(option, &DTK_CORE_NAMESPACE::DSettingsOption::dataChanged,
386  rightWidget, [ = ](const QString & dataType, const QVariant & value) {
387  updateData(dataType, value);
388  });
389 
390  return DSettingsWidgetFactory::createStandardItem(translateContext, option, rightWidget);
391 }
