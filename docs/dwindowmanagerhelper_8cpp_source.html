697 QList<DForeignWindow *> DWindowManagerHelper::currentWorkspaceWindows() const
698 {
699  D_DC(DWindowManagerHelper);
700 
701  for (QWindow *w : d->windowList) {
702  w->deleteLater();
703  }
704 
705  d->windowList.clear();
706 
707  QList<WId> currentApplicationWindowList;
708  const QWindowList &list = qApp->allWindows();
709 
710  currentApplicationWindowList.reserve(list.size());
711 
712  for (auto window : list) {
713  if (window->property("_q_foreignWinId").isValid()) continue;
714 
715  currentApplicationWindowList.append(window->winId());
716  }
717 
718  QVector<quint32> wmClientList = currentWorkspaceWindowIdList();
719 
720  for (WId wid : wmClientList) {
721  if (currentApplicationWindowList.contains(wid))
722  continue;
723 
724  if (DForeignWindow *w = DForeignWindow::fromWinId(wid)) {
725  d->windowList << w;
726  }
727  }
728 
729  return d->windowList;
730 }
