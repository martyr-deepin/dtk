32 bool Style::drawToolButtonLabelControl( const QStyleOption* option, QPainter* painter, const QWidget* widget ) const
33 {
34  if (isTabBarToolButton(widget)) {
35  return drawScrollButtonLabelControl(option, painter, widget);
36  }
37 
38  // cast option and check
39  const QStyleOptionToolButton* toolButtonOption( qstyleoption_cast<const QStyleOptionToolButton*>(option) );
40 
41  // copy rect and palette
42  const QRect& rect = option->rect;
43  const QPalette& palette = option->palette;
44 
45  // state
46  const State& state( option->state );
47  const bool enabled( state & State_Enabled );
48  const bool sunken( state & (State_On | State_Sunken) );
49  const bool mouseOver( enabled && (option->state & State_MouseOver) );
50  const bool flat( state & State_AutoRaise );
51 
52  // focus flag is set to match the background color in either renderButtonFrame or renderToolButtonFrame
53  bool hasFocus( false );
54  if( flat ) hasFocus = enabled && !mouseOver && (option->state & State_HasFocus);
55  else hasFocus = enabled && !mouseOver && (option->state & (State_HasFocus|State_Sunken) );
56 
57  const bool hasArrow( toolButtonOption->features & QStyleOptionToolButton::Arrow );
58  const bool hasIcon( !( hasArrow || toolButtonOption->icon.isNull() ) );
59  const bool hasText( !toolButtonOption->text.isEmpty() );
60 
61  // contents
62  QRect contentsRect( rect );
63  if( sunken && !flat ) contentsRect.translate( 1, 1 );
64 
65  // icon size
66  const QSize iconSize( toolButtonOption->iconSize );
67 
68  // text size
69  int textFlags( Qt::TextHideMnemonic );
70  const QSize textSize( option->fontMetrics.size( textFlags, toolButtonOption->text ) );
71 
72  // adjust text and icon rect based on options
73  QRect iconRect;
74  QRect textRect;
75 
76  if( hasText && ( !(hasArrow||hasIcon) || toolButtonOption->toolButtonStyle == Qt::ToolButtonTextOnly ) )
77  {
78 
79  // text only
80  textRect = contentsRect;
81  textFlags |= Qt::AlignCenter;
82 
83  } else if( (hasArrow||hasIcon) && (!hasText || toolButtonOption->toolButtonStyle == Qt::ToolButtonIconOnly ) ) {
84 
85  // icon only
86  iconRect = contentsRect;
87 
88  } else if( toolButtonOption->toolButtonStyle == Qt::ToolButtonTextUnderIcon ) {
89 
90  const int contentsHeight( iconSize.height() + textSize.height() + Metrics::ToolButton_ItemSpacing );
91  iconRect = QRect( QPoint( contentsRect.left() + (contentsRect.width() - iconSize.width())/2, contentsRect.top() + (contentsRect.height() - contentsHeight)/2 ), iconSize );
92  textRect = QRect( QPoint( contentsRect.left() + (contentsRect.width() - textSize.width())/2, iconRect.bottom() + Metrics::ToolButton_ItemSpacing + 1 ), textSize );
93  textFlags |= Qt::AlignCenter;
94 
95  } else {
96  const int contentsWidth( iconSize.width() + textSize.width() + Metrics::ToolButton_ItemSpacing );
97  iconRect = QRect( QPoint( contentsRect.left() + (contentsRect.width() - contentsWidth )/2, contentsRect.top() + (contentsRect.height() - iconSize.height())/2 ), iconSize );
98 
99  textRect = QRect( QPoint( iconRect.right() + Metrics::ToolButton_ItemSpacing + 1, contentsRect.top() + (contentsRect.height() - textSize.height())/2 ), textSize );
100 
101  // handle right to left layouts
102  iconRect = visualRect( option->direction, option->rect, iconRect );
103  textRect = visualRect( option->direction, option->rect, textRect );
104 
105  textFlags |= Qt::AlignLeft | Qt::AlignVCenter;
106 
107  }
108 
109  // make sure there is enough room for icon
110  if( iconRect.isValid() ) iconRect = centerRect( iconRect, iconSize );
111 
112  // render arrow or icon
113  if( hasArrow && iconRect.isValid() )
114  {
115 
116  QStyleOptionToolButton copy( *toolButtonOption );
117  copy.rect = iconRect;
118  switch( toolButtonOption->arrowType )
119  {
120  case Qt::LeftArrow: drawPrimitive( PE_IndicatorArrowLeft, &copy, painter, widget ); break;
121  case Qt::RightArrow: drawPrimitive( PE_IndicatorArrowRight, &copy, painter, widget ); break;
122  case Qt::UpArrow: drawPrimitive( PE_IndicatorArrowUp, &copy, painter, widget ); break;
123  case Qt::DownArrow: drawPrimitive( PE_IndicatorArrowDown, &copy, painter, widget ); break;
124  default: break;
125  }
126 
127  } else if( hasIcon && iconRect.isValid() ) {
128 
129  // icon state and mode
130  const QIcon::State iconState( sunken ? QIcon::On : QIcon::Off );
131  QIcon::Mode iconMode;
132  if( !enabled ) iconMode = QIcon::Disabled;
133  else if( !flat && hasFocus ) iconMode = QIcon::Selected;
134  else if( mouseOver && flat ) iconMode = QIcon::Active;
135  else iconMode = QIcon::Normal;
136 
137  const QPixmap pixmap = toolButtonOption->icon.pixmap( iconSize, iconMode, iconState );
138  drawItemPixmap( painter, iconRect, Qt::AlignCenter, pixmap );
139 
140  }
141 
142  // render text
143  if( hasText && textRect.isValid() )
144  {
145 
146  QPalette::ColorRole textRole( QPalette::ButtonText );
147  if( flat ) textRole = (hasFocus&&sunken&&!mouseOver) ? QPalette::HighlightedText: QPalette::WindowText;
148  else if( mouseOver ) textRole = QPalette::HighlightedText;
149 
150  painter->setFont(toolButtonOption->font);
151  drawItemText( painter, textRect, textFlags, palette, enabled, toolButtonOption->text, textRole );
152  }
153 
154  return true;
155 }
