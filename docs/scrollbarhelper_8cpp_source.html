31 QRect Style::scrollbarSubControlRect(const QStyleOptionComplex *opt, QStyle::SubControl sc, const QWidget *widget) const
32 {
33  QRect ret(0, 0, 0, 0);
34 
35  if (const QStyleOptionSlider *scrollbar = qstyleoption_cast<const QStyleOptionSlider *>(opt)) {
36  const bool enabled(opt->state & QStyle::State_Enabled);
37  const bool mouseOver(opt->state & QStyle::State_MouseOver);
38  const QRect scrollBarRect = scrollbar->rect;
39  int sbextent = 0;
40  if (!proxy()->styleHint(SH_ScrollBar_Transient, scrollbar, widget))
41  sbextent = proxy()->pixelMetric(PM_ScrollBarExtent, scrollbar, widget);
42  int maxlen = ((scrollbar->orientation == Qt::Horizontal) ?
43  scrollBarRect.width() : scrollBarRect.height());
44  int sliderlen;
45 
46  // calculate slider length
47  if (scrollbar->maximum != scrollbar->minimum) {
48  uint range = scrollbar->maximum - scrollbar->minimum;
49  sliderlen = (qint64(scrollbar->pageStep) * maxlen) / (range + scrollbar->pageStep);
50 
51  int slidermin = proxy()->pixelMetric(PM_ScrollBarSliderMin, scrollbar, widget);
52  if (sliderlen < slidermin || range > INT_MAX / 2)
53  sliderlen = slidermin;
54  if (sliderlen > maxlen)
55  sliderlen = maxlen;
56  } else {
57  sliderlen = maxlen;
58  }
59 
60  int sliderstart = sliderPositionFromValue(scrollbar->minimum,
61  scrollbar->maximum,
62  scrollbar->sliderPosition,
63  maxlen - sliderlen,
64  scrollbar->upsideDown);
65 
66  switch (sc) {
67  case SC_ScrollBarSubLine: // top/left button
68  if (scrollbar->orientation == Qt::Horizontal) {
69  int buttonWidth = qMin(scrollBarRect.width() / 2, sbextent);
70  ret.setRect(0, 0, buttonWidth, scrollBarRect.height());
71  } else {
72  int buttonHeight = qMin(scrollBarRect.height() / 2, sbextent);
73  ret.setRect(0, 0, scrollBarRect.width(), buttonHeight);
74  }
75  break;
76  case SC_ScrollBarAddLine: // bottom/right button
77  if (scrollbar->orientation == Qt::Horizontal) {
78  int buttonWidth = qMin(scrollBarRect.width()/2, sbextent);
79  ret.setRect(scrollBarRect.width() - buttonWidth, 0, buttonWidth, scrollBarRect.height());
80  } else {
81  int buttonHeight = qMin(scrollBarRect.height()/2, sbextent);
82  ret.setRect(0, scrollBarRect.height() - buttonHeight, scrollBarRect.width(), buttonHeight);
83  }
84  break;
85  case SC_ScrollBarSubPage: // between top/left button and slider
86  if (scrollbar->orientation == Qt::Horizontal)
87  ret.setRect(sbextent, 0, sliderstart - sbextent, scrollBarRect.height());
88  else
89  ret.setRect(0, sbextent, scrollBarRect.width(), sliderstart - sbextent);
90  break;
91  case SC_ScrollBarAddPage: // between bottom/right button and slider
92  if (scrollbar->orientation == Qt::Horizontal)
93  ret.setRect(sliderstart + sliderlen, 0,
94  maxlen - sliderstart - sliderlen + sbextent, scrollBarRect.height());
95  else
96  ret.setRect(0, sliderstart + sliderlen, scrollBarRect.width(),
97  maxlen - sliderstart - sliderlen + sbextent);
98  break;
99  case SC_ScrollBarGroove:
100  if (scrollbar->orientation == Qt::Horizontal)
101  ret.setRect(sbextent, 0, scrollBarRect.width() - sbextent * 2,
102  scrollBarRect.height());
103  else
104  ret.setRect(0, sbextent, scrollBarRect.width(),
105  scrollBarRect.height() - sbextent * 2);
106  break;
107  case SC_ScrollBarSlider:
108  if (scrollbar->orientation == Qt::Horizontal) {
109  if (mouseOver && enabled)
110  ret.setRect(sliderstart, 2, sliderlen, scrollBarRect.height() - 4);
111  else
112  ret.setRect(sliderstart, scrollBarRect.height() / 3, sliderlen, scrollBarRect.height() / 3);
113  } else {
114  if (mouseOver && enabled)
115  ret.setRect(2, sliderstart, scrollBarRect.width() - 4, sliderlen);
116  else
117  ret.setRect(scrollBarRect.width() / 3, sliderstart, scrollBarRect.width() / 3, sliderlen);
118  }
119  break;
120  default:
121  break;
122  }
123  ret = visualRect(scrollbar->direction, scrollBarRect, ret);
124  }
125 
126  return ret;
127 }

129 bool Style::drawScrollBarSliderControl(const QStyleOption *option, QPainter *painter, const QWidget *widget) const
130 {
131  const QStyleOptionSlider *scrollBar = qstyleoption_cast<const QStyleOptionSlider *>(option);
132  QObject *styleObject = option->styleObject;
133 
134  bool wasActive = false;
135 
136  if (styleObject && styleObject && proxy()->styleHint(SH_ScrollBar_Transient, option, widget)) {
137  Q_D(const Style);
138 
139  qreal opacity = 0.0;
140  bool shouldExpand = false;
141 
142  int oldPos = styleObject->property("_q_stylepos").toInt();
143  int oldMin = styleObject->property("_q_stylemin").toInt();
144  int oldMax = styleObject->property("_q_stylemax").toInt();
145  QRect oldRect = styleObject->property("_q_stylerect").toRect();
146  int oldState = styleObject->property("_q_stylestate").toInt();
147  uint oldActiveControls = styleObject->property("_q_stylecontrols").toUInt();
148 
149  // a scrollbar is transient when the the scrollbar itself and
150  // its sibling are both inactive (ie. not pressed/hovered/moved)
151  bool transient = !scrollBar->activeSubControls && !(option->state & State_On);
152 
153  if (!transient ||
154  oldPos != scrollBar->sliderPosition ||
155  oldMin != scrollBar->minimum ||
156  oldMax != scrollBar->maximum ||
157  oldRect != scrollBar->rect ||
158  oldState != static_cast<int>(scrollBar->state) ||
159  oldActiveControls != scrollBar->activeSubControls) {
160 
161  // if the scrollbar is transient or its attributes, geometry or
162  // state has changed, the opacity is reset back to 100% opaque
163  opacity = 1.0;
164 
165  styleObject->setProperty("_q_stylepos", scrollBar->sliderPosition);
166  styleObject->setProperty("_q_stylemin", scrollBar->minimum);
167  styleObject->setProperty("_q_stylemax", scrollBar->maximum);
168  styleObject->setProperty("_q_stylerect", scrollBar->rect);
169  styleObject->setProperty("_q_stylestate", static_cast<int>(scrollBar->state));
170  styleObject->setProperty("_q_stylecontrols", static_cast<uint>(scrollBar->activeSubControls));
171 
172 #ifndef QT_NO_ANIMATION
173  DScrollbarStyleAnimation *anim = qobject_cast<DScrollbarStyleAnimation *>(d->animation(styleObject));
174  if (transient) {
175  if (!anim) {
176  anim = new DScrollbarStyleAnimation(DScrollbarStyleAnimation::Deactivating, styleObject);
177  d->startAnimation(anim);
178  } else if (anim->mode() == DScrollbarStyleAnimation::Deactivating) {
179  // the scrollbar was already fading out while the
180  // state changed -> restart the fade out animation
181  anim->setCurrentTime(0);
182  }
183  } else if (anim && anim->mode() == DScrollbarStyleAnimation::Deactivating) {
184  d->stopAnimation(styleObject);
185  }
186 #endif // !QT_NO_ANIMATION
187  }
188 
189 #ifndef QT_NO_ANIMATION
190  DScrollbarStyleAnimation *anim = qobject_cast<DScrollbarStyleAnimation *>(d->animation(styleObject));
191 
192  if (anim && anim->mode() == DScrollbarStyleAnimation::Deactivating) {
193  // once a scrollbar was active (hovered/pressed), it retains
194  // the active look even if it's no longer active while fading out
195  if (oldActiveControls)
196  anim->setActive(true);
197 
198  wasActive = anim->wasActive();
199  opacity = anim->currentValue();
200  }
201 
202  shouldExpand = (scrollBar->activeSubControls || wasActive);
203  if (shouldExpand) {
204  if (!anim && !oldActiveControls) {
205  // Start expand animation only once and when entering
206  anim = new DScrollbarStyleAnimation(DScrollbarStyleAnimation::Activating, styleObject);
207  d->startAnimation(anim);
208  }
209  }
210  painter->setOpacity(opacity);
211 #endif // !QT_NO_ANIMATION
212  }
213 
214  if (qFuzzyIsNull(painter->opacity()))
215  return true;
216 
217  painter->setRenderHint(QPainter::Antialiasing);
218 
219  QRectF rect = option->rect;
220 
221  const QBrush &background = m_palette->brush(PaletteExtended::ScrollBar_HandleBrush, option, 0, Qt::lightGray);
222  const QBrush &border = m_palette->brush(PaletteExtended::ScrollBar_HandleBorderBrush, option, 0, Qt::transparent);
223  qreal radius = scrollBar->orientation == Qt::Horizontal ? rect.height() / 2.0 : rect.width() / 2.0;
224  PainterHelper::drawRoundedRect(painter, rect, radius, radius, Qt::AbsoluteSize, background, Metrics::Painter_PenWidth, border);
225 
226  return true;
227 }
