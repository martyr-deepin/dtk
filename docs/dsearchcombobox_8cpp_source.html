61 void DSearchComboBox::showPopup()
62 {
63  Q_D(DSearchComboBox);
64  QComboBoxPrivate *dd = reinterpret_cast<QComboBoxPrivate *>(qGetPtrHelper(d_ptr));
65  QComboBox::showPopup();
66 
67  if(auto layout = static_cast<QBoxLayout*>(dd->container->layout())) {
68  if (!isEditable() && !d->searchEdit) {
69  //因为dd->container容器在QComboBox::showPopup()
70  //中才被new 所以布局等在此实现
71  d->searchEdit = new DSearchEdit;
72  d->searchEdit->setFixedHeight(d->searchEdit->sizeHint().height());
73 
74  QHBoxLayout *searlayout = new QHBoxLayout;
75  searlayout->setContentsMargins(layoutMargin, layoutMargin, layoutMargin, 0);
76  layout->setMargin(0);
77  layout->setSpacing(0);
78  searlayout->addWidget(d->searchEdit);
79  layout->insertLayout(0, searlayout);
80 
81  d->proxyModel = new QSortFilterProxyModel;
82  d->proxyModel->setSourceModel(model());
83 
84  //Qt源码中modle的父对象为this就会delete
85  view()->model()->setParent(view());
86  setModel(d->proxyModel);
87 
88  connect(d->searchEdit, &DSearchEdit::textChanged, this, [ = ] (const QString & text){
89  d->proxyModel->setFilterRegExp(QRegExp(text, Qt::CaseInsensitive,
90  QRegExp::FixedString));
91  });
92  }
93  }
94 
95  if (d->searchEdit &&!isEditable()) {
96  //可编辑状态隐藏DSearchEdit
97  if (d->searchEdit->isVisible() != isEditable())
98  d->searchEdit->setVisible(!isEditable());
99  d->height = qMax(dd->container->geometry().height() + layoutMargin, d->height);
100  QRect containerRect = dd->container->geometry();
101  containerRect.setHeight(d->height);
102  dd->container->setGeometry(containerRect);
103  }
104 }
