562 class Q_DECL_HIDDEN CreatorWindowEventFile : public QObject {
563 public:
564  CreatorWindowEventFile(QObject *par= nullptr): QObject(par){}
565 
566 public:
567  bool eventFilter(QObject *watched, QEvent *event) override {
568  if (event->type() == QEvent::PlatformSurface) {
569  QPlatformSurfaceEvent *se = static_cast<QPlatformSurfaceEvent*>(event);
570  if (se->surfaceEventType() == QPlatformSurfaceEvent::SurfaceCreated) { // 若收到此信号， 则 WinID 已被创建
571  initWindowRadius(qobject_cast<QWindow *>(watched));
572  deleteLater();
573  }
574  }
575 
576  return QObject::eventFilter(watched, event);
577  }

637 inline DPlatformHandle::WMBlurArea operator *(const DPlatformHandle::WMBlurArea &area, qreal scale)
638 {

641

696 bool DPlatformHandle::setWindowBlurAreaByWM(QWindow *window, const QVector<DPlatformHandle::WMBlurArea> &area)
697 {
698  if (!window) {
699  return false;
700  }
701 
702  if (isEnabledDXcb(window)) {
703  setWindowProperty(window, _windowBlurAreas, QVariant::fromValue(*(reinterpret_cast<const QVector<quint32>*>(&area))));
704 
705  return true;
706  }
707 
708  QFunctionPointer setWmBlurWindowBackgroundArea = Q_NULLPTR;
709 
710 #if QT_VERSION >= QT_VERSION_CHECK(5, 4, 0)
711  setWmBlurWindowBackgroundArea = qApp->platformFunction(_setWmBlurWindowBackgroundArea);
712 #endif
713 
714  if (!setWmBlurWindowBackgroundArea) {
715  qWarning("setWindowBlurAreaByWM is not support");
716 
717  return false;
718  }
719 
720  QSurfaceFormat format = window->format();
721 
722  format.setAlphaBufferSize(8);
723  window->setFormat(format);
724 
725  const qreal device_ratio = window->devicePixelRatio();
726 
727  if (qFuzzyCompare(device_ratio, 1.0)) {
728  return reinterpret_cast<bool(*)(const quint32, const QVector<WMBlurArea>&)>(setWmBlurWindowBackgroundArea)(window->winId(), area);
729  }
730 
731  QVector<WMBlurArea> new_areas;
732 
733  new_areas.reserve(area.size());
734 
735  for (const WMBlurArea &a : area) {
736  new_areas.append(a * device_ratio);
737  }
738 
739  return reinterpret_cast<bool(*)(const quint32, const QVector<WMBlurArea>&)>(setWmBlurWindowBackgroundArea)(window->winId(), new_areas);
740 }

742 inline QPainterPath operator *(const QPainterPath &path, qreal scale)
743 {
744  if (qFuzzyCompare(1.0, scale))
745  return path;
746 
747  QPainterPath new_path = path;
748 
749  for (int i = 0; i < path.elementCount(); ++i) {
750  const QPainterPath::Element &e = path.elementAt(i);
751 
752  new_path.setElementPositionAt(i, qRound(e.x * scale), qRound(e.y * scale));
753  }
754 
755  return new_path;

885 bool DPlatformHandle::setWindowWallpaperParaByWM(QWindow *window, const QRect &area, WallpaperScaleMode sMode, WallpaperFillMode fMode)
886 {
887  if (!window) {
888  return false;

890 
891  QFunctionPointer setWmWallpaperParameter = Q_NULLPTR;
892 
893 #if QT_VERSION >= QT_VERSION_CHECK(5, 4, 0)
894  setWmWallpaperParameter = qApp->platformFunction(_setWmWallpaperParameter);
895 #endif
896 
897  if (!setWmWallpaperParameter) {
898  qWarning("setWindowWallpaperParaByWM is not support");
899 
900  return false;
901  }
902 
903  QSurfaceFormat format = window->format();
904 
905  format.setAlphaBufferSize(8);
906  window->setFormat(format);
907 
908  quint32 bMode = sMode | fMode;
909

911  window->setProperty("_d_dxcb_wallpaper", QVariant::fromValue(QPair<QRect, int>(area, bMode)));
912 
913  if (!DWindowManagerHelper::instance()->hasComposite() && !window->handle()) {
914  return true;
915  } else {
916  qWarning() << "because the window handle has been created, so 2D mode will have no effect";
917  }
918 
919  const qreal device_ratio = window->devicePixelRatio();
920  if (qFuzzyCompare(device_ratio, 1.0) || !area.isValid()) {
921  return reinterpret_cast<bool(*)(const quint32, const QRect&, const quint32)>(setWmWallpaperParameter)(window->winId(), area, bMode);
922  }
923 
924  QRect new_area(area.x() * device_ratio,
925  area.y() * device_ratio,
926  area.width() * device_ratio,
927  area.height() * device_ratio);
928 
929  return reinterpret_cast<bool(*)(const quint32, const QRect&, const quint32)>(setWmWallpaperParameter)(window->winId(), new_area, bMode);

1160 bool DPlatformHandle::eventFilter(QObject *obj, QEvent *event)
1161 {
1162  if (obj == m_window) {
1163  if (event->type() == QEvent::DynamicPropertyChange) {
1164  QDynamicPropertyChangeEvent *e = static_cast<QDynamicPropertyChangeEvent *>(event);
1165 
1166  if (e->propertyName() == _windowRadius) {
1167  Q_EMIT windowRadiusChanged();
1168  } else if (e->propertyName() == _borderWidth) {
1169  Q_EMIT borderWidthChanged();
1170  } else if (e->propertyName() == _borderColor) {
1171  Q_EMIT borderColorChanged();
1172  } else if (e->propertyName() == _shadowRadius) {
1173  Q_EMIT shadowRadiusChanged();
1174  } else if (e->propertyName() == _shadowOffset) {
1175  Q_EMIT shadowOffsetChanged();
1176  } else if (e->propertyName() == _shadowColor) {
1177  Q_EMIT shadowColorChanged();
1178  } else if (e->propertyName() == _clipPath) {
1179  Q_EMIT clipPathChanged();
1180  } else if (e->propertyName() == _frameMask) {
1181  Q_EMIT frameMaskChanged();
1182  } else if (e->propertyName() == _frameMargins) {
1183  Q_EMIT frameMarginsChanged();
1184  } else if (e->propertyName() == _translucentBackground) {
1185  Q_EMIT translucentBackgroundChanged();
1186  } else if (e->propertyName() == _enableSystemResize) {
1187  Q_EMIT enableSystemResizeChanged();
1188  } else if (e->propertyName() == _enableSystemMove) {
1189  Q_EMIT enableSystemMoveChanged();
1190  } else if (e->propertyName() == _enableBlurWindow) {
1191  Q_EMIT enableBlurWindowChanged();
1192  } else if (e->propertyName() == _autoInputMaskByClipPath) {
1193  Q_EMIT autoInputMaskByClipPathChanged();
1194  }
1195  }
1196  }
1197 
1198  return false;
1199 }
