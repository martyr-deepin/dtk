121 static QImage dropShadow(const QPixmap &px, qreal radius, const QColor &color)
122 {
123  if (px.isNull())
124  return QImage();
125 
126  QImage tmp(px.size() + QSize(radius * 2, radius * 2), QImage::Format_ARGB32_Premultiplied);
127  tmp.fill(0);
128  QPainter tmpPainter(&tmp);
129  tmpPainter.setCompositionMode(QPainter::CompositionMode_Source);
130  tmpPainter.drawPixmap(QPoint(radius, radius), px);
131  tmpPainter.end();
132 
133  // blur the alpha channel
134  QImage blurred(tmp.size(), QImage::Format_ARGB32_Premultiplied);
135  blurred.fill(0);
136  QPainter blurPainter(&blurred);
137  qt_blurImage(&blurPainter, tmp, radius, false, true);
138  blurPainter.end();
139 
140  if (color == QColor(Qt::black))
141  return blurred;
142 
143  tmp = blurred;
144 
145  // blacken the image...
146  tmpPainter.begin(&tmp);
147  tmpPainter.setCompositionMode(QPainter::CompositionMode_SourceIn);
148  tmpPainter.fillRect(tmp.rect(), color);
149  tmpPainter.end();
150 
151  return tmp;
152 }

286 void drawRoundedRect(QPainter *pa, const QRect &rect, qreal xRadius, qreal yRadius, Corners corners, Qt::SizeMode mode)
287 {
288  QRectF r = rect.normalized();
289 
290  if (r.isNull())
291  return;
292 
293  if (mode == Qt::AbsoluteSize) {
294  qreal w = r.width() / 2;
295  qreal h = r.height() / 2;
296 
297  if (w == 0) {
298  xRadius = 0;
299  } else {
300  xRadius = 100 * qMin(xRadius, w) / w;
301  }
302  if (h == 0) {
303  yRadius = 0;
304  } else {
305  yRadius = 100 * qMin(yRadius, h) / h;
306  }
307  } else {
308  if (xRadius > 100) // fix ranges
309  xRadius = 100;
310 
311  if (yRadius > 100)
312  yRadius = 100;
313  }
314 
315  if (xRadius <= 0 || yRadius <= 0) { // add normal rectangle
316  pa->drawRect(r);
317  return;
318  }
319 
320  QPainterPath path;
321  qreal x = r.x();
322  qreal y = r.y();
323  qreal w = r.width();
324  qreal h = r.height();
325  qreal rxx2 = w * xRadius / 100;
326  qreal ryy2 = h * yRadius / 100;
327 
328  path.arcMoveTo(x, y, rxx2, ryy2, 180);
329 
330  if (corners & TopLeftCorner) {
331  path.arcTo(x, y, rxx2, ryy2, 180, -90);
332  } else {
333  path.lineTo(r.topLeft());
334  }
335 
336  if (corners & TopRightCorner) {
337  path.arcTo(x + w - rxx2, y, rxx2, ryy2, 90, -90);
338  } else {
339  path.lineTo(r.topRight());
340  }
341 
342  if (corners & BottomRightCorner) {
343  path.arcTo(x + w - rxx2, y + h - ryy2, rxx2, ryy2, 0, -90);
344  } else {
345  path.lineTo(r.bottomRight());
346  }
347 
348  if (corners & BottomLeftCorner) {
349  path.arcTo(x, y + h - ryy2, rxx2, ryy2, 270, -90);
350  } else {
351  path.lineTo(r.bottomLeft());
352  }
353 
354  path.closeSubpath();
355  pa->drawPath(path);
356 }
