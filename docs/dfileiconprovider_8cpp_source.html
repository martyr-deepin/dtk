105 QIcon DFileIconProviderPrivate::getFilesystemIcon(const QFileInfo &info) const
106 {
107 #ifdef USE_GTK_PLUS_2_0
108  if (gnome_vfs_init && gnome_icon_lookup_sync && gtk_icon_theme_get_default) {
109  gnome_vfs_init();
110  GtkIconTheme *theme = gtk_icon_theme_get_default();
111  QByteArray fileurl = QUrl::fromLocalFile(info.absoluteFilePath()).toEncoded();
112  char *icon_name = gnome_icon_lookup_sync(theme,
113  NULL,
114  fileurl.data(),
115  NULL,
116  GNOME_ICON_LOOKUP_FLAGS_NONE,
117  NULL);
118  QString iconName = QString::fromUtf8(icon_name);
119  g_free(icon_name);
120  if (iconName.startsWith(QLatin1Char('/'))) {
121  return QIcon(iconName);
122  }
123  return fromTheme(iconName);
124  }
125 #endif
126 
127  const QMimeType &db = QMimeDatabase().mimeTypeForFile(info);
128  const QIcon &icon = fromTheme(db.iconName());
129 
130  if (!icon.isNull()) {
131  return icon;
132  }
133 
134  return fromTheme(db.genericIconName());
135 }

137 QIcon DFileIconProviderPrivate::fromTheme(QString iconName) const
138 {
139  QIcon icon = QIcon::fromTheme(iconName);
140 
141  if (Q_LIKELY(!icon.isNull())) {
142  return icon;
143  }
144 
145  if (iconName == "application-vnd.debian.binary-package") {
146  iconName = "application-x-deb";
147  } else if (iconName == "application-vnd.rar") {
148  iconName = "application-zip";
149  } else if (iconName == "application-vnd.ms-htmlhelp") {
150  iconName = "chmsee";
151  } else {
152  return icon;
153  }
154 
155  icon = QIcon::fromTheme(iconName);
156 
157  return icon;
158 }
