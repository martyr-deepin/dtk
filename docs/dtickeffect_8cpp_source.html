73 void DTickEffect::draw(QPainter *painter)
74 {
75  D_D(DTickEffect);
76 
77  QPoint offset;
78  QPixmap pixmap;
79 
80  if (sourceIsPixmap())
81  pixmap = sourcePixmap(Qt::LogicalCoordinates, &offset, QGraphicsEffect::NoPad);
82  else
83  pixmap = sourcePixmap(Qt::DeviceCoordinates, &offset, QGraphicsEffect::NoPad);
84 
85  const QPoint p { d->runAnimation->currentValue().toPoint() };
86  const QSize size { pixmap.size() / d->content->devicePixelRatioF() };
87 
88 
89  switch (d->direction) {
90  case LeftToRight:
91  offset = QPoint(-size.width() + p.x(), p.y());
92  break;
93  case RightToLeft:
94  offset = QPoint(size.width() + p.x(), p.y());
95  break;
96  case TopToBottom:
97  offset = QPoint(p.x(), -size.height() + p.y());
98  break;
99  case BottomToTop:
100  offset = QPoint(p.x(), size.height() + p.y());
101  break;
102  default:
103  break;
104  }
105 
106  painter->drawPixmap(d->runAnimation->currentValue().toPoint(), pixmap);
107  painter->drawPixmap(offset, pixmap);
108 }

221 void DTickEffectPrivate::initDirection()
222 {
223  switch (direction) {
224  case DTickEffect::LeftToRight:
225  runAnimation->setStartValue(QPoint(content->x(), content->y()));
226  runAnimation->setEndValue(QPoint(content->width(), content->y()));
227  runAnimation->setDuration((content->width() / fixPixel) * 1000);
228  break;
229  case DTickEffect::RightToLeft:
230  runAnimation->setStartValue(QPoint(content->x(), content->y()));
231  runAnimation->setEndValue(QPoint(-content->width(), content->y()));
232  runAnimation->setDuration((content->width() / fixPixel) * 1000);
233  break;
234  case DTickEffect::TopToBottom:
235  runAnimation->setStartValue(QPoint(content->x(), content->y()));
236  runAnimation->setEndValue(QPoint(content->x(), content->height()));
237  runAnimation->setDuration((content->height() / fixPixel) * 1000);
238  break;
239  case DTickEffect::BottomToTop:
240  runAnimation->setStartValue(QPoint(content->x(), content->y()));
241  runAnimation->setEndValue(QPoint(content->x(), -content->height()));
242  runAnimation->setDuration((content->height() / fixPixel) * 1000);
243  break;
244  default:
245  break;
246  }
247 }
