100 DPlatformTheme *DGuiApplicationHelperPrivate::initWindow(QWindow *window) const
101 {
102  // 如果appTheme还未初始化，应当先初始化appTheme
103  if (appTheme == systemTheme) {
104  // 此时QGuiApplication必须是已经初始化完成的状态
105  Q_ASSERT(QGuiApplicationPrivate::is_app_running);
106  // 初始程序级别的主题对象
107  const_cast<DGuiApplicationHelperPrivate*>(this)->_q_initApplicationTheme(true);
108  }
109 
110  DPlatformTheme *theme = new DPlatformTheme(window->winId(), appTheme);
111  window->setProperty(WINDOW_THEME_KEY, QVariant::fromValue(theme));
112  theme->setParent(window); // 跟随窗口销毁
113 
114  auto onWindowThemeChanged = [window] {
115  qGuiApp->postEvent(window, new QEvent(QEvent::ThemeChange));
116  };
117

119  window->connect(theme, &DPlatformTheme::activeColorChanged, window, onWindowThemeChanged);
120  window->connect(theme, &DPlatformTheme::paletteChanged, window, onWindowThemeChanged);
121 
122  return theme;

125 void DGuiApplicationHelperPrivate::_q_initApplicationTheme(bool notifyChange)
126 {
127  if (appTheme && appTheme != systemTheme)
128  return;
129 
130  appTheme = new DPlatformTheme(DPlatformHandle::windowLeader(), systemTheme);
131  QGuiApplication *app = qGuiApp;
132  auto onAppThemeChanged = std::bind(&DGuiApplicationHelperPrivate::notifyAppThemeChanged, this, app, false);
133  // 监听与程序主题相关的改变
134  QObject::connect(appTheme, &DPlatformTheme::themeNameChanged, app, onAppThemeChanged);
135  QObject::connect(appTheme, &DPlatformTheme::activeColorChanged, app, onAppThemeChanged);
136  QObject::connect(appTheme, &DPlatformTheme::paletteChanged, app, onAppThemeChanged);
137 
138  // appTheme在此之前可能由systemTheme所代替被使用，此时在创建appTheme
139  // 并初始化之后，应当发送信号通知程序主题的改变
140  if (notifyChange && appTheme->isValid()) {
141  notifyAppThemeChanged(app);
142  }

240 {

242  : base * (1 + increment / 100.0);

277 QColor DGuiApplicationHelper::adjustColor(const QColor &base,
278  qint8 hueFloat, qint8 saturationFloat, qint8 lightnessFloat,
279  qint8 redFloat, qint8 greenFloat, qint8 blueFloat, qint8 alphaFloat)
280 {
281  // 按HSL格式调整
282  int H, S, L, A;
283  base.getHsl(&H, &S, &L, &A);
284 
285  H = H > 0 ? adjustColorValue(H, hueFloat, 359) : H;
286  S = adjustColorValue(S, saturationFloat);
287  L = adjustColorValue(L, lightnessFloat);
288  A = adjustColorValue(A, alphaFloat);
289 
290  QColor new_color = QColor::fromHsl(H, S, L, A);
291 
292  // 按RGB格式调整
293  int R, G, B;
294  new_color.getRgb(&R, &G, &B);
295 
296  R = adjustColorValue(R, redFloat);
297  G = adjustColorValue(G, greenFloat);
298  B = adjustColorValue(B, blueFloat);
299 
300  new_color.setRgb(R, G, B, A);
301 
302  return new_color;
303 }

311 QColor DGuiApplicationHelper::blendColor(const QColor &substrate, const QColor &superstratum)
312 {
313  QColor c2 = superstratum.toRgb();
314 
315  if (c2.alpha() >= 255)
316  return c2;
317 
318  QColor c1 = substrate.toRgb();
319  qreal c1_weight = 1 - c2.alphaF();
320 
321  int r = c1_weight * c1.red() + c2.alphaF() * c2.red();
322  int g = c1_weight * c1.green() + c2.alphaF() * c2.green();
323  int b = c1_weight * c1.blue() + c2.alphaF() * c2.blue();
324 
325  return QColor(r, g, b, c1.alpha());
326 }

409 DPalette DGuiApplicationHelper::standardPalette(DGuiApplicationHelper::ColorType type)
410 {
411  static const DPalette *light_palette = nullptr, *dark_palette = nullptr;
412  static const DPalette *alpha_light_palette = nullptr, *alpha_dark_palette = nullptr;
413 
414  if (type == LightType) {
415  if (Q_UNLIKELY(DGuiApplicationHelperPrivate::compositingColor)) {
416  if (Q_LIKELY(alpha_light_palette)) {
417  return *alpha_light_palette;
418  }
419  }
420 
421  if (Q_LIKELY(light_palette)) {
422  return *light_palette;
423  }
424  } else if (type == DarkType) {
425  if (Q_UNLIKELY(DGuiApplicationHelperPrivate::compositingColor)) {
426  if (Q_LIKELY(alpha_dark_palette)) {
427  return *alpha_dark_palette;
428  }
429  }
430 
431  if (Q_LIKELY(dark_palette)) {
432  return *dark_palette;
433  }
434  } else {
435  return DPalette();
436  }
437 
438  DPalette *pa;
439  const QColor *qcolor_list, *dcolor_list;
440 
441  if (type == DarkType) {
442  pa = new DPalette();
443 
444  if (DGuiApplicationHelperPrivate::compositingColor)
445  alpha_dark_palette = pa;
446  else
447  dark_palette = pa;
448 
449  qcolor_list = dark_qpalette;
450  dcolor_list = dark_dpalette;
451  } else {
452  pa = new DPalette();
453 
454  if (DGuiApplicationHelperPrivate::compositingColor)
455  alpha_light_palette = pa;
456  else
457  light_palette = pa;
458 
459  qcolor_list = light_qpalette;
460  dcolor_list = light_dpalette;
461  }
462 
463  for (int i = 0; i < DPalette::NColorRoles; ++i) {
464  QPalette::ColorRole role = static_cast<QPalette::ColorRole>(i);
465  QColor color = qcolor_list[i];
466 
467  // 处理半透明色
468  if (DGuiApplicationHelperPrivate::compositingColor) {
469  switch (role) {
470  case QPalette::Window:
471  color = type == LightType ? adjustColor(color, 0, 0, 0, 0, 0, 0, -20) : adjustColor(color, 0, 0, -10, 0, 0, 0, -20);
472  break;
473  case QPalette::Base:
474  color = adjustColor(color, 0, 0, 0, 0, 0, 0, -20);
475  break;
476  case QPalette::WindowText:
477  case QPalette::Text:
478  color = adjustColor(color, 0, 0, type == LightType ? -20 : +20, 0, 0, 0, -20);
479  break;
480  case QPalette::ButtonText:
481  color = type == LightType ? adjustColor(color, 0, 0, -20, 0, 0, 0, -20) : adjustColor(color, 0, 0, +20, 0, 0, 0, 0);
482  break;
483  case QPalette::Button:
484  case QPalette::Light:
485  case QPalette::Mid:
486  case QPalette::Midlight:
487  case QPalette::Dark:
488  color = adjustColor(color, 0, 0, -20, 0, 0, 0, -40);
489  break;
490  default:
491  break;
492  }
493  }
494 
495  pa->setColor(DPalette::Active, role, color);
496  generatePaletteColor(*pa, role, type);
497  }
498 
499  for (int i = 0; i < DPalette::NColorTypes; ++i) {
500  DPalette::ColorType role = static_cast<DPalette::ColorType>(i);
501  QColor color = dcolor_list[i];
502 
503  // 处理半透明色
504  if (DGuiApplicationHelperPrivate::compositingColor) {
505  switch (role) {
506  case DPalette::ItemBackground:
507  color = adjustColor(color, 0, 0, 100, 0, 0, 0, type == LightType ? -80 : -90);
508  break;
509  case DPalette::TextTitle:
510  color = adjustColor(color, 0, 0, -20, 0, 0, 0, -20);
511  break;
512  case DPalette::TextTips:
513  color = type == LightType ? adjustColor(color, 0, 0, -40, 0, 0, 0, -40) : adjustColor(color, 0, 0, +40, 0, 0, 0, -50);
514  break;
515  default:
516  break;
517  }
518  }
519 
520  pa->setColor(DPalette::Active, role, color);
521  generatePaletteColor(*pa, role, type);
522  }
523 
524  return *const_cast<const DPalette*>(pa);
525 }

527 template<typename M>
528 static void generatePaletteColor_helper(DPalette &base, M role, DGuiApplicationHelper::ColorType type)
529 {
530  if (type == DGuiApplicationHelper::UnknownType) {
531  type = DGuiApplicationHelper::toColorType(base);
532  }
533 
534  QColor disable_mask_color, inactive_mask_color;
535 
536  if (type == DGuiApplicationHelper::DarkType) {
537  disable_mask_color = dark_qpalette[QPalette::Window];
538  inactive_mask_color = dark_qpalette[QPalette::Window];
539  disable_mask_color.setAlphaF(0.7);
540  inactive_mask_color.setAlphaF(0.6);
541  } else {
542  disable_mask_color = light_qpalette[QPalette::Window];
543  inactive_mask_color = light_qpalette[QPalette::Window];
544  disable_mask_color.setAlphaF(0.6);
545  inactive_mask_color.setAlphaF(0.4);
546  }
547 
548  const QColor &color = base.color(QPalette::Normal, role);
549  base.setColor(QPalette::Disabled, role, DGuiApplicationHelper::blendColor(color, disable_mask_color));
550 
551  QPalette::ColorRole qr = static_cast<QPalette::ColorRole>(role);
552  if (qr == QPalette::Text) {
553  // disable text color olny -60% alpha
554  base.setColor(QPalette::Disabled, role, DGuiApplicationHelper::adjustColor(color, 0, 0, 0, 0, 0, 0, -60));
555  }
556 
557  if (DGuiApplicationHelperPrivate::useInactiveColor)
558  base.setColor(QPalette::Inactive, role, DGuiApplicationHelper::blendColor(color, inactive_mask_color));
559  else
560  base.setColor(QPalette::Inactive, role, color);
561 }

569 void DGuiApplicationHelper::generatePaletteColor(DPalette &base, QPalette::ColorRole role, DGuiApplicationHelper::ColorType type)
570 {
571  if (role == QPalette::Window) {
572  const QBrush &window = base.brush(QPalette::Normal, role);
573  base.setBrush(QPalette::Disabled, role, window);
574  base.setBrush(QPalette::Inactive, role, window);
575  return;
576  } else if (role == QPalette::Highlight && toColorType(base) == DarkType) {
577  // 暗色模式下的高亮色亮度要降低10%，避免太突兀
578  QColor highlight = base.highlight().color();
579 
580  if (highlight.isValid()) {
581  base.setColor(QPalette::Highlight, adjustColor(highlight, 0, 0, -20, 0, 0, 0, 0));
582  }
583  }
584 
585  generatePaletteColor_helper(base, role, type);
586 }

604 void DGuiApplicationHelper::generatePalette(DPalette &base, ColorType type)
605 {
606  // 先判断调色板的色调
607  if (type == UnknownType) {
608  type = toColorType(base);
609  }
610 
611  for (int i = 0; i < QPalette::NColorRoles; ++i) {
612  QPalette::ColorRole role = static_cast<QPalette::ColorRole>(i);
613  generatePaletteColor(base, role, type);
614  }
615 
616  for (int i = 0; i < QPalette::NColorRoles; ++i) {
617  DPalette::ColorType role = static_cast<DPalette::ColorType>(i);
618  generatePaletteColor(base, role, type);
619  }
620 }

627 DPalette DGuiApplicationHelper::fetchPalette(const DPlatformTheme *theme)
628 {
629  DPalette base_palette;
630  const QByteArray theme_name = theme->themeName();
631  ColorType type = LightType;
632 
633  // 深色主题
634  if (theme_name.endsWith("dark")) {
635  type = DarkType;
636  }
637 
638  bool ok = false;
639  base_palette = theme->fetchPalette(standardPalette(type), &ok);
640  const QColor &active_color = theme->activeColor();
641 
642  if (active_color.isValid()) {
643  base_palette.setColor(QPalette::Normal, QPalette::Highlight, active_color);
644 
645  // ok为true时会整体处理palette的所有颜色
646  if (!ok) {
647  generatePaletteColor(base_palette, QPalette::Highlight, type);
648  }
649  }
650 
651  if (ok) {
652  generatePalette(base_palette, type);
653  }
654 
655  return base_palette;
656 }

729 DPalette DGuiApplicationHelper::applicationPalette() const
730 {
731  D_DC(DGuiApplicationHelper);
732 
733  if (d->appPalette) {
734  return *d->appPalette;
735  }
736 
737  ColorType type = UnknownType;
738 
739  // 如果应用程序自己设置过palette，则以这个palette为基础获取DPalette
740  if (qGuiApp && qGuiApp->testAttribute(Qt::AA_SetPalette)) {
741  type = toColorType(qGuiApp->palette());
742  } else {
743  type = d->paletteType;
744  }
745 
746  // 如果自定义了palette的类型，则直接返回对应的标准DPalette

866 bool DGuiApplicationHelper::setSingleInstance(const QString &key, DGuiApplicationHelper::SingleScope singleScope)
867 {
868  bool new_server = !_d_singleServer.exists();
869 
870  if (_d_singleServer->isListening()) {
871  _d_singleServer->close();
872  }
873 
874  QString socket_key = "_d_dtk_single_instance_";
875 
876  switch (singleScope) {
877  case GroupScope:
878  _d_singleServer->setSocketOptions(QLocalServer::GroupAccessOption);
879 #ifdef Q_OS_LINUX
880  socket_key += QString("%1_").arg(getgid());
881 #endif
882  break;
883  case WorldScope:
884  _d_singleServer->setSocketOptions(QLocalServer::WorldAccessOption);
885  break;
886  default:
887  _d_singleServer->setSocketOptions(QLocalServer::UserAccessOption);
888 #ifdef Q_OS_LINUX
889  socket_key += QString("%1_").arg(getuid());
890 #endif
891  break;
892  }
893 
894  socket_key += key;
895 
896  // 通知别的实例
897  QLocalSocket socket;
898  socket.connectToServer(socket_key);
899 
900  // 等待到有效数据时认为server实例有效
901  if (socket.waitForConnected(DGuiApplicationHelperPrivate::waitTime) && socket.waitForReadyRead(DGuiApplicationHelperPrivate::waitTime)) {
902  // 读取数据
903  qint8 version;
904  qint64 pid;
905  QStringList arguments;
906 
907  QDataStream ds(&socket);
908  ds >> version >> pid >> arguments;
909  qInfo() << "Process is started: pid=" << pid << "arguments=" << arguments;
910 
911  // 把自己的信息告诉第一个实例
912  ds << _d_singleServerVersion << qApp->applicationPid() << qApp->arguments();
913  socket.flush();
914 
915  return false;
916  }
917 
918  if (!_d_singleServer->listen(socket_key)) {
919  return false;
920  }
921 
922  if (new_server) {
923  QObject::connect(_d_singleServer, &QLocalServer::newConnection, qApp, [] {
924  QLocalSocket *instance = _d_singleServer->nextPendingConnection();
925  // 先发送数据告诉新的实例自己收到了它的请求
926  QDataStream ds(instance);
927  ds << _d_singleServerVersion // 协议版本
928  << qApp->applicationPid() // 进程id
929  << qApp->arguments(); // 启动时的参数
930 
931  QObject::connect(instance, &QLocalSocket::readyRead, qApp, [instance] {
932  // 读取数据
933  QDataStream ds(instance);
934 
935  qint8 version;
936  qint64 pid;
937  QStringList arguments;
938 
939  ds >> version >> pid >> arguments;
940  instance->close();
941 
942  qInfo() << "New instance: pid=" << pid << "arguments=" << arguments;
943 
944  // 通知新进程的信息
945  if (_globalHelper.exists())
946  Q_EMIT _globalHelper->helper->newProcessInstance(pid, arguments);
947  });
948 
949  instance->flush(); //发送数据给新的实例
950  });
951  }
952 
953  return true;
954 }
