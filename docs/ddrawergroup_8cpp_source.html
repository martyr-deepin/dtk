108 void DDrawerGroup::addExpand(DDrawer *expand, int id)
109 {
110  D_D(DDrawerGroup);
111  if (d->expandMap.values().indexOf(expand) == -1){
112  if (id == -1){
113  int maxId = -1;
114  Q_FOREACH (int tmp, d->expandMap.keys()) {
115  maxId = qMax(tmp, maxId);
116  }
117 
118  id = maxId + 1;
119  }
120  d->expandMap.insert(id, expand);
121 
122  connect(expand, &DDrawer::expandChange, this, &DDrawerGroup::onExpandChanged);
123  }
124 }

181 void DDrawerGroup::onExpandChanged(bool v)
182 {
183  D_D(DDrawerGroup);
184  DDrawer *dSender = qobject_cast<DDrawer *>(sender());
185  if (!dSender || dSender == checkedExpand() || !v)
186  return;
187 
188  d->checkedMap.clear();
189  int checkedIndex = d->expandMap.values().indexOf(dSender);
190  d->checkedMap.insert(d->expandMap.keys().at(checkedIndex), dSender);
191 
192  QList<DDrawer *> expandList = d->expandMap.values();
193  for (int i = 0; i < expandList.count(); i ++){
194  DDrawer *target = expandList.at(i);
195  if (target != dSender && target->expand())
196  target->setExpand(false);
197  }
198 }
