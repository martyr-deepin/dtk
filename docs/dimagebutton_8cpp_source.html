721 void DImageButtonPrivate::updateIcon()
722 {
723  D_Q(DImageButton);
724 
725  switch (m_state) {
726  case DImageButton::Hover:
727  q->setProperty("state", "hover");
728  if (!m_hoverPic.isEmpty()) { q->setPixmap(loadPixmap(m_hoverPic)); }
729  break;
730  case DImageButton::Press:
731  q->setProperty("state", "press");
732  if (!m_pressPic.isEmpty()) { q->setPixmap(loadPixmap(m_pressPic)); }
733  break;
734  case DImageButton::Checked:
735  q->setProperty("state", "checked");
736  if (!m_checkedPic.isEmpty()) { q->setPixmap(loadPixmap(m_checkedPic)); }
737  break;
738  case DImageButton::Disabled:
739  q->setProperty("state", "disabled");
740  if (!m_disabledPic.isEmpty()) { q->setPixmap(loadPixmap(m_disabledPic)); }
741  break;
742  default:
743  q->setProperty("state", "");
744  if (!m_normalPic.isEmpty()) { q->setPixmap(loadPixmap(m_normalPic)); }
745  break;
746  }
747 
748  q->setAlignment(Qt::AlignCenter);
749 
750  Q_EMIT q->stateChanged();
751 }

764 QPixmap DImageButtonPrivate::loadPixmap(const QString &path)
765 {
766  D_Q(DImageButton);
767 
768  qreal ratio = 1.0;
769 
770  const qreal devicePixelRatio = q->devicePixelRatioF();
771 
772  QPixmap pixmap;
773 
774  if (!qFuzzyCompare(ratio, devicePixelRatio)) {
775  QImageReader reader;
776  reader.setFileName(qt_findAtNxFile(path, devicePixelRatio, &ratio));
777  if (reader.canRead()) {
778  reader.setScaledSize(reader.size() * (devicePixelRatio / ratio));
779  pixmap = QPixmap::fromImage(reader.read());
780  pixmap.setDevicePixelRatio(devicePixelRatio);
781  }
782  } else {
783  pixmap.load(path);
784  }
785 
786  return pixmap;
787 }
