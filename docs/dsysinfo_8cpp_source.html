100 void DSysInfoPrivate::ensureDeepinInfo()
101 {
102  if (static_cast<int>(deepinType) >= 0)
103  return;
104 
105  QFile file("/etc/deepin-version");
106 
107  if (!file.open(QFile::ReadOnly)) {
108  deepinType = DSysInfo::UnknownDeepin;
109 
110  return;
111  }
112 
113  char buf[1024];
114  int buf_length = 0;
115 
116  Q_FOREVER {
117  buf_length = file.readLine(buf, sizeof(buf));
118 
119  if (buf_length < 0)
120  break;
121 
122  const QByteArray line(buf, buf_length);
123  const QByteArrayList &list = line.split('=');
124 
125  if (list.count() != 2) {
126  continue;
127  }
128 
129  const auto key_value = qMakePair(list.first().trimmed(), list.last().trimmed());
130 
131  if (key_value.first == "Version") {
132  deepinVersion = key_value.second;
133  } else if (line.startsWith("Type")) {
134  if (key_value.first == "Type") {
135  deepinTypeMap[QString()] = QString::fromLatin1(key_value.second);
136  } else if (key_value.first.at(4) == '[' && key_value.first.at(key_value.first.size() - 1) == ']') {
137  const QByteArray &language = key_value.first.mid(5, key_value.first.size() - 6);
138 
139  if (!language.isEmpty()) {
140  deepinTypeMap[QString::fromLatin1(language)] = QString::fromUtf8(key_value.second);
141  }
142  }
143  } else if (key_value.first == "Edition") {
144  deepinEdition = QString::fromUtf8(key_value.second);
145  } else if (key_value.first == "Copyright") {
146  deepinCopyright = QString::fromUtf8(key_value.second);
147  }
148 
149  if (!deepinTypeMap.isEmpty() && !deepinEdition.isEmpty() && !deepinCopyright.isEmpty()) {
150  break;
151  }
152  }
153 
154  file.close();
155 
156  const QString &deepin_type = deepinTypeMap[QString()];
157 
158  if (deepin_type.isEmpty()) {
159  deepinType = DSysInfo::UnknownDeepin;
160  } else if (deepin_type == "Desktop") {
161  deepinType = DSysInfo::DeepinDesktop;
162  } else if (deepin_type == "Professional") {
163  deepinType = DSysInfo::DeepinProfessional;
164  } else if (deepin_type == "Server") {
165  deepinType = DSysInfo::DeepinServer;
166  } else if (deepin_type == "Personal") {
167  deepinType = DSysInfo::DeepinPersonal;
168  } else {
169  deepinType = DSysInfo::UnknownDeepin;
170  }
171 
172  const QString distributionInfoFile(DSysInfo::distributionInfoPath());
173  // Generic DDE distribution info
174  distributionInfo.reset(new DDesktopEntry(distributionInfoFile));
175  QSettings distributionInfo(distributionInfoFile, QSettings::IniFormat); // TODO: treat as `.desktop` format instead of `.ini`
176 }

178 bool DSysInfoPrivate::ensureOsVersion()
179 {
180  if (minVersion.A > 0)
181  return true;
182 
183  DDesktopEntry entry("/etc/os-version");
184  majorVersion = entry.stringValue("MajorVersion", "Version");
185  minorVersion = entry.stringValue("MinorVersion", "Version");
186  Q_ASSERT(minorVersion.length() == 4);
187 
188  // A-BC-D
189  bool ok = false;
190  uint minv = minorVersion.toUInt(&ok);
191  if (ok) {
192  minVersion.D = minv % 10;
193  } else if (minorVersion.length() > 0) {
194  // 0-9...A-Z
195  minVersion.D = 10 + static_cast<uint>(minorVersion.right(1).data()->toLatin1() - 'A');
196  }
197  uint minVer = minorVersion.left(3).toUInt();
198  minVersion.BC = minVer % 100;
199  minVer /= 100;
200  minVersion.A = minVer % 10;
201 
202  // ABCDE.xyz
203  QString osb = entry.stringValue("OsBuild", "Version");
204  QStringList osbs = osb.split(".");
205  Q_ASSERT(osbs.size() == 2 && osbs.value(0).size() == 5);
206  uint left = osbs.value(0).trimmed().toUInt(&ok);
207  Q_ASSERT(ok);
208  if (ok) {
209  osBuild.E = left % 10;
210  left /= 10;
211  osBuild.D = left % 10;
212  left /= 10;
213  osBuild.C = left % 10; // default C is 0
214  left /= 10;
215  osBuild.B = left % 10;
216  left /= 10;
217  osBuild.A = left % 10;
218  }
219 
220  // xyz
221  osBuild.xyz = osbs.value(1).trimmed().toUInt(&ok);
222 
223  return ok;
224 }

226 static QString unquote(const QByteArray &value)
227 {
228  if (value.at(0) == '"' || value.at(0) == '\'') {
229  return QString::fromLatin1(value.mid(1, value.size() - 2));
230  }
231 
232  return QString::fromLatin1(value);
233 }

235 static bool readEtcFile(DSysInfoPrivate *info, const char *filename,
236  const QByteArray &idKey, const QByteArray &versionKey, const QByteArray &prettyNameKey)
237 {
238 
239  QFile file(QString::fromLatin1(filename));
240 
241  if (!file.open(QIODevice::ReadOnly)) {
242  return false;
243  }
244 
245  quint8 valid_data_count = 0;
246  char buf[1024];
247 
248  while (valid_data_count < 3) {
249  int buf_length = file.readLine(buf, sizeof(buf));
250 
251  if (buf_length < 0)
252  break;
253 
254  const QByteArray line(buf, buf_length - 1);
255 
256  if (info->productTypeString.isEmpty() && line.startsWith(idKey)) {
257  const QByteArray value(line.constData() + idKey.size());
258  info->productTypeString = unquote(value);
259  ++valid_data_count;
260  continue;
261  }
262 
263  if (info->prettyName.isEmpty() && line.startsWith(prettyNameKey)) {
264  const QByteArray value(line.constData() + prettyNameKey.size());
265  info->prettyName = unquote(value);
266  ++valid_data_count;
267  continue;
268  }
269 
270  if (info->productVersion.isEmpty() && line.startsWith(versionKey)) {
271  const QByteArray value(line.constData() + versionKey.size());
272  info->productVersion = unquote(value);
273  ++valid_data_count;
274  continue;
275  }
276  }
277 
278  file.close();
279 
280  return valid_data_count != 0;
281 }

297 void DSysInfoPrivate::ensureReleaseInfo()
298 {
299  if (productType >= 0) {
300  return;
301  }
302 
303 #ifdef Q_OS_LINUX
304  readOsRelease(this);
305  readLsbRelease(this);
306 
307  if (productTypeString.isEmpty()) {
308  productType = DSysInfo::UnknownType;
309  } else {
310  switch (productTypeString.at(0).unicode()) {
311  case 'd':
312  case 'D':
313  if (productTypeString.compare("deepin", Qt::CaseInsensitive) == 0) {
314  productType = DSysInfo::Deepin;
315  } else if (productTypeString.compare("debian", Qt::CaseInsensitive) == 0) {
316  productType = DSysInfo::Debian;
317  }
318  break;
319  case 'a':
320  case 'A':
321  if (productTypeString.compare("arch", Qt::CaseInsensitive) == 0)
322  productType = DSysInfo::ArchLinux;
323  break;
324  case 'c':
325  case 'C':
326  if (productTypeString.compare("centos", Qt::CaseInsensitive) == 0)
327  productType = DSysInfo::CentOS;
328  break;
329  case 'f':
330  case 'F':
331  if (productTypeString.compare("fedora", Qt::CaseInsensitive) == 0)
332  productType = DSysInfo::Fedora;
333  break;
334  case 'l':
335  case 'L':
336  if (productTypeString.compare("linuxmint", Qt::CaseInsensitive) == 0)
337  productType = DSysInfo::LinuxMint;
338  break;
339  case 'm':
340  case 'M':
341  if (productTypeString.compare("manjaro", Qt::CaseInsensitive) == 0)
342  productType = DSysInfo::Manjaro;
343  break;
344  case 'o':
345  case 'O':
346  if (productTypeString.compare("opensuse", Qt::CaseInsensitive) == 0)
347  productType = DSysInfo::openSUSE;
348  break;
349  case 's':
350  case 'S':
351  if (productTypeString.compare("sailfishos", Qt::CaseInsensitive) == 0)
352  productType = DSysInfo::SailfishOS;
353  break;
354  case 'u':
355  case 'U':
356  if (productTypeString.compare("ubuntu", Qt::CaseInsensitive) == 0) {
357  productType = DSysInfo::Ubuntu;
358  } else if (productTypeString.compare("uos", Qt::CaseInsensitive) == 0 || productTypeString.compare("UnionTech OS", Qt::CaseInsensitive) == 0) {
359  productType = DSysInfo::Uos;
360  }
361  break;
362  default:
363  productType = DSysInfo::UnknownType;
364  break;
365  }
366  }
367 #endif

370 void DSysInfoPrivate::ensureComputerInfo()
371 {
372  if (memoryAvailableSize >= 0)
373  return;
374 
375 #ifdef Q_OS_LINUX
376  struct utsname u;
377  if (uname(&u) == 0)
378  computerName = QString::fromLatin1(u.nodename);
379 
380  QFile file("/proc/cpuinfo");
381 
382  if (file.open(QFile::ReadOnly)) {
383  QMap<QString, QString> map = parseInfoFile(file);
384  if (map.contains("Processor")) {
385  // arm-cpuinfo hw_kirin-cpuinfo
386  cpuModelName = map.value("Processor");
387  } else if (map.contains("model name")) {
388  // cpuinfo
389  cpuModelName = map.value("model name");
390  } else if (map.contains("cpu model")) {
391  // loonson3-cpuinfo sw-cpuinfo
392  cpuModelName = map.value("cpu model");
393  }
394 
395  file.close();
396  }
397 
398  memoryAvailableSize = get_phys_pages() * sysconf(_SC_PAGESIZE);
399 
400  // Getting Memory Installed Size
401  // TODO: way to not dept on lshw?
402  if (!QStandardPaths::findExecutable("lshw").isEmpty()) {
403  QProcess lshw;
404 
405  lshw.start("lshw", {"-c", "memory", "-json", "-sanitize"}, QIODevice::ReadOnly);
406 
407  if (!lshw.waitForFinished()) {
408  return;
409  }
410 
411  const QByteArray &lshwInfoJson = lshw.readAllStandardOutput();
412  QJsonArray lshwResultArray = QJsonDocument::fromJson(lshwInfoJson).array();
413  if (!lshwResultArray.isEmpty()) {
414  QJsonValue memoryHwInfo = lshwResultArray.first();
415  QString id = memoryHwInfo.toObject().value("id").toString();
416  Q_ASSERT(id == "memory");
417  memoryInstalledSize = memoryHwInfo.toObject().value("size").toDouble(); // TODO: check "units" is "bytes" ?
418  }
419  }
420 
421  // Getting Disk Size
422  const QString &deviceName = QStorageInfo::root().device();
423  QProcess lsblk;
424 
425  lsblk.start("lsblk", {"-Jlpb", "-oNAME,KNAME,PKNAME,SIZE"}, QIODevice::ReadOnly);
426 
427  if (!lsblk.waitForFinished()) {
428  return;
429  }
430 
431  const QByteArray &diskStatusJson = lsblk.readAllStandardOutput();
432  QJsonDocument diskStatus = QJsonDocument::fromJson(diskStatusJson);
433  QJsonValue diskStatusJsonValue = diskStatus.object().value("blockdevices");
434  QMap<QString, QPair<QString, qulonglong>> deviceParentAndSizeMap;
435 
436  if (!diskStatusJsonValue.isUndefined()) {
437  QJsonArray diskStatusArray = diskStatusJsonValue.toArray();
438  QString keyName;
439 
440  for (const QJsonValue oneValue : diskStatusArray) {
441  QString name = oneValue.toObject().value("name").toString();
442  QString kname = oneValue.toObject().value("kname").toString();
443  QString pkname = oneValue.toObject().value("pkname").toString();
444  qulonglong size = oneValue.toObject().value("size").toVariant().toULongLong();
445 
446  if (keyName.isNull() && deviceName == name) {
447  keyName = kname;
448  }
449 
450  deviceParentAndSizeMap[kname] = QPair<QString, qulonglong>(pkname, size);
451  }
452 
453  while (!deviceParentAndSizeMap[keyName].first.isNull()) {
454  keyName = deviceParentAndSizeMap[keyName].first;
455  }
456 
457  diskSize = deviceParentAndSizeMap[keyName].second;
458  }
459 #endif
460 }

462 QMap<QString, QString> DSysInfoPrivate::parseInfoFile(QFile &file)
463 {
464  char buf[1024];
465  qint64 lineLength = 0;
466  QMap<QString, QString> map;
467  do {
468  lineLength = file.readLine(buf, sizeof(buf));
469  QString s(buf);
470  if (s.contains(':')) {
471  QStringList list = s.split(':');
472  if (list.size() == 2) {
473  map.insert(list.first().trimmed(), list.back().trimmed());
474  }
475  }
476  } while (lineLength >= 0);
477  return map;
478 }
