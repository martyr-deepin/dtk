63 void DGraphicsGlowEffect::draw(QPainter *painter)
64 {
65  // if nothing to show outside the item, just draw source
66  if ((blurRadius() + distance()) <= 0) {
67  drawSource(painter);
68  return;
69  }
70 
71  PixmapPadMode mode = QGraphicsEffect::PadToEffectiveBoundingRect;
72  QPoint offset;
73  const QPixmap sourcePx = sourcePixmap(Qt::DeviceCoordinates, &offset, mode);
74 
75  // return if no source
76  if (sourcePx.isNull()) {
77  return;
78  }
79 
80  qreal restoreOpacity = painter->opacity();
81  painter->setOpacity(m_opacity);
82 
83  // save world transform
84  QTransform restoreTransform = painter->worldTransform();
85  painter->setWorldTransform(QTransform());
86 
87  // Calculate size for the background image
88  QSize scaleSize(sourcePx.size().width() + 2 * distance(),
89  sourcePx.size().height() + 2 * distance());
90 
91  QImage tmpImg(scaleSize, QImage::Format_ARGB32_Premultiplied);
92  QPixmap scaled = sourcePx.scaled(scaleSize);
93  tmpImg.fill(0);
94  QPainter tmpPainter(&tmpImg);
95  tmpPainter.setCompositionMode(QPainter::CompositionMode_Source);
96  tmpPainter.drawPixmap(QPointF(-distance(), -distance()), scaled);
97  tmpPainter.end();
98 
99  // blur the alpha channel
100  QImage blurred(tmpImg.size(), QImage::Format_ARGB32_Premultiplied);
101  blurred.fill(0);
102  QPainter blurPainter(&blurred);
103  qt_blurImage(&blurPainter, tmpImg, blurRadius(), false, true);
104  blurPainter.end();
105 
106  tmpImg = blurred;
107 
108  // blacken the image...
109  tmpPainter.begin(&tmpImg);
110  tmpPainter.setCompositionMode(QPainter::CompositionMode_SourceIn);
111  tmpPainter.fillRect(tmpImg.rect(), color());
112  tmpPainter.end();
113 
114  // draw the blurred shadow...
115  painter->drawImage(offset, tmpImg);
116 
117  // draw the actual pixmap...
118  painter->drawPixmap(offset, sourcePx, QRectF());
119 
120  // restore world transform
121  painter->setWorldTransform(restoreTransform);
122  painter->setOpacity(restoreOpacity);
123 }
