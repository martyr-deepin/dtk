72 DPalette DApplicationHelper::palette(const QWidget *widget, const QPalette &base) const
73 {
74  DPalette palette;
75 
76  if (!widget) {
77  return applicationPalette();
78  }
79 
80  do {
81  // 先从缓存中取数据
82  if (d->paletteCache.contains(widget)) {
83  palette = d->paletteCache.value(widget);
84  break;
85  }
86 
87  if (QWidget *parent = widget->parentWidget()) {
88  palette = this->palette(parent, base);
89  } else {
90  palette = applicationPalette();
91  }
92 
93  // 判断widget对象有没有被设置过palette
94  if (widget->testAttribute(Qt::WA_SetPalette)) {
95  // 存在自定义palette时应该根据其自定义的palette获取对应色调的DPalette
96  const QPalette &wp = widget->palette();
97 
98  // 判断控件自己的palette色调是否和要继承调色板色调一致
99  if (toColorType(palette) != toColorType(wp)) {
100  // 不一致时则fallback到标准的palette
101  palette = standardPalette(toColorType(wp));
102  }
103  }
104 
105  // 缓存数据
106  d->paletteCache.insert(widget, palette);
107  // 关注控件palette改变的事件
108  const_cast<QWidget*>(widget)->installEventFilter(const_cast<DApplicationHelper*>(this));
109  } while (false);
110 
111  palette.QPalette::operator =(base.resolve() ? base : widget->palette());
112 
113  return palette;
114 }

156 bool DApplicationHelper::eventFilter(QObject *watched, QEvent *event)
157 {
158  if (Q_UNLIKELY(event->type() == QEvent::PaletteChange)) {
159  if (QWidget *widget = qobject_cast<QWidget*>(watched)) {
160  if (!widget->property("_d_set_palette").toBool()) {
161  // 清理缓存
162  d->paletteCache.remove(widget);
163  }
164  }
165  } else if (Q_UNLIKELY(event->type() == QEvent::Destroy)) {
166  if (QWidget *widget = qobject_cast<QWidget *>(watched)) {
167  if (d->paletteCache.contains(widget)) {
168  // 清理缓存
169  d->paletteCache.remove(widget);
170  }
171  }
172  }
173 
174  return DGuiApplicationHelper::eventFilter(watched, event);
175 }
