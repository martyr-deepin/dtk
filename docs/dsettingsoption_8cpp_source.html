331 void DSettingsOptionPrivate::parseJson(const QString &prefixKey, const QJsonObject &option)
332 {
333 // Q_Q(Option);
334  key = option.value("key").toString();
335  Q_ASSERT(!key.isEmpty());
336  Q_ASSERT(!prefixKey.isEmpty());
337  key = prefixKey + "." + key;
338  name = option.value("name").toString();
339 
340  canReset = !option.contains("reset") ? true : option.value("reset").toBool();
341  defalutValue = option.value("default").toVariant();
342  hidden = !option.contains("hide") ? false : option.value("hide").toBool();
343  viewType = option.value("type").toString();
344 
345  QStringList revserdKeys;
346  revserdKeys << "key" << "name" << "reset"
347  << "default" << "hide" << "type";
348 
349  auto allKeys = option.keys();
350  for (auto key : revserdKeys) {
351  allKeys.removeAll(key);
352  }
353 
354  for (auto key : allKeys) {
355  auto value = option.value(key);
356  if (value.isArray()) {
357  QStringList stringlist;
358  for (auto va : value.toArray()) {
359  stringlist << QString("%1").arg(va.toString());
360  }
361  datas.insert(key, stringlist);
362  } else {
363  datas.insert(key, value.toVariant());
364  }
365  }
366 }
