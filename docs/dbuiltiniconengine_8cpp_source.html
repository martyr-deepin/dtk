54  QString pmcKey(const QSize &size, QIcon::Mode mode, QIcon::State state)
55  {
56  return QLatin1String("$qt_icon_")
57  + filename + QLatin1String("_")
58  + QString::number((((((qint64(size.width()) << 11) | size.height()) << 11) | mode) << 4) | state, 16);

61  void genIconTypeIcon(QPixmap &pm, QIcon::Mode mode)
62  {
63  if (type == IconType && qobject_cast<QGuiApplication *>(QCoreApplication::instance())) {
64  const QPixmap generated = QGuiApplicationPrivate::instance()->applyQIconStyleHelper(mode, pm);
65  if (!generated.isNull())
66  pm = generated;
67  }
68  }

70  QPixmap pixmap(const QSize &size, QIcon::Mode mode, QIcon::State state) override {
71  Q_UNUSED(state)
72 
73  QPixmap pm;
74  QString pmckey(pmcKey(size, mode, state));
75  if (QPixmapCache::find(pmckey, pm)) {
76  genIconTypeIcon(pm, mode);
77  return pm;
78  }
79 
80  // png 会读取前8个 bit 对比，需要关闭重新打开，否则同一个 icon 不同 state/mode 会再次读取时报错导致 pixmap 为空。
81  // 所以此处只对 DirImageEntry(isdir==true) 进行判断不再重新调用 setFileName
82  if (Q_UNLIKELY(!reader.device()) || !QFileInfo(filename).isDir()) {
83  reader.setFileName(filename);
84  }
85 
86  if (dir.type == QIconDirInfo::Scalable)
87  reader.setScaledSize(size);
88 
89  pm = QPixmap::fromImageReader(&reader);
90  if (!pm.isNull())
91  QPixmapCache::insert(pmckey, pm);
92 
93  genIconTypeIcon(pm, mode);
94  return pm;
95  }
96 
97  Type type;
98  QImageReader reader;

106  static QString getIconFile(const QString &key, const QDir &dir, const QString &suffix)
107  {
108  if (dir.exists(key + "." + suffix)) {
109  return dir.filePath(key + "." + suffix);
110  }
111 
112  int _index = key.indexOf('_');
113 
114  if (_index > 0) {
115  const QString &mode = key.left(_index);
116 
117  if (dir.exists(mode + "." + suffix)) {
118  return dir.filePath(mode + "." + suffix);
119  }
120 
121  const QString &state = key.mid(_index);
122 
123  if (dir.exists("normal" + state + "." + suffix)) {
124  return dir.filePath("normal" + state + "." + suffix);
125  }
126  }
127 
128  return dir.filePath("normal." + suffix);
129  }
130 
131  QPixmap pixmap(const QSize &size, QIcon::Mode mode, QIcon::State state) override {
132  if (iconFileMap.isEmpty()) {
133  const QString &suffix = QFileInfo(filename).suffix();
134  QDir dir(filename);
135 
136  iconFileMap[QIcon::Disabled << 8 | QIcon::On] = getIconFile("disabled_on", dir, suffix);
137  iconFileMap[QIcon::Disabled << 8 | QIcon::Off] = getIconFile("disabled_off", dir, suffix);
138  iconFileMap[QIcon::Active << 8 | QIcon::On] = getIconFile("active_on", dir, suffix);
139  iconFileMap[QIcon::Active << 8 | QIcon::Off] = getIconFile("active_off", dir, suffix);
140  iconFileMap[QIcon::Selected << 8 | QIcon::On] = getIconFile("selected_on", dir, suffix);
141  iconFileMap[QIcon::Selected << 8 | QIcon::Off] = getIconFile("selected_off", dir, suffix);
142  iconFileMap[QIcon::Normal << 8 | QIcon::On] = getIconFile("normal_on", dir, suffix);
143  iconFileMap[QIcon::Normal << 8 | QIcon::Off] = getIconFile("normal_off", dir, suffix);
144  }
145 
146  reader.setFileName(iconFileMap.value(mode << 8 | state));
147 
148  return ImageEntry::pixmap(size, mode, state);
149  }
150 
151  QMap<qint16, QString> iconFileMap;
152 };

182 QSize DBuiltinIconEngine::actualSize(const QSize &size, QIcon::Mode mode,
183  QIcon::State state)
184 {
185  Q_UNUSED(mode);
186  Q_UNUSED(state);
187 
188  ensureLoaded();
189 
190  QIconLoaderEngineEntry *entry = QIconLoaderEngine::entryForSize(m_info, size);
191  if (entry) {
192  const QIconDirInfo &dir = entry->dir;
193  if (dir.type == QIconDirInfo::Scalable) {
194  return size;
195  } else {
196  int result = qMin<int>(dir.size, qMin(size.width(), size.height()));
197  return QSize(result, result);
198  }
199  }
200  return QSize(0, 0);
201 }

215 void DBuiltinIconEngine::paint(QPainter *painter, const QRect &rect,
216  QIcon::Mode mode, QIcon::State state)
217 {
218  ensureLoaded();
219 
220  QSize pixmapSize = rect.size();
221  qreal scale = 1;
222  if (painter->device())
223  scale = painter->device()->devicePixelRatioF();
224 
225  pixmapSize *= scale;
226 
227  QIconLoaderEngineEntry *entry = QIconLoaderEngine::entryForSize(m_info, pixmapSize);
228  if (!entry)
229  return;
230 
231  // 如果有 background 则绘制背景图先
232  QString bgFileName = entry->filename + QStringLiteral(".background");
233 
234  if (QFile::exists(bgFileName)) {
235  QIcon(bgFileName).paint(painter, rect, Qt::AlignCenter, mode, state);
236  }
237 
238  QPixmap pm = entry->pixmap(pixmapSize, mode, state);
239  ImageEntry::Type type = static_cast<ImageEntry *>(entry)->type;
240  if (type == ImageEntry::TextType || (type == ImageEntry::ActionType && mode != QIcon::Normal)) {
241  QPainter pa(&pm);
242  pa.setCompositionMode(QPainter::CompositionMode_SourceIn);
243  pa.fillRect(pm.rect(), painter->pen().brush());
244  }
245 
246  pm.setDevicePixelRatio(scale);
247  painter->drawPixmap(rect, pm);
248 }

284 QThemeIconInfo DBuiltinIconEngine::loadIcon(const QString &iconName, uint key)
285 {
286  QThemeIconInfo info;
287  info.iconName = iconName;
288 
289  QString theme_name = (key == DGuiApplicationHelper::DarkType ? "dark" : "light");
290  QStringList iconDirList {
291  QString("%1/%2/texts").arg(BUILTIN_ICON_PATH, theme_name),
292  QString("%1/%2/actions").arg(BUILTIN_ICON_PATH, theme_name),
293  QString("%1/%2/icons").arg(BUILTIN_ICON_PATH, theme_name),
294  QString("%1/texts").arg(BUILTIN_ICON_PATH),
295  QString("%1/actions").arg(BUILTIN_ICON_PATH),
296  QString("%1/icons").arg(BUILTIN_ICON_PATH)
297  };
298 
299  QDir dir;
300  dir.setFilter(QDir::Files | QDir::Dirs | QDir::NoDotAndDotDot);
301 
302  for (int i = 0; i < iconDirList.count(); ++i) {
303  dir.setPath(iconDirList.at(i));
304 
305  if (!dir.exists()) {
306  continue;
307  }
308 
309  ImageEntry::Type type = static_cast<ImageEntry::Type>(i % 3);
310 
311  for (const QFileInfo &icon_file_info : dir.entryInfoList()) {
312  const QString &file_name = icon_file_info.fileName();
313 
314  // do not load background file
315  if (!file_name.startsWith(iconName) || file_name.endsWith(QStringLiteral(".background")))
316  continue;
317 
318  // 先找到px所在位置
319  int size_str_pos = iconName.length() + 1;
320  int px_str_pos = file_name.indexOf("px.", size_str_pos + 1);
321 
322  if (px_str_pos < 0)
323  continue;
324 
325  bool ok = false;
326  int size = file_name.mid(size_str_pos, px_str_pos - size_str_pos).toInt(&ok);
327 
328  if (Q_UNLIKELY(!ok || size <= 0)) {
329  continue;
330  }
331 
332  ImageEntry *entry = icon_file_info.isDir() ? new DirImageEntry(type) : new ImageEntry(type);
333  entry->filename = icon_file_info.absoluteFilePath();
334  entry->dir.path = icon_file_info.absolutePath();
335  entry->dir.size = size;
336  entry->dir.type = icon_file_info.suffix().startsWith("svg") ? QIconDirInfo::Scalable : QIconDirInfo::Fixed;
337  info.entries.append(entry);
338  }
339 
340  // 已经找到图标时不再继续
341  if (!info.entries.isEmpty()) {
342  break;
343  }
344  }
345 
346  return info;
347 }

354 void DBuiltinIconEngine::virtual_hook(int id, void *data)
355 {
356  ensureLoaded();
357 
358  switch (id) {
359  case QIconEngine::AvailableSizesHook:
360  {
361  QIconEngine::AvailableSizesArgument &arg
362  = *reinterpret_cast<QIconEngine::AvailableSizesArgument*>(data);
363  const int N = m_info.entries.size();
364  QList<QSize> sizes;
365  sizes.reserve(N);
366 
367  // Gets all sizes from the DirectoryInfo entries
368  for (int i = 0; i < N; ++i) {
369  const QIconLoaderEngineEntry *entry = m_info.entries.at(i);
370  int size = entry->dir.size;
371  sizes.append(QSize(size, size));
372  }
373  arg.sizes.swap(sizes); // commit
374  }
375  break;
376  case QIconEngine::IconNameHook:
377  {
378  QString &name = *reinterpret_cast<QString*>(data);
379  name = m_info.iconName;
380  }
381  break;
382  case QIconEngine::IsNullHook:
383  {
384  *reinterpret_cast<bool*>(data) = m_info.entries.isEmpty();
385  }
386  break;
387  case QIconEngine::ScaledPixmapHook:
388  {
389  QIconEngine::ScaledPixmapArgument &arg = *reinterpret_cast<QIconEngine::ScaledPixmapArgument*>(data);
390  // QIcon::pixmap() multiplies size by the device pixel ratio.
391  const int integerScale = qCeil(arg.scale);
392  QIconLoaderEngineEntry *entry = QIconLoaderEngine::entryForSize(m_info, arg.size / integerScale, integerScale);
393  arg.pixmap = entry ? entry->pixmap(arg.size, arg.mode, arg.state) : QPixmap();
394  }
395  break;
396  default:
397  QIconEngine::virtual_hook(id, data);
398  }
399 }

401 void DBuiltinIconEngine::ensureLoaded()
402 {
403  if (Q_LIKELY(m_followSystemTheme) && Q_UNLIKELY(DGuiApplicationHelper::instance()->themeType() != m_key)) {
404  m_initialized = false;
405  // 记录当前使用的主题类型
406  m_key = DGuiApplicationHelper::instance()->themeType();
407  }
408 
409  if (Q_LIKELY(m_initialized)) {
410  return;
411  }
412  // 标记为已初始化
413  m_initialized = true;
414 
415  qDeleteAll(m_info.entries);
416  m_info.entries.clear();
417  m_info.iconName.clear();
418 
419  Q_ASSERT(m_info.entries.size() == 0);
420  m_info = loadIcon(m_iconName, m_key);
421 }
