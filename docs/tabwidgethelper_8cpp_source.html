28 bool Style::drawFrameTabWidgetPrimitive( const QStyleOption* option, QPainter* painter, const QWidget* widget ) const
29 {
30  Q_UNUSED(widget)
31 
32  // cast option and check
33  const QStyleOptionTabWidgetFrame *tabOption(qstyleoption_cast<const QStyleOptionTabWidgetFrame *>(option));
34  if( !tabOption ) return true;
35 
36  // do nothing if tabbar is hidden
37  const bool isQtQuickControl( false );
38  if( tabOption->tabBarSize.isEmpty() && !isQtQuickControl ) return true;
39 
40  // adjust rect to handle overlaps
41  QRect rect( option->rect );
42 
43  const QRect tabBarRect( tabOption->tabBarRect );
44  const QSize tabBarSize( tabOption->tabBarSize );
45  Corners corners = AllCorners;
46 
47  // adjust corners to deal with oversized tabbars
48  switch( tabOption->shape )
49  {
50  case QTabBar::RoundedNorth:
51  case QTabBar::TriangularNorth:
52  if( isQtQuickControl ) rect.adjust( -1, -1, 1, 0 );
53  if( tabBarSize.width() >= rect.width() - 2*Metrics::Frame_FrameRadius ) corners &= ~CornersTop;
54  if( tabBarRect.left() < rect.left() + Metrics::Frame_FrameRadius ) corners &= ~CornerTopLeft;
55  if( tabBarRect.right() > rect.right() - Metrics::Frame_FrameRadius ) corners &= ~CornerTopRight;
56  break;
57 
58  case QTabBar::RoundedSouth:
59  case QTabBar::TriangularSouth:
60  if( isQtQuickControl ) rect.adjust( -1, 0, 1, 1 );
61  if( tabBarSize.width() >= rect.width()-2*Metrics::Frame_FrameRadius ) corners &= ~CornersBottom;
62  if( tabBarRect.left() < rect.left() + Metrics::Frame_FrameRadius ) corners &= ~CornerBottomLeft;
63  if( tabBarRect.right() > rect.right() - Metrics::Frame_FrameRadius ) corners &= ~CornerBottomRight;
64  break;
65 
66  case QTabBar::RoundedWest:
67  case QTabBar::TriangularWest:
68  if( isQtQuickControl ) rect.adjust( -1, 0, 0, 0 );
69  if( tabBarSize.height() >= rect.height()-2*Metrics::Frame_FrameRadius ) corners &= ~CornersLeft;
70  if( tabBarRect.top() < rect.top() + Metrics::Frame_FrameRadius ) corners &= ~CornerTopLeft;
71  if( tabBarRect.bottom() > rect.bottom() - Metrics::Frame_FrameRadius ) corners &= ~CornerBottomLeft;
72  break;
73 
74  case QTabBar::RoundedEast:
75  case QTabBar::TriangularEast:
76  if( isQtQuickControl ) rect.adjust( 0, 0, 1, 0 );
77  if( tabBarSize.height() >= rect.height()-2*Metrics::Frame_FrameRadius ) corners &= ~CornersRight;
78  if( tabBarRect.top() < rect.top() + Metrics::Frame_FrameRadius ) corners &= ~CornerTopRight;
79  if( tabBarRect.bottom() > rect.bottom() - Metrics::Frame_FrameRadius ) corners &= ~CornerBottomRight;
80  break;
81 
82  default: break;
83  }
84 
85  // define colors
86 // const QPalette& palette( option->palette );
87  const QColor background;
88  const QColor outline( Qt::gray );
89 
90  painter->setRenderHint( QPainter::Antialiasing );
91 
92  QRectF frameRect( rect.adjusted( 1, 1, -1, -1 ) );
93  qreal radius( GeometryUtils::frameRadius() );
94 
95  // set pen
96  if( outline.isValid() )
97  {
98 
99  painter->setPen( outline );
100  frameRect.adjust( 0.5, 0.5, -0.5, -0.5 );
101  radius = qMax( radius-1, qreal( 0.0 ) );
102 
103  } else painter->setPen( Qt::NoPen );
104 
105  // set brush
106  if( background.isValid() ) painter->setBrush( background );
107  else painter->setBrush( Qt::NoBrush );
108 
109  // render
110  QPainterPath path( PainterHelper::roundedPath( frameRect, corners, radius ) );
111  QWidget *current_widget = qobject_cast<const QTabWidget*>(widget)->currentWidget();
112  QColor fill_color = painter->pen().color();
113 
114  fill_color.setAlphaF(0.2);
115  fill_color = PainterHelper::colorBlend(option->palette.color(QPalette::Window), fill_color);
116 
117  if (current_widget) {
118  painter->fillPath(path, fill_color);
119 
120  QPainterPath inside_path;
121  QRectF rect = current_widget->parentWidget()->geometry();
122 
123  inside_path.addRoundedRect(rect.adjusted(5.5, 5.5, -5.5, -5.5), radius, radius);
124 
125  painter->fillPath(inside_path, option->palette.color(QPalette::Window));
126  painter->drawPath(inside_path);
127  }
128 
129  QRectF selectedTabRect = QRectF(tabOption->selectedTabRect).adjusted(1.5, 0.5, -0.5, -0.5);
130  QPainterPath selectedtabBarButtonPath(PainterHelper::roundedPath(selectedTabRect, CornerTopLeft | CornerTopRight, radius));
131  QPainterPath tabBarButtonPath;
132 
133  tabBarButtonPath.addRoundedRect(tabOption->tabBarRect.adjusted(1, 0, -1, 0), radius, radius);
134 
135  painter->fillPath(tabBarButtonPath, Qt::gray);
136  painter->drawPath( path );
137  painter->fillPath(selectedtabBarButtonPath, fill_color);
138  painter->drawPath(selectedtabBarButtonPath);
139  painter->setPen(fill_color);
140  selectedTabRect.adjust(1, 0, -1, 0);
141  painter->drawLine(selectedTabRect.bottomLeft(), selectedTabRect.bottomRight());
142 
143  return true;
144 }
