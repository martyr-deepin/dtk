250 void DSettingsGroupPrivate::parseJson(const QString &prefixKey, const QJsonObject &group)
251 {
252  Q_Q(DSettingsGroup);
253  key = group.value("key").toString();
254  Q_ASSERT(!key.isEmpty());
255  key = prefixKey.isEmpty() ? key : prefixKey + "." + key;
256  name = group.value("name").toString();
257  hide = group.value("hide").toBool();
258 
259  for (auto optionJson : group.value("options").toArray()) {
260  auto optionObject = optionJson.toObject();
261  auto option = DSettingsOption::fromJson(key, optionObject);
262  options.insert(option->key(), option);
263  childOptions.insert(option->key(), option);
264  childOptionKeys << option->key();
265  }
266 
267  auto subGroups = group.value("groups").toArray();
268  for (auto subGroup : subGroups) {
269  auto child = DSettingsGroup::fromJson(key, subGroup.toObject());
270  child->setParentGroup(q);
271  childGroups.insert(child->key(), child);
272  childGroupKeys << child->key();
273 
274  for (auto option : child->options()) {
275  options.insert(option->key(), option);
276  }
277  }
278 
279 }
