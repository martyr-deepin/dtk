137 void QSvgIconEnginePrivate::loadDataForModeAndState(DSvgRenderer *renderer, QIcon::Mode mode, QIcon::State state)
138 {
139  QByteArray buf;
140  const QIcon::State oppositeState = state == QIcon::Off ? QIcon::On : QIcon::Off;
141  if (svgBuffers) {
142  buf = svgBuffers->value(hashKey(mode, state));
143  if (buf.isEmpty())
144  buf = svgBuffers->value(hashKey(QIcon::Normal, state));
145  if (buf.isEmpty())
146  buf = svgBuffers->value(hashKey(QIcon::Normal, oppositeState));
147  }
148  if (!buf.isEmpty()) {
149 #ifndef QT_NO_COMPRESS
150  buf = qUncompress(buf);
151 #endif
152  renderer->load(buf);
153  } else {
154  QString svgFile = svgFiles.value(hashKey(mode, state));
155  if (svgFile.isEmpty())
156  svgFile = svgFiles.value(hashKey(QIcon::Normal, state));
157  if (svgFile.isEmpty())
158  svgFile = svgFiles.value(hashKey(QIcon::Normal, oppositeState));
159  if (!svgFile.isEmpty())
160  renderer->load(svgFile);
161  }
162 }

186 QPixmap QSvgIconEngine::pixmap(const QSize &size, QIcon::Mode mode,
187  QIcon::State state)
188 {
189  QPixmap pm;
190 
191  QString pmckey(d->pmcKey(size, mode, state));
192  if (Q_LIKELY(QPixmapCache::find(pmckey, &pm)))
193  return pm;
194 
195  if (d->addedPixmaps) {
196  pm = d->addedPixmaps->value(d->hashKey(mode, state));
197  if (!pm.isNull() && pm.size() == size)
198  return pm;
199  }
200 
201  const QIcon::State oppositeState = state == QIcon::Off ? QIcon::On : QIcon::Off;
202  QString svgFile = d->svgFiles.value(d->hashKey(mode, state));
203  if (svgFile.isEmpty())
204  svgFile = d->svgFiles.value(d->hashKey(QIcon::Normal, state));
205  if (svgFile.isEmpty())
206  svgFile = d->svgFiles.value(d->hashKey(QIcon::Normal, oppositeState));
207 
208  QString cacheFile;
209 
210  if (Q_LIKELY(!svgFile.startsWith(":/") && QFile::exists(svgFile))) {
211  static const QString &cachePath = getIconCachePath();
212 
213  if (Q_LIKELY(!cachePath.isEmpty())) {
214  // svgFiles[-1] 是为了取到额外的cache key,例如会跟随颜色变化的svg图标，将会添加此特殊的标记
215  const QByteArray cacheKey = svgFile.toLocal8Bit() + d->svgFiles[-1].toLocal8Bit();
216  const QString &svgFileSha1 = QString::fromLatin1(QCryptographicHash::hash(cacheKey, QCryptographicHash::Sha1).toHex());
217  cacheFile = QStringLiteral("%1/%2.png").arg(cachePath).arg(svgFileSha1);
218  }
219  }
220 
221  const QFileInfo cacheFileInfo(cacheFile);
222  if (Q_LIKELY(cacheFileInfo.exists())) {
223  const QFileInfo svgFileInfo(svgFile);
224 
225  if (Q_UNLIKELY(svgFileInfo.lastModified() != cacheFileInfo.lastModified())) {
226  // clear invalid cache file
227  QFile::remove(cacheFile);
228  } else {
229  qCDebug(lcDSvg()) << "found cache file:" << cacheFile << ", for:" << svgFile;
230 
231  QImageReader ir(cacheFile);
232  const QSize &cacheSize = ir.size();
233 
234  if (Q_LIKELY(cacheSize.isValid() && cacheSize.width() >= size.width())) {
235  pm = QPixmap::fromImage(ir.read().scaledToWidth(size.width()));
236  } else { // 当对同一个图标文件有更大的size要求时，应当继续从源svg文件重新渲染图标并更新缓存文件
237  qCDebug(lcDSvg()) << "cache image size less then target size, cache size:" << cacheSize << ", request size:" << size;
238 
239  if (Q_UNLIKELY(!ir.canRead())) {
240  qCWarning(lcDSvg()) << "can't read the cache image:" << cacheFile << "file size:" << cacheFileInfo.size() << ", permissions:" << cacheFileInfo.permissions();
241  QFile::remove(cacheFile);
242  }
243  }
244  }
245  }
246 
247  if (Q_UNLIKELY(pm.isNull())) {
248  DSvgRenderer renderer;
249  d->loadDataForModeAndState(&renderer, mode, state);
250  if (Q_UNLIKELY(!renderer.isValid()))
251  return pm;
252 
253  QSize actualSize = renderer.defaultSize();
254  if (actualSize.width() < size.width())
255  actualSize.scale(size, Qt::KeepAspectRatio);
256 
257  if (Q_UNLIKELY(actualSize.isEmpty()))
258  return pm;
259 
260  const QImage image = renderer.toImage(actualSize);
261 
262  if (Q_LIKELY(!image.isNull() && !cacheFile.isEmpty())) {
263  QtConcurrent::run(QThreadPool::globalInstance(), [image, cacheFile, svgFile] {
264  QSaveFile file(cacheFile);
265  // 增加cache文件能被成功保存的概率
266  file.setDirectWriteFallback(true);
267  if (file.open(QFile::WriteOnly)) {
268  if (image.save(&file, "png", 80) && file.commit()) {
269  QFileInfo svgFileInfo(svgFile);
270  // 不能直接使用QSaveFile改写文件的时间
271  QFile file(cacheFile);
272  if (!file.open(QFile::ReadWrite) || !file.setFileTime(svgFileInfo.lastModified(), QFileDevice::FileModificationTime)) {
273  qCWarning(lcDSvg()) << "set cache file modified date time failed, error message:" << file.errorString()
274  << ", cache file:" << cacheFile << ", svg file:" << svgFile;
275  }
276  } else {
277  file.cancelWriting();
278  qCWarning(lcDSvg()) << "save cache image failed, cache file:" << cacheFile << ", svg file:" << svgFile;
279  }
280  } else {
281  qCWarning(lcDSvg()) << "open cache file failed, error message:" << file.errorString()
282  << ", cache file:" << cacheFile << ", svg file:" << svgFile;
283  }
284  });
285  }
286 
287  // 缩放图标到目标大小
288  pm = QPixmap::fromImage(image.scaledToWidth(size.width()));
289  }
290 
291  if (Q_LIKELY(qobject_cast<QGuiApplication *>(QCoreApplication::instance()))) {
292  const QPixmap generated = QGuiApplicationPrivate::instance()->applyQIconStyleHelper(mode, pm);
293  if (Q_LIKELY(!generated.isNull()))
294  pm = generated;
295  }
296 
297  if (Q_LIKELY(!pm.isNull()))
298  QPixmapCache::insert(pmckey, pm);
299 
300  return pm;
301 }

315 static FileType fileType(const QFileInfo &fi)
316 {
317  const QString &abs = fi.absoluteFilePath();
318  if (abs.endsWith(QLatin1String(".svg"), Qt::CaseInsensitive))
319  return SvgFile;
320  if (abs.endsWith(QLatin1String(".svgz"), Qt::CaseInsensitive)
321  || abs.endsWith(QLatin1String(".svg.gz"), Qt::CaseInsensitive)) {
322  return CompressedSvgFile;
323  }
324 #ifndef QT_NO_MIMETYPE
325  const QString &mimeTypeName = QMimeDatabase().mimeTypeForFile(fi).name();
326  if (mimeTypeName == QLatin1String("image/svg+xml"))
327  return SvgFile;
328  if (mimeTypeName == QLatin1String("image/svg+xml-compressed"))
329  return CompressedSvgFile;
330 #endif // !QT_NO_MIMETYPE
331  return OtherFile;
332 }

334 void QSvgIconEngine::addFile(const QString &fileName, const QSize &,
335  QIcon::Mode mode, QIcon::State state)
336 {
337  if (!fileName.isEmpty()) {
338  const QFileInfo fi(fileName);
339  const QString abs = fi.absoluteFilePath();
340  const FileType type = fileType(fi);
341 #ifndef QT_NO_COMPRESS
342  if (type == SvgFile || type == CompressedSvgFile) {
343 #else
344  if (type == SvgFile) {
345 #endif
346  DSvgRenderer renderer(abs);
347  if (renderer.isValid()) {
348  d->stepSerialNum();
349  d->svgFiles.insert(d->hashKey(mode, state), abs);
350  }
351  } else if (type == OtherFile) {
352  QPixmap pm(abs);
353  if (!pm.isNull())
354  addPixmap(pm, mode, state);
355  }
356  }

379 bool QSvgIconEngine::read(QDataStream &in)
380 {
381  d = new QSvgIconEnginePrivate;
382  d->svgBuffers = new QHash<int, QByteArray>;
383 
384  if (in.version() >= QDataStream::Qt_4_4) {
385  int isCompressed;
386  QHash<int, QString> fileNames; // For memoryoptimization later
387  in >> fileNames >> isCompressed >> *d->svgBuffers;
388 #ifndef QT_NO_COMPRESS
389  if (!isCompressed) {
390  for (auto it = d->svgBuffers->begin(), end = d->svgBuffers->end(); it != end; ++it)
391  it.value() = qCompress(it.value());
392  }
393 #else
394  if (isCompressed) {
395  qWarning("QSvgIconEngine: Can not decompress SVG data");
396  d->svgBuffers->clear();
397  }
398 #endif
399  int hasAddedPixmaps;
400  in >> hasAddedPixmaps;
401  if (hasAddedPixmaps) {
402  d->addedPixmaps = new QHash<int, QPixmap>;
403  in >> *d->addedPixmaps;
404  }
405 
406  d->svgFiles = fileNames;
407  }
408  else {
409  QPixmap pixmap;
410  QByteArray data;
411  uint mode;
412  uint state;
413  int num_entries;
414 
415  in >> data;
416  if (!data.isEmpty()) {
417 #ifndef QT_NO_COMPRESS
418  data = qUncompress(data);
419 #endif
420  if (!data.isEmpty())
421  d->svgBuffers->insert(d->hashKey(QIcon::Normal, QIcon::Off), data);
422  }
423  in >> num_entries;
424  for (int i=0; i<num_entries; ++i) {
425  if (in.atEnd())
426  return false;
427  in >> pixmap;
428  in >> mode;
429  in >> state;
430  // The pm list written by 4.3 is buggy and/or useless, so ignore.
431  //addPixmap(pixmap, QIcon::Mode(mode), QIcon::State(state));
432  }
433  }
434 
435  return true;
436 }

439 bool QSvgIconEngine::write(QDataStream &out) const
440 {
441  if (out.version() >= QDataStream::Qt_4_4) {
442  int isCompressed = 0;
443 #ifndef QT_NO_COMPRESS
444  isCompressed = 1;
445 #endif
446  QHash<int, QByteArray> svgBuffers;
447  if (d->svgBuffers)
448  svgBuffers = *d->svgBuffers;
449  for (auto it = d->svgFiles.cbegin(), end = d->svgFiles.cend(); it != end; ++it) {
450  QByteArray buf;
451  QFile f(it.value());
452  if (f.open(QIODevice::ReadOnly))
453  buf = f.readAll();
454 #ifndef QT_NO_COMPRESS
455  buf = qCompress(buf);
456 #endif
457  svgBuffers.insert(it.key(), buf);
458  }
459  out << d->svgFiles << isCompressed << svgBuffers;
460  if (d->addedPixmaps)
461  out << (int)1 << *d->addedPixmaps;
462  else
463  out << (int)0;
464  }
465  else {
466  QByteArray buf;
467  if (d->svgBuffers)
468  buf = d->svgBuffers->value(d->hashKey(QIcon::Normal, QIcon::Off));
469  if (buf.isEmpty()) {
470  QString svgFile = d->svgFiles.value(d->hashKey(QIcon::Normal, QIcon::Off));
471  if (!svgFile.isEmpty()) {
472  QFile f(svgFile);
473  if (f.open(QIODevice::ReadOnly))
474  buf = f.readAll();
475  }
476  }
477 #ifndef QT_NO_COMPRESS
478  buf = qCompress(buf);
479 #endif
480  out << buf;
481  // 4.3 has buggy handling of added pixmaps, so don't write any
482  out << (int)0;
483  }
484  return true;
485 }
