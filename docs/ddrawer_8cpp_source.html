51 void DDrawerPrivate::init()
52 {
53  DDrawer *qq = q_func();
54  qq->setSizePolicy(QSizePolicy::Preferred, QSizePolicy::Maximum);
55  QVBoxLayout *mainLayout = new QVBoxLayout(qq);
56  mainLayout->setContentsMargins(0, 0, 0, 0);
57  mainLayout->setSpacing(0);
58 
59  m_headerLayout = new QVBoxLayout();
60  m_headerLayout->setContentsMargins(0, 0, 0, 0);
61  m_headerLayout->setAlignment(Qt::AlignCenter);
62 
63  m_hSeparator = new DHorizontalLine();
64  m_bottom_separator = new DHorizontalLine;
65  m_bottom_separator->hide();
66  m_hSeparator->setAccessibleName("DDrawerHorizontalSeparator");
67  m_bottom_separator->setAccessibleName("DDrawerBottomSeparator");
68 
69  qq->connect(qq, &DDrawer::expandChange, m_bottom_separator, &QWidget::setVisible);
70 
71  m_contentLoader = new ContentBox();
72  m_contentLoader->setFixedHeight(0); // default to not expanded.
73  m_contentLoader->setAccessibleName("DDrawerContentBox");
74 
75  m_boxWidget = new DVBoxWidget;
76  m_boxWidget->setAccessibleName("DDrawerVBoxWidget");
77  m_contentLayout = m_boxWidget->layout();
78 
79  QVBoxLayout *layout_contentLoader = new QVBoxLayout(m_contentLoader);
80 
81  layout_contentLoader->setMargin(0);
82  layout_contentLoader->setSpacing(0);
83  layout_contentLoader->addWidget(m_boxWidget);
84  layout_contentLoader->addStretch();
85 
86  m_animation = new QPropertyAnimation(m_contentLoader, "height", qq);
87  m_animation->setDuration(200);
88  m_animation->setEasingCurve(QEasingCurve::InSine);
89  qq->connect(m_animation, &QPropertyAnimation::valueChanged, qq, [qq] {
90  qq->setFixedHeight(qq->sizeHint().height());
91  });
92 
93  mainLayout->addLayout(m_headerLayout);
94  mainLayout->addWidget(m_hSeparator);
95  mainLayout->addWidget(m_contentLoader);
96  mainLayout->addWidget(m_bottom_separator);
97 
98  qq->setLayout(mainLayout);
99 
100  qq->connect(m_boxWidget, &DBoxWidget::sizeChanged, qq, [this] {
101  if (m_expand) {
102  int endHeight = 0;
103  endHeight = m_boxWidget->height();
104 
105  m_animation->setStartValue(m_contentLoader->height());
106  m_animation->setEndValue(endHeight);
107  m_animation->stop();
108  m_animation->start();
109  }
110  });
111 }

222 void DDrawer::setExpand(bool value)
223 {
224  Q_D(DDrawer);
225 
226  if (d->m_expand == value) {
227  return;
228  }
229 
230  d->m_expand = value;
231  Q_EMIT expandChange(value);
232 
233  if (value) {
234  d->m_animation->setStartValue(0);
235  d->m_animation->setEndValue(d->m_boxWidget->height());
236  } else {
237  d->m_animation->setStartValue(d->m_boxWidget->height());
238  d->m_animation->setEndValue(0);
239  }
240 
241  d->m_animation->stop();
242  d->m_animation->start();
243 }
