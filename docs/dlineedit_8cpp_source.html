167 void DLineEdit::setLeftWidgets(const QList<QWidget *> &list)
168 {
169  Q_D(DLineEdit);
170 
171  if (d->leftWidget != nullptr) {
172  d->leftWidget->hide();
173  d->leftWidget->deleteLater();
174  d->leftWidget = nullptr;
175  }
176 
177  if (list.isEmpty())
178  return;
179 
180  d->leftWidget = new QWidget;
181  QHBoxLayout *layout = new QHBoxLayout(d->leftWidget);
182  layout->setContentsMargins(0, 0, 0, 0);
183  d->hLayout->insertWidget(0, d->leftWidget);
184 
185  QList<QWidget *>::const_iterator itor;
186 
187  for (itor = list.constBegin(); itor != list.constEnd(); itor++) {
188  layout->addWidget(*itor);
189  }
190 
191  d->leftWidget->adjustSize();
192 }

402 bool DLineEdit::eventFilter(QObject *watched, QEvent *event)
403 {
404  D_D(DLineEdit);
405 
406  if (event->type() == QEvent::FocusIn) {
407  Q_EMIT focusChanged(true);
408  } else if (event->type() == QEvent::FocusOut) {
409  Q_EMIT focusChanged(false);
410  } else if (watched == lineEdit() && event->type() == QEvent::KeyPress) {
411  QKeyEvent *pKeyEvent = static_cast<QKeyEvent *>(event);
412  if (pKeyEvent == QKeySequence::SelectAll) {
413  QApplication::clipboard()->setText(lineEdit()->text(), QClipboard::Mode::Selection);
414  }
415  } else if (watched == lineEdit() && event->type() == QEvent::ContextMenu && lineEdit()->contextMenuPolicy() == Qt::DefaultContextMenu) {
416  QLineEdit *le = static_cast<QLineEdit *>(watched);
417  if (!le->isEnabled() || le->echoMode() == QLineEdit::Password ||
418  (!textToSpeechIsEnabled() && !textToTranslateIsEnabled() && !speechToTextIsEnabled())) {
419  return QWidget::eventFilter(watched, event);
420  }
421 
422  QDBusInterface testSpeech("com.iflytek.aiassistant",
423  "/aiassistant/tts",
424  "com.iflytek.aiassistant.tts",
425  QDBusConnection::sessionBus());
426  //测试朗读接口是否开启
427  QDBusReply<bool> speechReply = testSpeech.call(QDBus::AutoDetect, "getTTSEnable");
428 
429  QDBusInterface testReading("com.iflytek.aiassistant",
430  "/aiassistant/tts",
431  "com.iflytek.aiassistant.tts",
432  QDBusConnection::sessionBus());
433  //测试朗读是否在进行
434  QDBusReply<bool> readingReply = testReading.call(QDBus::AutoDetect, "isTTSInWorking");
435 
436  QDBusInterface testTranslate("com.iflytek.aiassistant",
437  "/aiassistant/trans",
438  "com.iflytek.aiassistant.trans",
439  QDBusConnection::sessionBus());
440  //测试翻译接口是否开启
441  QDBusReply<bool> translateReply = testTranslate.call(QDBus::AutoDetect, "getTransEnable");
442 
443  QDBusInterface testSpeechToText("com.iflytek.aiassistant",
444  "/aiassistant/iat",
445  "com.iflytek.aiassistant.iat",
446  QDBusConnection::sessionBus());
447  //测试听写接口是否开启
448  QDBusReply<bool> speechToTextReply = testSpeechToText.call(QDBus::AutoDetect, "getIatEnable");
449 
450  QLineEdit *pLineEdit = static_cast<QLineEdit*>(watched);
451  QMenu *menu = pLineEdit->createStandardContextMenu();
452 
453  connect(menu, &QMenu::triggered, this, [pLineEdit](QAction *pAction) {
454  if (pAction->text().startsWith(QLineEdit::tr("Select All"))) {
455  QApplication::clipboard()->setText(pLineEdit->text(), QClipboard::Mode::Selection);
456  }
457  });
458 
459  //朗读,翻译,听写都没有开启，则弹出默认菜单
460  if (!speechReply.value() && !translateReply.value() && !speechToTextReply.value()) {
461  menu->popup(static_cast<QContextMenuEvent*>(event)->globalPos());
462  event->accept();
463  return true;
464  }
465 
466  menu->addSeparator();
467 
468  if (speechReply.value() && textToSpeechIsEnabled()) {
469  QAction *pAction_1 = nullptr;
470  if (readingReply.value()) {
471  pAction_1 = menu->addAction(QCoreApplication::translate("DLineEdit", "Stop reading"));
472  } else {
473  pAction_1 = menu->addAction(QCoreApplication::translate("DLineEdit", "Text to Speech"));
474  }
475 
476  //没有选中文本，则菜单置灰色
477  if (!pLineEdit->hasSelectedText()) {
478  pAction_1->setEnabled(false);
479  }
480 
481  connect(pAction_1, &QAction::triggered, this, [] {
482  QDBusInterface speechInterface("com.iflytek.aiassistant",
483  "/aiassistant/deepinmain",
484  "com.iflytek.aiassistant.mainWindow",
485  QDBusConnection::sessionBus());
486 
487  if (speechInterface.isValid()) {
488  speechInterface.call(QDBus::BlockWithGui, "TextToSpeech");//此函在第一次调用时朗读，在朗读状态下再次调用为停止朗读
489  } else {
490  qWarning() << "[DLineEdit] TextToSpeech ERROR";
491  }
492  });
493  }
494 
495  if (translateReply.value() && textToTranslateIsEnabled()) {
496  QAction *pAction_2 = menu->addAction(QCoreApplication::translate("DLineEdit", "Translate"));
497 
498  //没有选中文本，则菜单置灰色
499  if (!pLineEdit->hasSelectedText()) {
500  pAction_2->setEnabled(false);
501  }
502 
503  connect(pAction_2, &QAction::triggered, this, [] {
504  QDBusInterface translationInterface("com.iflytek.aiassistant",
505  "/aiassistant/deepinmain",
506  "com.iflytek.aiassistant.mainWindow",
507  QDBusConnection::sessionBus());
