87 bool Style::drawPushButtonLabel(const QStyleOption *option, QPainter *painter, const QWidget *widget) const
88 {
89  // cast option and check
90  const QStyleOptionButton* buttonOption( qstyleoption_cast<const QStyleOptionButton*>( option ) );
91  if( !buttonOption ) return true;
92 
93  // copy rect and palette
94  const QRect& rect( option->rect );
95 
96  // state
97  const QStyle::State& state( option->state );
98  const bool enabled( state & QStyle::State_Enabled );
99  const bool sunken( state & (QStyle::State_On | QStyle::State_Sunken) );
100  const bool mouseOver(option->state & QStyle::State_MouseOver);
101  const bool hasFocus(option->state & QStyle::State_HasFocus);
102  const bool flat( buttonOption->features & QStyleOptionButton::Flat );
103  const bool suggest(
104 #ifdef DTKWIDGET_CLASS_DSuggestButton
105  buttonOption->features & DStyleOptionButton::SuggestButton
106 #else
107  false
108 #endif
109  );
110 
111  // content
112  const bool hasText( !buttonOption->text.isEmpty() );
113  const bool hasIcon( ( flat || !hasText ) && !buttonOption->icon.isNull() );
114 
115  // contents
116  QRect contentsRect( rect );
117 
118  // menu arrow
119  /*
120  if( buttonOption->features & QStyleOptionButton::HasMenu )
121  {
122 
123  // define rect
124  QRect arrowRect( contentsRect );
125  arrowRect.setLeft( contentsRect.right() - Metrics::MenuButton_IndicatorWidth + 1 );
126  arrowRect = GeometryUtils::centerRect( arrowRect, Metrics::MenuButton_IndicatorWidth, Metrics::MenuButton_IndicatorWidth );
127 
128  contentsRect.setRight( arrowRect.left() - Metrics::Button_ItemSpacing - 1 );
129  contentsRect.adjust( Metrics::Button_MarginWidth, 0, 0, 0 );
130 
131  arrowRect = widget->style()->visualRect( option->direction, option->rect, arrowRect );
132 
133  // define color
134  const QColor arrowColor( _helper->arrowColor( palette, textRole ) );
135  _helper->renderArrow( painter, arrowRect, arrowColor, ArrowDown );
136 
137  }
138  */
139 
140  // icon size
141  QSize iconSize;
142  if( hasIcon )
143  {
144  iconSize = buttonOption->iconSize;
145  if( !iconSize.isValid() )
146  {
147  const int metric( proxy()->pixelMetric( QStyle::PM_SmallIconSize, option, widget ) );
148  iconSize = QSize( metric, metric );
149  }
150  }
151 
152  // text size
153  int textFlags( Qt::AlignCenter );
154  const QSize textSize( option->fontMetrics.size( textFlags, buttonOption->text ) );
155 
156  if (styleHint(SH_UnderlineShortcut, buttonOption, widget))
157  textFlags |= Qt::TextShowMnemonic;
158  else
159  textFlags |= Qt::TextHideMnemonic;
160 
161  // adjust text and icon rect based on options
162  QRect iconRect;
163  QRect textRect;
164 
165  if( hasText && !hasIcon ) textRect = contentsRect;
166  else if( hasIcon && !hasText ) iconRect = contentsRect;
167  else {
168 
169  const int contentsWidth( iconSize.width() + textSize.width() + Metrics::Button_ItemSpacing );
170  iconRect = QRect( QPoint( contentsRect.left() + (contentsRect.width() - contentsWidth )/2, contentsRect.top() + (contentsRect.height() - iconSize.height())/2 ), iconSize );
171  textRect = QRect( QPoint( iconRect.right() + Metrics::ToolButton_ItemSpacing + 1, contentsRect.top() + (contentsRect.height() - textSize.height())/2 ), textSize );
172 
173  }
174 
175  // handle right to left
176  if( iconRect.isValid() ) iconRect = proxy()->visualRect( option->direction, option->rect, iconRect );
177  if( textRect.isValid() ) textRect = proxy()->visualRect( option->direction, option->rect, textRect );
178 
179  // make sure there is enough room for icon
180  if( iconRect.isValid() ) iconRect = GeometryUtils::centerRect( iconRect, iconSize );
181 
182  // render icon
183  if( hasIcon && iconRect.isValid() ) {
184 
185  // icon state and mode
186  const QIcon::State iconState( sunken ? QIcon::On : QIcon::Off );
187  QIcon::Mode iconMode;
188  if( !enabled ) iconMode = QIcon::Disabled;
189  else if( !flat && hasFocus ) iconMode = QIcon::Selected;
190  else if( mouseOver && flat ) iconMode = QIcon::Active;
191  else iconMode = QIcon::Normal;
192 
193  buttonOption->icon.paint(painter, iconRect, Qt::AlignCenter, iconMode, iconState);
194  }
195 
196  // render text
197  if( hasText && textRect.isValid() ) {
198  if (suggest) {
199  painter->setPen(m_palette->brush(PaletteExtended::PushButton_SuggestTextColor, option).color());
200  } else {
201  painter->setPen(m_palette->brush(PaletteExtended::PushButton_TextColor, option).color());
202  }
203 
204  if (buttonOption->features & QStyleOptionButton::HasMenu)
205  textRect = textRect.adjusted(0, 0, -proxy()->pixelMetric(PM_MenuButtonIndicator, buttonOption, widget), 0);
206  painter->drawText(textRect, textFlags, buttonOption->text);
207  }
208 
209  return true;
210 }

212 bool Style::drawPushButtonFrame(QPainter* painter, const QRect& rect, const QBrush& brush, const QBrush& outline, const QColor& shadow, const QWidget *widget) const
213 {
214  Q_UNUSED(shadow)
215 
216  // setup painter
217  painter->setRenderHint( QPainter::Antialiasing, true );
218 
219  const qreal radius( GeometryUtils::frameRadius() );
220  const QMarginsF margins(0.5, 0.5, 0.5, 0.5);
221 
222  QPainterPath path;
223  path.addRoundedRect(QRectF(rect).marginsRemoved(margins), radius, radius);
224 
225  // ButtonTuples are QPushButton groups that needs to be taken care of,
226  // the right corners of the LeftButton are not rounded, and the left
227  // corners of the RightButton are not rounded either.
228  if (widget) {
229  const QString className = widget->metaObject()->className();
230  const QRectF rf = QRectF(rect);
231  if (className == "dcc::widgets::LeftButton") {
232  QPainterPath rightHalf;
233  rightHalf.addRect(rf.right() - radius, rf.y() + margins.top(), radius, rf.height() - margins.top() - margins.bottom());
234  path = path.united(rightHalf);
235  } else if (className == "dcc::widgets::RightButton") {
236  QPainterPath leftHalf;
237  leftHalf.addRect(rf.left(), rf.y() + margins.top(), radius, rf.height() - margins.top() - margins.bottom());
238  path = path.united(leftHalf);
239  }
240  }
241 
242  PainterHelper::drawPath(painter, path, brush, Metrics::Painter_PenWidth, outline);
243 
244  return true;
245 }
