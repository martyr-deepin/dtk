675 void DImageButtonPrivate::updateIcon()
676 {
677  D_Q(DImageButton);
678 
679  switch (m_state) {
680  case DImageButton::Hover:
681  q->setProperty("state", "hover");
682  if (!m_hoverPic.isEmpty()) { q->setPixmap(loadPixmap(m_hoverPic)); }
683  break;
684  case DImageButton::Press:
685  q->setProperty("state", "press");
686  if (!m_pressPic.isEmpty()) { q->setPixmap(loadPixmap(m_pressPic)); }
687  break;
688  case DImageButton::Checked:
689  q->setProperty("state", "checked");
690  if (!m_checkedPic.isEmpty()) { q->setPixmap(loadPixmap(m_checkedPic)); }
691  break;
692  case DImageButton::Disabled:
693  q->setProperty("state", "disabled");
694  if (!m_disabledPic.isEmpty()) { q->setPixmap(loadPixmap(m_disabledPic)); }
695  break;
696  default:
697  q->setProperty("state", "");
698  if (!m_normalPic.isEmpty()) { q->setPixmap(loadPixmap(m_normalPic)); }
699  break;
700  }
701 
702  q->setAlignment(Qt::AlignCenter);
703 
704  Q_EMIT q->stateChanged();
705 }

718 QPixmap DImageButtonPrivate::loadPixmap(const QString &path)
719 {
720  D_Q(DImageButton);
721 
722  qreal ratio = 1.0;
723 
724  const qreal devicePixelRatio = q->devicePixelRatioF();
725 
726  QPixmap pixmap;
727 
728  if (!qFuzzyCompare(ratio, devicePixelRatio)) {
729  QImageReader reader;
730  reader.setFileName(qt_findAtNxFile(path, devicePixelRatio, &ratio));
731  if (reader.canRead()) {
732  reader.setScaledSize(reader.size() * (devicePixelRatio / ratio));
733  pixmap = QPixmap::fromImage(reader.read());
734  pixmap.setDevicePixelRatio(devicePixelRatio);
735  }
736  } else {
737  pixmap.load(path);
738  }
739 
740  return pixmap;
741 }
