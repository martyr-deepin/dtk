226 void DSettingsGroupPrivate::parseJson(const QString &prefixKey, const QJsonObject &group)
227 {
228  Q_Q(DSettingsGroup);
229  key = group.value("key").toString();
230  Q_ASSERT(!key.isEmpty());
231  key = prefixKey.isEmpty() ? key : prefixKey + "." + key;
232  name = group.value("name").toString();
233  hide = group.value("hide").toBool();
234 
235  for (auto optionJson : group.value("options").toArray()) {
236  auto optionObject = optionJson.toObject();
237  auto option = DSettingsOption::fromJson(key, optionObject);
238  option->setParent(q);
239  options.insert(option->key(), option);
240  childOptions.insert(option->key(), option);
241  childOptionKeys << option->key();
242  }
243 
244  auto subGroups = group.value("groups").toArray();
245  for (auto subGroup : subGroups) {
246  auto child = DSettingsGroup::fromJson(key, subGroup.toObject());
247  child->setParent(q);
248  child->setParentGroup(q);
249  childGroups.insert(child->key(), child);
250  childGroupKeys << child->key();
251 
252  for (auto option : child->options()) {
253  options.insert(option->key(), option);
254  }
255  }
256 
257 }
