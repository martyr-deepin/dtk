55 TEST_F(TDGuiApplicationHelper, testFunction)
56 {
57  QColor testColor(Qt::red);
58  QColor adjustedColor = helper->adjustColor(testColor, 0, 0, 0, 0, 0, 0, -20);
59  ASSERT_NE(testColor, adjustedColor);
60  ASSERT_TRUE(adjustedColor.isValid());
61 
62  QColor disBlendColor = Qt::black;
63  QColor blendedColor = helper->blendColor(testColor, disBlendColor);
64  ASSERT_NE(testColor, blendedColor);
65  ASSERT_TRUE(blendedColor.isValid());
66 
67  DPalette tPalette;
68  helper->generatePaletteColor(tPalette, QPalette::Window, DGuiApplicationHelper::ColorType::LightType);
69  ASSERT_EQ(tPalette.brush(QPalette::Disabled, QPalette::Window), tPalette.brush(QPalette::Normal, QPalette::Window));
70 
71  tPalette.setColor(DPalette::Background, Qt::black);
72  helper->generatePaletteColor(tPalette, QPalette::Highlight, DGuiApplicationHelper::ColorType::DarkType);
73  ASSERT_TRUE(tPalette.highlight().color().isValid());
74 
75  // 初始化调色板为默认值
76  helper->generatePalette(tPalette);
77 
78  ASSERT_EQ(helper->isXWindowPlatform(), DGuiApplicationHelper::testAttribute(DGuiApplicationHelper::IsXWindowPlatform));
79  ASSERT_EQ(helper->isTabletEnvironment(), DGuiApplicationHelper::testAttribute(DGuiApplicationHelper::IsTableEnvironment));
80 
81  ASSERT_TRUE(helper->applicationTheme());
82  ASSERT_TRUE(helper->systemTheme());
83  ASSERT_NE(helper->applicationPalette(), tPalette);
84  qGuiApp->setAttribute(Qt::AA_SetPalette, false);
85  helper->setPaletteType(DGuiApplicationHelper::DarkType);
86  ASSERT_NE(helper->applicationPalette(), tPalette);
87 
88  helper->setApplicationPalette(tPalette);
89  ASSERT_EQ(helper->applicationPalette(), tPalette);
90 
91  ASSERT_TRUE(helper->fontManager());
92  ASSERT_EQ(helper->toColorType(QColor(Qt::white)), DGuiApplicationHelper::LightType);
93  ASSERT_EQ(helper->toColorType(QColor(Qt::black)), DGuiApplicationHelper::DarkType);
94  ASSERT_EQ(helper->themeType(), DGuiApplicationHelper::DarkType);
95  ASSERT_EQ(helper->paletteType(), DGuiApplicationHelper::DarkType);
96 }
