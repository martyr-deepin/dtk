883 void Logger::write(const QDateTime& timeStamp, LogLevel logLevel, const char* file, int line, const char* function, const char* category,
884  const QString& message, bool fromLocalInstance)
885 {
886  Q_D(Logger);
887 
888  QMutexLocker locker(&d->loggerMutex);
889 
890  QString logCategory = QString::fromLatin1(category);
891  if (logCategory.isNull() && !d->defaultCategory.isNull())
892  logCategory = d->defaultCategory;
893 
894  bool wasWritten = false;
895  bool isGlobalInstance = this == globalInstance();
896  bool linkedToGlobal = isGlobalInstance && d->categories.value(logCategory, false);
897 
898  if (!logCategory.isNull())
899  {
900  QList<AbstractAppender*> appenders = d->categoryAppenders.values(logCategory);
901  if (appenders.length() == 0)
902  {
903  if (logCategory != d->defaultCategory && !linkedToGlobal && !fromLocalInstance)
904  std::cerr << "No appenders assotiated with category " << qPrintable(logCategory) << std::endl;
905  }
906  else
907  {
908  Q_FOREACH (AbstractAppender* appender, appenders)
909  appender->write(timeStamp, logLevel, file, line, function, logCategory, message);
910  wasWritten = true;
911  }
912  }
913 
914  // the default category is linked to the main logger appenders
915  // global logger instance also writes all linked categories to the main appenders
916  if (logCategory.isNull() || logCategory == d->defaultCategory || linkedToGlobal)
917  {
918  if (!d->appenders.isEmpty())
919  {
920  Q_FOREACH (AbstractAppender* appender, d->appenders)
921  appender->write(timeStamp, logLevel, file, line, function, logCategory, message);
922  wasWritten = true;
923  }
924  else
925  {
926  static bool noAppendersWarningShown = false;
927  if (!noAppendersWarningShown)
928  {
929 #if defined(Q_OS_ANDROID)
930  __android_ write(ANDROID_LOG_WARN, "Logger", "No appenders registered with logger");
931 #else
932  std::cerr << "No appenders registered with logger" << std::endl;
933 #endif
934  noAppendersWarningShown = true;
935  }
936  }
937  }
938 
939  // local logger instances send category messages to the global instance
940  if (!logCategory.isNull() && !isGlobalInstance)
941  globalInstance()->write(timeStamp, logLevel, file, line, function, logCategory.toLatin1(), message, true);
942 
943  if (!wasWritten && !fromLocalInstance)
944  {
945  // Fallback
946 #if defined(Q_OS_ANDROID)
947  QString result = QString(QLatin1String("<%2> %3")).arg(AbstractStringAppender::stripFunctionName(function)).arg(message);
948  __android_log_write(AndroidAppender::androidLogPriority(logLevel), "Logger", qPrintable(result));
949 #else
950  QString result = QString(QLatin1String("[%1] <%2> %3")).arg(levelToString(logLevel), -7)
951  .arg(AbstractStringAppender::stripFunctionName(function)).arg(message);
952  std::cerr << qPrintable(result) << std::endl;
953 #endif
954  }
955 
956  if (logLevel == Logger::Fatal)
957  abort();
958 }
