135 bool DSysInfoPrivate::splitA_BC_DMode()
136 {
137  // A-BC-D
138  bool ok = false;
139  uint minv = minorVersion.toUInt(&ok);
140  if (ok) {
141  minVersion.D = minv % 10;
142  } else if (minorVersion.length() > 0) {
143  const QString D = minorVersion.right(1);
144  if (D.contains(QRegExp("[0-9A-Z]"))) {
145  // 0-9...A-Z
146  minVersion.D = 10 + static_cast<uint>(D.data()->toLatin1() - 'A');
147  } else {
148  qWarning() << "invalid minorVersion";
149  minVersion.D = 0;
150  }
151  }
152  uint minVer = minorVersion.left(3).toUInt();
153  minVersion.BC = minVer % 100;
154  minVer /= 100;
155  minVersion.A = minVer % 10;
156  minVersion.type = MinVersion::A_BC_D;
157  return ok;
158 }

160 void DSysInfoPrivate::ensureDeepinInfo()
161 {
162  if (static_cast<int>(deepinType) >= 0)
163  return;
164 
165  QFile file("/etc/deepin-version");
166 
167  if (!file.open(QFile::ReadOnly)) {
168  deepinType = DSysInfo::UnknownDeepin;
169 
170  return;
171  }
172 
173  char buf[1024];
174  int buf_length = 0;
175 
176  Q_FOREVER {
177  buf_length = file.readLine(buf, sizeof(buf));
178 
179  if (buf_length < 0)
180  break;
181 
182  const QByteArray line(buf, buf_length);
183  const QByteArrayList &list = line.split('=');
184 
185  if (list.count() != 2) {
186  continue;
187  }
188 
189  const auto key_value = qMakePair(list.first().trimmed(), list.last().trimmed());
190 
191  if (key_value.first == "Version") {
192  deepinVersion = key_value.second;
193  } else if (line.startsWith("Type")) {
194  if (key_value.first == "Type") {
195  deepinTypeMap[QString()] = QString::fromLatin1(key_value.second);
196  } else if (key_value.first.at(4) == '[' && key_value.first.at(key_value.first.size() - 1) == ']') {
197  const QByteArray &language = key_value.first.mid(5, key_value.first.size() - 6);
198 
199  if (!language.isEmpty()) {
200  deepinTypeMap[QString::fromLatin1(language)] = QString::fromUtf8(key_value.second);
201  }
202  }
203  } else if (key_value.first == "Edition") {
204  deepinEdition = QString::fromUtf8(key_value.second);
205  } else if (key_value.first == "Copyright") {
206  deepinCopyright = QString::fromUtf8(key_value.second);
207  }
208 
209  if (!deepinTypeMap.isEmpty() && !deepinEdition.isEmpty() && !deepinCopyright.isEmpty()) {
210  break;
211  }
212  }
213 
214  file.close();
215 
216  const QString &deepin_type = deepinTypeMap[QString()];
217 
218  if (deepin_type.isEmpty()) {
219  deepinType = DSysInfo::UnknownDeepin;
220  } else if (deepin_type == "Desktop") {
221  deepinType = DSysInfo::DeepinDesktop;
222  } else if (deepin_type == "Professional") {
223  deepinType = DSysInfo::DeepinProfessional;
224  } else if (deepin_type == "Server") {
225  deepinType = DSysInfo::DeepinServer;
226  } else if (deepin_type == "Personal") {
227  deepinType = DSysInfo::DeepinPersonal;
228  } else {
229  deepinType = DSysInfo::UnknownDeepin;
230  }
231 }

233 bool DSysInfoPrivate::ensureOsVersion()
234 {
235 #ifndef OS_VERSION_TEST_FILE // 测试时总是重新读取文件
236  if (osBuild.A > 0)
237  return true;
238 #endif
239 
240  DDesktopEntry entry(OS_VERSION_FILE);
241  bool ok = false;
242 
243  // 先获取版本信息
244  // ABCDE.xyz
245  QString osb = entry.stringValue("OsBuild", "Version");
246  QStringList osbs = osb.split(".");
247  Q_ASSERT(osbs.size() == 2 && osbs.value(0).size() == 5);
248  uint left = osbs.value(0).trimmed().toUInt(&ok);
249  Q_ASSERT(ok);
250  if (ok) {
251  osBuild.E = left % 10;
252  left /= 10;
253  osBuild.D = left % 10;
254  left /= 10;
255  osBuild.C = left % 10; // default C is 0
256  left /= 10;
257  osBuild.B = left % 10;
258  left /= 10;
259  osBuild.A = left % 10;
260  }
261 
262  // xyz
263  osBuild.xyz = osbs.value(1).trimmed().toUInt(&ok);
264 
265  majorVersion = entry.stringValue("MajorVersion", "Version");
266  minorVersion = entry.stringValue("MinorVersion", "Version");
267 
268  switch (osBuild.D) {
269  case 7: {
270  // 家庭版使用“完整版本号编码-X.Y.Z”的形式
271  const QStringList &versionList = minorVersion.split('.');
272  if (versionList.isEmpty()) {
273  // 如果读取失败直接返回为空
274  qWarning() << "no minorVersion";
275  return false;
276  } else if (versionList.length() == 2) {
277  // Z为0
278  minVersion.X = versionList.first().toUInt();
279  minVersion.Y = versionList.last().toUInt();
280  minVersion.Z = 0;
281  } else if (versionList.length() == 3) {
282  // X.Y.Z都存在
283  minVersion.X = versionList.at(0).toUInt();
284  minVersion.Y = versionList.at(1).toUInt();
285  minVersion.Z = versionList.at(2).toUInt();
286  }
287  minVersion.type = MinVersion::X_Y_Z;
288  } break;
289 
290  case 3: {
291  // 社区版使用“完整版本号编码-A.B.C”的形式
292  bool a_bc_dMode = false;
293  const QStringList &versionList = minorVersion.split('.');
294  if (versionList.isEmpty()) {
295  // 如果读取失败直接返回为空
296  qWarning() << "no minorVersion";
297  return false;
298  } else if (versionList.length() == 1) {
299  QString modeVersion = versionList.first();
300  if (modeVersion.length() == 2) {
301  //A.B.C模式且B C 为0
302  minVersion.A = modeVersion.toUInt();
303  minVersion.B = 0;
304  minVersion.C = 0;
305  } else {
306  // A_BC_D模式
307  splitA_BC_DMode();
308  a_bc_dMode = true;
309  }
310  } else if (versionList.length() == 2) {
311  // C为0
312  minVersion.A = versionList.first().toUInt();
313  minVersion.B = versionList.last().toUInt();
314  minVersion.C = 0;
315  } else if (versionList.length() == 3) {
316  // A.B.C都存在
317  minVersion.A = versionList.at(0).toUInt();
318  minVersion.B = versionList.at(1).toUInt();
319  minVersion.C = versionList.at(2).toUInt();
320  }
321 
322  if (!a_bc_dMode)
323  minVersion.type = MinVersion::A_B_C;
324  } break;
325  default: {
326  // A-BC-D
327  ok = splitA_BC_DMode();
328  } break;
329  }
330  return ok;
331 }

333 static QString unquote(const QByteArray &value)
334 {
335  if (value.at(0) == '"' || value.at(0) == '\'') {
336  return QString::fromLatin1(value.mid(1, value.size() - 2));
337  }
338 
339  return QString::fromLatin1(value);
340 }

342 static bool readEtcFile(DSysInfoPrivate *info, const char *filename,
343  const QByteArray &idKey, const QByteArray &versionKey, const QByteArray &prettyNameKey)
344 {
345 
346  QFile file(QString::fromLatin1(filename));
347 
348  if (!file.open(QIODevice::ReadOnly)) {
349  return false;
350  }
351 
352  quint8 valid_data_count = 0;
353  char buf[1024];
354 
355  while (valid_data_count < 3) {
356  int buf_length = file.readLine(buf, sizeof(buf));
357 
358  if (buf_length < 0)
359  break;
360 
361  const QByteArray line(buf, buf_length - 1);
362 
363  if (info->productTypeString.isEmpty() && line.startsWith(idKey)) {
364  const QByteArray value(line.constData() + idKey.size());
365  info->productTypeString = unquote(value);
366  ++valid_data_count;
367  continue;
368  }
369 
370  if (info->prettyName.isEmpty() && line.startsWith(prettyNameKey)) {
371  const QByteArray value(line.constData() + prettyNameKey.size());
372  info->prettyName = unquote(value);
373  ++valid_data_count;
374  continue;
375  }
376 
377  if (info->productVersion.isEmpty() && line.startsWith(versionKey)) {
378  const QByteArray value(line.constData() + versionKey.size());
379  info->productVersion = unquote(value);
380  ++valid_data_count;
381  continue;
382  }
383  }
384 
385  file.close();
386 
387  return valid_data_count != 0;
388 }

404 void DSysInfoPrivate::ensureReleaseInfo()
405 {
406  if (productType >= 0) {
407  return;
408  }
409 
410 #ifdef Q_OS_LINUX
411  readOsRelease(this);
412  readLsbRelease(this);
413 
414  if (productTypeString.isEmpty()) {
415  productType = DSysInfo::UnknownType;
416  } else {
417  switch (productTypeString.at(0).unicode()) {
418  case 'd':
419  case 'D':
420  if (productTypeString.compare("deepin", Qt::CaseInsensitive) == 0) {
421  productType = DSysInfo::Deepin;
422  } else if (productTypeString.compare("debian", Qt::CaseInsensitive) == 0) {
423  productType = DSysInfo::Debian;
424  }
425  break;
426  case 'a':
427  case 'A':
428  if (productTypeString.compare("arch", Qt::CaseInsensitive) == 0)
429  productType = DSysInfo::ArchLinux;
430  break;
431  case 'c':
432  case 'C':
433  if (productTypeString.compare("centos", Qt::CaseInsensitive) == 0)
434  productType = DSysInfo::CentOS;
435  break;
436  case 'f':
437  case 'F':
438  if (productTypeString.compare("fedora", Qt::CaseInsensitive) == 0)
439  productType = DSysInfo::Fedora;
440  break;
441  case 'l':
442  case 'L':
443  if (productTypeString.compare("linuxmint", Qt::CaseInsensitive) == 0)
444  productType = DSysInfo::LinuxMint;
445  break;
446  case 'm':
447  case 'M':
448  if (productTypeString.compare("manjaro", Qt::CaseInsensitive) == 0)
449  productType = DSysInfo::Manjaro;
450  break;
451  case 'o':
452  case 'O':
453  if (productTypeString.compare("opensuse", Qt::CaseInsensitive) == 0)
454  productType = DSysInfo::openSUSE;
455  break;
456  case 's':
457  case 'S':
458  if (productTypeString.compare("sailfishos", Qt::CaseInsensitive) == 0)
459  productType = DSysInfo::SailfishOS;
460  break;
461  case 'u':
462  case 'U':
463  if (productTypeString.compare("ubuntu", Qt::CaseInsensitive) == 0) {
464  productType = DSysInfo::Ubuntu;
465  } else if (productTypeString.compare("uos", Qt::CaseInsensitive) == 0 || productTypeString.compare("UnionTech OS", Qt::CaseInsensitive) == 0) {
466  productType = DSysInfo::Uos;
467  }
468  break;
469  default:
470  productType = DSysInfo::UnknownType;
471  break;
472  }
473  }
474 #endif

484 QMap<QString, QString> DSysInfoPrivate::parseInfoFile(QFile &file)
485 {
486  char buf[1024];
487  qint64 lineLength = 0;
488  QMap<QString, QString> map;
489  do {
490  lineLength = file.readLine(buf, sizeof(buf));
491  QString s(buf);
492  if (s.contains(':')) {
493  QStringList list = s.split(':');
494  if (list.size() == 2) {
495  map.insert(list.first().trimmed(), list.back().trimmed());
496  }
497  }
498  } while (lineLength >= 0);
499  return map;
500 }

1062  QString pkname = oneValue.toObject().value("pkname").toString();

1064

1066  keyName = kname;
1067  }

1073  keyName = deviceParentAndSizeMap[keyName].first;
