534 void DSliderPrivate::updtateTool(int value)
535 {
536  Q_Q(DSlider);
537 
538  if (!tipvalue)
539  return;
540 
541  int min = q->minimum();
542  int max = q->maximum();
543  QStyleOptionSlider opt;
544  const QRectF rectHandle = q->style()->subControlRect(QStyle::CC_Slider, &opt, QStyle::SC_SliderHandle, slider); //滑块
545 
546  double x = ((value - min) * 1.0 / (max -min)) * (slider->rect().width() - rectHandle.width());
547  double y = slider->y() + slider->height();
548 
549  //x是以实际滑槽为起始坐标，而非DSlider为参考坐标（实际滑槽长度 == Slider长度 - 滑块长度）
550  //气泡位置
551  if (slider->invertedAppearance()) {
552  tipvalue->move(slider->width() - x - (tipvalue->width() + rectHandle.width()) / 2.0 , y);
553  } else {
554  tipvalue->move(x + (rectHandle.width() - tipvalue->width()) / 2.0, y);
555  }
556 
557  int shadowMarge = DStyle::pixelMetric(q->style(), DStyle::PM_FloatingWidgetShadowMargins) - 1;
558 
559  //气泡贴边
560  if (tipvalue->x() < 0) {
561  tipvalue->move(0 - shadowMarge, y);
562  }
563 
564  if (tipvalue->x() + tipvalue->width() > slider->width()) {
565  tipvalue->move(slider->width() - tipvalue->width() + shadowMarge, y);
566  }
567 
568  tipvalue->raise();
569  tipvalue->adjustSize();
570 }

801 void SliderStrip::paintEvent(QPaintEvent *event)
802 {
803  Q_UNUSED(event)
804  QPainter pa(this);
805  auto dpa = DPaletteHelper::instance()->palette(this);
806  QPen penLine = QPen(dpa.color(DPalette::ObviousBackground), 1);
807  QPen penNumber = QPen(dpa.textTips(), 1);
808 
809  int tickSize = fontMetrics().height() / 2;
810  int offsetSize = style()->pixelMetric(QStyle::PM_SliderLength, nullptr, this) / 2;
811  int margin = DStyle::pixelMetric(style(), DStyle::PM_FocusBorderSpacing) + \
812  DStyle::pixelMetric(style(), DStyle::PM_FocusBorderSpacing);
813  qreal startX = 0, startY = 0, endX = 0, endY = 0;
814  int width = this->width(), height = this->height(); //slider宽高
815  int paragraph = scaleInfo.count(); //刻度个数
816  qreal average = 0; //每一段的距离
817  qreal textPos = 0; //字体位置
818 
819  QSlider* slider = static_cast<DSlider *>(this->parent())->slider();
820  for (int i = 0; i < list.count(); i++) {
821  qreal percentage = (list.at(i) - slider->minimum()) * 1.0 / (slider->maximum() - slider->minimum());
822  pa.setPen(penLine);
823 
824  if (orient == Qt::Horizontal) {
825  qreal sliderX = percentage * width;
826  if (tick == QSlider::TicksAbove)
827  pa.drawLine(QPointF(sliderX + offsetSize, height), QPointF(sliderX + offsetSize, height - tickSize));
828  if (tick == QSlider::TicksBelow)
829  pa.drawLine(QPointF(sliderX + offsetSize, 0), QPointF(sliderX + offsetSize, tickSize));
830  } else {
831  qreal sliderX = percentage * height;
832  if (tick == QSlider::TicksLeft)
833  pa.drawLine(QPointF(width - tickSize, height - (sliderX + offsetSize)), QPointF(width, height - (sliderX + offsetSize)));
834  if (tick == QSlider::TicksRight)
835  pa.drawLine(QPointF(0, height - (sliderX + offsetSize)), QPointF(tickSize, height - (sliderX + offsetSize)));
836  }
837  }
838 
839  if (scaleInfo.isEmpty())
840  return;
841 
842  if (orient == Qt::Horizontal) {
843  width -= 2 * offsetSize + margin * 2;
844  startX += offsetSize + margin;
845  endX = startX;
846  average = width / static_cast<qreal>(paragraph - 1);
847 
848  if (tick == QSlider::TicksBelow) {
849  endY = tickSize;
850  textPos = endY;
851  } else if (tick == QSlider::TicksAbove) {
852  startY = height;
853  endY = startY - tickSize;
854  }
855 
856  pa.setPen(penLine);
857  pa.drawLine(QPointF(startX, startY), QPointF(endX, endY));
858  pa.setPen(penNumber);
859  pa.drawText(QRectF(endX, textPos, width, height - tickSize), Qt::AlignLeft, scaleInfo[0]);
860 
861  for (int i = 1; i < paragraph - 1; i++) {
862  startX += average;
863  endX = startX;
864  pa.setPen(penLine);
865  pa.drawLine(QPointF(startX, startY), QPointF(endX, endY));
866  pa.setPen(penNumber);
867  pa.drawText(QRectF(endX - width / 2, textPos, width, height - tickSize), Qt::AlignHCenter, scaleInfo[i]);
868  }
869 
870  if (paragraph > 1) {
871  startX += average;
872  endX = startX;
873  pa.setPen(penLine);
874  pa.drawLine(QPointF(startX, startY), QPointF(endX, endY));
875  pa.setPen(penNumber);
876  pa.drawText(QRectF(endX - width, textPos, width, height - tickSize), Qt::AlignRight, scaleInfo[paragraph - 1]);
877  }
878  } else {
879  startY = offsetSize + margin;
880  height -= offsetSize * 2 + margin * 2;
881  endY = startY;
882  average = height / static_cast<qreal>(paragraph - 1);
883  Qt::Alignment text_flags = Qt::AlignVCenter;
884 
885  if (tick == QSlider::TicksLeft) {
886  startX = width;
887  endX = startX - tickSize;
888  textPos = 0;
889  text_flags |= Qt::AlignRight;
890  } else if (tick == QSlider::TicksRight) {
891  endX = tickSize;
892  textPos = endX;
893  text_flags |= Qt::AlignLeft;
894  }
895 
896  pa.setPen(penLine);
897  pa.drawLine(QPointF(startX, startY), QPointF(endX, endY));
898  pa.setPen(penNumber);
899  pa.drawText(QRectF(textPos, endY - average / 2 + offsetSize / 2, width - tickSize, average), text_flags, scaleInfo[0]);
900 
901  for (int i = 1; i < paragraph - 1; i++) {
902  startY += average;
903  endY = startY;
904  pa.setPen(penLine);
905  pa.drawLine(QPointF(startX, startY), QPointF(endX, endY));
906  pa.setPen(penNumber);
907  pa.drawText(QRectF(textPos, endY - average / 2, width - tickSize, average), text_flags, scaleInfo[i]);
908  }
909 
910  if (paragraph > 1) {
911  startY += average;
912  endY = startY;
913  pa.setPen(penLine);
914  pa.drawLine(QPointF(startX, startY), QPointF(endX, endY));
915  pa.setPen(penNumber);
916  pa.drawText(QRectF(textPos, endY - average / 2 - offsetSize / 2, width - tickSize, average), text_flags, scaleInfo[paragraph - 1]);
917  }
918  }
919 }
